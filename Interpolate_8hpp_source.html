93  template <typename Metavariables, typename ParallelComponent>
94  void operator()(const TimeStepId& time_id, const Mesh<VolumeDim>& mesh,
95  const typename Tensors::type&... tensors,
96  Parallel::GlobalCache<Metavariables>& cache,
97  const ElementId<VolumeDim>& array_index,
98  const ParallelComponent* const /*meta*/) const noexcept {
99  Variables<tmpl::list<Tensors...>> interp_vars(mesh.number_of_grid_points());
100  const auto copy_to_variables = [&interp_vars](const auto tensor_tag_v,
101  const auto& tensor) noexcept {
102  using tensor_tag = tmpl::type_from<decltype(tensor_tag_v)>;
103  get<tensor_tag>(interp_vars) = tensor;
104  return 0;
105  };
106  (void) copy_to_variables; // GCC warns unused variable if Tensors is empty.
107  expand_pack(copy_to_variables(tmpl::type_<Tensors>{}, tensors)...);
108 
109  // Send volume data to the Interpolator, to trigger interpolation.
110  auto& interpolator =
111  *::Parallel::get_parallel_component<Interpolator<Metavariables>>(cache)
112  .ckLocalBranch();
113  Parallel::simple_action<Actions::InterpolatorReceiveVolumeData>(
114  interpolator, time_id, ElementId<VolumeDim>(array_index), mesh,
115  interp_vars);
116 
117  // Tell the interpolation target that it should interpolate.
118  auto& target = Parallel::get_parallel_component<
119  InterpolationTarget<Metavariables, InterpolationTargetTag>>(cache);
120  Parallel::simple_action<
121  Actions::AddTemporalIdsToInterpolationTarget<InterpolationTargetTag>>(
122  target, std::vector<TimeStepId>{time_id});
123  }
124 };
