51 /// \ingroup DataBoxTagsGroup
52 /// \ingroup ComputationalDomainGroup
53 /// \brief Compute tag for representing items computed on a set of interfaces.
54 /// Can be retrieved using `Tags::Interface<DirectionsTag, Tag>`
55 ///
56 /// The computed object will be a map from a ::Direction to the item type of
57 /// `Tag` (i.e. `Tag::type`), with the set of directions being those produced by
58 /// `DirectionsTag`. `Tag::function` will be applied separately to the data on
59 /// each interface. If some of the compute item's inputs (i.e.
60 /// `Tag::argument_tags`) should be taken from the volume rather than the
61 /// interface, they should be specified in the typelist `Tag::volume_tags`.
62 ///
63 /// \tparam DirectionsTag the simple tag labeling the set of Directions
64 /// \tparam Tag the compute tag to apply on each interface
65 template <typename DirectionsTag, typename Tag>
66 struct InterfaceCompute : Interface<DirectionsTag, typename Tag::base>,
67  db::ComputeTag {
68  static_assert(db::is_simple_tag_v<DirectionsTag>);
69  static_assert(db::is_compute_tag_v<Tag>,
70  "Cannot use a non compute item as an interface compute item.");
71  using base = Interface<DirectionsTag, typename Tag::base>;
72  static constexpr size_t volume_dim = DirectionsTag::volume_dim;
73  using return_type =
74  std::unordered_map<::Direction<volume_dim>, typename Tag::type>;
75 
76  using forwarded_argument_tags =
77  InterfaceHelpers_detail::get_interface_argument_tags<Tag, DirectionsTag>;
78  using argument_tags =
79  tmpl::push_front<forwarded_argument_tags, DirectionsTag>;
80 
81  using volume_tags = get_volume_tags<Tag>;
82  static_assert(
83  tmpl::size<tmpl::list_difference<volume_tags, argument_tags>>::value == 0,
84  "volume_tags contains tags not in argument_tags");
85 
86  template <typename... ArgTypes>
87  static constexpr void function(
88  const gsl::not_null<return_type*> result,
89  const std::unordered_set<::Direction<volume_dim>>& directions,
90  const ArgTypes&... args) noexcept {
91  for (const auto& direction : directions) {
92  Interface_detail::evaluate_compute_item<Tag, volume_tags>(
93  make_not_null(&(*result)[direction]), direction,
94  forwarded_argument_tags{}, args...);
95  }
96  }
97 };

99 /// \ingroup DataBoxTagsGroup
100 /// \ingroup ComputationalDomainGroup
101 /// \brief Compute tag for representing a compute item that slices data from
102 /// the volume to a set of interfaces.
103 ///
104 /// The computed object will be a map from a ::Direction to the item type of
105 /// `Tag` (i.e. `Tag::type`), with the set of directions being those produced by
106 /// `DirectionsTag`. `Tag::type` must be a `Tensor` or a `Variables`.
107 /// Retrievable from the DataBox using `Tags::Interface<DirectionsTag, Tag>`
108 ///
109 /// \requires `Tag` correspond to a `Tensor` or a `Variables`
110 ///
111 /// \tparam DirectionsTag the simple tag labeling the set of Directions
112 /// \tparam Tag the simple tag for the volume data to be sliced
113 template <typename DirectionsTag, typename Tag>
114 struct Slice : Interface<DirectionsTag, Tag>, db::ComputeTag {
115  static_assert(db::is_simple_tag_v<DirectionsTag>);
116  static_assert(db::is_simple_tag_v<Tag>);
117  using base = Interface<DirectionsTag, Tag>;
118  static constexpr size_t volume_dim = DirectionsTag::volume_dim;
119 
120  using return_type =
121  std::unordered_map<::Direction<volume_dim>, typename Tag::type>;
122 
123  static constexpr void function(
124  const gsl::not_null<return_type*> sliced_vars,
125  const ::Mesh<volume_dim>& mesh,
126  const std::unordered_set<::Direction<volume_dim>>& directions,
127  const typename Tag::type& variables) noexcept {
128  for (const auto& direction : directions) {
129  data_on_slice(make_not_null(&((*sliced_vars)[direction])), variables,
130  mesh.extents(), direction.dimension(),
131  index_to_slice_at(mesh.extents(), direction));
132  }
133  }
134 
135  using argument_tags = tmpl::list<Mesh<volume_dim>, DirectionsTag, Tag>;
136  using volume_tags = tmpl::list<Mesh<volume_dim>, Tag>;
137 };

160 /// \ingroup DataBoxTagsGroup
161 /// \ingroup ComputationalDomainGroup
162 /// Computes the `VolumeDim-1` dimensional mesh on an interface from the volume
163 /// mesh. `Tags::InterfaceCompute<Dirs, InterfaceMesh<VolumeDim>>` is
164 /// retrievable as `Tags::Interface<Dirs, Mesh<VolumeDim>>` from the DataBox.
165 template <size_t VolumeDim>
166 struct InterfaceMesh : db::ComputeTag, Tags::Mesh<VolumeDim - 1> {
167  using base = Tags::Mesh<VolumeDim - 1>;
168  using return_type = typename base::type;
169  using argument_tags = tmpl::list<Direction<VolumeDim>, Mesh<VolumeDim>>;
170  static constexpr auto function(
171  const gsl::not_null<return_type*> mesh,
172  const ::Direction<VolumeDim>& direction,
173  const ::Mesh<VolumeDim>& volume_mesh) noexcept {
174  *mesh = volume_mesh.slice_away(direction.dimension());
175  }
176  using volume_tags = tmpl::list<Mesh<VolumeDim>>;
177 };
