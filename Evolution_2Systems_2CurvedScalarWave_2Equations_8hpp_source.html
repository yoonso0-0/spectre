28 /*!
29  * \brief Compute the time derivative of the evolved variables of the
30  * first-order scalar wave system on a curved background.
31  *
32  * The evolution equations for the first-order scalar wave system are given by
33  * \cite Holst2004wt :
34  *
35  * \f{align}
36  * \partial_t\Psi = & (1 + \gamma_1) \beta^k \partial_k \Psi - \alpha \Pi -
37  * \gamma_1 \beta^k \Phi_k \\
38  *
39  * \partial_t\Pi = & - \alpha \gamma^{ij}\partial_i\Phi_j + \beta^k \partial_k
40  * \Pi + \gamma_1 \gamma_2 \beta^k \partial_k \Psi + \alpha \Gamma^i -
41  * \gamma^{ij} \Phi_i \partial_j \alpha
42  * + \alpha K \Pi - \gamma_1 \gamma_2 \beta^k \Phi_k\\
43  *
44  * \partial_t\Phi_i = & - \alpha \partial_i \Pi + \beta^k \partial_k \Phi +
45  * \gamma_2 \alpha \partial_i \Psi - \Pi
46  * \partial_i \alpha + \Phi_k \partial_i \beta^j - \gamma_2 \alpha \Phi_i\\
47  * \f}
48  *
49  * where \f$\Psi\f$ is the scalar field, \f$\Pi\f$ is the
50  * conjugate momentum to \f$\Psi\f$, \f$\Phi_i=\partial_i\Psi\f$ is an
51  * auxiliary variable, \f$\alpha\f$ is the lapse, \f$\beta^k\f$ is the shift,
52  * \f$ \gamma_{ij} \f$ is the spatial metric, \f$ K \f$ is the trace of the
53  * extrinsic curvature, and \f$ \Gamma^i \f$ is the trace of the Christoffel
54  * symbol of the second kind. \f$\gamma_1, \gamma_2\f$ are constraint damping
55  * parameters.
56  */
57 template <size_t Dim>
58 struct ComputeDuDt {
59  public:
60  template <template <class> class StepPrefix>
61  using return_tags =
62  tmpl::list<StepPrefix<Pi>, StepPrefix<Phi<Dim>>, StepPrefix<Psi>>;
63 
64  using argument_tags = tmpl::list<
65  Pi, Phi<Dim>, ::Tags::deriv<Psi, tmpl::size_t<Dim>, Frame::Inertial>,
66  ::Tags::deriv<Pi, tmpl::size_t<Dim>, Frame::Inertial>,
67  ::Tags::deriv<Phi<Dim>, tmpl::size_t<Dim>, Frame::Inertial>,
68  gr::Tags::Lapse<DataVector>,
69  gr::Tags::Shift<Dim, Frame::Inertial, DataVector>,
70  ::Tags::deriv<gr::Tags::Lapse<DataVector>, tmpl::size_t<Dim>,
71  Frame::Inertial>,
72  ::Tags::deriv<gr::Tags::Shift<Dim, Frame::Inertial, DataVector>,
73  tmpl::size_t<Dim>, Frame::Inertial>,
74  gr::Tags::InverseSpatialMetric<Dim, Frame::Inertial, DataVector>,
75  gr::Tags::TraceSpatialChristoffelSecondKind<Dim, Frame::Inertial,
76  DataVector>,
77  gr::Tags::TraceExtrinsicCurvature<DataVector>, Tags::ConstraintGamma1,
78  Tags::ConstraintGamma2>;

98 /*!
99  * \brief Compute fluxes for the scalar-wave system in curved spacetime.
100  *
101  * \details The expressions for fluxes is obtained from \cite Holst2004wt by
102  * taking the principal part of equations 15, 23, and 24, and replacing
103  * derivatives \f$ \partial_k \f$ with the unit normal \f$ n_k \f$ (c.f. Gauss'
104  * theorem). This gives:
105  *
106  * \f{align*}
107  * F(\psi) &= -(1 + \gamma_1) \beta^k n_k \psi \\
108  * F(\Pi) &= - \beta^k n_k \Pi + \alpha g^{ki}n_k \Phi_{i}
109  * - \gamma_1 \gamma_2 \beta^k n_k \Psi \\
110  * F(\Phi_{i}) &= - \beta^k n_k \Phi_{i} + \alpha n_i \Pi - \gamma_2 \alpha n_i
111  * \psi \f}
112  *
113  * where \f$\psi\f$ is the scalar field, \f$\Pi\f$ is its conjugate
114  * momentum, \f$ \Phi_{i} \f$ is an auxiliary field defined as the spatial
115  * derivative of \f$\psi\f$, \f$\alpha\f$ is the lapse, \f$ \beta^k \f$ is the
116  * shift, \f$ g^{ki} \f$ is the inverse spatial metric, and \f$ \gamma_1,
117  * \gamma_2\f$ are constraint damping parameters. Note that the last term in
118  * \f$F(\Pi)\f$ will be identically zero, as it is necessary to set
119  * \f$\gamma_1\gamma_2=0\f$ in order to make this first-order formulation of the
120  * curved scalar wave system symmetric hyperbolic.
121  */
122 template <size_t Dim>
123 struct ComputeNormalDotFluxes {
124  public:
125  using argument_tags =
126  tmpl::list<Pi, Phi<Dim>, Psi, Tags::ConstraintGamma1,
127  Tags::ConstraintGamma2, gr::Tags::Lapse<>,
128  gr::Tags::Shift<Dim>, gr::Tags::InverseSpatialMetric<Dim>,
129  ::Tags::Normalized<domain::Tags::UnnormalizedFaceNormal<
130  Dim, Frame::Inertial>>>;
131 
132  static void apply(
133  gsl::not_null<Scalar<DataVector>*> pi_normal_dot_flux,
134  gsl::not_null<tnsr::i<DataVector, Dim>*> phi_normal_dot_flux,
135  gsl::not_null<Scalar<DataVector>*> psi_normal_dot_flux,
136  const Scalar<DataVector>& pi, const tnsr::i<DataVector, Dim>& phi,
137  const Scalar<DataVector>& psi, const Scalar<DataVector>& gamma1,
138  const Scalar<DataVector>& gamma2, const Scalar<DataVector>& lapse,
139  const tnsr::I<DataVector, Dim>& shift,
140  const tnsr::II<DataVector, Dim>& inverse_spatial_metric,
141  const tnsr::i<DataVector, Dim>& interface_unit_normal) noexcept;
142 };

144 /*!
145  * \ingroup NumericalFluxesGroup
146  * \brief Computes the upwind flux for scalar-waves in curved spacetime.
147  *
148  * \details The upwind flux is given by \cite Teukolsky2015ega :
149  *
150  * \f[
151  * G = S\left(\Lambda^+ S^{-1} U^{\rm int}
152  * + \Lambda^- S^{-1} U^{\rm ext}\right)
153  * = S \left(\Lambda^+ \hat{U}^{\rm int}
154  * + \Lambda^- \hat{U}^{\rm ext}\right),
155  * \f]
156  *
157  * where
158  *
159  * - \f$G\f$ is the numerical upwind flux dotted with the interface normal;
160  * - \f$U\f$ is a vector of all evolved variables;
161  * - \f$S\f$ is a matrix whose columns are the eigenvectors of the
162  * characteristic matrix for the evolution system. It maps the
163  * evolved variables to characteristic variables \f$\hat{U}\f$, s.t.
164  * \f$\hat{U} := S^{-1}\cdot U\f$; and
165  * - \f$\Lambda^\pm\f$ are diagonal matrices containing
166  * positive / negative eigenvalues of the same matrix as its elements.
167  *
168  * The superscripts \f${\rm int}\f$ and \f${\rm ext}\f$ on \f$U\f$ indicate
169  * that the corresponding set of variables at the element interface have been
170  * taken from the _interior_ or _exterior_ of the element. Exterior of the
171  * element is naturally the interior of its neighboring element. Therefore,
172  * \f$\hat{U}^{\rm int}:= S^{-1}U^{\rm int}\f$ are the characteristic variables
173  * at the element interface computed using evolved variables from the interior
174  * of the element, and \f$\hat{U}^{\rm ext}= S^{-1}U^{\rm ext}\f$ are the
175  * same computed from evolved variables taken from the element exterior, i.e.
176  * the neighboring element.
177  * The sign of characteristic speeds indicates the direction of propagation of
178  * the corresponding characteristic field with respect to the interface normal
179  * that the field has been computed along (with negative speeds indicating
180  * incoming characteristics, and with positive speeds indicating outgoing
181  * characteristics). Therefore, \f$\Lambda^+\f$ contains characterstic speeds
182  * for variables that are outgoing from the element at the interface, and
183  * \f$\Lambda^-\f$ contains characteristic speeds for variables that are
184  * incoming to the element at the interface. An ambiguity naturally arises
185  * as to which set of evolved variables (''interior'' or ''exterior'') to use
186  * when computing these speeds. We compute both and use their average, i.e.
187  * \f$\lambda^{\rm avg} = (\lambda^{\rm int} + \lambda^{\rm ext})/2\f$, to
188  * populate \f$\Lambda^\pm\f$.
189  *
190  *
191  * This function computes the upwind flux as follows:
192  * -# Computes internal and external characteristic variables and speeds using
193  * evolved variables from
194  * both the element interior and exterior.
195  * -# Computes the average characteristic speeds and constructs
196  * \f$\Lambda^{\pm}\f$.
197  * -# Computes the upwind flux as a weigted sum of external and internal
198  * characteristic variables
199  *
200  * \f[
201  * G = S\left(w_{\rm ext} \hat{U}^{\rm ext}
202  * + w_{\rm int} \hat{U}^{\rm int}\right),
203  * \f]
204  *
205  * with weights \f$w_{\rm ext} = \Theta(-\Lambda)\cdot\Lambda\f$, and
206  * \f$w_{\rm int} = \Theta(\Lambda)\cdot\Lambda\f$, where \f$\Theta\f$ is the
207  * step function centered at zero, \f$\Lambda = \Lambda^+ + \Lambda^-\f$, and
208  * the dot operator \f$(\cdot)\f$ indicates an element-wise product.
209  *
210  * \warning With the averaging of characteristic speeds, this flux does not
211  * satisfy the generalized Rankine-Hugoniot conditions
212  * \f${G}^{\rm int}(\hat{U}^{\rm int}, \hat{U}^{\rm ext})
213  * = - {G}^{\rm ext}(\hat{U}^{\rm ext}, \hat{U}^{\rm int})\f$,
214  * which enforces that the flux leaving the element is equal to the flux
215  * entering the neighboring element, and vice versa. This condition is
216  * important for a well-balanced scheme, and so please use this flux with
217  * caution.
218  */
