63 /*!
64  * \brief Computes the time derivative for a DG time step.
65  *
66  * Computes the volume fluxes, the divergence of the fluxes and all additional
67  * interior contributions to the time derivatives (both nonconservative products
68  * and source terms). The internal mortar data is also computed.
69  *
70  * The general first-order hyperbolic evolution equation solved for conservative
71  * systems is:
72  *
73  * \f{align*}{
74  * \frac{\partial u_\alpha}{\partial \hat{t}}
75  * + \partial_{i}
76  * \left(F^i_\alpha - v^i_g u_\alpha\right)
77  * = S_\alpha-u_\alpha\partial_i v^i_g,
78  * \f}
79  *
80  * where \f$F^i_{\alpha}\f$ are the fluxes when the mesh isn't moving,
81  * \f$v^i_g\f$ is the velocity of the mesh, \f$u_{\alpha}\f$ are the evolved
82  * variables, \f$S_{\alpha}\f$ are the source terms, and \f$\hat{t}\f$ is the
83  * time in the logical frame. For evolution equations that do not have any
84  * fluxes and only nonconservative products we evolve:
85  *
86  * \f{align*}{
87  * \frac{\partial u_\alpha}{\partial \hat{t}}
88  * +\left(B^i_{\alpha\beta}-v^i_g \delta_{\alpha\beta}
89  * \right)\partial_{i}u_\beta = S_\alpha.
90  * \f}
91  *
92  * Finally, for equations with both conservative terms and nonconservative
93  * products we use:
94  *
95  * \f{align*}{
96  * \frac{\partial u_\alpha}{\partial \hat{t}}
97  * + \partial_{i}
98  * \left(F^i_\alpha - v^i_g u_\alpha\right)
99  * +B^i_{\alpha\beta}\partial_{i}u_\beta
100  * = S_\alpha-u_\alpha\partial_i v^i_g,
101  * \f}
102  *
103  * where \f$B^i_{\alpha\beta}\f$ is the matrix for the nonconservative products.
104  *
105  * ### Volume Terms
106  *
107  * The mesh velocity is added to the flux automatically if the mesh is moving.
108  * That is,
109  *
110  * \f{align*}{
111  * F^i_{\alpha}\to F^i_{\alpha}-v^i_{g} u_{\alpha}
112  * \f}
113  *
114  * The source terms are also altered automatically by adding:
115  *
116  * \f{align*}{
117  * -u_\alpha \partial_i v^i_g,
118  * \f}
119  *
120  * For systems with equations that only contain nonconservative products, the
121  * following mesh velocity is automatically added to the time derivative:
122  *
123  * \f{align*}{
124  * v^i_g \partial_i u_\alpha,
125  * \f}
126  *
127  * \note The term is always added in the `Frame::Inertial` frame, and the plus
128  * sign arises because we add it to the time derivative.
129  *
130  * Here are examples of the `TimeDerivative` struct used to compute the volume
131  * time derivative. This struct is what the type alias
132  * `System::compute_volume_time_derivative` points to. The time derivatives are
133  * as `gsl::not_null` first, then the temporary tags as `gsl::not_null`,
134  * followed by the `argument_tags`. These type aliases are given by
135  *
136  * \snippet ComputeTimeDerivativeImpl.tpp dt_ta
137  *
138  * for the examples. For a conservative system without primitives the `apply`
139  * function would look like
140  *
141  * \snippet ComputeTimeDerivativeImpl.tpp dt_con
142  *
143  * For a nonconservative system it would be
144  *
145  * \snippet ComputeTimeDerivativeImpl.tpp dt_nc
146  *
147  * And finally, for a mixed conservative-nonconservative system with primitive
148  * variables
149  *
150  * \snippet ComputeTimeDerivativeImpl.tpp dt_mp
151  *
152  * Uses:
153  * - System:
154  * - `variables_tag`
155  * - `flux_variables`
156  * - `gradient_variables`
157  * - `compute_volume_time_derivative_terms`
158  *
159  * - DataBox:
160  * - Items in `system::compute_volume_time_derivative_terms::argument_tags`
161  * - `domain::Tags::MeshVelocity<Metavariables::volume_dim>`
162  * - `Metavariables::system::variables_tag`
163  * - `Metavariables::system::flux_variables`
164  * - `Metavariables::system::gradient_variables`
165  * - `domain::Tags::DivMeshVelocity`
166  * - `DirectionsTag`,
167  * - Required interface items for `Metavariables::system::normal_dot_fluxes`
168  *
169  * DataBox changes:
170  * - Adds: nothing
171  * - Removes: nothing
172  * - Modifies:
173  * - db::add_tag_prefix<Tags::Flux, variables_tag,
174  * tmpl::size_t<system::volume_dim>, Frame::Inertial>
175  * - `Tags::dt<system::variable_tags>`
176  * - Tags::Interface<
177  * DirectionsTag, db::add_tag_prefix<Tags::NormalDotFlux, variables_tag>>
178  * - `Tags::Mortars<typename BoundaryScheme::mortar_data_tag, VolumeDim>`
179  *
180  * ### Internal Boundary Terms
181  *
182  * Internal boundary terms are computed from the `System::boundary_correction`
183  * type alias. This type alias must point to a base class with
184  * `creatable_classes`. Each concrete boundary correction must specify:
185  *
186  * - type alias template `dg_package_field_tags`. These are what will be
187  * returned by `gsl::not_null` from the `dg_package_data` member function.
188  *
189  * - type alias template `dg_package_temporary_tags`. These are temporary tags
190  * that are projected to the face and then passed to the `dg_package_data`
191  * function.
192  *
193  * - type alias template `dg_package_primitive_tags`. These are the primitive
194  * variables (if any) that are projected to the face and then passed to
195  * `dg_package_data`.
196  *
197  * - type alias template `dg_package_volume_tags`. These are tags that are not
198  * projected to the interface and are retrieved directly from the `DataBox`.
199  * The equation of state for hydrodynamics systems is an example of what would
200  * be a "volume tag".
201  *
202  * A `static constexpr bool need_normal_vector` must be specified. If `true`
203  * then the normal vector is computed from the normal covector. This is
204  * currently not implemented.
205  *
206  * The `dg_package_data` function takes as arguments `gsl::not_null` of the
207  * `dg_package_data_field_tags`, then the projected evolved variables, the
208  * projected fluxes, the projected temporaries, the projected primitives, the
209  * unit normal covector, mesh velocity, normal dotted into the mesh velocity,
210  * the `volume_tags`, and finally the `dg::Formulation`. The `dg_package_data`
211  * function must compute all ingredients for the boundary correction, including
212  * mesh-velocity-corrected characteristic speeds. However, the projected fluxes
213  * passed in are \f$F^i - u v^i_g\f$ (the mesh velocity term is already
214  * included). The `dg_package_data` function must also return a `double` that is
215  * the maximum absolute characteristic speed over the entire face. This will be
216  * used for checking that the time step doesn't violate the CFL condition.
217  *
218  * Here is an example of the type aliases and `bool`:
219  *
220  * \snippet ComputeTimeDerivativeImpl.tpp bt_ta
221  *
222  * The normal vector requirement is:
223  *
224  * \snippet ComputeTimeDerivativeImpl.tpp bt_nnv
225  *
226  * For a conservative system with primitive variables and using the `TimeStepId`
227  * as a volume tag the `dg_package_data` function looks like:
228  *
229  * \snippet ComputeTimeDerivativeImpl.tpp bt_cp
230  *
231  * For a mixed conservative-nonconservative system with primitive variables and
232  * using the `TimeStepId` as a volume tag the `dg_package_data` function looks
233  * like:
234  *
235  * \snippet ComputeTimeDerivativeImpl.tpp bt_mp
236  *
237  * Uses:
238  * - System:
239  * - `boundary_correction`
240  * - `variables_tag`
241  * - `flux_variables`
242  * - `gradients_tags`
243  * - `compute_volume_time_derivative`
244  * - `has_primitive_and_conservative_vars`
245  * - `primitive_variables_tag` if system has primitive variables
246  *
247  * - DataBox:
248  * - `domain::Tags::Element<Dim>`
249  * - `domain::Tags::Mesh<Dim>`
250  * - `evolution::dg::Tags::MortarMesh<Dim>`
251  * - `evolution::dg::Tags::MortarSize<Dim>`
252  * - `evolution::dg::Tags::MortarData<Dim>`
253  * - `Tags::TimeStepId`
254  * - \code{.cpp}
255  * domain::Tags::Interface<domain::Tags::InternalDirections<Dim>,
256  * domain::Tags::Mesh<Dim - 1>>
257  * \endcode
258  * - \code{.cpp}
259  * domain::Tags::Interface<
260  * domain::Tags::InternalDirections<Dim>,
261  * ::Tags::Normalized<
262  * domain::Tags::UnnormalizedFaceNormal<Dim, Frame::Inertial>>>
263  * \endcode
264  * - \code{.cpp}
265  * domain::Tags::Interface<
266  * domain::Tags::InternalDirections<Dim>,
267  * domain::Tags::MeshVelocity<Dim, Frame::Inertial>>
268  * \endcode
269  * - `Metavariables::system::variables_tag`
270  * - `Metavariables::system::flux_variables`
271  * - `Metavariables::system::primitive_tags` if exists
272  * - `Metavariables::system::boundary_correction::dg_package_data_volume_tags`
273  *
274  * DataBox changes:
275  * - Adds: nothing
276  * - Removes: nothing
277  * - Modifies:
278  * - `evolution::dg::Tags::MortarData<Dim>`
279  */

301  /*
302  * Computes the volume terms for a discontinuous Galerkin scheme.
303  *
304  * The function does the following (in order):
305  *
306  * 1. Compute the partial derivatives of the `System::gradient_variables`.
307  *
308  * The partial derivatives are needed in the nonconservative product terms
309  * of the evolution equations. Any variable whose evolution equation does
310  * not contain a flux must contain a nonconservative product and the
311  * variable must be listed in the `System::gradient_variables` type alias.
312  * The partial derivatives are also needed for adding the moving mesh terms
313  * to the equations that do not have a flux term.
314  *
315  * 2. The volume time derivatives are calculated from
316  * `System::compute_volume_time_derivative_terms`
317  *
318  * The source terms and nonconservative products are contributed directly
319  * to the `dt_vars` arguments passed to the time derivative function, while
320  * the volume fluxes are computed into the `volume_fluxes` arguments. The
321  * divergence of the volume fluxes will be computed and added to the time
322  * derivatives later in the function.
323  *
324  * 3. If the mesh is moving the appropriate mesh velocity terms are added to
325  * the equations.
326  *
327  * For equations with fluxes this means that \f$-v^i_g u_\alpha\f$ is
328  * added to the fluxes and \f$-u_\alpha \partial_i v^i_g\f$ is added
329  * to the time derivatives. For equations without fluxes
330  * \f$v^i\partial_i u_\alpha\f$ is added to the time derivatives.
331  *
332  * 4. Compute flux divergence contribution and add it to the time derivatives.
333  *
334  * Either the weak or strong form can be used. Currently only the strong
335  * form is coded, but adding the weak form is quite easy.
336  *
337  * Note that the computation of the flux divergence and adding that to the
338  * time derivative must be done *after* the mesh velocity is subtracted
339  * from the fluxes.
340  */

367  // Helper function to get parameter packs so we can forward `Tensor`s instead
368  // of `Variables` to the boundary corrections. Returns the maximum absolute
369  // char speed on the face, which can be used for setting or monitoring the CFL
370  // without having to compute the speeds for each dimension in the volume.
371  // Whether using only the face speeds is accurate enough to guarantee
372  // stability is yet to be determined. However, if the CFL condition is
373  // violated on the boundaries we are definitely in trouble, so it can at least
374  // be used as a cheap diagnostic.
375  template <typename BoundaryCorrection, typename... PackagedFieldTags,

380  packaged_data,
381  const BoundaryCorrection& boundary_correction,
382  const Variables<tmpl::list<ProjectedFieldTags...>>& projected_fields,
383  const tnsr::i<DataVector, Dim, Frame::Inertial>& unit_normal_covector,
384  const boost::optional<tnsr::I<DataVector, Dim, Frame::Inertial>>&

399  return boundary_correction.dg_package_data(

431 template <typename Metavariables>
432 template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
433  typename ActionList, typename ParallelComponent>
434 std::tuple<db::DataBox<DbTagsList>&&>
435 ComputeTimeDerivative<Metavariables>::apply(
436  db::DataBox<DbTagsList>& box,
437  tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
438  Parallel::GlobalCache<Metavariables>& cache,
439  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
440  const ParallelComponent* const /*meta*/) noexcept { // NOLINT const
441  static constexpr size_t volume_dim = Metavariables::volume_dim;
442  using system = typename Metavariables::system;
443  using variables_tag = typename system::variables_tag;
444  using variables_tags = typename variables_tag::tags_list;
445  using partial_derivative_tags = typename system::gradient_variables;
446  using flux_variables = typename system::flux_variables;
447  using compute_volume_time_derivative_terms =
448  typename system::compute_volume_time_derivative_terms;
449 
450  const Mesh<volume_dim>& mesh = db::get<::domain::Tags::Mesh<volume_dim>>(box);
451  // Allocate the Variables classes needed for the time derivative
452  // computation.
453  //
454  // This is factored out so that we will be able to do ADER-DG/CG where a
455  // spacetime polynomial is constructed by solving implicit equations in time
456  // using a Picard iteration. A high-order initial guess is needed to
457  // efficiently construct the ADER spacetime solution. This initial guess is
458  // obtained using continuous RK methods, and so we will want to reuse
459  // buffers. Thus, the volume_terms function returns by reference rather than
460  // by value.
461  Variables<typename compute_volume_time_derivative_terms::temporary_tags>
462  temporaries{mesh.number_of_grid_points()};
463  Variables<db::wrap_tags_in<::Tags::Flux, flux_variables,
464  tmpl::size_t<volume_dim>, Frame::Inertial>>
465  volume_fluxes{mesh.number_of_grid_points()};
466  Variables<db::wrap_tags_in<::Tags::deriv, partial_derivative_tags,
467  tmpl::size_t<volume_dim>, Frame::Inertial>>
468  partial_derivs{mesh.number_of_grid_points()};
469 
470  volume_terms<volume_dim, compute_volume_time_derivative_terms>(
471  make_not_null(&box), make_not_null(&volume_fluxes),
472  make_not_null(&partial_derivs), make_not_null(&temporaries),
473  db::get<::dg::Tags::Formulation>(box), variables_tags{},
474  typename compute_volume_time_derivative_terms::argument_tags{});
475 
476  // The below if-else and fill_mortar_data_for_internal_boundaries are for
477  // compatibility with the current boundary schemes. Once the current
478  // boundary schemes are removed the code will be refactored to reflect that
479  // change.
480  if constexpr (not std::is_same_v<tmpl::list<>, flux_variables>) {
481  using flux_variables_tag = ::Tags::Variables<flux_variables>;
482  using fluxes_tag =
483  db::add_tag_prefix<::Tags::Flux, flux_variables_tag,
484  tmpl::size_t<Metavariables::volume_dim>,
485  Frame::Inertial>;
486 
487  // We currently set the fluxes in the DataBox to interface with the
488  // boundary correction communication code.
489  db::mutate<fluxes_tag>(make_not_null(&box),
490  [&volume_fluxes](const auto fluxes_ptr) noexcept {
491  *fluxes_ptr = volume_fluxes;
492  });
493  } else {
494  boundary_terms_nonconservative_products(make_not_null(&box));
495  }
496 
497  if constexpr (detail::has_boundary_correction_v<system>) {
498  const auto& boundary_correction =
499  db::get<evolution::Tags::BoundaryCorrection<system>>(box);
500  using derived_boundary_corrections =
501  typename std::decay_t<decltype(boundary_correction)>::creatable_classes;
502 
503  static_assert(
504  tmpl::all<derived_boundary_corrections, std::is_final<tmpl::_1>>::value,
505  "All createable classes for boundary corrections must be marked "
506  "final.");
507  tmpl::for_each<derived_boundary_corrections>(
508  [&boundary_correction, &box, &temporaries,
509  &volume_fluxes](auto derived_correction_v) noexcept {
510  using DerivedCorrection =
511  tmpl::type_from<decltype(derived_correction_v)>;
512  if (typeid(boundary_correction) == typeid(DerivedCorrection)) {
513  // Compute internal boundary quantities on the mortar for sides
514  // of the element that have neighbors, i.e. they are not an
515  // external side.
516  compute_internal_mortar_data<volume_dim>(
517  make_not_null(&box),
518  dynamic_cast<const DerivedCorrection&>(boundary_correction),
519  db::get<variables_tag>(box), volume_fluxes, temporaries);
520  }
521  });
522  } else {
523  // Compute internal boundary quantities
524  fill_mortar_data_for_internal_boundaries<
525  volume_dim, typename Metavariables::boundary_scheme>(
526  make_not_null(&box));
527 
528  send_data_for_fluxes<ParallelComponent>(make_not_null(&cache), box);
529  }
530  return {std::move(box)};
531 }

539  const gsl::not_null<db::DataBox<DbTagsList>*> box,
540  const gsl::not_null<Variables<tmpl::list<FluxVariablesTags...>>*>
541  volume_fluxes,
542  const gsl::not_null<Variables<tmpl::list<PartialDerivTags...>>*>
543  partial_derivs,
544  const gsl::not_null<Variables<tmpl::list<TemporaryTags...>>*> temporaries,
545  const ::dg::Formulation dg_formulation,
546 
547  tmpl::list<VariablesTags...> /*variables_tags*/,
548  tmpl::list<TimeDerivativeArgumentTags...> /*meta*/) noexcept {
549  static constexpr bool has_partial_derivs = sizeof...(PartialDerivTags) != 0;
550  static constexpr bool has_fluxes = sizeof...(FluxVariablesTags) != 0;
551  static_assert(
552  has_fluxes or has_partial_derivs,
553  "Must have either fluxes or partial derivatives in a "
554  "DG evolution scheme. This means the evolution system struct (usually in "
555  "Evolution/Systems/YourSystem/System.hpp) being used does not specify "
556  "any flux_variables or gradient_variables. Make sure the type aliases "
557  "are defined, and that at least one of them is a non-empty list of "
558  "tags.");
559 
560  using system = typename Metavariables::system;
561  using variables_tag = typename system::variables_tag;
562  using variables_tags = typename variables_tag::tags_list;
563  using partial_derivative_tags = typename system::gradient_variables;
564  using flux_variables = typename system::flux_variables;
565 
566  const Mesh<Dim>& mesh = db::get<::domain::Tags::Mesh<Dim>>(*box);
567  const auto& logical_to_inertial_inverse_jacobian = db::get<
568  ::domain::Tags::InverseJacobian<Dim, Frame::Logical, Frame::Inertial>>(
569  *box);
570  const auto& evolved_vars = db::get<variables_tag>(*box);
571 
572  // Compute d_i u_\alpha for nonconservative products
573  if constexpr (has_partial_derivs) {
574  partial_derivatives<partial_derivative_tags>(
575  partial_derivs, evolved_vars, mesh,
576  logical_to_inertial_inverse_jacobian);
577  }
578 
579  // Compute volume du/dt and fluxes
580 
581  // Compute volume terms that are unrelated to moving meshes
582  db::mutate<db::add_tag_prefix<::Tags::dt, variables_tag>>(
583  box,
584  [&mesh, &partial_derivs, &temporaries, &volume_fluxes](
585  const gsl::not_null<Variables<
586  db::wrap_tags_in<::Tags::dt, typename variables_tag::tags_list>>*>
587  dt_vars_ptr,
588  const auto&... args) noexcept {
589  // Silence compiler warnings since we genuinely don't always need all
590  // the vars but sometimes do. This warning shows up with empty parameter
591  // packs, which means the packs aren't "used" below in the
592  // ComputeVolumeTimeDerivatives::apply call.
593  (void)partial_derivs;
594  (void)volume_fluxes;
595  (void)temporaries;
596 
597  // For now just zero dt_vars. If this is a performance bottle neck we
598  // can re-evaluate in the future.
599  dt_vars_ptr->initialize(mesh.number_of_grid_points(), 0.0);
600 
601  ComputeVolumeTimeDerivatives::apply(
602  make_not_null(&get<::Tags::dt<VariablesTags>>(*dt_vars_ptr))...,
603  make_not_null(&get<FluxVariablesTags>(*volume_fluxes))...,
604  make_not_null(&get<TemporaryTags>(*temporaries))...,
605  get<PartialDerivTags>(*partial_derivs)..., args...);
606  },
607  db::get<TimeDerivativeArgumentTags>(*box)...);
608 
609  // Add volume terms for moving meshes
610  if (const auto& mesh_velocity =
611  db::get<::domain::Tags::MeshVelocity<Dim>>(*box);
612  static_cast<bool>(mesh_velocity)) {
613  db::mutate<db::add_tag_prefix<::Tags::dt, variables_tag>>(
614  box,
615  [&evolved_vars, &mesh_velocity, &volume_fluxes](
616  const gsl::not_null<Variables<db::wrap_tags_in<
617  ::Tags::dt, typename variables_tag::tags_list>>*>
618  dt_vars_ptr,
619  const boost::optional<Scalar<DataVector>>&
620  div_mesh_velocity) noexcept {
621  tmpl::for_each<flux_variables>([&div_mesh_velocity, &dt_vars_ptr,
622  &evolved_vars, &mesh_velocity,
623  &volume_fluxes](auto tag_v) noexcept {
624  // Modify fluxes for moving mesh
625  using var_tag = typename decltype(tag_v)::type;
626  using flux_var_tag =
627  db::add_tag_prefix<::Tags::Flux, var_tag, tmpl::size_t<Dim>,
628  Frame::Inertial>;
629  auto& flux_var = get<flux_var_tag>(*volume_fluxes);
630  // Loop over all independent components of flux_var
631  for (size_t flux_var_storage_index = 0;
632  flux_var_storage_index < flux_var.size();
633  ++flux_var_storage_index) {
634  // Get the flux variable's tensor index, e.g. (i,j) for a F^i of
635  // the spatial velocity (or some other spatial tensor).
636  const auto flux_var_tensor_index =
637  flux_var.get_tensor_index(flux_var_storage_index);
638  // Remove the first index from the flux tensor index, gets back
639  // (j)
640  const auto var_tensor_index =
641  all_but_specified_element_of(flux_var_tensor_index, 0);
642  // Set flux_index to (i)
643  const size_t flux_index = gsl::at(flux_var_tensor_index, 0);
644 
645  // We now need to index flux(i,j) -= u(j) * v_g(i)
646  flux_var[flux_var_storage_index] -=
647  get<var_tag>(evolved_vars).get(var_tensor_index) *
648  mesh_velocity->get(flux_index);
649  }
650 
651  // Modify time derivative (i.e. source terms) for moving mesh
652  auto& dt_var = get<::Tags::dt<var_tag>>(*dt_vars_ptr);
653  for (size_t dt_var_storage_index = 0;
654  dt_var_storage_index < dt_var.size(); ++dt_var_storage_index) {
655  // This is S -> S - u d_i v^i_g
656  dt_var[dt_var_storage_index] -=
657  get<var_tag>(evolved_vars)[dt_var_storage_index] *
658  get(*div_mesh_velocity);
659  }
660  });
661  },
662  db::get<::domain::Tags::DivMeshVelocity>(*box));
663 
664  // We add the mesh velocity to all equations that don't have flux terms.
665  // This doesn't need to be equal to the equations that have partial
666  // derivatives. For example, the scalar field evolution equation in
667  // first-order form does not have any partial derivatives but still needs
668  // the velocity term added. This is because the velocity term arises from
669  // transforming the time derivative.
670  using non_flux_tags = tmpl::list_difference<variables_tags, flux_variables>;
671 
672  db::mutate<db::add_tag_prefix<::Tags::dt, variables_tag>>(
673  box, [&mesh_velocity, &partial_derivs](
674  const gsl::not_null<Variables<db::wrap_tags_in<
675  ::Tags::dt, typename variables_tag::tags_list>>*>
676  dt_vars) noexcept {
677  tmpl::for_each<non_flux_tags>(
678  [&dt_vars, &mesh_velocity,
679  &partial_derivs](auto var_tag_v) noexcept {
680  using var_tag = typename decltype(var_tag_v)::type;
681  using dt_var_tag = ::Tags::dt<var_tag>;
682  using deriv_var_tag =
683  ::Tags::deriv<var_tag, tmpl::size_t<Dim>, Frame::Inertial>;
684 
685  const auto& deriv_var = get<deriv_var_tag>(*partial_derivs);
686  auto& dt_var = get<dt_var_tag>(*dt_vars);
687 
688  // Loop over all independent components of the derivative of the
689  // variable.
690  for (size_t deriv_var_storage_index = 0;
691  deriv_var_storage_index < deriv_var.size();
692  ++deriv_var_storage_index) {
693  // We grab the `deriv_tensor_index`, which would be e.g.
694  // `(i, a, b)`, so `(0, 2, 3)`
695  const auto deriv_var_tensor_index =
696  deriv_var.get_tensor_index(deriv_var_storage_index);
697  // Then we drop the derivative index (the first entry) to get
698  // `(a, b)` (or `(2, 3)`)
699  const auto dt_var_tensor_index =
700  all_but_specified_element_of(deriv_var_tensor_index, 0);
701  // Set `deriv_index` to `i` (or `0` in the example)
702  const size_t deriv_index = gsl::at(deriv_var_tensor_index, 0);
703  dt_var.get(dt_var_tensor_index) +=
704  mesh_velocity->get(deriv_index) *
705  deriv_var[deriv_var_storage_index];
706  }
707  });
708  });
709  }
710 
711  // Add the flux divergence term to du_\alpha/dt, which must be done
712  // after the corrections for the moving mesh are made.
713  if constexpr (has_fluxes) {
714  db::mutate<db::add_tag_prefix<::Tags::dt, variables_tag>>(
715  box,
716  [dg_formulation, &logical_to_inertial_inverse_jacobian, &mesh,
717  &volume_fluxes](const gsl::not_null<Variables<db::wrap_tags_in<
718  ::Tags::dt, typename variables_tag::tags_list>>*>
719  dt_vars_ptr) noexcept {
720  if (dg_formulation == ::dg::Formulation::StrongInertial) {
721  const Variables<tmpl::list<::Tags::div<FluxVariablesTags>...>>
722  div_fluxes = divergence(*volume_fluxes, mesh,
723  logical_to_inertial_inverse_jacobian);
724  tmpl::for_each<flux_variables>([&dt_vars_ptr, &div_fluxes](
725  auto var_tag_v) noexcept {
726  using var_tag = typename decltype(var_tag_v)::type;
727  auto& dt_var = get<::Tags::dt<var_tag>>(*dt_vars_ptr);
728  const auto& div_flux_var = get<::Tags::div<
729  ::Tags::Flux<var_tag, tmpl::size_t<Dim>, Frame::Inertial>>>(
730  div_fluxes);
731  for (size_t storage_index = 0; storage_index < dt_var.size();
732  ++storage_index) {
733  dt_var[storage_index] -= div_flux_var[storage_index];
734  }
735  });
736  } else if (dg_formulation == ::dg::Formulation::WeakInertial) {
737  ERROR("Weak inertial DG formulation not yet implemented");
738  } else {
739  ERROR("Unsupported DG formulation: " << dg_formulation);
740  }
741  });
742  } else {
743  (void)dg_formulation;
744  }
745 }

747 template <typename Metavariables>
748 template <size_t Dim, typename BoundaryCorrection, typename TemporaryTags,
749  typename DbTagsList, typename VariablesTags,
750  typename... PackageDataVolumeTags>
751 void ComputeTimeDerivative<Metavariables>::compute_internal_mortar_data(
752  const gsl::not_null<db::DataBox<DbTagsList>*> box,
753  const BoundaryCorrection& boundary_correction,
754  const Variables<VariablesTags>& volume_evolved_vars,
755  const Variables<db::wrap_tags_in<
756  ::Tags::Flux, typename Metavariables::system::flux_variables,
757  tmpl::size_t<Dim>, Frame::Inertial>>& volume_fluxes,
758  const Variables<TemporaryTags>& volume_temporaries) noexcept {
759  using system = typename Metavariables::system;
760  using variables_tags = typename system::variables_tag::tags_list;
761  using flux_variables = typename system::flux_variables;
762  using fluxes_tags = db::wrap_tags_in<::Tags::Flux, flux_variables,
763  tmpl::size_t<Metavariables::volume_dim>,
764  Frame::Inertial>;
765  using temporary_tags_for_face =
766  typename BoundaryCorrection::dg_package_data_temporary_tags;
767  using primitive_tags_for_face = typename detail::get_primitive_vars<
768  system::has_primitive_and_conservative_vars>::
769  template f<BoundaryCorrection>;
770  using mortar_tags_list = typename BoundaryCorrection::dg_package_field_tags;
771  static_assert(
772  std::is_same_v<VariablesTags, variables_tags>,
773  "The evolved variables passed in should match the evolved variables of "
774  "the system.");
775 
776  const Element<Dim>& element = db::get<domain::Tags::Element<Dim>>(*box);
777  const Mesh<Dim>& volume_mesh = db::get<domain::Tags::Mesh<Dim>>(*box);
778  const auto& face_meshes =
779  get<domain::Tags::Interface<domain::Tags::InternalDirections<Dim>,
780  domain::Tags::Mesh<Dim - 1>>>(*box);
781  const auto& mortar_meshes = db::get<Tags::MortarMesh<Dim>>(*box);
782  const auto& mortar_sizes = db::get<Tags::MortarSize<Dim>>(*box);
783  const TimeStepId& temporal_id = db::get<::Tags::TimeStepId>(*box);
784  const std::unordered_map<Direction<Dim>, tnsr::i<DataVector, Dim>>&
785  face_normals = db::get<domain::Tags::Interface<
786  domain::Tags::InternalDirections<Dim>,
787  ::Tags::Normalized<
788  domain::Tags::UnnormalizedFaceNormal<Dim, Frame::Inertial>>>>(
789  *box);
790  const std::unordered_map<Direction<Dim>,
791  boost::optional<tnsr::I<DataVector, Dim>>>&
792  face_mesh_velocities = db::get<domain::Tags::Interface<
793  domain::Tags::InternalDirections<Dim>,
794  domain::Tags::MeshVelocity<Dim, Frame::Inertial>>>(*box);
795 
796  Variables<tmpl::append<variables_tags, fluxes_tags, temporary_tags_for_face,
797  primitive_tags_for_face>>
798  fields_on_face{};
799  for (const auto& [direction, neighbors_in_direction] : element.neighbors()) {
800  (void)neighbors_in_direction; // unused variable
801  // In order to reduce memory allocations we handle both the upper and
802  // lower neighbors in each direction together since computing the
803  // contributions to the faces is guaranteed to require the same number of
804  // grid points.
805  if (direction.side() == Side::Upper and
806  element.neighbors().count(direction.opposite()) != 0) {
807  continue;
808  }
809 
810  const auto internal_mortars =
811  [&box, &boundary_correction, &element, &face_mesh_velocities,
812  &face_normals, &fields_on_face, &mortar_meshes, &mortar_sizes,
813  &temporal_id, &volume_evolved_vars, &volume_fluxes,
814  &volume_temporaries,
815  &volume_mesh](const Mesh<Dim - 1>& face_mesh,
816  const Direction<Dim>& local_direction) noexcept {
817  // We may not need to bring the volume fluxes or temporaries to the
818  // boundary since that depends on the specific boundary correction we
819  // are using. Silence compilers warnings about them being unused.
820  (void)volume_fluxes;
821  (void)volume_temporaries;
822  // We only need the box for volume tags and primitive tags. Silence
823  // unused variable compiler warnings in the case where we don't have
824  // both of those.
825  (void)box;
826 
827  // This helper does the following:
828  //
829  // 1. Use a helper function to get data onto the faces. Done either by
830  // slicing (Gauss-Lobatto points) or interpolation (Gauss points).
831  // This is done using the `project_contiguous_data_to_boundary` and
832  // `project_tensors_to_boundary` functions.
833  //
834  // 2. Invoke the boundary correction to get the packaged data. Note
835  // that this is done on the *face* and NOT the mortar.
836  //
837  // 3. Project the packaged data onto the DG mortars (these might need
838  // re-projection onto subcell mortars later).
839 
840  // Perform step 1
841  project_contiguous_data_to_boundary(make_not_null(&fields_on_face),
842  volume_evolved_vars, volume_mesh,
843  local_direction);
844  if constexpr (tmpl::size<fluxes_tags>::value != 0) {
845  project_contiguous_data_to_boundary(make_not_null(&fields_on_face),
846  volume_fluxes, volume_mesh,
847  local_direction);
848  }
849  if constexpr (tmpl::size<temporary_tags_for_face>::value != 0) {
850  project_tensors_to_boundary<temporary_tags_for_face>(
851  make_not_null(&fields_on_face), volume_temporaries, volume_mesh,
852  local_direction);
853  }
854  if constexpr (system::has_primitive_and_conservative_vars and
855  tmpl::size<primitive_tags_for_face>::value != 0) {
856  project_tensors_to_boundary<primitive_tags_for_face>(
857  make_not_null(&fields_on_face),
858  db::get<typename system::primitive_variables_tag>(*box),
859  volume_mesh, local_direction);
860  }
861 
862  // Perform step 2
863  Variables<mortar_tags_list> packaged_data{
864  face_mesh.number_of_grid_points()};
865  // The DataBox is passed in for retrieving the `volume_tags`
866  const double max_abs_char_speed_on_face = dg_package_data(
867  make_not_null(&packaged_data), boundary_correction,
868  fields_on_face, face_normals.at(local_direction),
869  face_mesh_velocities.at(local_direction), *box,
870  typename BoundaryCorrection::dg_package_data_volume_tags{});
871  (void)max_abs_char_speed_on_face;
872 
873  // Perform step 3
874  const auto& neighbors_in_local_direction =
875  element.neighbors().at(local_direction);
876  for (const auto& neighbor : neighbors_in_local_direction) {
877  const auto mortar_id = std::make_pair(local_direction, neighbor);
878  const auto& mortar_mesh = mortar_meshes.at(mortar_id);
879  const auto& mortar_size = mortar_sizes.at(mortar_id);
880 
881  // Project the data from the face to the mortar.
882  // Where no projection is necessary we `std::move` the data directly
883  // to avoid a copy. We can't move the data or modify it in-place
884  // when projecting, because in that case the face may touch two
885  // mortars so we need to keep the data around.
886  auto boundary_data_on_mortar =
887  ::dg::needs_projection(face_mesh, mortar_mesh, mortar_size)
888  // NOLINTNEXTLINE(bugprone-use-after-move)
889  ? ::dg::project_to_mortar(packaged_data, face_mesh,
890  mortar_mesh, mortar_size)
891  : std::move(packaged_data);
892 
893  // Store the boundary data on this side of the mortar in a way that
894  // is agnostic to the type of boundary correction used. This
895  // currently requires an additional allocation that could be
896  // eliminated either by:
897  //
898  // 1. Having non-owning Variables
899  //
900  // 2. Allow stealing the allocation out of a Variables (and
901  // inserting an allocation).
902  std::vector<double> type_erased_boundary_data_on_mortar{
903  boundary_data_on_mortar.data(),
904  boundary_data_on_mortar.data() +
905  boundary_data_on_mortar.size()};
906  db::mutate<Tags::MortarData<Dim>>(
907  box, [&face_mesh, &mortar_id, &temporal_id,
908  &type_erased_boundary_data_on_mortar](
909  const auto mortar_data_ptr) noexcept {
910  mortar_data_ptr->at(mortar_id).insert_local_mortar_data(
911  temporal_id, face_mesh,
912  std::move(type_erased_boundary_data_on_mortar));
913  });
914  }
915  };
916 
917  if (fields_on_face.number_of_grid_points() !=
918  face_meshes.at(direction).number_of_grid_points()) {
919  fields_on_face.initialize(
920  face_meshes.at(direction).number_of_grid_points());
921  }
922  internal_mortars(face_meshes.at(direction), direction);
923 
924  if (element.neighbors().count(direction.opposite()) != 0) {
925  if (fields_on_face.number_of_grid_points() !=
926  face_meshes.at(direction.opposite()).number_of_grid_points()) {
927  fields_on_face.initialize(
928  face_meshes.at(direction.opposite()).number_of_grid_points());
929  }
930  internal_mortars(face_meshes.at(direction.opposite()),
931  direction.opposite());
932  }
933  }
934 }
