55 template <size_t Dim>
56 struct ExportCoordinates {
57  template <typename ParallelComponent, typename DbTagsList,
58  typename ArrayIndex>
59  static std::pair<observers::TypeOfObservation, observers::ObservationKey>
60  register_info(const db::DataBox<DbTagsList>& /*box*/,
61  const ArrayIndex& /*array_index*/) noexcept {
62  return {observers::TypeOfObservation::Volume,
63  observers::ObservationKey("ObserveCoords")};
64  }
65 
66  template <
67  typename DbTagsList, typename... InboxTags, typename Metavariables,
68  typename ArrayIndex, typename ActionList, typename ParallelComponent,
69  Requires<tmpl::list_contains_v<DbTagsList, domain::Tags::Mesh<Dim>>> =
70  nullptr>
71  static std::tuple<db::DataBox<DbTagsList>&&> apply(
72  db::DataBox<DbTagsList>& box,
73  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
74  const Parallel::GlobalCache<Metavariables>& cache,
75  const ArrayIndex& array_index, const ActionList /*meta*/,
76  const ParallelComponent* const /*meta*/) {
77  const double time = get<Tags::Time>(box);
78 
79  const auto& mesh = get<domain::Tags::Mesh<Dim>>(box);
80  const auto& inertial_coordinates =
81  db::get<domain::Tags::Coordinates<Dim, Frame::Inertial>>(box);
82  const std::string element_name = MakeString{} << ElementId<Dim>(array_index)
83  << '/';
84  // Collect volume data
85  // Remove tensor types, only storing individual components
86  std::vector<TensorComponent> components;
87  components.reserve(Dim + 1);
88  for (size_t d = 0; d < Dim; d++) {
89  components.emplace_back(element_name + "InertialCoordinates_" +
90  inertial_coordinates.component_name(
91  inertial_coordinates.get_tensor_index(d)),
92  inertial_coordinates.get(d));
93  }
94  // Also output the determinant of the inverse jacobian, which measures
95  // the expansion and compression of the grid
96  const auto& det_inv_jac =
97  db::get<domain::Tags::DetInvJacobian<Frame::Logical, Frame::Inertial>>(
98  box);
99  components.emplace_back(
100  element_name +
101  db::tag_name<domain::Tags::DetInvJacobian<Frame::Logical,
102  Frame::Inertial>>(),
103  get(det_inv_jac));
104  // Send data to volume observer
105  auto& local_observer =
106  *Parallel::get_parallel_component<observers::Observer<Metavariables>>(
107  cache)
108  .ckLocalBranch();
109  Parallel::simple_action<observers::Actions::ContributeVolumeData>(
110  local_observer, observers::ObservationId(time, "ObserveCoords"),
111  std::string{"/element_data"},
112  observers::ArrayComponentId(
113  std::add_pointer_t<ParallelComponent>{nullptr},
114  Parallel::ArrayIndex<ElementId<Dim>>(array_index)),
115  std::move(components), mesh.extents(), mesh.basis(), mesh.quadrature());
116  return std::forward_as_tuple(std::move(box));
117  }
118 };

151 struct FindGlobalMinimumGridSpacing {
152  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
153  size_t Dim, typename ActionList,
154  typename ParallelComponent>
155  static std::tuple<db::DataBox<DbTagsList>&&> apply(
156  db::DataBox<DbTagsList>& box,
157  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
158  const Parallel::GlobalCache<Metavariables>& cache,
159  const ElementId<Dim>& element_id, const ActionList /*meta*/,
160  const ParallelComponent* const /*meta*/) {
161  const auto& element_proxy =
162  Parallel::get_parallel_component<ParallelComponent>(cache)[element_id];
163  const auto& singleton_proxy = Parallel::get_parallel_component<
164  SingletonParallelComponent<Metavariables>>(cache);
165 
166  Parallel::ReductionData<
167  Parallel::ReductionDatum<double, funcl::AssertEqual<>>,
168  Parallel::ReductionDatum<double, funcl::Min<>>>
169  time_and_elemental_min_grid_spacing(
170  get<Tags::Time>(box),
171  get<domain::Tags::MinimumGridSpacing<Dim, Frame::Inertial>>(box));
172  Parallel::contribute_to_reduction<Actions::PrintGlobalMinimumGridSpacing>(
173  std::move(time_and_elemental_min_grid_spacing), element_proxy,
174  singleton_proxy);
175 
176  return {std::move(box)};
177  }
178 };

235  static Phase determine_next_phase(
236  const Phase& current_phase,
237  const Parallel::CProxy_GlobalCache<
238  Metavariables>& /*cache_proxy*/) noexcept {
239  switch (current_phase) {
240  case Phase::Initialization:
241  return Phase::RegisterWithObserver;
242  case Phase::RegisterWithObserver:
243  return Phase::Export;
244  case Phase::Export:
245  return Phase::Exit;
246  case Phase::Exit:
247  ERROR(
248  "Should never call determine_next_phase with the current phase "
249  "being 'Exit'");
250  default:
251  ERROR("Unknown type of phase.");
252  }
