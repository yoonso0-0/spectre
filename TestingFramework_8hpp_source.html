70 /*!
71  * \ingroup TestingFrameworkGroup
72  * \brief Set a default tolerance for floating-point number comparison
73  *
74  * \details
75  * Catch's default (relative) tolerance for comparing floating-point numbers is
76  * `std::numeric_limits<float>::%epsilon() * 100`, or roughly \f$10^{-5}\f$.
77  * This tolerance is too loose for checking many scientific algorithms that
78  * rely on double precision floating-point accuracy, so we provide a tighter
79  * tighter tolerance through the `approx` static object.
80  *
81  * \example
82  * \snippet Test_TestingFramework.cpp approx_test
83  */
84 // clang-tidy: static object creation may throw exception
85 static Approx approx = // NOLINT
86  Approx::custom() // NOLINT
87  .epsilon(std::numeric_limits<double>::epsilon() * 100) // NOLINT
88  .scale(1.0); // NOLINT

91  * \ingroup TestingFrameworkGroup
92  * \brief A wrapper around Catch's CHECK macro that checks approximate
93  * equality of the two entries in a std::complex. For efficiency, no function
94  * forwarding is performed, just a pair of `CHECK`s inline

161 template <typename T>
162 struct check_iterable_approx<
163  T, Requires<not tt::is_maplike_v<T> and tt::is_iterable_v<T> and
164  not tt::is_a_v<std::unordered_set, T>>> {
165  // clang-tidy: non-const reference
166  static void apply(const T& a, const T& b, Approx& appx = approx) { // NOLINT
167  CAPTURE(a);
168  CAPTURE(b);
169  auto a_it = a.begin();
170  auto b_it = b.begin();
171  CHECK(a_it != a.end());
172  CHECK(b_it != b.end());
173  while (a_it != a.end() and b_it != b.end()) {
174  check_iterable_approx<std::decay_t<decltype(*a_it)>>::apply(*a_it, *b_it,
175  appx);
176  ++a_it;
177  ++b_it;
178  }
179  {
180  INFO("Iterable is longer in first argument than in second argument");
181  CHECK(a_it == a.end());
182  }
183  {
184  INFO("Iterable is shorter in first argument than in second argument");
185  CHECK(b_it == b.end());
186  }
187  }
188 };

200 template <typename T>
201 struct check_iterable_approx<
202  T, Requires<tt::is_maplike_v<T> and tt::is_iterable_v<T>>> {
203  // clang-tidy: non-const reference
204  static void apply(const T& a, const T& b, Approx& appx = approx) { // NOLINT
205  CAPTURE(a);
206  CAPTURE(b);
207  for (const auto& kv : a) {
208  const auto& key = kv.first;
209  try {
210  const auto& a_value = kv.second;
211  const auto& b_value = b.at(key);
212  CAPTURE(key);
213  check_iterable_approx<std::decay_t<decltype(a_value)>>::apply(
214  a_value, b_value, appx);
215  } catch (const std::out_of_range&) {
216  INFO("Missing key in second container: " << key);
217  CHECK(false);
218  }
219  }
220 
221  for (const auto& kv : b) {
222  const auto& key = kv.first;
223  try {
224  a.at(key);
225  // We've checked that the values match above.
226  } catch (const std::out_of_range&) {
227  INFO("Missing key in first container: " << key);
228  CHECK(false);
229  }
230  }
231  }

249 template <typename M>
250 struct check_matrix_approx {
251  // clang-tidy: non-const reference
252  static void apply(const M& a, const M& b,
253  Approx& appx = approx) { // NOLINT
254  // This implementation is for a column-major matrix. It does not trivially
255  // generalize to a row-major matrix because the iterator
256  // `blaze::DynamicMatrix<T, SO>::cbegin(i)` traverses either a row or a
257  // column, and takes its index as argument.
258  CHECK(a.columns() == b.columns());
259  for (size_t j = 0; j < a.columns(); j++) {
260  CAPTURE(a);
261  CAPTURE(b);
262  auto a_it = a.cbegin(j);
263  auto b_it = b.cbegin(j);
264  CHECK(a_it != a.cend(j));
265  CHECK(b_it != b.cend(j));
266  while (a_it != a.cend(j) and b_it != b.cend(j)) {
267  check_iterable_approx<std::decay_t<decltype(*a_it)>>::apply(
268  *a_it, *b_it, appx);
269  ++a_it;
270  ++b_it;
271  }
272  {
273  INFO("Column " << j
274  << " of the first matrix is longer than that of the "
275  "second matrix.");
276  CHECK(a_it == a.end(j));
277  }
278  {
279  INFO("Column " << j
280  << " of the first matrix is shorter than that of the "
281  "second matrix.");
282  CHECK(b_it == b.end(j));
283  }
284  }
285  }

313 /*!
314  * \ingroup TestingFrameworkGroup
315  * \brief Mark a test to be checking an ASSERT
316  *
317  * \details
318  * Testing error handling is just as important as testing functionality. Tests
319  * that are supposed to exit with an error must be annotated with the attribute
320  * \code
321  * // [[OutputRegex, The regex that should be found in the output]]
322  * \endcode
323  * Note that the regex only needs to be a sub-expression of the error message,
324  * that is, there are implicit wildcards before and after the string.
325  *
326  * In order to test ASSERT's properly the test must also fail for release
327  * builds. This is done by adding this macro at the beginning for the test.
328  *
329  * \example
330  * \snippet Test_Time.cpp example_of_error_test
331  */
332 #ifdef SPECTRE_DEBUG
333 #define ASSERTION_TEST() \
334  do { \
335  ERROR_TEST(); \
336  } while (false)
337 #else
