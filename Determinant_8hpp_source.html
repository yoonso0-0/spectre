132  * \ingroup TensorGroup
133  * \brief Computes the determinant of a rank-2 Tensor `tensor`.
134  *
135  * \requires That `tensor` be a rank-2 Tensor, with both indices sharing the
136  * same dimension and type.
137  */
138 template <typename T, typename Symm, typename Index0, typename Index1>
139 void determinant(
140  const gsl::not_null<Scalar<T>*> det_tensor,
141  const Tensor<T, Symm, index_list<Index0, Index1>>& tensor) noexcept {
142  static_assert(Index0::dim == Index1::dim,
143  "Cannot take the determinant of a Tensor whose Indices are not "
144  "of the same dimensionality.");
145  static_assert(Index0::index_type == Index1::index_type,
146  "Taking the determinant of a mixed Spatial and Spacetime index "
147  "Tensor is not allowed since it's not clear what that means.");
148  get(*det_tensor) = detail::DeterminantImpl<Symm, Index0>::apply(tensor);
149 }

152 Scalar<T> determinant(

154  Scalar<T> result{};
