19 /*!
20  * \ingroup ObserversGroup
21  * \brief The group parallel component that is responsible for reducing data
22  * to be observed.
23  *
24  * Once the data from all elements on the processing element (usually a core)
25  * has been collected, it is copied (not sent over the network) to the local
26  * nodegroup parallel component, `ObserverWriter`, for writing to disk.
27  */
28 template <class Metavariables>
29 struct Observer {
30  using chare_type = Parallel::Algorithms::Group;
31  using metavariables = Metavariables;
32  using phase_dependent_action_list = tmpl::list<Parallel::PhaseActions<
33  typename metavariables::Phase, metavariables::Phase::Initialization,
34  tmpl::list<::Actions::SetupDataBox, Actions::Initialize<Metavariables>,
35  Initialization::Actions::RemoveOptionsAndTerminatePhase>>>;
36  using initialization_tags = Parallel::get_initialization_tags<
37  Parallel::get_initialization_actions_list<phase_dependent_action_list>>;
38 
39  static void execute_next_phase(
40  const typename Metavariables::Phase /*next_phase*/,
41  Parallel::CProxy_GlobalCache<
42  Metavariables>& /*global_cache*/) noexcept {}
43 };
