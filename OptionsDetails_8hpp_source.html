34 // Traverses the group hierarchy of `Tag`, returning the topmost group that is
35 // a subgroup of `Root`. Directly returns `Tag` if it has no group.
36 // This means that the returned type is always the direct child node of `Root`
37 // that contains `Tag` in its hierarchy.
38 // If `Root` is not in the group hierarchy of `Tag`, this function returns the
39 // topmost group of `Tag` (meaning that `Root` is treated as the root of its
40 // group hierarchy).
41 template <typename Tag, typename Root, typename = std::void_t<>>
42 struct find_subgroup {
43  using type = Tag;
44 };

48  std::void_t<typename Tag::group>> {
49  using type = Tag;
50 };

53 struct find_subgroup<Tag, Root, std::void_t<typename Tag::group>> {
54  using type = typename find_subgroup<typename Tag::group, Root>::type;

173 template <typename Tag, typename OptionList>
174 struct print_impl<Tag, OptionList,
175  Requires<tmpl::list_contains_v<OptionList, Tag>>> {
176  static std::string apply() noexcept {
177  std::ostringstream ss;
178  ss << " " << name<Tag>() << ":\n"
179  << " " << "type=" << yaml_type<typename Tag::type>::value();
180  if constexpr (has_suggested<Tag>::value) {
181  if constexpr (tt::is_a_v<std::unique_ptr, typename Tag::type>) {
182  call_with_dynamic_type<
183  void, typename Tag::type::element_type::creatable_classes>(
184  Tag::suggested_value().get(), [&ss](const auto* derived) noexcept {
185  ss << "\n suggested=" << std::boolalpha
186  << pretty_type::short_name<decltype(*derived)>();
187  });
188  } else {
189  ss << "\n suggested="
190  << (MakeString{} << std::boolalpha << Tag::suggested_value());
191  }
192  }
193  if constexpr (has_lower_bound<Tag>::value) {
194  ss << "\n min=" << (MakeString{} << Tag::lower_bound());
195  }
196  if constexpr (has_upper_bound<Tag>::value) {
197  ss << "\n max=" << (MakeString{} << Tag::upper_bound());
198  }
199  if constexpr (has_lower_bound_on_size<Tag>::value) {
200  ss << "\n min size=" << Tag::lower_bound_on_size();
201  }
202  if constexpr (has_upper_bound_on_size<Tag>::value) {
203  ss << "\n max size=" << Tag::upper_bound_on_size();
204  }
205  ss << "\n" << wrap_text(Tag::help, 77, " ") << "\n\n";
206  return ss.str();
207  }
208 };

286 template <typename K, typename V>
287 struct wrap_create_types_impl<std::map<K, V>> {
288  template <typename Metavariables>
289  using wrapped_type = std::map<wrap_create_types<K, Metavariables>,
290  wrap_create_types<V, Metavariables>>;
291 
292  static auto unwrap(std::map<K, V> wrapped) {
293  using UnwrappedK = decltype(unwrap_create_types<K>(std::declval<K>()));
294  using UnwrappedV = decltype(unwrap_create_types<V>(std::declval<V>()));
295  std::map<UnwrappedK, UnwrappedV> result;
296  for (auto it = wrapped.begin(); it != wrapped.end();) {
297  auto node = wrapped.extract(it++);
298  result.emplace(unwrap_create_types<K>(std::move(node.key())),
299  unwrap_create_types<V>(std::move(node.mapped())));
300  }
301  return result;
302  }
303 };

347  std::integer_sequence<size_t, Is...> /*meta*/) {
348  using UnwrappedT = decltype(unwrap_create_types<T>(std::declval<T>()));
349  static_cast<void>(wrapped); // Work around broken GCC warning
