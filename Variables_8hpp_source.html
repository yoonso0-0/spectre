46 /*!
47  * \ingroup DataStructuresGroup
48  * \brief A Variables holds a contiguous memory block with Tensors pointing
49  * into it.
50  *
51  * The `Tags` are `struct`s that must have a public type alias `type` whose
52  * value must be a `Tensor<DataVector, ...>`, a `static' method `name()` that
53  * returns a `std::string` of the tag name, and must derive off of
54  * `db::SimpleTag`. In general, they should be DataBoxTags that are not compute
55  * items. For example,
56  *
57  * \snippet Helpers/DataStructures/TestTags.hpp simple_variables_tag
58  *
59  * Prefix tags can also be stored and their format is:
60  *
61  * \snippet Helpers/DataStructures/TestTags.hpp prefix_variables_tag
62  *
63  * #### Design Decisions
64  *
65  * The `Variables` class is designed to hold several different `Tensor`s
66  * performing one memory allocation for all the `Tensor`s. The advantage is that
67  * memory allocations are quite expensive, especially in a parallel environment.
68  *
69  * `Variables` stores the data it owns in a `std::unique_ptr<double[],
70  * decltype(&free)>` instead of a `std::vector` because allocating the
71  * `unique_ptr` with `malloc` allows us to avoid initializing the memory
72  * completely in release mode when no value is passed to the constructor.
73  * Additionally, if the macro `SPECTRE_NAN_INIT` is defined, initialization with
74  * `NaN`s is done even in release mode.
75  */

247  Variables<SubsetOfTags> sub_vars(number_of_grid_points());
248  tmpl::for_each<SubsetOfTags>([&](const auto tag_v) noexcept {

489 template <typename... Tags>
490 void Variables<tmpl::list<Tags...>>::initialize(
491  const size_t number_of_grid_points) noexcept {
492  if (number_of_grid_points_ != number_of_grid_points) {
493  number_of_grid_points_ = number_of_grid_points;
494  size_ = number_of_grid_points * number_of_independent_components;
495  if (size_ > 0) {
496  // clang-tidy: cppcoreguidelines-no-malloc
497  variable_data_impl_.reset(static_cast<value_type*>(
498  malloc(size_ * sizeof(value_type)))); // NOLINT
499 #if defined(SPECTRE_DEBUG) || defined(SPECTRE_NAN_INIT)
500  std::fill(variable_data_impl_.get(), variable_data_impl_.get() + size_,
501  make_signaling_NaN<value_type>());
502 #endif // SPECTRE_DEBUG
503  add_reference_variable_data();
504  }
505  }
