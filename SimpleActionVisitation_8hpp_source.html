47 template <typename Invokable, typename ParallelComponent, typename ThisVariant,
48  typename... Variants, typename... Args,
49  Requires<is_apply_callable_v<Invokable, ParallelComponent,
50  std::add_lvalue_reference_t<ThisVariant>,
51  Args&&...>> = nullptr>
52 void simple_action_visitor_helper(boost::variant<Variants...>& box,
53  const gsl::not_null<int*> iter,
54  const gsl::not_null<bool*> already_visited,
55  Args&&... args) {
56  if (box.which() == *iter and not*already_visited) {
57  try {
58  static_assert(
59  std::is_same<void,
60  decltype(Invokable::template apply<ParallelComponent>(
61  std::declval<ThisVariant&>(),
62  std::declval<Args>()...))>::value,
63  "A simple action must return void.");
64  Invokable::template apply<ParallelComponent>(boost::get<ThisVariant>(box),
65  std::forward<Args>(args)...);
66  } catch (std::exception& e) {
67  ERROR("Fatal error: Failed to call simple Action '"
68  << pretty_type::get_name<Invokable>() << "' on iteration '" << iter
69  << "' with DataBox type '" << pretty_type::get_name<ThisVariant>()
70  << "'\nThe exception is: '" << e.what() << "'\n");
71  }
72  *already_visited = true;
73  }
74  (*iter)++;
75 }

77 template <typename Invokable, typename ParallelComponent, typename ThisVariant,
78  typename... Variants, typename... Args,
79  Requires<not is_apply_callable_v<
80  Invokable, ParallelComponent,
81  std::add_lvalue_reference_t<ThisVariant>, Args&&...>> = nullptr>
82 void simple_action_visitor_helper(boost::variant<Variants...>& box,
83  const gsl::not_null<int*> iter,
84  const gsl::not_null<bool*> already_visited,
85  Args&&... /*args*/) {
86  if (box.which() == *iter and not*already_visited) {
87  std::string args_list{};
88  const auto helper = [&args_list](auto arg_v) {
89  args_list +=
90  pretty_type::get_name<typename decltype(arg_v)::type>() + ", ";
91  };
92  EXPAND_PACK_LEFT_TO_RIGHT(helper(tmpl::type_<Args>{}));
93  if (sizeof...(Args) > 0) {
94  args_list.pop_back();
95  args_list.pop_back();
96  }
97  ERROR("\nCannot call apply function of '"
98  << pretty_type::get_name<Invokable>() << "' with DataBox type '"
99  << pretty_type::get_name<ThisVariant>() << "' and arguments '"
100  << args_list << "'.\n"
101  << "If the argument types to the apply function match, then it is "
102  "possible that the apply function has non-deducible template "
103  "parameters. This could occur from removing an apply function "
104  "argument and forgetting to remove its associated template "
105  "parameters.\n");
106  }
107  (*iter)++;
108 }

110 /*!
111  * \brief Calls an `Invokable`'s `apply` static member function with the current
112  * type in the `boost::variant`.
113  *
114  * The simple action can only be invoked if it accesses members of the DataBox
115  * that are guaranteed to be present at the time of invocation. This can lead to
116  * race conditions since simple actions can be invoked at any time during
117  * execution at any part of any phase. As a concrete example of such a race
118  * condition, consider a parallel component with iterable actions A, B, and C
119  * where B adds something to the DataBox (call it G) and C removes it. If a
120  * simple action S that uses G is called on the parallel component the call will
121  * succeed if S is invoked between B and C, and fail otherwise.
122  */
123 template <typename Invokable, typename ParallelComponent, typename... Variants,
124  typename... Args>
125 void simple_action_visitor(boost::variant<Variants...>& box, Args&&... args) {
126  // iter is the current element of the variant in the "for loop"
127  int iter = 0;
128  // already_visited ensures that only one visitor is invoked
129  bool already_visited = false;
130  EXPAND_PACK_LEFT_TO_RIGHT(
131  simple_action_visitor_helper<Invokable, ParallelComponent, Variants>(
132  box, &iter, &already_visited, std::forward<Args>(args)...));
133 }
