33  * \ingroup ComputationalDomainGroup
34  * \brief A wrapper around a vector of Blocks that represent the computational
35  * domain.
36  */
37 template <size_t VolumeDim>
38 class Domain {
39  public:
40  explicit Domain(std::vector<Block<VolumeDim>> blocks) noexcept;
41 
42  /*!
43  * Create a Domain using CoordinateMaps to encode the Orientations.
44  * This constructor does not support periodic boundary conditions.
45  *
46  * \details A constructor that does not require the user to provide a corner
47  * numbering scheme. Constructs a global corner numbering for each pair
48  * of abutting Blocks from their maps alone. The numbering is used to
49  * set up the corresponding Orientation, and then is discarded; the
50  * next pair of blocks uses a new global corner numbering, and so on,
51  * until all pairs of abutting Blocks have had their Orientations
52  * determined. For more information on setting up domains, see the
53  * [domain creation tutorial](\ref tutorial_domain_creation).
54  *
55  */
56  explicit Domain(std::vector<std::unique_ptr<domain::CoordinateMapBase<
57  Frame::Logical, Frame::Inertial, VolumeDim>>>
58  maps) noexcept;
59 
60  /*!
61  * Create a Domain using a corner numbering scheme to encode the Orientations,
62  * with an optional parameter that encodes periodic boundary conditions.
63  *
64  * \details Each element of `corners_of_all_blocks` contains the corner
65  * numbering of that block's corners according to the global corner number
66  * scheme. The details of the corner numbering scheme are described in the
67  * [tutorial](@ref tutorial_orientations). `identifications` is for imposing
68  * periodic boundary conditions on the domain. To identify faces,
69  * `identifications` should contain the PairOfFaces containing the corners of
70  * each pair of faces that you wish to identify with one another. For more
71  * information on setting up domains, see the
72  * [domain creation tutorial](@ref tutorial_domain_creation).
73  *
74  * \requires `maps.size() == corners_of_all_blocks.size()`, and
75  * `identifications.size()` is even.
76  */
77  Domain(std::vector<std::unique_ptr<domain::CoordinateMapBase<
78  Frame::Logical, Frame::Inertial, VolumeDim>>>
79  maps,
80  const std::vector<std::array<size_t, two_to_the(VolumeDim)>>&
81  corners_of_all_blocks,
82  const std::vector<PairOfFaces>& identifications = {}) noexcept;
