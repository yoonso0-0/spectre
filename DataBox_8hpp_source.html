836 /*!
837  * \ingroup DataBoxGroup
838  * \brief Allows changing the state of one or more non-computed elements in
839  * the DataBox
840  *
841  * `mutate()`'s first argument is the DataBox from which to retrieve the tags
842  * `MutateTags`. The objects corresponding to the `MutateTags` are then passed
843  * to `invokable`, which is a lambda or a function object taking as many
844  * arguments as there are `MutateTags` and with the arguments being of types
845  * `gsl::not_null<db::item_type<MutateTags>*>...`. Inside the `invokable` no
846  * items can be retrieved from the DataBox `box`. This is to avoid confusing
847  * subtleties with order of evaluation of compute items, as well as dangling
848  * references. If an `invokable` needs read access to items in `box` they should
849  * be passed as additional arguments to `mutate`. Capturing them by reference in
850  * a lambda does not work because of a bug in GCC 6.3 and earlier. For a
851  * function object the read-only items can also be stored as const references
852  * inside the object by passing `db::get<TAG>(t)` to the constructor.
853  *
854  * \example
855  * \snippet Test_DataBox.cpp databox_mutate_example
856  */
857 template <typename... MutateTags, typename TagList, typename Invokable,
858  typename... Args>
859 void mutate(const gsl::not_null<DataBox<TagList>*> box, Invokable&& invokable,
860  Args&&... args) noexcept {
861  static_assert(
862  tmpl2::flat_all_v<
863  detail::has_unique_matching_tag_v<TagList, MutateTags>...>,
864  "One of the tags being mutated could not be found in the DataBox or "
865  "is a base tag identifying more than one tag.");
866  static_assert(not tmpl2::flat_any_v<tmpl::list_contains_v<
867  typename DataBox<TagList>::compute_with_subitems_tags,
868  detail::first_matching_tag<TagList, MutateTags>>...>,
869  "Cannot mutate a compute item");
870  if (UNLIKELY(box->mutate_locked_box_)) {
871  ERROR(
872  "Unable to mutate a DataBox that is already being mutated. This "
873  "error occurs when mutating a DataBox from inside the invokable "
874  "passed to the mutate function.");
875  }
876  box->mutate_locked_box_ = true;
877  invokable(
878  make_not_null(&box->template get_deferred<
879  detail::first_matching_tag<TagList, MutateTags>>()
880  .mutate())...,
881  std::forward<Args>(args)...);
882  using mutate_tags_list =
883  tmpl::list<detail::first_matching_tag<TagList, MutateTags>...>;
884  // For all the tags in the DataBox, check if one of their subtags is
885  // being mutated and if so add the parent to the list of tags
886  // being mutated. Then, remove any tags that would be passed
887  // multiple times.
888  using extra_mutated_tags = tmpl::list_difference<
889  tmpl::filter<TagList,
890  tmpl::bind<tmpl::found, Subitems<tmpl::_1>,
891  tmpl::pin<tmpl::bind<tmpl::list_contains,
892  tmpl::pin<mutate_tags_list>,
893  tmpl::_1>>>>,
894  mutate_tags_list>;
895  // Extract the subtags inside the MutateTags and reset compute items
896  // depending on those too.
897  using full_mutated_items =
898  tmpl::append<detail::expand_subitems<mutate_tags_list>,
899  extra_mutated_tags>;
900 
901  using first_compute_items_to_reset = tmpl::remove_duplicates<
902  tmpl::transform<tmpl::filter<typename DataBox<TagList>::edge_list,
903  tmpl::bind<tmpl::list_contains,
904  tmpl::pin<full_mutated_items>,
905  tmpl::get_source<tmpl::_1>>>,
906  tmpl::get_destination<tmpl::_1>>>;
907 
908  EXPAND_PACK_LEFT_TO_RIGHT(
909  box->template mutate_subitem_tags_in_box<MutateTags>(
910  typename Subitems<MutateTags>::type{}));
911  box->template reset_compute_items_after_mutate(
912  first_compute_items_to_reset{});
913 
914  box->mutate_locked_box_ = false;
915 }

1056 SPECTRE_ALWAYS_INLINE constexpr auto create_from(Box&& box,
1057  Args&&... args) noexcept {
1058  static_assert(tmpl::size<AddTags>::value == sizeof...(Args),
1059  "Must pass in as many arguments as AddTags to db::create_from");

1061  // 1. Full list of old tags, and the derived tags list of the RemoveTags
1062  using old_box_tags = typename std::decay_t<Box>::tags_list;
1063  static_assert(
1064  tmpl::all<RemoveTags, has_unique_matching_tag<tmpl::pin<old_box_tags>,
1065  tmpl::_1>>::value,
1066  "One of the tags being removed could not be found in the DataBox or "
1067  "is a base tag identifying more than one tag.");
1068  using remove_tags = tmpl::transform<
1069  RemoveTags,
1070  tmpl::bind<first_matching_tag, tmpl::pin<old_box_tags>, tmpl::_1>>;
1071 
1072  // 2. Expand simple remove tags and compute remove tags
1073  using compute_tags_to_remove =
1074  tmpl::filter<remove_tags, db::is_compute_tag<tmpl::_1>>;
1075  using compute_tags_to_remove_with_subitems =
1076  expand_subitems<compute_tags_to_remove>;
1077  using simple_tags_to_remove =
1078  tmpl::list_difference<remove_tags, compute_tags_to_remove>;
1079  using simple_tags_to_remove_with_subitems =
1080  expand_subitems<simple_tags_to_remove>;
1081 
1082  // 3. Expand AddTags (these are just the simple tags)
1083  using simple_tags_to_add_with_subitems = expand_subitems<AddTags>;
1084  using compute_tags_to_add_with_subitems = expand_subitems<AddComputeTags>;
1085 
1086  // 4. Create new list of tags by removing all the remove tags, and adding all
1087  // the AddTags, including subitems
1088  using simple_tags_to_keep =
1089  tmpl::list_difference<typename std::decay_t<Box>::simple_item_tags,
1090  simple_tags_to_remove_with_subitems>;
1091  using new_simple_tags =
1092  tmpl::append<simple_tags_to_keep, simple_tags_to_add_with_subitems>;
1093 
1094  // 5. Create the list of compute items with the RemoveTags removed
1095  using compute_tags_to_keep = tmpl::list_difference<
1096  typename std::decay_t<Box>::compute_with_subitems_tags,
1097  compute_tags_to_remove_with_subitems>;
1098 
1099  // 6. List of the old tags that are being kept
1100  using old_tags_to_keep =

1245  * `A1` and `a2` of type `A2`, then
1246  * \code

1248  * \endcode

1254  * `sizeof...(Args)` arguments of types `Args...`,
1255  * \code
1256  * result = func(box, db::get<Tags>(box)..., args...);
1257  * \endcode
