68 /*!
69  * \brief Computational structs for evaluating the hypersurface integrals during
70  * CCE evolution. These are compatible with use in `db::mutate_apply`.
71  *
72  * \details
73  * The integral evaluated and the corresponding inputs required depend on the
74  * CCE quantity being computed. In any of these, the only mutated tag is `Tag`,
75  * where the result of the integration is placed. The supported `Tag`s act in
76  * the following ways:
77  * - If the `Tag` is `Tags::BondiBeta` or `Tags::BondiU`, the integral to be
78  * evaluated is simply \f[ \partial_y f = A, \f] where \f$A\f$ is retrieved with
79  * `Tags::Integrand<Tag>`.
80  * - If the `Tag` is `Tags::BondiQ` or `Tags::BondiW`, the integral to be
81  * evaluated is \f[ (1 - y) \partial_y f + 2 f = A + (1 - y) B, \f] where
82  * \f$A\f$ is retrieved with `Tags::PoleOfIntegrand<Tag>` and \f$B\f$ is
83  * retrieved with `Tags::RegularIntegrand<Tag>`.
84  * - If `Tag` is `Tags::BondiH`, the integral to be evaluated is:
85  *
86  * \f[
87  * (1 - y) \partial_y f + L f + L^\prime \bar{f} = A + (1 - y) B,
88  * \f]
89  *
90  * for \f$f\f$, where \f$A\f$ is retrieved with `Tags::PoleOfIntegrand<Tag>`,
91  * \f$B\f$ is retrieved with `Tags::RegularIntegrand<Tag>`, \f$L\f$ is retrieved
92  * with `Tags::LinearFactor<Tag>`, and \f$L^\prime\f$ is retrieved with
93  * `Tags::LinearFactorForConjugate<Tag>`. The presence of \f$L\f$ and
94  * \f$L^\prime\f$ ensure that the only current method we have for evaluating the
95  * \f$H\f$ hypersurface equation is a direct linear solve, rather than the
96  * spectral matrix multiplications which are available for the other integrals.
97  *
98  * In each case, the boundary value at the world tube for the integration is
99  * retrieved from `BoundaryPrefix<Tag>`.
100  *
101  * Additional type aliases `boundary_tags` and `integrand_tags` are provided for
102  * template processing of the required input tags necessary for these functions.
103  * These type aliases are `tmpl::list`s with the subsets of `argument_tags` from
104  * specific other parts of the CCE computation. Because they play different
105  * roles, and have different extents, it is better for tag management to give
106  * separated lists for the dependencies.
107  */
108 template <template <typename> class BoundaryPrefix, typename Tag>
109 struct RadialIntegrateBondi {
110  using boundary_tags = tmpl::list<BoundaryPrefix<Tag>>;
111  using integrand_tags = tmpl::list<Tags::Integrand<Tag>>;
112 
113  using return_tags = tmpl::list<Tag>;
114  using argument_tags =
115  tmpl::append<integrand_tags, boundary_tags,
116  tmpl::list<Tags::LMax, Tags::NumberOfRadialPoints>>;
117  static void apply(
118  gsl::not_null<
119  Scalar<SpinWeighted<ComplexDataVector, Tag::type::type::spin>>*>
120  integral_result,
121  const Scalar<SpinWeighted<ComplexDataVector, Tag::type::type::spin>>&
122  integrand,
123  const Scalar<SpinWeighted<ComplexDataVector, Tag::type::type::spin>>&
124  boundary,
125  size_t l_max, size_t number_of_radial_points) noexcept;
126 };
