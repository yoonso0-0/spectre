64 /*!
65  * \ingroup DiscontinuousGalerkinGroup
66  * \brief Boundary contributions for a first-order DG scheme with local
67  * time-stepping
68  *
69  * This class is the local time-stepping equivalent to the
70  * `dg::FirstOrderScheme::FirstOrderScheme`. Notable differences are:
71  *
72  * - Boundary contributions are added to the `VariablesTag` directly.
73  * - We use the `Tags::BoundaryHistory` on mortars.
74  * - We need to store the face-normal magnitude in the boundary history, as
75  * opposed to the non-LTS case where we can retrieve it when needed.
76  */
77 template <size_t Dim, typename VariablesTag, typename DtVariablesTag,
78  typename NumericalFluxComputerTag, typename TemporalIdTag,
79  typename TimeStepperTag>
80 struct FirstOrderSchemeLts {
81  private:
82  using base = FirstOrderScheme<Dim, VariablesTag, DtVariablesTag,
83  NumericalFluxComputerTag, TemporalIdTag>;
84 
85  public:
86  static constexpr size_t volume_dim = Dim;
87  using variables_tag = VariablesTag;
88  using numerical_flux_computer_tag = NumericalFluxComputerTag;
89  using NumericalFlux = typename NumericalFluxComputerTag::type;
90  using temporal_id_tag = TemporalIdTag;
91  using receive_temporal_id_tag = ::Tags::Next<temporal_id_tag>;
92  using time_stepper_tag = TimeStepperTag;
93 
94  using magnitude_of_face_normal_tag =
95  ::Tags::Magnitude<domain::Tags::UnnormalizedFaceNormal<volume_dim>>;
96  using BoundaryData = dg::SimpleBoundaryData<
97  typename base::BoundaryData::field_tags,
98  tmpl::push_back<typename base::BoundaryData::extra_data_tags,
99  magnitude_of_face_normal_tag>>;
100  using boundary_data_computer = detail::boundary_data_computer_lts_impl<
101  volume_dim, variables_tag, numerical_flux_computer_tag, BoundaryData>;
102 
103  using mortar_data_tag = ::Tags::BoundaryHistory<BoundaryData, BoundaryData,
104  typename variables_tag::type>;
105 
106  using return_tags =
107  tmpl::list<variables_tag, ::Tags::Mortars<mortar_data_tag, Dim>>;
108  using argument_tags =
109  tmpl::list<domain::Tags::Mesh<Dim>,
110  ::Tags::Mortars<domain::Tags::Mesh<Dim - 1>, Dim>,
111  ::Tags::Mortars<::Tags::MortarSize<Dim - 1>, Dim>,
112  NumericalFluxComputerTag, time_stepper_tag, ::Tags::TimeStep>;
113 
114  static void apply(
115  const gsl::not_null<typename variables_tag::type*> variables,
116  const gsl::not_null<typename ::Tags::Mortars<mortar_data_tag, Dim>::type*>
117  all_mortar_data,
118  const Mesh<Dim>& volume_mesh,
119  const typename ::Tags::Mortars<domain::Tags::Mesh<Dim - 1>, Dim>::type&
120  mortar_meshes,
121  const typename ::Tags::Mortars<::Tags::MortarSize<Dim - 1>, Dim>::type&
122  mortar_sizes,
123  const NumericalFlux& normal_dot_numerical_flux_computer,
124  const typename time_stepper_tag::type::element_type& time_stepper,
125  const TimeDelta& time_step) noexcept {
126  // Iterate over all mortars
127  for (auto& mortar_id_and_data : *all_mortar_data) {
128  // Retrieve mortar data
129  const auto& mortar_id = mortar_id_and_data.first;
130  auto& mortar_data = mortar_id_and_data.second;
131  const auto& direction = mortar_id.first;
132  const size_t dimension = direction.dimension();
133 
134  const auto& mortar_mesh = mortar_meshes.at(mortar_id);
135  const auto& mortar_size = mortar_sizes.at(mortar_id);
136  const auto face_mesh = volume_mesh.slice_away(dimension);
137  const size_t extent_perpendicular_to_face =
138  volume_mesh.extents(dimension);
139 
140  // This lambda must only capture quantities that are
141  // independent of the simulation state.
142  const auto coupling = [&face_mesh, &mortar_mesh, &mortar_size,
143  &extent_perpendicular_to_face,
144  &normal_dot_numerical_flux_computer](
145  const BoundaryData& local_data,
146  const BoundaryData& remote_data) noexcept {
147  return boundary_flux(
148  local_data, remote_data, normal_dot_numerical_flux_computer,
149  get<magnitude_of_face_normal_tag>(local_data.extra_data),
150  extent_perpendicular_to_face, face_mesh, mortar_mesh, mortar_size);
151  };
152 
153  const auto lifted_data = time_stepper.compute_boundary_delta(
154  coupling, make_not_null(&mortar_data), time_step);
155 
156  // Add the flux contribution to the volume data
157  add_slice_to_data(variables, lifted_data, volume_mesh.extents(),
158  dimension,
159  index_to_slice_at(volume_mesh.extents(), direction));
160  }
161  }
162 };
