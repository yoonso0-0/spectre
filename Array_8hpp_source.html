26 /// \brief A `std::array` implementation with partial C++20 support
27 ///
28 /// \warning This is not a standard-compliant C++20 `std::array` implementation.
29 /// We provide this implementation because we need the `constexpr operator==`
30 /// from C++20. Note that other C++20 changes, such as the replacement of the
31 /// other comparison operators in favor of `operator<=>`, are not implemented.
32 /// This class can be removed when we support C++20.
33 template <typename T, size_t Size>
34 struct array {
35  using value_type = T;
36  using reference = value_type&;
37  using const_reference = const value_type&;
38  using iterator = value_type*;
39  using const_iterator = const value_type*;
40  using pointer = value_type*;
41  using const_pointer = const value_type*;
42  using size_type = size_t;
43  using difference_type = std::ptrdiff_t;
44 
45  // clang-tidy: mark explicit. We want implicit conversion
46  constexpr operator std::array<T, Size>() const noexcept { // NOLINT
47  return detail::convert_to_array(data_, std::make_index_sequence<Size>{});
48  }
49 
50  constexpr iterator begin() noexcept {
51  return iterator(data_); // NOLINT
52  }
53  constexpr const_iterator begin() const noexcept {
54  return const_iterator(data_); // NOLINT
55  }
56  constexpr iterator end() noexcept {
57  return iterator(data_ + Size); // NOLINT
58  }
59  constexpr const_iterator end() const noexcept {
60  return const_iterator(data_ + Size); // NOLINT
61  }
62 
63  constexpr const_iterator cbegin() const noexcept { return begin(); }
64  constexpr const_iterator cend() const noexcept { return end(); }
65 
66  constexpr size_type size() const noexcept { return Size; }
67  constexpr size_type max_size() const noexcept { return Size; }
68  constexpr bool empty() const noexcept { return Size == 0; }
69 
70  constexpr reference operator[](const size_type i) noexcept {
71  return data_[i]; // NOLINT
72  }
73  constexpr const_reference operator[](const size_type i) const noexcept {
74  return data_[i]; // NOLINT
75  }
76 
77  constexpr reference at(const size_type i) noexcept {
78  return data_[i]; // NOLINT
79  }
80  constexpr const_reference at(const size_type i) const noexcept {
81  return data_[i]; // NOLINT
82  }
83 
84  constexpr reference front() { return data_[0]; }
85  constexpr const_reference front() const { return data_[0]; }
86  constexpr reference back() { return data_[Size > 0 ? Size - 1 : 0]; }
87  constexpr const_reference back() const {
88  return data_[Size > 0 ? Size - 1 : 0];
89  }
90 
91  constexpr value_type* data() noexcept { return data_; }
92  constexpr const value_type* data() const noexcept { return data_; }
93 
94  // NOLINTNEXTLINE(modernize-avoid-c-arrays)
95  value_type data_[Size > 0 ? Size : 1];
96 };

115 template <typename InputIter1, typename InputIter2,
116  typename BinaryPred = detail::Equal<>>
117 constexpr bool equal(InputIter1 first1, InputIter1 last1, InputIter2 first2,
118  BinaryPred pred = detail::Equal<>{}) {
119  for (; first1 != last1; ++first1, ++first2) {
120  if (not pred(*first1, *first2)) {
121  return false;
122  }
123  }
124  return true;
125 }

127 template <typename InputIter1, typename InputIter2,
128  typename BinaryPred = std::less_equal<>>
129 constexpr bool lexicographical_compare(InputIter1 first1, InputIter1 last1,
130  InputIter2 first2, InputIter2 last2,
131  BinaryPred pred = std::less_equal<>{}) {
132  for (; first2 != last2; ++first1, ++first2) {
133  if (first1 == last1 or pred(*first1, *first2)) {
134  return true;
135  } else if (pred(*first2, *first1)) {
136  return false;
137  }
138  }
139  return false;
140 }
