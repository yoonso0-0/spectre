33 /*!
34  * \brief Initialize the damped harmonic gauge, either with or without a rollon
35  * function.
36  *
37  * See `GeneralizedHarmonic::gauges::damped_harmonic()` for details on the
38  * condition.
39  *
40  * When the non-rollon gauge is initialized, Eq. 9 and 10 of
41  * \cite Lindblom2005qh are used to compute the time derivatives for the lapse
42  * and shift to compute the new \f$\Pi_{ab}\f$.
43  *
44  * \warning The rollon gauge does not support moving meshes.
45  *
46  * When the rollon gauge is used:
47  *
48  * DataBox:
49  * - Uses:
50  * - `domain::Tags::Domain<Dim, Frame::Inertial>`
51  * - `domain::CoordinateMaps::Tags::CoordinateMap<Dim,
52  * Frame::Grid, Frame::Inertial>`
53  * - `domain::Tags::InverseJacobian<Dim, Frame::Logical, Frame::Inertial>`
54  * - `domain::Tags::Mesh<Dim>`
55  * - `gr::Tags::SpacetimeMetric<Dim, Frame::Inertial>`
56  * - `GeneralizedHarmonic::Tags::Pi<Dim, Frame::Inertial>`
57  * - `GeneralizedHarmonic::Tags::Phi<Dim, Frame::Inertial>`
58  * - Adds:
59  * - `GeneralizedHarmonic::Tags::InitialGaugeH<Dim, Frame::Inertial>`
60  * - `GeneralizedHarmonic::Tags::SpacetimeDerivInitialGaugeH<Dim,
61  * Frame::Inertial>`
62  * - `GeneralizedHarmonic::Tags::GaugeH<Dim, Frame::Inertial>`
63  * - `GeneralizedHarmonic::Tags::SpacetimeDerivGaugeH<Dim, Frame::Inertial>`
64  * - Removes: nothing
65  * - Modifies: nothing
66  *
67  * When the non-rollon gauge is used:
68  *
69  * DataBox:
70  * - Uses:
71  * - `Initialization::Tags::InitialTime`
72  * - `domain::Tags::ElementMap<Dim, Frame::Grid>`
73  * - `domain::CoordinateMaps::Tags::CoordinateMap<Dim,
74  * Frame::Grid, Frame::Inertial>`
75  * - `domain::Tags::Coordinates<Dim, Frame::Logical>`
76  * - `domain::Tags::FunctionsOfTime`
77  * - `gr::Tags::SpacetimeMetric<Dim, Frame::Inertial>`
78  * - `GeneralizedHarmonic::Tags::Pi<Dim, Frame::Inertial>`
79  * - `GeneralizedHarmonic::Tags::Phi<Dim, Frame::Inertial>`
80  * - Adds:
81  * - `GeneralizedHarmonic::Tags::GaugeH<Dim, Frame::Inertial>`
82  * - `GeneralizedHarmonic::Tags::SpacetimeDerivGaugeH<Dim, Frame::Inertial>`
83  * - Removes: nothing
84  * - Modifies:
85  * - `GeneralizedHarmonic::Tags::Pi<Dim, Frame::Inertial>`
86  *
87  * \note This action relies on the `SetupDataBox` aggregated initialization
88  * mechanism, so `Actions::SetupDataBox` must be present in the `Initialization`
89  * phase action list prior to this action.
90  */

194  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
195  typename ArrayIndex, typename ActionList,
196  typename ParallelComponent>
197  static auto apply(db::DataBox<DbTagsList>& box,
198  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
199  const Parallel::GlobalCache<Metavariables>& /*cache*/,
200  const ArrayIndex& /*array_index*/,
201  const ActionList /*meta*/,
202  const ParallelComponent* const /*meta*/) noexcept {
203  if constexpr (UseRollon) {
204  if (not db::get<domain::CoordinateMaps::Tags::CoordinateMap<
205  Metavariables::volume_dim, Frame::Grid, Frame::Inertial>>(box)
206  .is_identity()) {
207  ERROR(
208  "Cannot use the damped harmonic rollon gauge with a moving mesh "
209  "because the rollon is not implemented for a moving mesh. The "
210  "issue is that the initial H_a needs to be in the grid frame, and "
211  "then transformed to the inertial frame at each time step. "
212  "Transforming the spacetime derivative requires spacetime "
213  "Hessians, which are not implemented for the maps and there is "
214  "currently no plan to add them because we do not need them for "
215  "anything else.");
216  }
217  const auto& inverse_jacobian =
218  db::get<domain::Tags::InverseJacobian<Dim, Frame::Logical, frame>>(
219  box);
220 
221  auto [initial_gauge_h, initial_d4_gauge_h] = impl_rollon(
222  db::get<gr::Tags::SpacetimeMetric<Dim, Frame::Inertial, DataVector>>(
223  box),
224  db::get<GeneralizedHarmonic::Tags::Pi<Dim, Frame::Inertial>>(box),
225  db::get<GeneralizedHarmonic::Tags::Phi<Dim, Frame::Inertial>>(box),
226  db::get<domain::Tags::Mesh<Dim>>(box), inverse_jacobian);
227 
228  // Finally, update gauge related quantities in the box
229  Initialization::mutate_assign<simple_tags>(make_not_null(&box),
230  std::move(initial_gauge_h),
231  std::move(initial_d4_gauge_h));
232  return std::make_tuple(std::move(box));
233  } else {
234  const double initial_time =
235  db::get<::Initialization::Tags::InitialTime>(box);
236  const auto inertial_coords =
237  db::get<::domain::CoordinateMaps::Tags::CoordinateMap<
238  Metavariables::volume_dim, Frame::Grid, Frame::Inertial>>(box)(
239  db::get<::domain::Tags::ElementMap<Metavariables::volume_dim,
240  Frame::Grid>>(box)(
241  db::get<domain::Tags::Coordinates<Dim, Frame::Logical>>(box)),
242  initial_time, db::get<::domain::Tags::FunctionsOfTime>(box));
243 
244  db::mutate<GeneralizedHarmonic::Tags::Pi<Dim, Frame::Inertial>>(
245  make_not_null(&box), &new_pi_from_gauge_h,
246  db::get<gr::Tags::SpacetimeMetric<Dim, Frame::Inertial, DataVector>>(
247  box),
248  db::get<GeneralizedHarmonic::Tags::Phi<Dim, Frame::Inertial>>(box),
249  inertial_coords,
250  db::get<
251  GeneralizedHarmonic::gauges::Tags::DhGaugeParameters<UseRollon>>(
252  box));
253 
254  // Add gauge tags
255  return std::make_tuple(std::move(box));
256  }
257  }
258 
259  private:

262  impl_rollon(
263  const tnsr::aa<DataVector, Dim, Frame::Inertial>& spacetime_metric,
