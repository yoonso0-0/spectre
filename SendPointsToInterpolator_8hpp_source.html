17 /// \ingroup ActionsGroup
18 /// \brief Sets up points on an `InterpolationTarget` at a new `temporal_id`
19 /// and sends these points to an `Interpolator`.
20 ///
21 /// Uses:
22 /// - DataBox:
23 /// - `domain::Tags::Domain<3>`
24 ///
25 /// DataBox changes:
26 /// - Adds: nothing
27 /// - Removes: nothing
28 /// - Modifies:
29 /// - `Tags::IndicesOfFilledInterpPoints`
30 /// - `Tags::IndicesOfInvalidInterpPoints`
31 /// - `Tags::InterpolatedVars<InterpolationTargetTag, TemporalId>`
32 ///
33 /// For requirements on InterpolationTargetTag, see InterpolationTarget
34 template <typename InterpolationTargetTag>
35 struct SendPointsToInterpolator {
36  template <
37  typename ParallelComponent, typename DbTags, typename Metavariables,
38  typename ArrayIndex, typename TemporalId,
39  Requires<tmpl::list_contains_v<DbTags, Tags::TemporalIds<TemporalId>>> =
40  nullptr>
41  static void apply(db::DataBox<DbTags>& box,
42  Parallel::GlobalCache<Metavariables>& cache,
43  const ArrayIndex& /*array_index*/,
44  const TemporalId& temporal_id) noexcept {
45  auto coords = InterpolationTarget_detail::block_logical_coords<
46  InterpolationTargetTag>(box, tmpl::type_<Metavariables>{}, temporal_id);
47  InterpolationTarget_detail::set_up_interpolation<InterpolationTargetTag>(
48  make_not_null(&box), temporal_id, coords);
49  auto& receiver_proxy =
50  Parallel::get_parallel_component<Interpolator<Metavariables>>(cache);
51  Parallel::simple_action<Actions::ReceivePoints<InterpolationTargetTag>>(
52  receiver_proxy, temporal_id, std::move(coords));
53  }
54 };
