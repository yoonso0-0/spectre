232  (void)std::initializer_list<char>{((void)args, '0')...};

362 template <typename Metavariables>
363 void Main<Metavariables>::
364  allocate_array_components_and_execute_initialization_phase() noexcept {
365  ASSERT(current_phase_ == Metavariables::Phase::Initialization,
366  "Must be in the Initialization phase.");
367  using array_component_list =
368  tmpl::filter<component_list,
369  Parallel::is_array_proxy<tmpl::bind<
370  Parallel::proxy_from_parallel_component, tmpl::_1>>>;
371  tmpl::for_each<array_component_list>([this](
372  auto parallel_component_v) noexcept {
373  using parallel_component = tmpl::type_from<decltype(parallel_component_v)>;
374  parallel_component::allocate_array(
375  global_cache_proxy_,
376  Parallel::create_from_options<Metavariables>(
377  options_, typename parallel_component::initialization_tags{}));
378  });
379 
380  // Free any resources from the initial option parsing.
381  options_ = decltype(options_){};
382 
383  tmpl::for_each<component_list>([this](auto parallel_component_v) noexcept {
384  using parallel_component = tmpl::type_from<decltype(parallel_component_v)>;
385  Parallel::get_parallel_component<parallel_component>(
386  *(global_cache_proxy_.ckLocalBranch()))
387  .start_phase(current_phase_);
388  });
389  CkStartQD(CkCallback(CkIndex_Main<Metavariables>::execute_next_phase(),
390  this->thisProxy));
391 }

393 template <typename Metavariables>
394 void Main<Metavariables>::execute_next_phase() noexcept {
395  current_phase_ = Metavariables::determine_next_phase(
396  current_phase_, global_cache_proxy_);
397  if (Metavariables::Phase::Exit == current_phase_) {
398  Informer::print_exit_info();
399  Parallel::exit();
400  }
401  tmpl::for_each<component_list>([this](auto parallel_component) noexcept {
402  tmpl::type_from<decltype(parallel_component)>::execute_next_phase(
403  current_phase_, global_cache_proxy_);
404  });
405  CkStartQD(CkCallback(CkIndex_Main<Metavariables>::execute_next_phase(),
406  this->thisProxy));
407 }
