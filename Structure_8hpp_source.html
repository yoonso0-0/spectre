25 template <size_t Size>
26 constexpr size_t number_of_independent_components(
27  const std::array<int, Size>& symm,
28  const std::array<size_t, Size>& dims) noexcept {
29  if constexpr (Size == 0) {
30  (void)symm;
31  (void)dims;
32 
33  return 1;
34  } else if constexpr (Size == 1) {
35  (void)symm;
36 
37  return dims[0];
38  } else {
39  size_t max_element = 0;
40  for (size_t i = 0; i < Size; ++i) {
41  // clang-tidy: internals of assert(), don't need gsl::at in constexpr
42  // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-array-to-pointer-decay,cppcoreguidelines-pro-bounds-constant-array-index)
43  assert(symm[i] > 0);
44  // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
45  max_element = std::max(static_cast<size_t>(ce_abs(symm[i])), max_element);
46  }
47  assert(max_element > 0); // NOLINT
48  size_t total_independent_components = 1;
49  for (size_t symm_index = 1; symm_index <= max_element; ++symm_index) {
50  size_t number_of_indices_with_symm = 0;
51  size_t dim_of_index = 0;
52  for (size_t i = 0; i < Size; ++i) {
53  if (static_cast<size_t>(symm[i]) == symm_index) { // NOLINT
54  ++number_of_indices_with_symm;
55  dim_of_index = dims[i]; // NOLINT
56  }
57  }
58  assert(dim_of_index > 0); // NOLINT
59  assert(number_of_indices_with_symm > 0); // NOLINT
60  if (dim_of_index - 1 > number_of_indices_with_symm) {
61  total_independent_components *=
62  falling_factorial(dim_of_index + number_of_indices_with_symm - 1,
63  number_of_indices_with_symm) /
64  factorial(number_of_indices_with_symm);
65  } else {
66  total_independent_components *=
67  falling_factorial(dim_of_index + number_of_indices_with_symm - 1,
68  dim_of_index - 1) /
69  factorial(dim_of_index - 1);
70  }
71  }
72  return total_independent_components;
73  }
74 }

99 // them so that when the max constexpr steps is reached on clang it is reached
100 // in this function rather than in array.
101 template <size_t Rank>
102 constexpr size_t index_to_swap_with(
103  const cpp20::array<size_t, Rank>& tensor_index,
104  const cpp20::array<int, Rank>& sym, size_t index_to_swap_with,
105  const size_t current_index) noexcept {
106  // If you encounter infinite loop compilation errors here you are
107  // constructing very large Tensor's. If you are sure Tensor is
108  // the correct data structure you can extend the compiler limit
109  // by passing the flag -fconstexpr-steps=<SOME LARGER VALUE>
110  while (true) { // See source code comment on line above this one for fix
111  if (Rank == index_to_swap_with) {
112  return current_index;
113  } else if (tensor_index[current_index] <
114  tensor_index[index_to_swap_with] and
115  sym[current_index] == sym[index_to_swap_with]) {
116  return index_to_swap_with;
117  }
118  index_to_swap_with++;
119  }
120 }

122 template <size_t Size, size_t SymmSize>
123 constexpr cpp20::array<size_t, Size> canonicalize_tensor_index(
124  cpp20::array<size_t, Size> tensor_index,
125  const cpp20::array<int, SymmSize>& symm) noexcept {
126  for (size_t i = 0; i < Size; ++i) {
127  const size_t temp = tensor_index[i];
128  const size_t swap = index_to_swap_with(tensor_index, symm, i, i);
129  tensor_index[i] = tensor_index[swap];
130  tensor_index[swap] = temp;
131  }
132  return tensor_index;
133 }

135 template <size_t Rank>
136 constexpr size_t compute_collapsed_index(
137  const cpp20::array<size_t, Rank>& tensor_index,
138  const cpp20::array<size_t, Rank> dims) noexcept {
139  size_t collapsed_index = 0;
140  for (size_t i = Rank - 1; i < Rank; --i) {
141  collapsed_index = tensor_index[i] + dims[i] * collapsed_index;
142  }
143  return collapsed_index;
144 }

146 template <typename Symm, size_t NumberOfComponents>
147 constexpr auto compute_collapsed_to_storage(
148  const cpp20::array<size_t, tmpl::size<Symm>::value>&
149  index_dimensions) noexcept {
150  if constexpr (tmpl::size<Symm>::value != 0) {
151  cpp20::array<size_t, NumberOfComponents> collapsed_to_storage{};
152  auto tensor_index =
153  convert_to_cpp20_array(make_array<tmpl::size<Symm>::value>(size_t{0}));
154  size_t count{0};
155  for (auto& current_storage_index : collapsed_to_storage) {
156  // Compute canonical tensor_index, which, for symmetric get_tensor_index
157  // is in decreasing numerical order, e.g. (3,2) rather than (2,3).
158  const auto canonical_tensor_index = canonicalize_tensor_index(
159  tensor_index, make_cpp20_array_from_list<Symm>());
160  // If the tensor_index was already in the canonical form, then it must be
161  // a new unique entry and we add it to collapsed_to_storage_ as a new
162  // integer, thus increasing the size_. Else, the StorageIndex has already
163  // been determined so we look it up in the existing collapsed_to_storage
164  // table.
165  if (tensor_index == canonical_tensor_index) {
166  current_storage_index = count;
167  ++count;
168  } else {
169  current_storage_index = collapsed_to_storage[compute_collapsed_index(
170  canonical_tensor_index, index_dimensions)];
171  }
172  // Move to the next tensor_index.
173  increment_tensor_index(tensor_index, index_dimensions);
174  }
175  return collapsed_to_storage;
176  } else {
177  (void)index_dimensions;
178 
179  return cpp20::array<size_t, 1>{{0}};
180  }
181 }

183 template <typename Symm, size_t NumIndComps, size_t NumComps>
184 constexpr auto compute_storage_to_tensor(
185  const cpp20::array<size_t, NumComps>& collapsed_to_storage,
186  const cpp20::array<size_t, tmpl::size<Symm>::value>&
187  index_dimensions) noexcept {
188  if constexpr (tmpl::size<Symm>::value > 0) {
189  constexpr size_t rank = tmpl::size<Symm>::value;
190  cpp20::array<cpp20::array<size_t, rank>, NumIndComps> storage_to_tensor{};
191  cpp20::array<size_t, rank> tensor_index =
192  convert_to_cpp20_array(make_array<rank>(size_t{0}));
193  for (const auto& current_storage_index : collapsed_to_storage) {
194  storage_to_tensor[current_storage_index] = canonicalize_tensor_index(
195  tensor_index, make_cpp20_array_from_list<Symm>());
196  increment_tensor_index(tensor_index, index_dimensions);
197  }
198  return storage_to_tensor;
199  } else {
200  (void)collapsed_to_storage;
201  (void)index_dimensions;
202 
203  return cpp20::array<cpp20::array<size_t, 1>, 1>{
204  {cpp20::array<size_t, 1>{{0}}}};
205  }
206 }

219 template <size_t NumIndices>
220 struct ComponentNameImpl {
221  template <typename Structure, typename T>
222  static std::string apply(
223  const std::array<T, NumIndices>& tensor_index,
224  const std::array<std::string, NumIndices>& axis_labels) {
225  const size_t storage_index = Structure::get_storage_index(tensor_index);
226  std::array<std::string, Structure::rank()> labels = axis_labels;
227  constexpr auto index_dim = Structure::dims();
228  for (size_t i = 0; i < Structure::rank(); ++i) {
229  if (gsl::at(labels, i).length() == 0) {
230  if (gsl::at(Structure::index_types(), i) == IndexType::Spacetime) {
231  switch (gsl::at(index_dim, i)) {
232  case 2:
233  gsl::at(labels, i) = "tx";
234  break;
235  case 3:
236  gsl::at(labels, i) = "txy";
237  break;
238  case 4:
239  gsl::at(labels, i) = "txyz";
240  break;
241  default:
242  ERROR("Tensor dim["
243  << i
244  << "] must be 1,2,3, or 4 for default axis_labels. "
245  "Either pass a string or extend the function.");
246  }
247  } else {
248  switch (gsl::at(index_dim, i)) {
249  case 1:
250  gsl::at(labels, i) = "x";
251  break;
252  case 2:
253  gsl::at(labels, i) = "xy";
254  break;
255  case 3:
256  gsl::at(labels, i) = "xyz";
257  break;
258  default:
259  ERROR("Tensor dim["
260  << i
261  << "] must be 1,2, or 3 for default axis_labels. "
262  "Either pass a string or extend the function.");
263  }
264  }
265  } else {
266  if (gsl::at(axis_labels, i).length() != gsl::at(index_dim, i)) {
267  ERROR("Dimension mismatch: Tensor has dim = "
268  << gsl::at(index_dim, i) << ", but you specified "
269  << gsl::at(axis_labels, i).length() << " different labels in "
270  << gsl::at(axis_labels, i));
271  }
272  }
273  }
274  // Create string labeling get_tensor_index
275  std::stringstream ss;
276  const auto canonical_tensor_index =
277  Structure::get_canonical_tensor_index(storage_index);
278  for (size_t r = 0; r < Structure::rank(); ++r) {
279  ss << gsl::at(labels, r)[gsl::at(canonical_tensor_index, r)];
280  }
281  return ss.str();
282  }

294 /// \ingroup TensorGroup
295 /// A lookup table between each tensor_index and storage_index
296 ///
297 /// 1. tensor_index: (a, b, c,...). There are Dim^rank tensor_index's
298 /// 2. collapsed_index: a + Dim * (b + Dim * (c + ...)), there are Dim^rank
299 /// unique collapsed indices and there is a 1-1 map between
300 /// a tensor_index and a collapsed_index.
301 /// 3. storage_index: index into the storage vector of the Tensor. This depends
302 /// on symmetries of the tensor, rank and dimensionality.
303 /// There are size storage_index's.
304 /// \tparam Symm the symmetry of the Tensor
305 /// \tparam Indices list of tensor_index's giving the dimensionality and frame
306 /// of the index
307 template <typename Symm, typename... Indices>
308 struct Structure {
309  static_assert(
310  TensorMetafunctions::check_index_symmetry_v<Symm, Indices...>,
311  "Cannot construct a Tensor with a symmetric pair that are not the same.");
312  static_assert(tmpl::size<Symm>::value == sizeof...(Indices),
313  "The number of indices in Symmetry do not match the number of "
314  "indices given to the Structure.");
315  static_assert(
316  tmpl2::flat_all_v<tt::is_tensor_index_type<Indices>::value...>,
317  "All Indices passed to Structure must be of type TensorIndexType.");
318 
319  using index_list = tmpl::list<Indices...>;
320 
321  SPECTRE_ALWAYS_INLINE static constexpr size_t rank() noexcept {
322  return sizeof...(Indices);
323  }
324 
325  SPECTRE_ALWAYS_INLINE static constexpr size_t size() noexcept {
326  constexpr auto number_of_independent_components =
327  ::Tensor_detail::number_of_independent_components(
328  make_array_from_list<
329  tmpl::conditional_t<sizeof...(Indices) != 0, Symm, int>>(),
330  make_array_from_list<tmpl::conditional_t<sizeof...(Indices) != 0,
331  index_list, size_t>>());
332  return number_of_independent_components;

338  make_array_from_list<tmpl::conditional_t<sizeof...(Indices) != 0,
339  index_list, size_t>>());
340  return number_of_components;
