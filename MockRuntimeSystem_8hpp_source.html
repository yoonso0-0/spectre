34 // `get_initialization` computes the type of the TaggedTuple that holds the
35 // initial simple tags for the DataBox, as well as the
36 // `initialize_databox_action` so that the user interface to the Action Testing
37 // does not require the user to redundantly specify this information.
38 // `get_initialization` also performs various sanity checks for the user.
39 template <typename Component>
40 struct get_initialization {
41  using phase = typename Component::metavariables::Phase;
42 
43  template <typename T>
44  struct is_initialization_phase {
45  using type =
46  std::integral_constant<bool, T::phase == phase::Initialization>;
47  };
48 
49  using initialization_pdal_list =
50  tmpl::filter<typename Component::phase_dependent_action_list,
51  is_initialization_phase<tmpl::_1>>;
52  static_assert(
53  tmpl::size<initialization_pdal_list>::value == 1,
54  "Must have exactly one Initialization PhaseDependentActionList");
55 
56  using initialization_pdal = typename tmpl::front<initialization_pdal_list>;
57 
58  static_assert(
59  tt::is_a_v<ActionTesting::InitializeDataBox,
60  tmpl::front<typename initialization_pdal::action_list>>,
61  "The first action in the initialization phase must be "
62  "ActionTesting::InitializeDataBox, even if the simple and "
63  "compute tags are empty.");
64 
65  using initialize_databox_action =
66  tmpl::front<typename initialization_pdal::action_list>;
67 
68  using InitialValues = typename initialize_databox_action::InitialValues;
69 };

279  /// Instead of the next call to `next_action` applying the next action in
280  /// the action list, force the next action to be `Action`
281  template <typename Component, typename Action>
282  void force_next_action_to_be(
283  const typename Component::array_index& array_index) noexcept {
284  static_assert(
285  tmpl::list_contains_v<
286  typename MockDistributedObject<Component>::all_actions_list,
287  Action>,
288  "Cannot force a next action that is not in the action list of the "
289  "parallel component. See the first template parameter of "
290  "'force_next_action_to_be' for the component and the second template "
291  "parameter for the action.");
292  bool found_matching_phase = false;
293  const auto invoke_for_phase =
294  [this, &array_index, &found_matching_phase](auto phase_dep_v) noexcept {
295  using PhaseDep = decltype(phase_dep_v);
296  constexpr typename Metavariables::Phase phase = PhaseDep::type::phase;
297  using actions_list = typename PhaseDep::type::action_list;
298  auto& distributed_object =
299  this->mock_distributed_objects<Component>().at(array_index);
300  if (distributed_object.get_phase() == phase) {
301  found_matching_phase = true;
302  distributed_object.force_next_action_to_be(
303  tmpl::conditional_t<
304  std::is_same<tmpl::no_such_type_,
305  tmpl::index_of<actions_list, Action>>::value,
306  std::integral_constant<size_t, 0>,
307  tmpl::index_of<actions_list, Action>>::value);
308  }
309  };
310  tmpl::for_each<typename MockDistributedObject<
311  Component>::phase_dependent_action_lists>(invoke_for_phase);
312  if (not found_matching_phase) {
313  ERROR(
314  "Could not find any actions in the current phase for the component '"
315  << pretty_type::short_name<Component>()
316  << "'. Maybe you are not in the phase you expected to be in? The "
317  "integer value corresponding to the current phase is "
318  << static_cast<int>(this->mock_distributed_objects<Component>()
319  .at(array_index)
320  .get_phase()));
321  }
322  }
