29 /// \ingroup UtilitiesGroup
30 /// \brief Compute the continued fraction representation of a number
31 ///
32 /// The template argument may be a fraction type or a floating point
33 /// type. If the expansion is computed exactly then it will be the
34 /// shorter of the expansions for the given value.
35 template <class T>
36 class ContinuedFraction {
37  template <typename U, typename = typename is_fraction<U>::type>
38  struct value_type_helper;
39 
40  template <typename U>
41  struct value_type_helper<U, std::false_type> {
42  using type = int64_t;
43  };
44 
45  template <typename U>
46  struct value_type_helper<U, std::true_type> {
47  using type = std::decay_t<decltype(std::declval<U>().numerator())>;
48  };
49 
50  public:
51  /// The decayed return type of T::numerator() for a fraction,
52  /// int64_t for other types.
53  using value_type = typename value_type_helper<T>::type;
54 
55  explicit ContinuedFraction(const T& value) noexcept
56  : term_(ifloor(value)),
57  remainder_(value - term_),
58  error_([this, &value]() noexcept {
59  using std::abs;
60  using std::max;
61  // For any non-fundamental type, epsilon() returns a
62  // default-constructed value, which should be zero for
63  // fractions.
64  return max(abs(value), abs(remainder_)) *
65  std::numeric_limits<T>::epsilon();
66  }()) {}
67 
68  /// Obtain the current element in the expansion
69  value_type operator*() const noexcept { return term_; }
70 
71  /// Check if the expansion is incomplete. If T is a fraction type
72  /// this will be true until the full exact representation of the
73  /// fraction is produced. If T is a floating point type this will
74  /// be true until the estimated numerical error is larger than the
75  /// remaining error in the representation.
76  explicit operator bool() const noexcept { return not done_; }
77 
78  /// Advance to the next element in the expansion
79  ContinuedFraction& operator++() noexcept {
80  // Terminate when remainder_ is consistent with zero.
81  if (remainder_ == 0 or error_ > remainder_) {
82  done_ = true;
83  return *this;
84  }
85  remainder_ = 1 / remainder_;
86  error_ *= square(remainder_);
87  term_ = ifloor(remainder_);
88  remainder_ -= term_;
89  return *this;
90  }
91 
92  private:
93  template <typename U>
94  static value_type ifloor(const U& x) noexcept {
95  if constexpr (is_fraction_v<U>) {
96  return static_cast<value_type>(x.numerator() / x.denominator());
97  } else {
98  return static_cast<value_type>(std::floor(x));
99  }
100  }
101 
102  value_type term_;
103  T remainder_;
104  // Estimate of error in the term.
105  T error_;
106  bool done_{false};
107 };

109 /// \ingroup UtilitiesGroup
110 /// \brief Sum a continued fraction
111 ///
112 /// \tparam Fraction the result type, which must be a fraction type
113 template <class Fraction>
114 class ContinuedFractionSummer {
115  public:
116  using Term_t = std::decay_t<decltype(std::declval<Fraction>().numerator())>;
117 
118  /// Sum of the supplied continued fraction terms
119  Fraction value() const noexcept { return Fraction(numerator_, denominator_); }
120 
121  /// Insert a new term. Terms should be supplied from most to least
122  /// significant.
123  void insert(Term_t term) noexcept {
124  const auto new_numerator = term * numerator_ + prev_numerator_;
125  prev_numerator_ = numerator_;
126  numerator_ = new_numerator;
127 
128  const auto new_denominator = term * denominator_ + prev_denominator_;
129  prev_denominator_ = denominator_;
130  denominator_ = new_denominator;
131  }
132 
133  private:
134  Term_t numerator_{1};
135  Term_t denominator_{0};
136  Term_t prev_numerator_{0};
137  Term_t prev_denominator_{1};
138 };

140 /// \ingroup UtilitiesGroup
141 /// \brief Find the fraction in the supplied interval with the
142 /// smallest denominator
143 ///
144 /// The endpoints are considered to be in the interval. The order of
145 /// the arguments is not significant. The answer is unique as long as
146 /// the interval has length less than 1; for longer intervals, an
147 /// integer in the range will be returned.
148 template <typename Fraction, typename T1, typename T2>
149 Fraction simplest_fraction_in_interval(const T1& end1,
150  const T2& end2) noexcept {
151  ContinuedFractionSummer<Fraction> result;
152  using Term_t = typename decltype(result)::Term_t;
153  ContinuedFraction<T1> cf1(end1);
154  ContinuedFraction<T2> cf2(end2);
155  using InputTerm_t = std::common_type_t<typename decltype(cf1)::value_type,
156  typename decltype(cf2)::value_type>;
157  InputTerm_t term1 = *cf1;
158  InputTerm_t term2 = *cf2;
159  for (; cf1 and cf2; term1 = *++cf1, term2 = *++cf2) {
160  if (term1 != term2) {
161  if (++cf1) {
162  ++term1;
163  }
164  if (++cf2) {
165  ++term2;
166  }
167  using std::min;
168  result.insert(gsl::narrow<Term_t>(min(term1, term2)));
169  return result.value();
170  }
171  result.insert(gsl::narrow<Term_t>(term1));
172  }
173  return result.value();
174 }
