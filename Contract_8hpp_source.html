67 template <size_t I, size_t Index1, size_t Index2, typename... LhsIndices,
68  typename T, typename S>
69 static SPECTRE_ALWAYS_INLINE decltype(auto) compute_contraction(S tensor_index,
70  const T& t1) {
71  if constexpr (I == 0) {
72  tensor_index[Index1] = 0;
73  tensor_index[Index2] = 0;
74  return t1.template get<LhsIndices...>(tensor_index);
75  } else {
76  tensor_index[Index1] = I;
77  tensor_index[Index2] = I;
78  return t1.template get<LhsIndices...>(tensor_index) +
79  compute_contraction<I - 1, Index1, Index2, LhsIndices...>(
80  tensor_index, t1);
81  }
82 }

142  template <size_t I, size_t Rank>
143  SPECTRE_ALWAYS_INLINE void fill_contracting_tensor_index(
144  std::array<size_t, Rank>& tensor_index_in,
145  const std::array<size_t, num_tensor_indices>& tensor_index) const {
146  if constexpr (I < FirstContractedIndexPos) {
147  tensor_index_in[I] = tensor_index[I];
148  fill_contracting_tensor_index<I + 1>(tensor_index_in, tensor_index);
149  } else if constexpr (I == FirstContractedIndexPos) {
150  // 10000 is for the slot that will be set later. Easy to debug.
151  tensor_index_in[I] = 10000;
152  fill_contracting_tensor_index<I + 1>(tensor_index_in, tensor_index);
153  } else if constexpr (I > FirstContractedIndexPos and
154  I <= SecondContractedIndexPos and I < Rank - 1) {
155  // tensor_index is Rank - 2 since it shouldn't be called for Rank 2 case
156  // 20000 is for the slot that will be set later. Easy to debug.
157  tensor_index_in[I] =
158  I == SecondContractedIndexPos ? 20000 : tensor_index[I - 1];
159  fill_contracting_tensor_index<I + 1>(tensor_index_in, tensor_index);
160  } else if constexpr (I > SecondContractedIndexPos and I < Rank - 1) {
161  // Left as Rank - 2 since it should never be called for the Rank 2 case
162  tensor_index_in[I] = tensor_index[I - 2];
163  fill_contracting_tensor_index<I + 1>(tensor_index_in, tensor_index);
164  } else if constexpr (I == SecondContractedIndexPos) {
165  tensor_index_in[I] = 20000;
166  } else {
167  tensor_index_in[I] = tensor_index[I - 2];
168  }
169  }

185  /// \brief Return the tensor multi-index of one uncontracted LHS component to
186  /// be summed to compute a contracted LHS component
187  ///
188  /// \details
189  /// Example: If we have RHS tensor \f$R^{a}{}_{abc}\f$ and we want to contract
190  /// it to the LHS tensor \f$L_{cb}\f$, then \f$L_{cb}\f$ represents the
191  /// contracted LHS, while \f$L^{a}{}_{acb}\f$ represents the uncontracted
192  /// LHS. This function takes a concrete contracted LHS multi-index as input,
193  /// representing the multi-index of a component of the contracted LHS that we
194  /// wish to compute. If `lhs_contracted_multi_index == [1, 2]`, this
195  /// represents \f$L_{12}\f$, the contracted LHS component we wish to compute.
196  /// In this case, we will need to sum \f$L^{a}{}_{a12}\f$ for all values of
197  /// \f$a\f$. `contracted_index_value` represents one such concrete value that
198  /// is filled in for \f$a\f$. In continuing the example, if
199  /// `contracted_index_value == 3`, then this function returns the multi-index
200  /// that represents \f$L^{3}{}_{312}\f$, which is `[3, 3, 1, 2]`. In this way,
201  /// what is constructed and returned is one such concrete multi-index of the
202  /// uncontracted LHS tensor to be summed as part of computing a component of
203  /// the contracted LHS tensor.
204  ///
205  /// \param lhs_contracted_multi_index the tensor multi-index of a contracted
206  /// LHS component to be computed
207  /// \param contracted_index_value the concrete value inserted for the indices
208  /// to contract
209  /// \return the tensor multi-index of one uncontracted LHS component to be
210  /// summed for computing the contracted LHS component at
211  /// `lhs_contracted_multi_index`
212  SPECTRE_ALWAYS_INLINE static constexpr std::array<
213  size_t, num_uncontracted_tensor_indices>
214  get_tensor_index_to_sum(
215  const std::array<size_t, num_tensor_indices>& lhs_contracted_multi_index,
216  const size_t contracted_index_value) noexcept {
217  std::array<size_t, num_uncontracted_tensor_indices>
218  contracting_tensor_index{};
219 
220  for (size_t i = 0; i < FirstContractedIndexPos; i++) {
221  gsl::at(contracting_tensor_index, i) =
222  gsl::at(lhs_contracted_multi_index, i);
223  }
224  contracting_tensor_index[FirstContractedIndexPos] = contracted_index_value;
225  for (size_t i = FirstContractedIndexPos + 1; i < SecondContractedIndexPos;
226  i++) {
227  gsl::at(contracting_tensor_index, i) =
228  gsl::at(lhs_contracted_multi_index, i - 1);
229  }
230  contracting_tensor_index[SecondContractedIndexPos] = contracted_index_value;
231  for (size_t i = SecondContractedIndexPos + 1;
232  i < num_uncontracted_tensor_indices; i++) {
233  gsl::at(contracting_tensor_index, i) =
234  gsl::at(lhs_contracted_multi_index, i - 2);
235  }
236  return contracting_tensor_index;
237  }

239  /// \brief Return the storage indices of the uncontracted LHS components to
240  /// be summed to compute a contracted LHS component
241  ///
242  /// \details
243  /// Example: If we have RHS tensor \f$R^{a}{}_{abc}\f$ and we want to contract
244  /// it to the LHS tensor \f$L_{cb}\f$, then \f$L_{cb}\f$ represents the
245  /// contracted LHS, while \f$L^{a}{}_{acb}\f$ represents the uncontracted
246  /// LHS. `I` represents the storage index of the component \f$L_{cb}\f$ for
247  /// some \f$c\f$ and \f$b\f$, an uncontracted LHS component that we wish to
248  /// compute. If `c == 1` and `b == 2`, then this function computes and returns
249  /// the list of storage indices of components \f$L^{a}{}_{a12}\f$ for all
250  /// values of \f$a\f$, i.e. the components to sum to compute the component
251  /// \f$L_{12}\f$.
252  ///
253  /// \tparam I the storage index of a contracted LHS component to be computed
254  /// \tparam UncontractedLhsStructure the Structure of the uncontracted LHS
255  /// tensor
256  /// \tparam ContractedLhsStructure the Structure of the contracted LHS tensor
257  /// \tparam Ints a sequence of integers from [0, dimension of contracted
258  /// indices)
259  /// \return the storage indices of the uncontracted LHS components to be
260  /// summed to compute a contracted LHS component
261  template <size_t I, typename UncontractedLhsStructure,
262  typename ContractedLhsStructure, size_t... Ints>
263  SPECTRE_ALWAYS_INLINE static constexpr std::array<size_t,
264  first_contracted_index::dim>
265  get_storage_indices_to_sum(
266  const std::index_sequence<Ints...>& /*dim_seq*/) noexcept {
267  constexpr std::array<size_t, num_tensor_indices>
268  lhs_contracted_multi_index =
269  ContractedLhsStructure::get_canonical_tensor_index(I);
270  return {{UncontractedLhsStructure::get_storage_index(
271  get_tensor_index_to_sum(lhs_contracted_multi_index, Ints))...}};
272  }

274  /// \brief Computes a mapping between the storage indices of the contracted
275  /// LHS components and the uncontracted LHS components to sum for a
276  /// contraction
277  ///
278  /// \details
279  /// Example: If we have RHS tensor \f$R^{a}{}_{abc}\f$ and we want to contract
280  /// it to the LHS tensor \f$L_{cb}\f$, then \f$L_{cb}\f$ represents the
281  /// contracted LHS, while \f$L^{a}{}_{acb}\f$ represents the uncontracted
282  /// LHS. This function computes and returns a mapping between the storage
283  /// indices of (1) each component of \f$L_{cb}\f$ and (2) the corresponding
284  /// lists of components of \f$L^{a}{}_{acb}\f$ to sum to compute each
285  /// component of \f$L_{cb}\f$.
286  ///
287  /// \tparam ContractedLhsNumComponents the number of components in the
288  /// contracted LHS tensor
289  /// \tparam UncontractedLhsStructure the Structure of the uncontracted LHS
290  /// tensor
291  /// \tparam ContractedLhsStructure the Structure of the contracted LHS tensor
292  /// \tparam Ints a sequence of integers from [0, `ContractedLhsNumComponents`)
293  /// \return a mapping between the storage indices of the contracted LHS
294  /// components and the uncontracted LHS components to sum for a contraction
295  template <size_t ContractedLhsNumComponents,
296  typename UncontractedLhsStructure, typename ContractedLhsStructure,
297  size_t... Ints>
298  SPECTRE_ALWAYS_INLINE static constexpr std::array<
299  std::array<size_t, first_contracted_index::dim>,
300  ContractedLhsNumComponents>
301  get_map_of_components_to_sum(
302  const std::index_sequence<Ints...>& /*index_seq*/) noexcept {
303  constexpr std::make_index_sequence<first_contracted_index::dim> dim_seq{};
304  return {{get_storage_indices_to_sum<Ints, UncontractedLhsStructure,
305  ContractedLhsStructure>(dim_seq)...}};

346  template <typename... UncontractedLhsTensorIndices>
347  struct ComputeContraction<tmpl::list<UncontractedLhsTensorIndices...>> {
348  /// \brief Computes the value of the component in the contracted LHS tensor
349  /// at a given storage index
350  ///
351  /// \details
352  /// This function recursively computes the value of the component in the
353  /// contracted LHS tensor at a given storage index by iterating over the
354  /// list of storage indices of uncontracted LHS components to sum. This list
355  /// is stored at `map_of_components_to_sum[lhs_storage_index]`, and the
356  /// current component being summed is at
357  /// `map_of_components_to_sum[lhs_storage_index][Index]`.
358  ///
359  /// \tparam UncontractedLhsStructure the Structure of the uncontracted LHS
360  /// tensor
361  /// \tparam ContractedLhsNumComponents the number of components in the
362  /// contracted LHS tensor
363  /// \tparam Index for a given list of uncontracted LHS storage indices whose
364  /// components are summed to compute a contracted LHS component, this is the
365  /// position of one such storage index in that list
366  /// \param map_of_components_to_sum a mapping between the storage indices of
367  /// the contracted LHS components and the uncontracted LHS components to sum
368  /// to compute the former
369  /// \param t1 the expression contained within the RHS contraction expression
370  /// \param lhs_storage_index the storage index of the LHS tensor component
371  /// to compute
372  /// \return the computed value of the component at `lhs_storage_index` in
373  /// the contracted LHS tensor
374  template <typename UncontractedLhsStructure,
375  size_t ContractedLhsNumComponents, size_t Index, typename T1>
376  static SPECTRE_ALWAYS_INLINE decltype(auto) apply(
377  const std::array<std::array<size_t, first_contracted_index::dim>,
378  ContractedLhsNumComponents>& map_of_components_to_sum,
379  const T1& t1, const size_t& lhs_storage_index) noexcept {
380  if constexpr (Index < first_contracted_index::dim - 1) {
381  // We have more than one component left to sum
382  return apply<UncontractedLhsStructure, ContractedLhsNumComponents,
383  Index + 1>(map_of_components_to_sum, t1,
384  lhs_storage_index) +
385  t1.template get<UncontractedLhsStructure,
386  UncontractedLhsTensorIndices...>(
387  gsl::at(gsl::at(map_of_components_to_sum, lhs_storage_index),
388  Index));
389  } else {
390  // We only have one final component to sum
391  return t1.template get<UncontractedLhsStructure,
392  UncontractedLhsTensorIndices...>(
393  gsl::at(gsl::at(map_of_components_to_sum, lhs_storage_index),
394  first_contracted_index::dim - 1));
395  }
396  }
397  };

399  /// \brief Return the value of the component of the contracted LHS tensor at a
400  /// given storage index
401  ///
402  /// \details
403  /// Given a RHS tensor to be contracted, the uncontracted LHS represents the
404  /// uncontracted RHS tensor arranged with the LHS's generic index order. The
405  /// contracted LHS represents the result of contracting this uncontracted
406  /// LHS. For example, if we have RHS tensor \f$R^{a}{}_{abc}\f$ and we want to
407  /// contract it to the LHS tensor \f$L_{cb}\f$, then \f$L_{cb}\f$ represents
408  /// the contracted LHS, while \f$L^{a}{}_{acb}\f$ represents the uncontracted
409  /// LHS. Note that the relative ordering of the LHS generic indices \f$c\f$
410  /// and \f$b\f$ in the contracted LHS is preserved in the uncontracted LHS.
411  ///
412  /// To compute a contraction, we need to get all the uncontracted LHS
413  /// components to sum. In the example above, this means that in order to
414  /// compute \f$L_{cb}\f$ for some \f$c\f$ and \f$b\f$, we need to sum the
415  /// components \f$L^{a}{}_{acb}\f$ for all values of \f$a\f$. This function
416  /// first constructs the list of generic indices (TensorIndexs) of the
417  /// uncontracted LHS, then uses a series of helper functions to compute a
418  /// mapping from (1) the storage indices of the components in the contracted
419  /// LHS tensor to (2) their corresponding lists of storage indices of
420  /// components in the uncontracted LHS tensor that need to be summed to
421  /// compute each contracted LHS component. Finally, the `ComputeContraction`
422  /// helper struct is used to compute the contracted component at
423  /// `lhs_storage_index` by leveraging this precomputed map's lists of indices
424  /// to sum for each contracted LHS component's storage index.
425  ///
426  /// \tparam LhsStructure the Structure of the contracted LHS tensor
427  /// \tparam LhsIndices the TensorIndexs of the contracted LHS tensor
428  /// \param lhs_storage_index the storage index of the contracted LHS tensor
429  /// component to retrieve
430  /// \return the value of the component at `lhs_storage_index` in the
431  /// contracted LHS tensor
432  template <typename LhsStructure, typename... LhsIndices>
433  SPECTRE_ALWAYS_INLINE decltype(auto) get(
434  const size_t lhs_storage_index) const {
435  constexpr size_t contracted_lhs_num_components = LhsStructure::size();
436  using uncontracted_lhs_tensorindex_list =
437  get_uncontracted_lhs_tensorindex_list<LhsIndices...>;
438  using uncontracted_lhs_structure =
439  typename LhsTensorSymmAndIndices<ArgsList,
440  uncontracted_lhs_tensorindex_list,
441  Symm, IndexList>::structure;
442 
443  constexpr std::make_index_sequence<contracted_lhs_num_components> map_seq{};
444 
445  // A map from contracted LHS storage indices to lists of uncontracted LHS
446  // storage indices of components to sum for contraction
447  constexpr std::array<std::array<size_t, first_contracted_index::dim>,
448  contracted_lhs_num_components>
449  map_of_components_to_sum =
450  get_map_of_components_to_sum<contracted_lhs_num_components,
451  uncontracted_lhs_structure,
452  LhsStructure>(map_seq);
453 
454  // This returns the value of the component stored at `lhs_storage_index` in
455  // the contracted LHS tensor
456  return ComputeContraction<uncontracted_lhs_tensorindex_list>::
457  template apply<uncontracted_lhs_structure,
458  contracted_lhs_num_components, 0>(
459  map_of_components_to_sum, t_, lhs_storage_index);
460  }
461 
462  private:
463  const std::conditional_t<std::is_base_of<Expression, T>::value, T,
464  TensorExpression<T, X, Symm, IndexList, ArgsList>>
465  t_;

468 /*!
469  * \ingroup TensorExpressionsGroup
470  * \brief Returns the positions of the first indices to contract in an
471  * expression
472  *
473  * \details Given a list of values that represent an expression's generic index
474  * encodings, this function looks to see if it can find a pair of values that
475  * encode one generic index and the generic index with opposite valence, such as
476  * `ti_A` and `ti_a`. This denotes a pair of indices that will need to be
477  * contracted. If there exists more than one such pair of indices in the
478  * expression, the first pair of values found will be returned.
479  *
480  * For example, if we have tensor \f$R^{ab}{}_{ab}\f$ represented by the tensor
481  * expression, `R(ti_A, ti_B, ti_a, ti_b)`, then this will return the positions
482  * of the pair of values encoding `ti_A` and `ti_a`, which would be (0, 2)
483  *
484  * @param tensorindex_values the TensorIndex values of a tensor expression
485  * @return the positions of the first pair of TensorIndex values to contract
486  */
487 template <size_t NumIndices>
488 SPECTRE_ALWAYS_INLINE static constexpr std::pair<size_t, size_t>
489 get_first_index_positions_to_contract(
490  const std::array<size_t, NumIndices>& tensorindex_values) noexcept {
491  for (size_t i = 0; i < tensorindex_values.size(); ++i) {
492  const size_t current_value = gsl::at(tensorindex_values, i);
493  const size_t opposite_value_to_find =
494  get_tensorindex_value_with_opposite_valence(current_value);
495  for (size_t j = i + 1; j < tensorindex_values.size(); ++j) {
496  if (opposite_value_to_find == gsl::at(tensorindex_values, j)) {
497  // We found both the lower and upper version of a generic index in the
498  // list of generic indices, so we return this pair's positions
499  return std::pair{i, j};
500  }
501  }
502  }
503  // We couldn't find a single pair of indices that needs to be contracted
504  return std::pair{std::numeric_limits<size_t>::max(),
505  std::numeric_limits<size_t>::max()};
506 }

508 /*!
509  * \ingroup TensorExpressionsGroup
510  * \brief Creates a contraction expression from a tensor expression if there are
511  * any indices to contract
512  *
513  * \details If there are no indices to contract, the input TensorExpression is
514  * simply returned. Otherwise, a contraction expression is created for
515  * contracting one pair of upper and lower indices. If there is more than one
516  * pair of indices to contract, subsequent contraction expressions are
517  * recursively created, nesting one contraction expression inside another.
518  *
519  * For example, if we have tensor \f$R^{ab}{}_{ab}\f$ represented by the tensor
520  * expression, `R(ti_A, ti_B, ti_a, ti_b)`, then one contraction expression is
521  * created to represent contracting \f$R^{ab}{}_ab\f$ to \f$R^b{}_b\f$, and a
522  * second to represent contracting \f$R^b{}_b\f$ to the scalar, \f$R\f$.
523  *
524  * @param t the TensorExpression to potentially contract
525  * @return the input tensor expression or a contraction expression of the input
526  */
527 template <typename T, typename X, typename Symm, typename IndexList,
528  typename... TensorIndices>
529 SPECTRE_ALWAYS_INLINE static constexpr auto contract(
530  const TensorExpression<T, X, Symm, IndexList, tmpl::list<TensorIndices...>>&
531  t) noexcept {
532  constexpr std::array<size_t, sizeof...(TensorIndices)> tensorindex_values = {
533  {TensorIndices::value...}};
534  constexpr std::pair first_index_positions_to_contract =
535  get_first_index_positions_to_contract(tensorindex_values);
536  constexpr std::pair no_indices_to_contract_sentinel{
537  std::numeric_limits<size_t>::max(), std::numeric_limits<size_t>::max()};
538 
539  if constexpr (first_index_positions_to_contract ==
540  no_indices_to_contract_sentinel) {
541  // There aren't any indices to contract, so we just return the input
542  return ~t;
543  } else {
544  // We have a pair of indices to be contract
545  return contract(
546  TensorContract<first_index_positions_to_contract.first,
547  first_index_positions_to_contract.second, T, X, Symm,
548  IndexList, tmpl::list<TensorIndices...>>{t});
549  }
550 }
