51 /// \brief post interpolation callback (see InterpolationTarget)
52 /// that does a FastFlow iteration and triggers another one until convergence.
53 ///
54 /// Assumes that InterpolationTargetTag contains an additional
55 /// struct called `post_horizon_find_callback`, which has a function
56 ///```
57 /// static void apply(const DataBox<DbTags>&,
58 /// const intrp::GlobalCache<Metavariables>&,
59 /// const Metavariables::temporal_id&) noexcept;
60 ///```
61 /// that is called if the FastFlow iteration has converged.
62 ///
63 /// Uses:
64 /// - Metavariables:
65 /// - `temporal_id`
66 /// - DataBox:
67 /// - `logging::Tags::Verbosity<InterpolationTargetTag>`
68 /// - `::gr::Tags::InverseSpatialMetric<3,Frame>`
69 /// - `::gr::Tags::ExtrinsicCurvature<3,Frame>`
70 /// - `::gr::Tags::SpatialChristoffelSecondKind<3,Frame>`
71 /// - `::ah::Tags::FastFlow`
72 /// - `StrahlkorperTags::Strahlkorper<Frame>`
73 ///
74 /// Modifies:
75 /// - DataBox:
76 /// - `::ah::Tags::FastFlow`
77 /// - `StrahlkorperTags::Strahlkorper<Frame>`
78 ///
79 /// This is an InterpolationTargetTag::post_interpolation_callback;
80 /// see InterpolationTarget for a description of InterpolationTargetTag.
81 template <typename InterpolationTargetTag, typename Frame>
82 struct FindApparentHorizon {
83  using observation_types = typename InterpolationTargetTag::
84  post_horizon_find_callback::observation_types;
85  template <typename DbTags, typename Metavariables, typename TemporalId>
86  static bool apply(
87  const gsl::not_null<db::DataBox<DbTags>*> box,
88  const gsl::not_null<Parallel::GlobalCache<Metavariables>*> cache,
89  const TemporalId& temporal_id) noexcept {
90  // Before doing anything else, deal with the possibility that some
91  // of the points might be outside of the Domain.
92  const auto& indices_of_invalid_pts =
93  db::get<Tags::IndicesOfInvalidInterpPoints<TemporalId>>(*box);
94  if (indices_of_invalid_pts.count(temporal_id) > 0 and
95  not indices_of_invalid_pts.at(temporal_id).empty()) {
96  ERROR("FindApparentHorizon: Found points that are not in any block");
97  }
98 
99  const auto& verbosity =
100  db::get<logging::Tags::Verbosity<InterpolationTargetTag>>(*box);
101  const auto& inv_g =
102  db::get<::gr::Tags::InverseSpatialMetric<3, Frame>>(*box);
103  const auto& ex_curv =
104  db::get<::gr::Tags::ExtrinsicCurvature<3, Frame>>(*box);
105  const auto& christoffel =
106  db::get<::gr::Tags::SpatialChristoffelSecondKind<3, Frame>>(*box);
107 
108  std::pair<FastFlow::Status, FastFlow::IterInfo> status_and_info;
109 
110  // Do a FastFlow iteration.
111  db::mutate<::ah::Tags::FastFlow, StrahlkorperTags::Strahlkorper<Frame>>(
112  box,
113  [&inv_g, &ex_curv, &christoffel, &status_and_info ](
114  const gsl::not_null<::FastFlow*> fast_flow,
115  const gsl::not_null<::Strahlkorper<Frame>*> strahlkorper) noexcept {
116  status_and_info = fast_flow->template iterate_horizon_finder<Frame>(
117  strahlkorper, inv_g, ex_curv, christoffel);
118  });
119 
120  // Determine whether we have converged, whether we need another step,
121  // or whether we have encountered an error.
122 
123  const auto& status = status_and_info.first;
124  const auto& info = status_and_info.second;
125  const auto has_converged = converged(status);
126 
127  if (verbosity > ::Verbosity::Quiet or
128  (verbosity > ::Verbosity::Silent and has_converged)) {
129  // The things printed out here are:
130  // its = current iteration number.
131  // R = min and max of residual over all prolonged grid points.
132  // |R| = L2 norm of residual, counting only L modes solved for.
133  // |R_mesh| = L2 norm of residual over prolonged grid points.
134  // r = min and max radius of trial horizon surface.
135  //
136  // Difference between |R| and |R_mesh|:
137  // The horizon is represented in a Y_lm expansion up to l=l_surface;
138  // the residual |R| represents the failure of that surface to satisfy
139  // the apparent horizon equation.
140  //
141  // However, at each iteration we also interpolate the horizon surface
142  // to a higher resolution ("prolongation"). The prolonged surface
143  // includes Ylm coefficents up to l=l_mesh, where l_mesh > l_surface.
144  // The residual computed on this higher-resolution surface is |R_mesh|.
145  //
146  // As iterations proceed, |R| should decrease until it reaches numerical
147  // roundoff error, because we are varying all the Y_lm coefficients up to
148  // l=l_surface to minimize the residual. However, |R_mesh| should
149  // eventually stop decreasing as iterations proceed, hitting a
150  // floor that represents the numerical truncation error of the solution.
151  //
152  // The convergence criterion looks at both |R| and |R_mesh|: Once
153  // |R| is small enough and |R_mesh| has stabilized, it is pointless
154  // to keep iterating (even though one could iterate until |R| reaches
155  // roundoff).
156  //
157  // Problems with convergence of the apparent horizon finder can often
158  // be diagnosed by looking at the behavior of |R| and |R_mesh| as a
159  // function of iteration.
160  Parallel::printf(
161  "%s: its=%d: %.1e<R<%.0e, |R|=%.1g, "
162  "|R_grid|=%.1g, %.4g<r<%.4g\n",
163  pretty_type::short_name<InterpolationTargetTag>(), info.iteration,
164  info.min_residual, info.max_residual, info.residual_ylm,
165  info.residual_mesh, info.r_min, info.r_max);
166  }
167 
168  if (status == FastFlow::Status::SuccessfulIteration) {
169  // Do another iteration of the same horizon search.
170  const auto& temporal_ids =
171  db::get<intrp::Tags::TemporalIds<TemporalId>>(*box);
172  auto& interpolation_target = Parallel::get_parallel_component<
173  intrp::InterpolationTarget<Metavariables, InterpolationTargetTag>>(
174  *cache);
175  Parallel::simple_action<
176  Actions::SendPointsToInterpolator<InterpolationTargetTag>>(
177  interpolation_target, temporal_ids.front());
178  // We return false because we don't want this iteration to clean
179  // up the volume data, since we are using it for the next iteration
180  // (i.e. the simple_action that we just called).
181  return false;
182  } else if (not has_converged) {
183  ERROR("Apparent horizon finder "
184  << pretty_type::short_name<InterpolationTargetTag>()
185  << " failed, reason = " << status);
186  }
187  // If we get here, the horizon finder has converged.
188 
189  // The interpolated variables
190  // ::Tags::Variables<InterpolationTargetTag::vars_to_interpolate_to_target>
191  // have been interpolated from the volume to the points on the
192  // prolonged_strahlkorper, not to the points on the actual
193  // strahlkorper. So here we do a restriction of these quantities onto
194  // the actual strahlkorper.
195 
196  // Type alias to make code more understandable.
197  using vars_tags =
198  typename InterpolationTargetTag::vars_to_interpolate_to_target;
199  db::mutate_apply<tmpl::list<::Tags::Variables<vars_tags>>,
200  tmpl::list<StrahlkorperTags::Strahlkorper<Frame>,
201  ::ah::Tags::FastFlow>>(
202  [](const gsl::not_null<Variables<vars_tags>*> vars,
203  const Strahlkorper<Frame>& strahlkorper,
204  const FastFlow& fast_flow) noexcept {
205  const size_t L_mesh = fast_flow.current_l_mesh(strahlkorper);
206  const auto prolonged_strahlkorper =
207  Strahlkorper<Frame>(L_mesh, L_mesh, strahlkorper);
208  auto new_vars = ::Variables<vars_tags>(
209  strahlkorper.ylm_spherepack().physical_size());
210 
211  tmpl::for_each<vars_tags>([
212  &strahlkorper, &prolonged_strahlkorper, &vars, &new_vars
213  ](auto tag_v) noexcept {
214  using tag = typename decltype(tag_v)::type;
215  const auto& old_var = get<tag>(*vars);
216  auto& new_var = get<tag>(new_vars);
217  auto old_iter = old_var.begin();
218  auto new_iter = new_var.begin();
219  for (; old_iter != old_var.end() and new_iter != new_var.end();
220  ++old_iter, ++new_iter) {
221  *new_iter = strahlkorper.ylm_spherepack().spec_to_phys(
222  prolonged_strahlkorper.ylm_spherepack().prolong_or_restrict(
223  prolonged_strahlkorper.ylm_spherepack().phys_to_spec(
224  *old_iter),
225  strahlkorper.ylm_spherepack()));
226  }
227  });
228  *vars = std::move(new_vars);
229  },
230  box);
231 
232  InterpolationTargetTag::post_horizon_find_callback::apply(*box, *cache,
233  temporal_id);
234 
235  // Prepare for finding horizon at a new time.
236  // For now, the initial guess for the new
237  // horizon is the result of the old one, so we don't need
238  // to modify the strahlkorper.
239  // Eventually we will hold more than one previous guess and
240  // will do time-extrapolation to set the next guess.
241  db::mutate<::ah::Tags::FastFlow>(
242  box, [](const gsl::not_null<::FastFlow*> fast_flow) noexcept {
243  fast_flow->reset_for_next_find();
244  });
245  // We return true because we are now done with all the volume data
246  // at this temporal_id, so we want it cleaned up.
247  return true;
248  }
249 };
250 } // namespace callbacks
251 } // namespace intrp
