19 /*!
20  * \ingroup DataStructuresGroup
21  * \brief A dynamically sized matrix of arbitrary type.
22  *
23  * \note This is a thin wrapper around `blaze::DynamicMatrix`. Please refer to
24  * the [Blaze
25  * documentation](https://bitbucket.org/blaze-lib/blaze/wiki/Matrices) for
26  * information on how to use it.
27  *
28  * \warning In Blaze 3.7, a combination of changes were made such that our use
29  * of `DynamicMatrix` winds up with non-optional padding associated with the
30  * length of vector registers on the target hardware, which often adds some
31  * number of zeros to the fastest-varying dimension. To be compatible with
32  * LAPACK calls in Blaze 3.7, it is necessary to use `blaze::columnMajor`
33  * ordering (the current default storage order for this custom class) and to
34  * provide `spacing()` to the LAPACK `LDA`, `LDB`, etc. parameters
35  * whenever using the `data()` pointer from this class. This is a defect that is
36  * expected to be solved by additional configuration options in future versions
37  * of Blaze.
38  */
39 template <typename T, bool SO = blaze::columnMajor>
40 class DenseMatrix : public blaze::DynamicMatrix<T, SO> {
41  public:
42  // Inherit constructors
43  using blaze::DynamicMatrix<T, SO>::DynamicMatrix;
44 
45  /// Charm++ serialization
46  // clang-tidy: runtime-references
47  void pup(PUP::er& p) noexcept { // NOLINT
48  auto rows = blaze::DynamicMatrix<T, SO>::rows();
49  auto columns = blaze::DynamicMatrix<T, SO>::columns();
50  p | rows;
51  p | columns;
52  if (p.isUnpacking()) {
53  blaze::DynamicMatrix<T, SO>::resize(rows, columns);
54  }
55  auto spacing = blaze::DynamicMatrix<T, SO>::spacing();
56  if (std::is_fundamental_v<T>) {
57  PUParray(p, blaze::DynamicMatrix<T, SO>::data(), columns * spacing);
58  } else {
59  for (size_t i = 0; i < columns * spacing; i++) {
60  p | blaze::DynamicMatrix<T, SO>::data()[i];
61  }
62  }
63  }
64 };

67 struct Options::create_from_yaml<DenseMatrix<T, SO>> {
68  template <typename Metavariables>
69  static DenseMatrix<T, SO> create(const Options::Option& options) {
70  const auto data = options.parse_as<std::vector<std::vector<T>>>();
71  const size_t num_rows = data.size();
72  size_t num_cols = 0;
73  if (num_rows > 0) {
74  num_cols = data[0].size();
75  }
76  DenseMatrix<T, SO> result(num_rows, num_cols);
77  // We don't use an iterator over the matrix here to make the code
78  // independent of the matrix storage order (see the [Blaze
79  // documentation](https://bitbucket.org/
80  // blaze-lib/blaze/wiki/Matrix%20Operations#!iterators)).
81  for (size_t i = 0; i < num_rows; i++) {
82  const auto& row = gsl::at(data, i);
83  if (row.size() != num_cols) {
84  PARSE_ERROR(options.context(),
85  "All matrix columns must have the same size.");
86  }
87  for (size_t j = 0; j < num_cols; j++) {
88  result(i, j) = gsl::at(row, j);
89  }
90  }
91  return result;
92  }
