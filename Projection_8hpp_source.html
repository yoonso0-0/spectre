22 /*!
23  * \brief The projection matrix from a 1D mortar to an element.
24  *
25  * \details
26  * The projection matrices returned by this function (and by
27  * projection_matrix_element_to_mortar()) define orthogonal projection
28  * operators between the spaces of functions on the boundary of the
29  * element and the mortar. These projections are usually the correct
30  * way to transfer data onto and off of the mortars.
31  *
32  * The half-interval projections are based on an equation derived by
33  * Saul. This shows that the projection from the spectral basis for
34  * the entire interval to the spectral basis for the upper half
35  * interval is
36  * \f{equation*}
37  * T_{jk} = \frac{2 j + 1}{2} 2^j \sum_{n=0}^{j-k} \binom{j}{k+n}
38  * \binom{(j + k + n - 1)/2}{j} \frac{(k + n)!^2}{(2 k + n + 1)! n!}
39  * \f}
40  */
41 const Matrix& projection_matrix_mortar_to_element(
42  MortarSize size, const Mesh<1>& element_mesh,
43  const Mesh<1>& mortar_mesh) noexcept;
