29 /*!
30  * \ingroup ActionsGroup
31  * \brief Stores the boundary data from the GH evolution in the
32  * `Cce::InterfaceManagers::GhInterfaceManager`, and sends to the
33  * `EvolutionComponent` (template argument) if the data fulfills a prior
34  * request.
35  *
36  * \details If the new data fulfills a prior request submitted to the
37  * `Cce::InterfaceManagers::GhInterfaceManager`, this will dispatch the result
38  * to `Cce::Actions::SendToEvolution<GhWorldtubeBoundary<Metavariables>,
39  * EvolutionComponent>` for sending the processed boundary data to
40  * the `EvolutionComponent`.
41  *
42  * \ref DataBoxGroup changes:
43  * - Adds: nothing
44  * - Removes: nothing
45  * - Modifies:
46  * - `Tags::GhInterfaceManager`
47  */
48 template <typename EvolutionComponent>
49 struct ReceiveGhWorldtubeData {
50  template <typename ParallelComponent, typename... DbTags,
51  typename Metavariables, typename ArrayIndex,
52  Requires<tmpl::list_contains_v<tmpl::list<DbTags...>,
53  Tags::GhInterfaceManager>> = nullptr>
54  static void apply(
55  db::DataBox<tmpl::list<DbTags...>>& box,
56  Parallel::GlobalCache<Metavariables>& cache,
57  const ArrayIndex& /*array_index*/, const TimeStepId& time,
58  const tnsr::aa<DataVector, 3>& spacetime_metric,
59  const tnsr::iaa<DataVector, 3>& phi, const tnsr::aa<DataVector, 3>& pi,
60  const tnsr::aa<DataVector, 3>& dt_spacetime_metric =
61  tnsr::aa<DataVector, 3>{},
62  const tnsr::iaa<DataVector, 3>& dt_phi = tnsr::iaa<DataVector, 3>{},
63  const tnsr::aa<DataVector, 3>& dt_pi =
64  tnsr::aa<DataVector, 3>{}) noexcept {
65  db::mutate<Tags::GhInterfaceManager>(
66  make_not_null(&box),
67  [&spacetime_metric, &phi, &pi, &dt_spacetime_metric, &dt_phi, &dt_pi,
68  &time, &cache](const gsl::not_null<
69  std::unique_ptr<InterfaceManagers::GhInterfaceManager>*>
70  interface_manager) noexcept {
71  (*interface_manager)
72  ->insert_gh_data(time, spacetime_metric, phi, pi,
73  dt_spacetime_metric, dt_phi, dt_pi);
74  const auto gh_data =
75  (*interface_manager)->retrieve_and_remove_first_ready_gh_data();
76  if (static_cast<bool>(gh_data)) {
77  Parallel::simple_action<Actions::SendToEvolution<
78  GhWorldtubeBoundary<Metavariables>, EvolutionComponent>>(
79  Parallel::get_parallel_component<
80  GhWorldtubeBoundary<Metavariables>>(cache),
81  get<0>(*gh_data), get<1>(*gh_data));
82  }
83  });
84  }
85 };
86 } // namespace Actions
