136  using rem_ref_value_type = typename std::remove_reference<value_type>::type;
137  using rem_ref_T = typename std::remove_reference<T>::type;
138  using is_lvalue_type = std::integral_constant<
139  bool,
140  std::is_lvalue_reference<T>::value or
141  std::is_same<std::reference_wrapper<rem_ref_value_type>,
142  rem_ref_T>::value or

144  rem_ref_value_type>::type>,
145  rem_ref_T>::value>;
146  return not std::is_reference<value_type>::value or
147  (std::is_lvalue_reference<value_type>::value and
148  is_lvalue_type::value) or
149  (std::is_rvalue_reference<value_type>::value and
150  not std::is_lvalue_reference<T>::value);
