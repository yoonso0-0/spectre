113 /// Reduction data to be written to disk.
114 ///
115 /// If we have `M` `ArrayComponentIds` registered with a given `Observer`
116 /// for a given `ObservationId`, then we expect the `Observer` to receive
117 /// `M` contributions with the given `ObservationId`. We combine the reduction
118 /// data as we receive it, so we only need one copy for each `ObservationId`.
119 template <class... ReductionDatums>
120 struct ReductionData : db::SimpleTag {
121  using type = std::unordered_map<observers::ObservationId,
122  Parallel::ReductionData<ReductionDatums...>>;
123  using names_tag = ReductionDataNames<ReductionDatums...>;
124 };
125 
126 /// Names of the reduction data to be written to disk.

129  using type =
130  std::unordered_map<observers::ObservationId, std::vector<std::string>>;
131  using data_tag = ReductionData<ReductionDatums...>;
132 };

171 /// \brief The name of the HDF5 file on disk into which volume data is written.
172 ///
173 /// By volume data we mean any data that is not written once across all nodes.
174 /// For example, data on a 2d surface written from a 3d simulation is considered
175 /// volume data, while an integral over the entire (or a subset of the) domain
176 /// is considered reduction data.
177 struct VolumeFileName : db::SimpleTag {
178  using type = std::string;
179  using option_tags = tmpl::list<::observers::OptionTags::VolumeFileName>;
180 
181  static constexpr bool pass_metavariables = false;
182  static std::string create_from_options(
183  const std::string& volume_file_name) noexcept {
184  return volume_file_name;
185  }
186 };
187 
188 /// \brief The name of the HDF5 file on disk into which reduction data is
189 /// written.
190 ///
191 /// By reduction data we mean any data that is written once across all nodes.
192 /// For example, an integral over the entire (or a subset of the) domain
193 /// is considered reduction data, while data on a 2d surface written from a 3d
194 /// simulation is considered volume data.
195 struct ReductionFileName : db::SimpleTag {
196  using type = std::string;
197  using option_tags = tmpl::list<::observers::OptionTags::ReductionFileName>;
198 
199  static constexpr bool pass_metavariables = false;
200  static std::string create_from_options(

203  }
