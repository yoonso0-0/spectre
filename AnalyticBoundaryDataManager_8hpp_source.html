80 template <typename Metavariables>
81 void AnalyticBoundaryDataManager::write_news(
82  Parallel::GlobalCache<Metavariables>& cache,
83  const double time) const noexcept {
84  const auto news = get<Tags::News>(
85  generator_->variables(l_max_, time, tmpl::list<Tags::News>{}));
86  const size_t observation_l_max = Parallel::get<Tags::ObservationLMax>(cache);
87  std::vector<double> data_to_write(2 * square(observation_l_max + 1) + 1);
88  std::vector<std::string> file_legend;
89  file_legend.reserve(2 * square(observation_l_max + 1) + 1);
90  file_legend.emplace_back("time");
91  for (int i = 0; i <= static_cast<int>(observation_l_max); ++i) {
92  for (int j = -i; j <= i; ++j) {
93  file_legend.push_back(MakeString{} << "Real Y_" << i << "," << j);
94  file_legend.push_back(MakeString{} << "Imag Y_" << i << "," << j);
95  }
96  }
97  const ComplexModalVector goldberg_modes =
98  Spectral::Swsh::libsharp_to_goldberg_modes(
99  Spectral::Swsh::swsh_transform(l_max_, 1, get(news)), l_max_)
100  .data();
101  data_to_write[0] = time;
102  for (size_t i = 0; i < square(observation_l_max + 1); ++i) {
103  data_to_write[2 * i + 1] = real(goldberg_modes[i]);
104  data_to_write[2 * i + 2] = imag(goldberg_modes[i]);
105  }
106  auto observer_proxy = Parallel::get_parallel_component<
107  observers::ObserverWriter<Metavariables>>(
108  cache)[static_cast<size_t>(Parallel::my_node())];
109  Parallel::threaded_action<observers::ThreadedActions::WriteSimpleData>(
110  observer_proxy, file_legend, data_to_write, "/expected_news"s);
111 }
112 } // namespace Cce
