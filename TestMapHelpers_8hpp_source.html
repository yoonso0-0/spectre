51 /// \ingroup TestingFrameworkGroup
52 /// \brief Given two coordinate maps (but not their types), check that the maps
53 /// are equal by evaluating them at a random set of points.
54 template <typename SourceFrame, typename TargetFrame, size_t VolumeDim>
55 void check_if_maps_are_equal(
56  const domain::CoordinateMapBase<SourceFrame, TargetFrame, VolumeDim>&
57  map_one,
58  const domain::CoordinateMapBase<SourceFrame, TargetFrame, VolumeDim>&
59  map_two,
60  const double time = std::numeric_limits<double>::quiet_NaN(),
61  const std::unordered_map<
62  std::string, std::unique_ptr<domain::FunctionsOfTime::FunctionOfTime>>&
63  functions_of_time = {}) noexcept {
64  MAKE_GENERATOR(gen);
65  std::uniform_real_distribution<> real_dis(-1, 1);
66 
67  for (size_t n = 0; n < 10; ++n) {
68  tnsr::I<double, VolumeDim, SourceFrame> source_point{};
69  for (size_t d = 0; d < VolumeDim; ++d) {
70  source_point.get(d) = real_dis(gen);
71  }
72  CAPTURE(source_point);
73  CHECK_ITERABLE_APPROX(map_one(source_point, time, functions_of_time),
74  map_two(source_point, time, functions_of_time));
75  CHECK_ITERABLE_APPROX(
76  map_one.jacobian(source_point, time, functions_of_time),
77  map_two.jacobian(source_point, time, functions_of_time));
78  CHECK_ITERABLE_APPROX(
79  map_one.inv_jacobian(source_point, time, functions_of_time),
80  map_two.inv_jacobian(source_point, time, functions_of_time));
81  }
82 }

99  * \brief Given a Map `map`, checks that the jacobian gives expected results
100  * when compared to the numerical derivative in each direction.
101  */
102 template <typename Map>
103 void test_jacobian(const Map& map,
104  const std::array<double, Map::dim>& test_point) noexcept {
105  // Our default approx value is too stringent for this test
106  Approx local_approx = Approx::custom().epsilon(1e-10).scale(1.0);
107  const double dx = 1e-4;
108  const auto jacobian = map.jacobian(test_point);
109  for (size_t i = 0; i < Map::dim; ++i) {
110  const auto numerical_deriv_i = numerical_derivative(map, test_point, i, dx);
111  for (size_t j = 0; j < Map::dim; ++j) {
112  INFO("i: " << i << " j: " << j);
113  CHECK(jacobian.get(j, i) == local_approx(gsl::at(numerical_deriv_i, j)));
114  }
115  }
116 }

146 /*!
147  * \ingroup TestingFrameworkGroup
148  * \brief Given a Map `map`, checks that the inverse jacobian and jacobian
149  * multiply together to produce the identity matrix
150  */
151 template <typename Map>
152 void test_inv_jacobian(
153  const Map& map, const std::array<double, Map::dim>& test_point) noexcept {
154  const auto jacobian = map.jacobian(test_point);
155  const auto inv_jacobian = map.inv_jacobian(test_point);
156 
157  const auto expected_identity = [&jacobian, &inv_jacobian]() {
158  std::array<std::array<double, Map::dim>, Map::dim> identity{};
159  for (size_t i = 0; i < Map::dim; ++i) {
160  for (size_t j = 0; j < Map::dim; ++j) {
161  gsl::at(gsl::at(identity, i), j) = 0.;
162  for (size_t k = 0; k < Map::dim; ++k) {
163  gsl::at(gsl::at(identity, i), j) +=
164  jacobian.get(i, k) * inv_jacobian.get(k, j);
165  }
166  }
167  }
168  return identity;
169  }();
170 
171  for (size_t i = 0; i < Map::dim; ++i) {
172  for (size_t j = 0; j < Map::dim; ++j) {
173  CHECK(gsl::at(gsl::at(expected_identity, i), j) ==
174  approx(i == j ? 1. : 0.));
175  }
176  }
177 }

244  * \brief Checks that the CoordinateMap `map` functions as expected when used as
245  * the template parameter to the `CoordinateMap` type.
246  */
247 template <typename Map, typename... Args>
248 void test_coordinate_map_implementation(const Map& map) noexcept {
249  const auto coord_map =
250  domain::make_coordinate_map<Frame::Logical, Frame::Grid>(map);
251  MAKE_GENERATOR(gen);
252  std::uniform_real_distribution<> real_dis(-1, 1);
253 
254  const auto test_point = [&gen, &real_dis] {
255  std::array<double, Map::dim> p{};
256  for (size_t i = 0; i < Map::dim; ++i) {
257  gsl::at(p, i) = real_dis(gen);
258  }
259  return p;
260  }();
261 
262  const auto test_point_tensor = [&test_point]() {
263  tnsr::I<double, Map::dim, Frame::Logical> point_as_tensor{};
264  for (size_t i = 0; i < Map::dim; ++i) {
265  point_as_tensor.get(i) = gsl::at(test_point, i);
266  }
267  return point_as_tensor;
268  }();
269 
270  for (size_t i = 0; i < Map::dim; ++i) {
271  CHECK(coord_map(test_point_tensor).get(i) ==
272  approx(gsl::at(map(test_point), i)));
273  for (size_t j = 0; j < Map::dim; ++j) {
274  CHECK(coord_map.jacobian(test_point_tensor).get(i, j) ==
275  map.jacobian(test_point).get(i, j));
276  CHECK(coord_map.inv_jacobian(test_point_tensor).get(i, j) ==
277  map.inv_jacobian(test_point).get(i, j));
278  }

282 /*!
283  * \ingroup TestingFrameworkGroup
284  * \brief Checks that the CoordinateMap `map` functions as expected when used
285  * with different argument types.
286  */
287 template <typename Map, typename... Args>
288 void test_coordinate_map_argument_types(
289  const Map& map, const std::array<double, Map::dim>& test_point,
290  const Args&... args) {
291  const auto make_array_data_vector = [](const auto& double_array) noexcept {
292  std::array<DataVector, Map::dim> result;
293  std::transform(double_array.begin(), double_array.end(), result.begin(),
294  [](const double x) noexcept {
295  return DataVector{x, x};
296  });
297  return result;
298  };
299  const auto add_reference_wrapper = [](const auto& unwrapped_array) noexcept {
300  using Arg = std::decay_t<decltype(unwrapped_array)>;
301  return make_array<std::reference_wrapper<const typename Arg::value_type>,
302  Map::dim>(unwrapped_array);
303  };
304 
305  {
306  const auto mapped_point = map(test_point, args...);
307  CHECK_ITERABLE_APPROX(map(add_reference_wrapper(test_point), args...),
308  mapped_point);
309  CHECK_ITERABLE_APPROX(map(make_array_data_vector(test_point), args...),
310  make_array_data_vector(mapped_point));
311  CHECK_ITERABLE_APPROX(
312  map(add_reference_wrapper(make_array_data_vector(test_point)), args...),
313  make_array_data_vector(mapped_point));
314  }
315 
316  // Here, time_args is a const auto& not const Args& because time_args
317  // is allowed to be different than Args (which was the reason for the
318  // overloader below that calls this function).
319  const auto check_jac =
320  [](const auto& make_arr_data_vec, const auto& add_ref_wrap,
321  const Map& the_map, const std::array<double, Map::dim>& point,
322  const auto&... time_args) noexcept {
323  const auto make_tensor_data_vector = [](const auto& double_tensor) {
324  using Arg = std::decay_t<decltype(double_tensor)>;
325  Tensor<DataVector, typename Arg::symmetry, typename Arg::index_list>
326  result;
327  std::transform(double_tensor.begin(), double_tensor.end(), result.begin(),
328  [](const double x) {
329  return DataVector{x, x};
330  });
331  return result;
332  };
333 
334  {
335  const auto expected = the_map.jacobian(point, time_args...);
336  CHECK_ITERABLE_APPROX(the_map.jacobian(add_ref_wrap(point), time_args...),
337  expected);
338  CHECK_ITERABLE_APPROX(
339  the_map.jacobian(make_arr_data_vec(point), time_args...),
340  make_tensor_data_vector(expected));
341  CHECK_ITERABLE_APPROX(
342  the_map.jacobian(add_ref_wrap(make_arr_data_vec(point)),
343  time_args...),
344  make_tensor_data_vector(expected));
345  }
346  {
347  const auto expected = the_map.inv_jacobian(point, time_args...);
348  CHECK_ITERABLE_APPROX(
349  the_map.inv_jacobian(add_ref_wrap(point), time_args...), expected);
350  CHECK_ITERABLE_APPROX(
351  the_map.inv_jacobian(make_arr_data_vec(point), time_args...),
352  make_tensor_data_vector(expected));
353  CHECK_ITERABLE_APPROX(
354  the_map.inv_jacobian(add_ref_wrap(make_arr_data_vec(point)),
355  time_args...),
356  make_tensor_data_vector(expected));
357  }
358 
359  return nullptr;
360  };
361 
362  const auto jac_overloader = make_overloader(
363  // The first two functions are passed through the jacobian
364  // overloader and the check_jac function due to gcc failing to deduce auto
365  [&check_jac](const auto& make_array_data_vec, const auto& add_ref_wrapper,
366  const Map& this_map,
367  const std::array<double, Map::dim>& this_point,
368  const std::false_type /*is_time_independent*/,
369  const Args&... /*the_args*/) noexcept {
370  check_jac(make_array_data_vec, add_ref_wrapper, this_map, this_point);
371  return nullptr;
372  },
373  [&check_jac](const auto& make_array_data_vec, const auto& add_ref_wrapper,
374  const Map& this_map,
375  const std::array<double, Map::dim>& this_point,
376  const std::true_type /*is_time_dependent*/,
377  const Args&... the_args) noexcept {
378  check_jac(make_array_data_vec, add_ref_wrapper, this_map, this_point,
379  the_args...);
380  return nullptr;
381  });
382 
383  jac_overloader(make_array_data_vector, add_reference_wrapper, map, test_point,
384  domain::is_jacobian_time_dependent_t<decltype(map), double>{},
385  args...);
386 }

398 /*!
399  * \ingroup TestingFrameworkGroup
400  * \brief Given a Map `map`, tests the map functions, including map inverse,
401  * jacobian, and inverse jacobian, for a series of points.
402  * These points are chosen in a dim-dimensonal cube of side 2 centered at
403  * the origin. The map is expected to be valid on the boundaries of the cube.
404  */
405 template <typename Map>
406 void test_suite_for_map_on_unit_cube(const Map& map) noexcept {
407  // Set up random number generator
408  MAKE_GENERATOR(gen);
409  std::uniform_real_distribution<> real_dis(-1.0, 1.0);
410 
411  std::array<double, Map::dim> origin{};
412  std::array<double, Map::dim> random_point{};
413  for (size_t i = 0; i < Map::dim; i++) {
414  gsl::at(origin, i) = 0.0;
415  gsl::at(random_point, i) = real_dis(gen);
416  }
417 
418  const auto test_helper =
419  [&origin, &random_point ](const auto& map_to_test) noexcept {
420  test_serialization(map_to_test);
421  CHECK_FALSE(map_to_test != map_to_test);
422  test_coordinate_map_argument_types(map_to_test, origin);
423 
424  test_jacobian(map_to_test, origin);
425  test_inv_jacobian(map_to_test, origin);
426  test_inverse_map(map_to_test, origin);
427 
428  for (VolumeCornerIterator<Map::dim> vci{}; vci; ++vci) {
429  test_jacobian(map_to_test, vci.coords_of_corner());
430  test_inv_jacobian(map_to_test, vci.coords_of_corner());
431  test_inverse_map(map_to_test, vci.coords_of_corner());
432  }
433 
434  test_jacobian(map_to_test, random_point);
435  test_inv_jacobian(map_to_test, random_point);
436  test_inverse_map(map_to_test, random_point);
437  };
438  test_helper(map);
439  const auto map2 = serialize_and_deserialize(map);
440  check_if_maps_are_equal(
441  domain::make_coordinate_map<Frame::Logical, Frame::Grid>(map),
442  domain::make_coordinate_map<Frame::Logical, Frame::Grid>(map2));
443  test_helper(map2);
444 }

446 /*!
447  * \ingroup TestingFrameworkGroup
448  * \brief Given a Map `map`, tests the map functions, including map inverse,
449  * jacobian, and inverse jacobian, for a series of points.
450  * These points are chosen in a sphere of radius `radius_of_sphere`, and the
451  * map is expected to be valid on the boundary of that sphere as well as
452  * in its interior. The flag `include_origin` indicates whether to test the
453  * map at the origin.
454  * This test works only in 3 dimensions.
455  */
456 template <typename Map>
457 void test_suite_for_map_on_sphere(
458  const Map& map, const bool include_origin = true,
459  const double radius_of_sphere = 1.0) noexcept {
460  static_assert(Map::dim == 3, "Works only for a 3d map");
461 
462  // Set up random number generator
463  MAKE_GENERATOR(gen);
464 
465  // If we don't include the origin, we want to use some finite inner
466  // boundary so that random points stay away from the origin.
467  // test_jacobian has a dx of 1.e-4 for finite-differencing, so here
468  // we pick a value larger than that.
469  const double inner_bdry = include_origin ? 0.0 : 5.e-3;
470 
471  std::uniform_real_distribution<> radius_dis(inner_bdry, radius_of_sphere);
472  std::uniform_real_distribution<> theta_dis(0, M_PI);
473  std::uniform_real_distribution<> phi_dis(0, 2.0 * M_PI);
474 
475  const double theta = theta_dis(gen);
476  CAPTURE(theta);
477  const double phi = phi_dis(gen);
478  CAPTURE(phi);
479  const double radius = radius_dis(gen);
480  CAPTURE(radius);
481 
482  const std::array<double, 3> random_point{{radius * sin(theta) * cos(phi),
483  radius * sin(theta) * sin(phi),
484  radius * cos(theta)}};
485 
486  const std::array<double, 3> random_bdry_point{
487  {radius_of_sphere * sin(theta) * cos(phi),
488  radius_of_sphere * sin(theta) * sin(phi),
489  radius_of_sphere * cos(theta)}};
490 
491  // This point is either the origin or (if include_origin is false)
492  // it is some random point on the inner boundary.
493  const std::array<double, 3> random_inner_bdry_point_or_origin{
494  {inner_bdry * sin(theta) * cos(phi),
495  inner_bdry * sin(theta) * sin(phi),
496  inner_bdry * cos(theta)}};
497 
498  const auto test_helper =
499  [&random_bdry_point, &random_inner_bdry_point_or_origin,
500  &random_point ](const auto& map_to_test) noexcept {
501  test_serialization(map_to_test);
502  CHECK_FALSE(map_to_test != map_to_test);
503 
504  test_coordinate_map_argument_types(map_to_test,
505  random_inner_bdry_point_or_origin);
506  test_jacobian(map_to_test, random_inner_bdry_point_or_origin);
507  test_inv_jacobian(map_to_test, random_inner_bdry_point_or_origin);
508  test_inverse_map(map_to_test, random_inner_bdry_point_or_origin);
509 
510  test_coordinate_map_argument_types(map_to_test, random_point);
511  test_jacobian(map_to_test, random_point);
512  test_inv_jacobian(map_to_test, random_point);
513  test_inverse_map(map_to_test, random_point);
514 
515  test_jacobian(map_to_test, random_bdry_point);
516  test_inv_jacobian(map_to_test, random_bdry_point);
517  test_inverse_map(map_to_test, random_bdry_point);
518  };
519  test_helper(map);
520  const auto map2 = serialize_and_deserialize(map);
521  check_if_maps_are_equal(
522  domain::make_coordinate_map<Frame::Logical, Frame::Grid>(map),
523  domain::make_coordinate_map<Frame::Logical, Frame::Grid>(map2));
524  test_helper(map2);
525 }

532 template <size_t VolumeDim>
