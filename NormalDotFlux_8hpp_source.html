25 /*!
26  * \brief Contract a surface normal covector with the first index of a flux
27  * tensor or variables
28  *
29  * \details
30  * Returns \f$n_i F^i_{j\ldots}\f$, where the flux tensor \f$F\f$ must have an
31  * upper spatial first index and may have arbitray extra indices.
32  */
33 template <size_t VolumeDim, typename Fr, typename Symm,
34  typename... RemainingIndices,
35  typename ResultTensor = Tensor<DataVector, tmpl::pop_front<Symm>,
36  index_list<RemainingIndices...>>>
37 void normal_dot_flux(
38  const gsl::not_null<ResultTensor*> normal_dot_flux,
39  const tnsr::i<DataVector, VolumeDim, Fr>& normal,
40  const Tensor<DataVector, Symm,
41  index_list<SpatialIndex<VolumeDim, UpLo::Up, Fr>,
42  RemainingIndices...>>& flux_tensor) noexcept {
43  for (auto it = normal_dot_flux->begin(); it != normal_dot_flux->end(); it++) {
44  const auto result_indices = normal_dot_flux->get_tensor_index(it);
45  *it = get<0>(normal) * flux_tensor.get(prepend(result_indices, size_t{0}));
46  for (size_t d = 1; d < VolumeDim; d++) {
47  *it += normal.get(d) * flux_tensor.get(prepend(result_indices, d));
48  }
49  }
50 }

52 template <typename TagsList, size_t VolumeDim, typename Fr>
53 auto normal_dot_flux(
54  const tnsr::i<DataVector, VolumeDim, Fr>& normal,
55  const Variables<
56  db::wrap_tags_in<::Tags::Flux, TagsList, tmpl::size_t<VolumeDim>, Fr>>&
57  fluxes) noexcept {
58  auto result = make_with_value<
59  Variables<db::wrap_tags_in<::Tags::NormalDotFlux, TagsList>>>(fluxes, 0.);
60  tmpl::for_each<TagsList>([&result, &fluxes,
61  &normal](auto local_tag) noexcept {
62  using tensor_tag = tmpl::type_from<decltype(local_tag)>;
63  normal_dot_flux(
64  make_not_null(&get<::Tags::NormalDotFlux<tensor_tag>>(result)), normal,
65  get<::Tags::Flux<tensor_tag, tmpl::size_t<VolumeDim>, Fr>>(fluxes));
66  });
67  return result;
68 }
