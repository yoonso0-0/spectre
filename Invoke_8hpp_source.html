15 // Allow 64 inline entry method calls before we fall back to Charm++. This is
16 // done to avoid blowing the stack.
17 inline bool max_inline_entry_methods_reached() noexcept {
18  thread_local size_t approx_stack_depth = 0;
19 #ifndef SPECTRE_PROFILING
20  approx_stack_depth++;
21  if (approx_stack_depth < 64) {
22  return false;
23  }
24  approx_stack_depth = 0;
25 #endif
26  return true;
27 }

38 /*!
39  * \ingroup ParallelGroup
40  * \brief Send the data `args...` to the algorithm running on `proxy`, and tag
41  * the message with the identifier `temporal_id`.
42  *
43  * If the algorithm was previously disabled, set `enable_if_disabled` to true to
44  * enable the algorithm on the parallel component.
45  */
46 template <
47  typename ReceiveTag, typename Proxy, typename ReceiveDataType,
48  Requires<detail::has_ckLocal_method<std::decay_t<Proxy>>::value> = nullptr>
49 void receive_data(Proxy&& proxy, typename ReceiveTag::temporal_id temporal_id,
50  ReceiveDataType&& receive_data,
51  const bool enable_if_disabled = false) noexcept {
52  auto* obj = proxy.ckLocal();
53  // Only elide the Charm++ RTS if the object is local and if we won't blow the
54  // stack by having too many recursive function calls.
55  if (obj != nullptr and not detail::max_inline_entry_methods_reached()) {
56  obj->template receive_data<ReceiveTag>(
57  std::move(temporal_id), std::forward<ReceiveDataType>(receive_data),
58  enable_if_disabled);
59  } else {
60  proxy.template receive_data<ReceiveTag>(
61  std::move(temporal_id), std::forward<ReceiveDataType>(receive_data),
62  enable_if_disabled);
63  }
64 }
65 
66 template <typename ReceiveTag, typename Proxy, typename ReceiveDataType,
67  Requires<not detail::has_ckLocal_method<std::decay_t<Proxy>>::value> =
68  nullptr>
69 void receive_data(Proxy&& proxy, typename ReceiveTag::temporal_id temporal_id,
70  ReceiveDataType&& receive_data,
71  const bool enable_if_disabled = false) noexcept {
72  proxy.template receive_data<ReceiveTag>(
73  std::move(temporal_id), std::forward<ReceiveDataType>(receive_data),
74  enable_if_disabled);
