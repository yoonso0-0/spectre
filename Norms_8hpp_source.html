18 namespace L2Norm_detail {
19 template <typename DataType, typename Symm, typename IndexList>
20 Scalar<DataType> pointwise_l2_norm_square(
21  const Tensor<DataType, Symm, IndexList>& tensor) noexcept {
22  auto pointwise_l2_normsq = make_with_value<Scalar<DataType>>(tensor, 0.);
23  for (auto tensor_element = tensor.begin(); tensor_element != tensor.end();
24  ++tensor_element) {
25  // In order to handle tensory symmetries, we multiply the square of each
26  // stored component with its multiplicity
27  get(pointwise_l2_normsq) +=
28  tensor.multiplicity(tensor_element) * square(*tensor_element);
29  }
30  return pointwise_l2_normsq;
31 }
32 } // namespace L2Norm_detail

35 /*!
36  * \ingroup TensorGroup
37  * \brief Compute point-wise Euclidean \f$L^2\f$-norm of arbitrary Tensors.
38  *
39  * \details
40  * At each grid point \f$p\f$ in the element, this function computes the
41  * point-wise Frobenius norm of a given Tensor with arbitrary rank. If the
42  * Tensor \f$A\f$ has rank \f$n\f$ and dimensionality \f$D\f$, then its
43  * Frobenius norm at point \f$p\f$ is computed as:
44  *
45  * \f{equation}
46  * ||A||_2(p) =
47  * \left(\sum^{D-1}_{i_1=0}\sum^{D-1}_{i_2=0}\cdots \sum^{D-1}_{i_n=0}
48  * |A_{i_1 i_2 \cdots i_n}(p)|^2 \right)^{1/2},
49  * \f}
50  *
51  * where both contra-variant and co-variant indices are shown as lower indices.
52  */
53 template <typename DataType, typename Symm, typename IndexList>
54 Scalar<DataType> pointwise_l2_norm(
55  const Tensor<DataType, Symm, IndexList>& tensor) noexcept {
56  return Scalar<DataType>{
57  sqrt(get(L2Norm_detail::pointwise_l2_norm_square(tensor)))};
58 }
59 template <typename DataType, typename Symm, typename IndexList>
60 void pointwise_l2_norm(
61  const gsl::not_null<Scalar<DataType>*> norm,
62  const Tensor<DataType, Symm, IndexList>& tensor) noexcept {
63  destructive_resize_components(norm, get_size(tensor[0].size()));
64  get(*norm) = sqrt(get(L2Norm_detail::pointwise_l2_norm_square(tensor)));
65 }

68 /*!
69  * \ingroup TensorGroup
70  * \brief Compute Euclidean \f$L^2\f$-norm of arbitrary Tensors reduced over an
71  * element.
72  *
73  * \details
74  * Computes the RMS value of the point-wise Frobenius norm of a given Tensor
75  * with arbitrary rank over all grid points in an element. If the Tensor \f$A\f$
76  * has rank \f$n\f$ and dimensionality \f$D\f$, and the element (of order
77  * \f$N\f$) has \f$N+1\f$ points, then its element-reduced Frobenius norm is
78  * computed as:
79  *
80  * \f{equation}
81  * ||A||_2 =
82  * \left(\frac{1}{N+1}\sum^{N}_{p=0}
83  * \left(\sum^{D-1}_{i_1=0}\sum^{D-1}_{i_2=0}\cdots
84  * \sum^{D-1}_{i_n=0} |A^p_{i_1 i_2 \cdots i_n}|^2 \right)
85  * \right)^{1/2},
86  * \f}
87  *
88  * where both contra-variant and co-variant indices are shown as lower indices,
89  * and \f$p\f$ indexes grid points in the element.
90  *
91  * \warning This function reduces the Frobenius norm over the element, not the
92  * whole domain.
93  */
94 template <typename DataType, typename Symm, typename IndexList>
95 double l2_norm(const Tensor<DataType, Symm, IndexList>& tensor) noexcept {
96  const auto pointwise_l2_normsq =
97  L2Norm_detail::pointwise_l2_norm_square(tensor);
98  using Plus = funcl::Plus<funcl::Identity>;
99  return sqrt(alg::accumulate(get(pointwise_l2_normsq), 0., Plus{}) /
100  tensor.begin()->size());
101 }
