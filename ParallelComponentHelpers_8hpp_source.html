164 // This struct provides a better error message if
165 // an unknown tag is requested from the GlobalCache.
166 template <typename GlobalCacheTag, typename ListOfPossibleTags>
167 struct matching_tag_helper {
168  using found_tags = tmpl::filter<ListOfPossibleTags,
169  std::is_base_of<tmpl::pin<GlobalCacheTag>, tmpl::_1>>;
170  static_assert(not std::is_same_v<found_tags, tmpl::list<>>,
171  "Trying to get a nonexistent tag from the GlobalCache. "
172  "To diagnose the problem, search for "
173  "'matching_tag_helper' in the error message. "
174  "The first template parameter of "
175  "'matching_tag_helper' is the requested tag, and "
176  "the second template parameter is a tmpl::list of all the "
177  "tags in the GlobalCache. One possible bug that may "
178  "lead to this error message is a missing or misspelled "
179  "const_global_cache_tags or mutable_global_cache_tags "
180  "type alias.");
181  static_assert(tmpl::size<found_tags>::value == 1,
182  "Found more than one matching tag. "
183  "To diagnose the problem, search for "
184  "'matching_tag_helper' in the error message. "
185  "The first template parameter of "
186  "'matching_tag_helper' is the requested tag, and "
187  "the second template parameter is a tmpl::list of all the "
188  "tags in the GlobalCache.");
189  using type = tmpl::front<found_tags>;
190 };
