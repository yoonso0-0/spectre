15 /// \ingroup UtilitiesGroup
16 /// \brief Callable struct which retrieves the `t.size()` for operand `t`. This
17 /// will cause a compiler error if no such function exists.
18 struct GetContainerSize {
19  template <typename T>
20  SPECTRE_ALWAYS_INLINE decltype(auto) operator()(const T& t) const noexcept {
21  return t.size();

96 /*!
97  * \ingroup UtilitiesGroup
98  * \brief Returns the `i`th element if `T` has a subscript operator, otherwise
99  * if `T` is fundamental or a `std::complex` of a fundamental type, returns
100  * `t`.
101  *
102  * \details This function also optionally takes the user-defined subscript
103  * function `at`, which can be used to specify a custom indexing function. For
104  * instance, for a type which is a `std::array` of a `std::array`s, the indexing
105  * function could be the below callable struct:
106  * \snippet Test_ContainerHelpers.cpp get_element_example_indexing_callable
107  *
108  * which would index the data structure in a manner in which the outer array
109  * index varies fastest. The indexing function must take as arguments the
110  * applicable container and a `size_t` index, in that order. This follows the
111  * convention of `gsl::at`.
112  * \note `std::complex` are regarded as non-indexable (despite a predictable
113  * memory layout), so this function acts as the identity on `std::complex` of
114  * fundamental types
115  */
116 template <typename T, typename SubscriptFunction = GetContainerElement>
117 SPECTRE_ALWAYS_INLINE decltype(auto) get_element(
118  T& t, const size_t i,
119  SubscriptFunction at = GetContainerElement{}) noexcept {
120  return ContainerHelpers_detail::ContainerImpls<(
121  tt::is_complex_of_fundamental_v<std::remove_cv_t<T>> or
122  std::is_fundamental_v<std::remove_cv_t<T>>)>::get_element(t, i, at);
123 }

125 /*!
126  * \ingroup UtilitiesGroup
127  * \brief Retrieve the size of `t` if `t.size()` is a valid expression,
128  * otherwise if `T` is fundamental or a `std::complex` of a fundamental type,
129  * returns 1.
130  *
131  * \details This function also optionally takes the user-defined `size`
132  * function, which can be used to specify a custom size function. For
133  * instance, for a type which is a `std::array` of a `std::array`, the size
134  * function could be the below callable struct:
135  * \snippet Test_ContainerHelpers.cpp get_size_example_size_callable
136  *
137  * The `size` function must take the single argument of the applicable
138  * container, and should return a `size_t`. This follows the convention of
139  * `std::size()` as of C++17.
140  * \note `std::complex` are regarded as non-indexable (despite a predictable
141  * memory layout), so this function will return 1 for a `std::complex` of a
142  * fundamental type
143  */
144 template <typename T, typename SizeFunction = GetContainerSize>
145 SPECTRE_ALWAYS_INLINE decltype(auto) get_size(
146  const T& t, SizeFunction size = GetContainerSize{}) noexcept {
147  return ContainerHelpers_detail::ContainerImpls<(
148  tt::is_complex_of_fundamental_v<std::remove_cv_t<T>> or
149  std::is_fundamental_v<std::remove_cv_t<T>>)>::get_size(t, size);
150 }

152 /*!
153  * \ingroup UtilitiesGroup
154  * \brief Checks the size of each component of the container, and resizes if
155  * necessary.
156  *
157  * \details This operation is not permitted when any of the components of the
158  * tensor is non-owning (see `VectorImpl` for ownership details).
159  * \note This utility should NOT be used when it is anticipated that the
160  * components will be the wrong size. In that case, suggest either manual
161  * checking or restructuring so that resizing is less common. The internal
162  * call uses `UNLIKELY` to perform the checks most quickly when resizing is
163  * unnecessary.
164  *
165  * \warning This is typically to be called on `Tensor`s, NOT (for instance)
166  * `DataVector`s. For derived classes of `VectorImpl`, this function will cause
167  * no resize. Instead, use the `VectorImpl` member function
168  * `VectorImpl::destructive_resize()`.
169  *
170  * \note This assumes that a range-based iterator will appropriately loop over
171  * the elements to resize, and that each resized element is either a fundamental
172  * type or a derived class of `VectorImpl`. If either of those assumptions needs
173  * to be relaxed, this function will need to be generalized.
174  */
175 template <typename Container,
176  typename DestructiveResizeFunction = ContainerDestructiveResize>
177 void SPECTRE_ALWAYS_INLINE destructive_resize_components(
178  // NOLINTNEXTLINE(readability-avoid-const-params-in-decls)
179  const gsl::not_null<Container*> container, const size_t new_size,
180  DestructiveResizeFunction destructive_resize =
181  ContainerDestructiveResize{}) noexcept {
182  for (auto& vector : *container) {
183  ContainerHelpers_detail::ContainerImpls<(
184  tt::is_complex_of_fundamental_v<
185  std::remove_cv_t<typename Container::value_type>> or
186  std::is_fundamental_v<
187  std::remove_cv_t<typename Container::value_type>>)>::
188  apply_destructive_resize(vector, new_size, destructive_resize);
189  }
190 }
