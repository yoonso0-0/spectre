36 /// \brief test construction and assignment of a `VectorType` with a `ValueType`
37 template <typename VectorType, typename ValueType>
38 void vector_test_construct_and_assign(
39  tt::get_fundamental_type_t<ValueType> low =
40  tt::get_fundamental_type_t<ValueType>{-100.0},
41  tt::get_fundamental_type_t<ValueType> high =
42  tt::get_fundamental_type_t<ValueType>{100.0}) noexcept {
43  MAKE_GENERATOR(gen);
44  UniformCustomDistribution<tt::get_fundamental_type_t<ValueType>> dist{low,
45  high};
46  UniformCustomDistribution<size_t> sdist{2, 20};
47 
48  const size_t size = sdist(gen);
49 
50  const VectorType size_constructed{size};
51  CHECK(size_constructed.size() == size);
52  const auto generated_value1 = make_with_random_values<ValueType>(
53  make_not_null(&gen), make_not_null(&dist));
54 
55  const VectorType value_size_constructed{size, generated_value1};
56  CHECK(value_size_constructed.size() == size);
57  alg::for_each(value_size_constructed, [
58  generated_value1, &value_size_constructed
59  ](typename VectorType::value_type element) noexcept {
60  CAPTURE(value_size_constructed);
61  CHECK(element == generated_value1);
62  });
63 
64  // random generation must use `make_with_random_values`, because stored value
65  // in vector type might be a non-fundamental type.
66  const auto generated_value2 = make_with_random_values<ValueType>(
67  make_not_null(&gen), make_not_null(&dist));
68  const auto generated_value3 = make_with_random_values<ValueType>(
69  make_not_null(&gen), make_not_null(&dist));
70 
71  VectorType initializer_list_constructed{
72  {static_cast<typename VectorType::value_type>(generated_value2),
73  static_cast<typename VectorType::value_type>(generated_value3)}};
74  CHECK(initializer_list_constructed.size() == 2);
75  CHECK(initializer_list_constructed.is_owning());
76  CHECK(gsl::at(initializer_list_constructed, 0) == generated_value2);
77  CHECK(gsl::at(initializer_list_constructed, 1) == generated_value3);
78 
79  typename VectorType::value_type raw_ptr[2] = {generated_value2,
80  generated_value3};
81  const VectorType pointer_size_constructed{
82  static_cast<typename VectorType::value_type*>(raw_ptr), 2};
83  CHECK(initializer_list_constructed == pointer_size_constructed);
84  CHECK_FALSE(initializer_list_constructed != pointer_size_constructed);
85 
86  test_copy_semantics(initializer_list_constructed);
87  auto initializer_list_constructed_copy = initializer_list_constructed;
88  CHECK(initializer_list_constructed_copy.is_owning());
89  CHECK(initializer_list_constructed_copy == pointer_size_constructed);
90  test_move_semantics(std::move(initializer_list_constructed),
91  initializer_list_constructed_copy);
92 
93  VectorType move_assignment_initialized;
94  move_assignment_initialized = std::move(initializer_list_constructed_copy);
95  CHECK(move_assignment_initialized.is_owning());
96 
97  VectorType move_constructed{std::move(move_assignment_initialized)};
98  CHECK(move_constructed.is_owning());
99  CHECK(move_constructed == pointer_size_constructed);
100 
101  // clang-tidy has performance complaints, and we're checking functionality
102  const VectorType copy_constructed{move_constructed}; // NOLINT
103  CHECK(copy_constructed.is_owning());
104  CHECK(copy_constructed == pointer_size_constructed);
105 
106  // check the destructive resize utility
107  const VectorType destructive_resize_check_copy = move_constructed;
108  move_constructed.destructive_resize(move_constructed.size());
109  CHECK(move_constructed == destructive_resize_check_copy);
110  move_constructed.destructive_resize(move_constructed.size() + 1);
111  CHECK(move_constructed != destructive_resize_check_copy);
112  CHECK(move_constructed.size() == destructive_resize_check_copy.size() + 1);
113 }

116 /// \brief test the serialization of a `VectorType` constructed with a
117 /// `ValueType`
118 template <typename VectorType, typename ValueType>
119 void vector_test_serialize(tt::get_fundamental_type_t<ValueType> low =
120  tt::get_fundamental_type_t<ValueType>{-100.0},
121  tt::get_fundamental_type_t<ValueType> high =
122  tt::get_fundamental_type_t<ValueType>{
123  100.0}) noexcept {
124  MAKE_GENERATOR(gen);
125  UniformCustomDistribution<tt::get_fundamental_type_t<ValueType>> dist{low,
126  high};
127  UniformCustomDistribution<size_t> sdist{2, 20};
128 
129  const size_t size = sdist(gen);
130  VectorType vector_test{size}, vector_control{size};
131  VectorType vector_ref;
132  const auto start_value = make_with_random_values<ValueType>(
133  make_not_null(&gen), make_not_null(&dist));
134  const auto value_difference = make_with_random_values<ValueType>(
135  make_not_null(&gen), make_not_null(&dist));
136  // generate_series is used to generate a pair of equivalent, but independently
137  // constructed, data sets to fill the vectors with.
138  ValueType current_value = start_value;
139  const auto generate_series = [&current_value, value_difference ]() noexcept {
140  return current_value += value_difference;
141  };
142  std::generate(vector_test.begin(), vector_test.end(), generate_series);
143  current_value = start_value;
144  std::generate(vector_control.begin(), vector_control.end(), generate_series);
145  // checks the vectors have been constructed as expected
146  CHECK(vector_control == vector_test);
147  CHECK(vector_test.is_owning());
148  CHECK(vector_control.is_owning());
149  const VectorType serialized_vector_test =
150  serialize_and_deserialize(vector_test);
151  // check that the vector is unaltered by serialize_and_deserialize
152  CHECK(vector_control == vector_test);
153  CHECK(serialized_vector_test == vector_control);
154  CHECK(serialized_vector_test.is_owning());
155  CHECK(serialized_vector_test.data() != vector_test.data());
156  CHECK(vector_test.is_owning());
157  // checks serialization for reference
158  vector_ref.set_data_ref(&vector_test);
159  CHECK(vector_test.is_owning());
160  CHECK_FALSE(vector_ref.is_owning());
161  CHECK(vector_ref == vector_test);
162  const VectorType serialized_vector_ref =
163  serialize_and_deserialize(vector_ref);
164  CHECK(vector_test.is_owning());
165  CHECK(vector_test == vector_control);
166  CHECK(vector_ref == vector_test);
167  CHECK(serialized_vector_ref == vector_test);
168  CHECK(serialized_vector_ref.is_owning());
169  CHECK(serialized_vector_ref.data() != vector_ref.data());
170  CHECK_FALSE(vector_ref.is_owning());
171 }

263 /// \ingroup TestingFrameworkGroup
264 /// \brief Test that assigning to a non-owning `VectorType` of the wrong size
265 /// appropriately generates an error.
266 ///
267 /// \details a calling function should be an `ASSERTION_TEST()` and check for
268 /// the string "Must copy into same size".
269 /// Three types of tests are provided and one must be provided as the first
270 /// function argument:
271 /// - `RefSizeErrorTestKind::Copy`: Checks that copy-assigning to a non-owning
272 /// `VectorType` from a `VectorType` with the wrong size generates an error.
273 /// - `RefSizeErrorTestKind::ExpressionAssign`: Checks that assigning to a
274 /// non-owning `VectorType` from an expression with alias `ResultType` of
275 /// `VectorType` with the wrong size generates an error
276 /// - `RefSizeErrorTestKind::Move`: Checks that move-assigning to a non-owning
277 /// `VectorType` from a `VectorType` with the wrong size generates an error.
278 template <typename VectorType,
279  typename ValueType = typename VectorType::ElementType>
280 void vector_ref_test_size_error(
281  RefSizeErrorTestKind test_kind,
282  tt::get_fundamental_type_t<ValueType> low =
283  tt::get_fundamental_type_t<ValueType>{-100.0},
284  tt::get_fundamental_type_t<ValueType> high =
285  tt::get_fundamental_type_t<ValueType>{100.0}) noexcept {
286  MAKE_GENERATOR(gen);
287  UniformCustomDistribution<tt::get_fundamental_type_t<ValueType>> dist{low,
288  high};
289  UniformCustomDistribution<size_t> sdist{2, 20};
290 
291  const size_t size = sdist(gen);
292  VectorType generated_vector = make_with_random_values<VectorType>(
293  make_not_null(&gen), make_not_null(&dist), VectorType{size});
294  VectorType ref_generated_vector;
295  ref_generated_vector.set_data_ref(&generated_vector);
296  const VectorType larger_generated_vector =
297  make_with_random_values<VectorType>(
298  make_not_null(&gen), make_not_null(&dist), VectorType{size + 1});
299  // each of the following options should error, the reference should have
300  // received the wrong size
301  if (test_kind == RefSizeErrorTestKind::Copy) {
302  ref_generated_vector = larger_generated_vector;
303  }
304  if (test_kind == RefSizeErrorTestKind::ExpressionAssign) {
305  ref_generated_vector = (larger_generated_vector + larger_generated_vector);
306  }
307  if (test_kind == RefSizeErrorTestKind::Move) {
308  ref_generated_vector = std::move(larger_generated_vector);
309  }
310 }

399 /// \ingroup TestingFrameworkGroup
400 /// \brief the set of test types that may be used for the math operations
401 ///
402 /// \details Three types of test are provided:
403 /// - `Normal` is used to indicate those tests which should be performed over
404 /// all combinations of the supplied vector type(s) and their value
405 /// types. This is useful for e.g. `+`.
406 ///
407 /// - `Strict` is used to indicate those tests which should be performed over
408 /// only sets of the vector type and compared to the same operation of the set
409 /// of its value type. This is useful for e.g. `atan2`, which cannot take a
410 /// `DataVector` and a double as arguments.
411 ///
412 /// - `Inplace` is used to indicate those tests which should be performed
413 /// maintaining the type of the left-hand side of the operator and not
414 /// including it in the combinations. Inplace operators such as `+=` have a
415 /// more restrictive condition on the type of the left hand side than do
416 /// simply `+`. (e.g. `double + complex<double>` compiles, but
417 /// `double += complex<double>` does not)
418 ///
419 /// - `GivenOrderOfArgumentsOnly` is used to indicate that the arguments given
420 /// should not be taken in any combination apart from the given combination.
421 /// This should be used for highly restrictive operations which are only
422 /// supported for certain type combinations.
423 enum TestKind { Normal, Strict, Inplace, GivenOrderOfArgumentsOnly };

509 // given the set of types of operands to test (`Operands`), and a set of
510 // reference wrappers (`Wraps`), make each operand with random values according
511 // to the bound from `Bounds`. Then, call
512 // `CHECK_CUSTOM_ELEMENT_WISE_FUNCTION_APPROX` to test the element wise `func`.
513 template <typename Function, typename... Bounds, typename... Wraps,
514  typename... Operands, size_t... Is>
515 void test_function_on_vector_operands(
516  const Function& function, const std::tuple<Bounds...>& bounds,
517  const std::tuple<Wraps...>& /*wraps*/,
518  const std::tuple<Operands...>& /*operands*/,
519  std::index_sequence<Is...> /*meta*/) noexcept {
520  MAKE_GENERATOR(generator);
521  UniformCustomDistribution<size_t> size_distribution{2, 5};

594 // functions to recursively assemble the arguments and wrappers for
595 // calling the operator tester `test_function_on_vector_operands`.
596 //
597 // `UniqueTypeList` is a tmpl::list which stores the set of unique types to
598 // test the functions with.
599 //
600 // `FirstOperand` is the first operand type, used when the Test is
601 // `TestKind::Inplace`, as inplace operators have specific demands on the
602 // first operand.
603 
604 // base case: the correct number of operand types has been obtained in
605 // `Operands`, so this calls the test function with the function, bounds,
606 // reference wrap, and operand type information.
607 template <TestKind Test, typename UniqueTypeList, typename FirstOperand,
608  typename... Operands, typename Function, typename... DistBounds,
609  typename... Wraps>
610 void assemble_test_function_arguments_and_execute_tests(
611  const Function& function, const std::tuple<DistBounds...>& bounds,
612  const std::tuple<Wraps...>& wraps, const std::tuple<Operands...> operands,
613  TestFunctionsWithVectorArgumentsChoices::Done /*meta*/) noexcept {

617 }

619 // general case: add an additional reference wrapper identification type from
620 // `WrapperList` to `wraps`, and an additional type from `UniqueTypeList`
621 // to `operands`, and recurse on each option.
622 template <TestKind Test, typename UniqueTypeList, typename FirstOperand,
623  typename... Operands, typename Function, typename... DistBounds,
624  typename... Wraps>
625 void assemble_test_function_arguments_and_execute_tests(
626  const Function& function, const std::tuple<DistBounds...>& bounds,
627  const std::tuple<Wraps...>& wraps, const std::tuple<Operands...>&& operands,
628  TestFunctionsWithVectorArgumentsChoices::Continuing
629  /*meta*/) noexcept {
630  tmpl::for_each<WrapperList>([&function, &bounds, &wraps,
631  &operands ](const auto x) noexcept {
632  tmpl::for_each<UniqueTypeList>([&function, &bounds, &wraps,
633  &operands ](const auto y) noexcept {
634  using next_vector = typename decltype(y)::type;
635  using next_wrap = typename decltype(x)::type;
636  assemble_test_function_arguments_and_execute_tests<Test, UniqueTypeList,
637  FirstOperand>(
638  function, bounds, std::tuple_cat(wraps, std::tuple<next_wrap>{}),
639  std::tuple_cat(operands, std::tuple<next_vector>{}),
640  detail::next_choice_for_test_functions_recursion<
641  tmpl::list<Wraps..., next_wrap>, tmpl::list<DistBounds...>,
642  Test>{});
643  });
644  });
645 }

647 // case of first operand and inplace test: the left hand operand for inplace
648 // tests cannot be const, so the reference wrapper must be chosen
649 // accordingly. Also, the left hand size type is fixed to be FirstOperand.
650 template <TestKind Test, typename UniqueTypeList, typename FirstOperand,
651  typename... Operands, typename Function, typename... DistBounds,
652  typename... Wraps>
653 void assemble_test_function_arguments_and_execute_tests(
654  const Function& function, const std::tuple<DistBounds...>& bounds,
655  const std::tuple<Wraps...>& /*wraps*/,
656  const std::tuple<Operands...>&& /*operands*/,
657  TestFunctionsWithVectorArgumentsChoices::FirstInplace /*meta*/) noexcept {
658  tmpl::for_each<NonConstWrapperList>([&function,
659  &bounds ](const auto x) noexcept {
660  using next_wrap = typename decltype(x)::type;
661  assemble_test_function_arguments_and_execute_tests<Test, UniqueTypeList,
662  FirstOperand>(
663  function, bounds, std::tuple<next_wrap>{}, std::tuple<FirstOperand>{},
664  detail::next_choice_for_test_functions_recursion<
665  tmpl::list<FirstOperand>, tmpl::list<DistBounds...>, Test>{});
666  });
667 }

671 template <TestKind Test, typename UniqueTypeList, typename FirstOperand,
672  typename... Operands, typename Function, typename... DistBounds,

687  tmpl::list<Wraps..., next_wrap>, tmpl::list<DistBounds...>,

689  });

692 // dispatch function for the individual tuples of functions and bounds for their
693 // respective distributions. This processes the requested set of vector types to
694 // test and passes the resulting request to the recursive argument assembly
695 // function assemble_test_function_arguments_and_execute_tests above
696 template <TestKind Test, typename VectorType0, typename... VectorTypes,
697  typename Function, typename... DistBounds>
698 void test_function_with_vector_arguments_impl(
699  const std::tuple<Function, std::tuple<DistBounds...>>&
700  function_and_argument_bounds) noexcept {
701  // The unique set of possible operands
702  using operand_type_list = tmpl::conditional_t<
703  Test == TestKind::Strict,
704  // if strict, the operand choices are just the vector types passed in
705  tmpl::remove_duplicates<tmpl::list<VectorType0, VectorTypes...>>,
706  // else, the operand choices include both the vectors and their element
707  // types
708  tmpl::remove_duplicates<tmpl::list<
709  VectorType0, VectorTypes..., get_vector_element_type_t<VectorType0>,
710  get_vector_element_type_t<VectorTypes>...>>>;
711 
712  using starting_operands =
713  tmpl::conditional_t<Test == TestKind::GivenOrderOfArgumentsOnly,
714  std::tuple<VectorType0, VectorTypes...>,
715  std::tuple<>>;
716  assemble_test_function_arguments_and_execute_tests<Test, operand_type_list,
717  VectorType0>(
718  get<0>(function_and_argument_bounds) /*function*/,
719  get<1>(function_and_argument_bounds) /*tuple of bounds*/, std::tuple<>{},
720  starting_operands{},
721  next_choice_for_test_functions_recursion<
722  tmpl::list<>, tmpl::list<DistBounds...>, Test>{});
723 }

726 /*!
727  * \ingroup TestingFrameworkGroup
728  * \brief General entry function for testing arbitrary math functions
729  * on vector types
730  *
731  * \details This utility tests all combinations of the operator on the type
732  * arguments, and all combinations of reference or constant reference wrappers
733  * on all arguments. In certain test cases (see below), it also tests using the
734  * vector type's `value_type`s in the operators as well (e.g. `DataVector +
735  * double`). This is very useful for quickly generating a lot of tests, but the
736  * number of tests scales exponentially in the number of arguments. Therefore,
737  * functions with many arguments can be time-consuming to
738  * run. 4-or-more-argument functions should be used only if completely necessary
739  * and with caution. Any number of vector types may be specified, and tests are
740  * run on all unique combinations of the provided. For instance, if only one
741  * type is provided, the tests will be run only on combinations of that single
742  * type and its `value_type`.
743  *
744  * \param tuple_of_functions_and_argument_bounds A tuple of tuples, in which
745  * the inner tuple contains first a function object followed by a tuple of
746  * 2-element arrays equal to the number of arguments, which represent the
747  * bounds for the random generation of the respective arguments. This system
748  * is provided for robust testing of operators like `/`, where the left-hand
749  * side has a different valid set of values than the right-hand-side.
750  *
751  * \tparam Test from the `TestKind` enum, determines whether the tests will
752  * be:
753  * - `TestKind::Normal`: executed on all combinations of arguments and value
754  * types
755  * - `TestKind::Strict`: executed on all combinations of arguments, for only
756  * the vector types
757  * - `TestKind::Inplace`: executed on all combinations of arguments after the
758  * first, so first is always the 'left hand side' of the operator. In this
759  * case, at least two `VectorTypes` must be specified, where the first is
760  * used only for the left-hand side.
761  * - `TestKind::GivenOrderOfArgumentsOnly`: executed on only the combination of
762  * arguments provided, in the order provided. In this case, the number of
763  * provided types in `typename VectorType0, typename... VectorTypes` must
764  * precisely match the number of arguments taken by the function.
765  *
766  * \tparam VectorType0 The first vector type for which combinations are
767  * tested. The first is accepted as a separate template argument for
768  * appropriately handling `Inplace` tests.
769  * \tparam VectorTypes The remaining types for which combinations are tested.
770  * Any number of types may be passed in, and the test will check the
771  * appropriate combinations of the vector types and (depending on the `Test`)
772  * the respective `value_type`s.
773  */
774 template <TestKind Test, typename VectorType0, typename... VectorTypes,
775  typename... FunctionsAndArgumentBounds>
776 void test_functions_with_vector_arguments(
777  const std::tuple<FunctionsAndArgumentBounds...>&
778  tuple_of_functions_and_argument_bounds) noexcept {
779  tuple_fold(
780  tuple_of_functions_and_argument_bounds,
781  [](const auto& function_and_argument_bounds) noexcept {
782  detail::test_function_with_vector_arguments_impl<Test, VectorType0,
783  VectorTypes...>(
784  function_and_argument_bounds);
785  });
786 }
