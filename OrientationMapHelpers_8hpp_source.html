51 /// \ingroup ComputationalDomainGroup
52 /// Orient variables to the data-storage order of a neighbor element with
53 /// the given orientation.
54 template <size_t VolumeDim, typename TagsList>
55 Variables<TagsList> orient_variables(
56  const Variables<TagsList>& variables, const Index<VolumeDim>& extents,
57  const OrientationMap<VolumeDim>& orientation_of_neighbor) noexcept {
58  // Skip work (aside from a copy) if neighbor is aligned
59  if (orientation_of_neighbor.is_aligned()) {
60  return variables;
61  }
62 
63  const size_t number_of_grid_points = extents.product();
64  ASSERT(variables.number_of_grid_points() == number_of_grid_points,
65  "Inconsistent `variables` and `extents`:\n"
66  " variables.number_of_grid_points() = "
67  << variables.number_of_grid_points()
68  << "\n"
69  " extents.product() = "
70  << extents.product());
71  Variables<TagsList> oriented_variables(number_of_grid_points);
72  const auto oriented_offset = OrientationMapHelpers_detail::oriented_offset(
73  extents, orientation_of_neighbor);
74  auto oriented_vars_view = gsl::make_span(oriented_variables);
75  OrientationMapHelpers_detail::orient_each_component(
76  make_not_null(&oriented_vars_view), gsl::make_span(variables),
77  number_of_grid_points, oriented_offset);
78 
79  return oriented_variables;
80 }

82 template <size_t VolumeDim, typename TagsList>
83 Variables<TagsList> orient_variables_on_slice(
84  const Variables<TagsList>& variables_on_slice,
85  const Index<VolumeDim - 1>& slice_extents, const size_t sliced_dim,
86  const OrientationMap<VolumeDim>& orientation_of_neighbor) noexcept {
87  // Skip work (aside from a copy) if neighbor slice is aligned
88  if (orientation_of_neighbor.is_aligned()) {
89  return variables_on_slice;
90  }
91 
92  const size_t number_of_grid_points = slice_extents.product();
93  ASSERT(variables_on_slice.number_of_grid_points() == number_of_grid_points,
94  "Inconsistent `variables_on_slice` and `slice_extents`:\n"
95  " variables_on_slice.number_of_grid_points() = "
96  << variables_on_slice.number_of_grid_points()
97  << "\n"
98  " slice_extents.product() = "
99  << slice_extents.product());
100  Variables<TagsList> oriented_variables(number_of_grid_points);
101  const auto oriented_offset =
102  OrientationMapHelpers_detail::oriented_offset_on_slice(
103  slice_extents, sliced_dim, orientation_of_neighbor);
104 
105  auto oriented_vars_view = gsl::make_span(oriented_variables);
106  OrientationMapHelpers_detail::orient_each_component(
107  make_not_null(&oriented_vars_view), gsl::make_span(variables_on_slice),
108  number_of_grid_points, oriented_offset);
109 
110  return oriented_variables;
111 }

114 /// \ingroup ComputationalDomainGroup
115 /// Orient data in a `std::vector<double>` representing one or more tensor
116 /// components.
117 ///
118 /// In most cases the `Variables` version of `orient_variables` should be
119 /// called. However, in some cases the tags and thus the type of the data being
120 /// sent is determined at runtime. In these cases the `std::vector` version of
121 /// `orient_variables` is useful. A concrete example of this is when hybridizing
122 /// DG with finite difference methods, where sometimes the data sent is both the
123 /// variables for reconstruction and the fluxes for either the DG or finite
124 /// difference scheme, while at other points only one of these three is sent.
