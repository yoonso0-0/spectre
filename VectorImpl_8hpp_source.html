30 /*!
31  * \ingroup DataStructuresGroup
32  * \brief Base class template for various DataVector and related types
33  *
34  * \details The `VectorImpl` class is the generic parent class for vectors
35  * representing collections of related function values, such as `DataVector`s
36  * for contiguous data over a computational domain.
37  *
38  * The `VectorImpl` does not itself define any particular mathematical
39  * operations on the contained values. The `VectorImpl` template class and the
40  * macros defined in `VectorImpl.hpp` assist in the construction of various
41  * derived classes supporting a chosen set of mathematical operations.
42  *
43  * In addition, the equivalence operator `==` is inherited from the underlying
44  * `PointerVector` type, and returns true if and only if the size and contents
45  * of the two compared vectors are equivalent.
46  *
47  * Template parameters:
48  * - `T` is the underlying stored type, e.g. `double`, `std::complex<double>`,
49  * `float`, etc.
50  * - `VectorType` is the type that should be associated with the VectorImpl
51  * during mathematical computations. In most cases, inherited types should
52  * have themselves as the second template argument, e.g.
53  * ```
54  * class DataVector : VectorImpl<double, DataVector> {
55  * ```
56  * The second template parameter communicates arithmetic type restrictions to
57  * the underlying Blaze framework. For example, if `VectorType` is
58  * `DataVector`, then the underlying architecture will prevent addition with a
59  * vector type whose `ResultType` (which is aliased to its `VectorType`) is
60  * `ModalVector`. Since `DataVector`s and `ModalVector`s represent data in
61  * different spaces, we wish to forbid several operations between them. This
62  * vector-type-tracking through an expression prevents accidental mixing of
63  * vector types in math expressions.
64  *
65  * \note
66  * - If created with size 0, then `data()` will return `nullptr`
67  * - If either `SPECTRE_DEBUG` or `SPECTRE_NAN_INIT` are defined, then the
68  * `VectorImpl` is default initialized to `signaling_NaN()`. Otherwise, the
69  * vector is filled with uninitialized memory for performance.
70  */

191  /*!
192  * \brief A common operation for checking the size and resizing a memory
193  * buffer if needed to ensure that it has the desired size. This operation is
194  * not permitted on a non-owning vector.
195  *
196  * \note This utility should NOT be used when it is anticipated that the
197  * supplied buffer will typically be the wrong size (in that case, suggest
198  * either manual checking or restructuring so that resizing is less common).
199  * This uses `UNLIKELY` to perform the check most quickly when the buffer
200  * needs no resizing, but will be slower when resizing is common.
201  */
202  void SPECTRE_ALWAYS_INLINE
203  destructive_resize(const size_t new_size) noexcept {
204  if (UNLIKELY(size() != new_size)) {
205  ASSERT(owning_,
206  MakeString{}
207  << "Attempting to resize a non-owning vector from size: "
208  << size() << " to size: " << new_size
209  << " but we may not destructively resize a non-owning vector");
210  // NOLINTNEXTLINE(modernize-avoid-c-arrays)
211  owned_data_ = std::unique_ptr<value_type[], decltype(&free)>{
212  new_size > 0
213  ? static_cast<value_type*>(malloc(new_size * sizeof(value_type)))
214  : nullptr,
215  &free};
216  reset_pointer_vector(new_size);
217  }
218  }

250 template <typename T, typename VectorType>
251 VectorImpl<T, VectorType>& VectorImpl<T, VectorType>::operator=(
252  const VectorImpl<T, VectorType>& rhs) noexcept {
253  if (this != &rhs) {
254  if (owning_) {
255  if (size() != rhs.size()) {
256  owned_data_.reset(rhs.size() > 0 ? static_cast<value_type*>(malloc(
257  rhs.size() * sizeof(value_type)))
258  : nullptr);
259  }
260  reset_pointer_vector(rhs.size());
261  } else {
262  ASSERT(rhs.size() == size(), "Must copy into same size, not "
263  << rhs.size() << " into " << size());
264  }
265  std::memcpy(data(), rhs.data(), size() * sizeof(value_type));
266  }
267  return *this;
268 }

354 template <typename T, typename VectorType>
355 void VectorImpl<T, VectorType>::pup(PUP::er& p) noexcept { // NOLINT
356  auto my_size = size();
357  p | my_size;
358  if (my_size > 0) {
359  if (p.isUnpacking()) {
360  owning_ = true;
361  owned_data_.reset(my_size > 0 ? static_cast<value_type*>(
362  malloc(my_size * sizeof(value_type)))
363  : nullptr);
364  reset_pointer_vector(my_size);
365  }
366  PUParray(p, data(), size());
367  }
368 }
