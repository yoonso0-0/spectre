39 /*!
40  * \ingroup UtilitiesGroup
41  * \brief Evaluate a polynomial \f$\sum_{p=0}^N c_p x^p\f$ with Horner's rule
42  *
43  * \param coeffs The polynomial coefficients \f$c_p\f$ ordered from constant to
44  * largest power
45  * \param x The polynomial variable \f$x\f$
46  *
47  * \tparam CoeffsIterable The type of the polynomial coefficients \p coeffs. Can
48  * be a `std::vector<double>` or `std::array<double>`, which means the
49  * coefficients are constant for all values in \p x. Each coefficient can also
50  * be a vector type of typically the same size as \p x, which means the
51  * coefficients vary with the elements in \p x.
52  * \tparam DataType The type of the polynomial variable \p x. Must support
53  * `make_with_value<DataType, DataType>`, as well as (elementwise) addition with
54  * `CoeffsIterable::value_type` and multiplication with `DataType`.
55  */
56 template <typename CoeffsIterable, typename DataType>
57 DataType evaluate_polynomial(const CoeffsIterable& coeffs,
58  const DataType& x) noexcept {
59  return std::accumulate(
60  coeffs.rbegin(), coeffs.rend(), make_with_value<DataType>(x, 0.),
61  [&x](const DataType& state, const auto& element) noexcept {
62  return state * x + element;
63  });
64 }

75 /*!
76  * \ingroup UtilitiesGroup
77  * \brief Smoothly interpolates from 0 to 1 between `lower_edge` and
78  * `upper_edge` with a Hermite polynomial of degree `2 * N + 1`.
79  *
80  * The smoothstep function is
81  *
82  * \f{align*}
83  * S_N(x) = \begin{cases}
84  * 0 &\quad \text{for} \quad x\leq x_0 \\
85  * \tilde{S}_N((x - x_0) / (x_1 - x_0))
86  * &\quad \text{for} \quad x_0 \leq x\leq x_1 \\
87  * 1 &\quad \text{for} \quad x_1\leq x \\
88  * \end{cases}
89  * \f}
90  *
91  * where \f$x_0\f$ is `lower_edge`, \f$x_1\f$ is `upper_edge`, and, up to
92  * \f$N=3\f$,
93  *
94  * \f{align*}
95  * \tilde{S}_0(x) &= x \\
96  * \tilde{S}_1(x) &= 3x^2 - 2x^3 \\
97  * \tilde{S}_2(x) &= 10x^3 - 15x^4 + 6x^5 \\
98  * \tilde{S}_3(x) &= 35x^4 - 84x^5 + 70x^6 - 20x^7
99  * \text{.}
100  * \f}
101  */
102 template <size_t N, typename DataType>
103 DataType smoothstep(const double lower_edge, const double upper_edge,
104  const DataType& arg) noexcept {
105  ASSERT(lower_edge < upper_edge,
106  "Requires lower_edge < upper_edge, but lower_edge="
107  << lower_edge << " and upper_edge=" << upper_edge);
108  using std::clamp;
109  return evaluate_polynomial(
110  []() noexcept -> std::array<double, 2 * N + 2> {
111  static_assert(N <= 3,
112  "The smoothstep function is currently only implemented "
113  "for N <= 3.");
114  if constexpr (N == 0) {
115  return {0., 1};
116  } else if constexpr (N == 1) {
117  return {0., 0., 3., -2};
118  } else if constexpr (N == 2) {
119  return {0., 0., 0., 10., -15., 6.};
120  } else if constexpr (N == 3) {
121  return {0., 0., 0., 0., 35., -84., 70., -20.};
122  }
123  }(),
124  static_cast<DataType>(clamp(
125  static_cast<DataType>((arg - lower_edge) / (upper_edge - lower_edge)),
126  0., 1.)));
127 }
