157 /*!
158  * \ingroup DataBoxTagsGroup
159  * \brief Compute the spatial derivatives of tags in a Variables
160  *
161  * Computes the spatial derivatives of the Tensors in the Variables represented
162  * by `VariablesTag` in the frame mapped to by `InverseJacobianTag`. To only
163  * take the derivatives of a subset of these Tensors you can set the
164  * `DerivTags` template parameter. It takes a `tmpl::list` of the desired
165  * tags and defaults to the full `tags_list` of the Variables.
166  *
167  * This tag may be retrieved via `::Tags::Variables<db::wrap_tags_in<deriv,
168  * DerivTags, Dim, deriv_frame>`.
169  */
170 template <typename VariablesTag, typename InverseJacobianTag,
171  typename DerivTags = typename VariablesTag::type::tags_list>
172 struct DerivCompute
173  : db::add_tag_prefix<
174  deriv, ::Tags::Variables<DerivTags>,
175  tmpl::size_t<
176  tmpl::back<typename InverseJacobianTag::type::index_list>::dim>,
177  typename tmpl::back<
178  typename InverseJacobianTag::type::index_list>::Frame>,
179  db::ComputeTag {
180  private:
181  using inv_jac_indices = typename InverseJacobianTag::type::index_list;
182  static constexpr auto Dim = tmpl::back<inv_jac_indices>::dim;
183  using deriv_frame = typename tmpl::back<inv_jac_indices>::Frame;
184 
185  public:
186  using base = db::add_tag_prefix<
187  deriv, ::Tags::Variables<DerivTags>,
188  tmpl::size_t<
189  tmpl::back<typename InverseJacobianTag::type::index_list>::dim>,
190  typename tmpl::back<
191  typename InverseJacobianTag::type::index_list>::Frame>;
192  using return_type = typename base::type;
193  static constexpr void (*function)(
194  gsl::not_null<return_type*>, const typename VariablesTag::type&,
195  const Mesh<Dim>&,
196  const InverseJacobian<DataVector, Dim, Frame::Logical, deriv_frame>&) =
197  partial_derivatives<DerivTags, typename VariablesTag::type::tags_list,
198  Dim, deriv_frame>;
199  using argument_tags =
200  tmpl::list<VariablesTag, domain::Tags::Mesh<Dim>, InverseJacobianTag>;
201 };
