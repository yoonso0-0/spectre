38 /// \brief A solid torus of points, useful, e.g., when measuring data from an
39 /// accretion disc.
40 ///
41 /// The torus's cross section (e.g., a cut at \f$\phi=0\f$) is a wedge-like
42 /// shape bounded by \f$r_{\text{min}} \le r \le r_{\text{max}}\f$ and
43 /// \f$\theta_{\text{min}} \le \theta \le \theta_{\text{max}}\f$.
44 ///
45 /// The grid points are located on surfaces of constant \f$r\f$, \f$\theta\f$,
46 /// and \f$\phi\f$. There are `NumberRadialPoints` points in the radial
47 /// direction between `MinRadius` and `MaxRadius` (including these endpoints);
48 /// `NumberThetaPoints` points in the \f$\theta\f$ direction between `MinTheta`
49 /// and `MaxTheta` (including these endpoints); `NumberPhiPoints` points in the
50 /// \f$\phi\f$ direction (with one point always at \f$\phi=0\f$).
51 ///
52 /// By default, the points follow a Legendre Gauss-Lobatto distribution in the
53 /// \f$r\f$ and \f$\theta\f$ directions, and a uniform distribution in the
54 /// \f$\phi\f$ direction. The distribution in the \f$r\f$ (and/or \f$\theta\f$)
55 /// direction can be made uniform using the `UniformRadialGrid` (and/or
56 /// `UniformThetaGrid`) option.
57 ///
58 /// The `target_points` form a 3D mesh ordered with \f$r\f$ varying fastest,
59 /// then \f$\theta\f$, and finally \f$\phi\f$ varying slowest.
60 ///
61 /// \note Input coordinates (radii, angles) are interpreted in the
62 /// `Frame::Inertial`

186 /// \brief Computes points in a wedge-sectioned torus.
187 ///
188 /// For requirements on InterpolationTargetTag, see InterpolationTarget
189 template <typename InterpolationTargetTag>
190 struct WedgeSectionTorus {
191  using const_global_cache_tags =
192  tmpl::list<Tags::WedgeSectionTorus<InterpolationTargetTag>>;
193  using is_sequential = std::false_type;
194 
195  template <typename Metavariables, typename DbTags>
196  static tnsr::I<DataVector, 3, Frame::Inertial> points(
197  const db::DataBox<DbTags>& box,
198  const tmpl::type_<Metavariables>& /*meta*/) noexcept {
199  const auto& options =
200  get<Tags::WedgeSectionTorus<InterpolationTargetTag>>(box);
201 
202  // Compute locations of constant r/theta/phi surfaces
203  const size_t num_radial = options.number_of_radial_points;
204  const DataVector radii_1d = [&num_radial, &options ]() noexcept {
205  DataVector result(num_radial);
206  if (options.use_uniform_radial_grid) {
207  // uniform point distribution
208  const double coefficient =
209  (options.max_radius - options.min_radius) / (num_radial - 1.0);
210  for (size_t r = 0; r < num_radial; ++r) {
211  result[r] = options.min_radius + coefficient * r;
212  }
213  } else {
214  // radial Legendre-Gauss-Lobatto distribution via linear rescaling
215  const double mean = 0.5 * (options.max_radius + options.min_radius);
216  const double diff = 0.5 * (options.max_radius - options.min_radius);
217  result =
218  mean + diff * Spectral::collocation_points<
219  Spectral::Basis::Legendre,
220  Spectral::Quadrature::GaussLobatto>(num_radial);
221  }
222  return result;
223  }
224  ();
225  const size_t num_theta = options.number_of_theta_points;
226  const DataVector thetas_1d = [&num_theta, &options ]() noexcept {
227  DataVector result(num_theta);
228  if (options.use_uniform_theta_grid) {
229  // uniform point distribution
230  const double coefficient =
231  (options.max_theta - options.min_theta) / (num_theta - 1.0);
232  for (size_t theta = 0; theta < num_theta; ++theta) {
233  result[theta] = options.min_theta + coefficient * theta;
234  }
235  } else {
236  // Legendre-Gauss-Lobatto point distribution (in theta)
237  const double mean = 0.5 * (options.max_theta + options.min_theta);
238  const double diff = 0.5 * (options.max_theta - options.min_theta);
239  result =
240  mean + diff * Spectral::collocation_points<
241  Spectral::Basis::Legendre,
242  Spectral::Quadrature::GaussLobatto>(num_theta);
243  }
244  return result;
245  }
246  ();
247  const size_t num_phi = options.number_of_phi_points;
248  const DataVector phis_1d = [&num_phi]() noexcept {
249  DataVector result(num_phi);
250  // We do NOT want a grid point at phi = 2pi, as this would duplicate the
251  // phi = 0 data. So, divide by num_phi rather than (n-1) as elsewhere.
252  const double coefficient = 2.0 * M_PI / num_phi;
253  for (size_t phi = 0; phi < num_phi; ++phi) {
254  result[phi] = coefficient * phi;
255  }
256  return result;
257  }
258  ();
259 
260  // Take tensor product to get full 3D r/theta/phi points
261  const size_t num_total = num_radial * num_theta * num_phi;
262  DataVector radii(num_total);
263  DataVector thetas(num_total);
264  DataVector phis(num_total);
265  for (size_t phi = 0; phi < num_phi; ++phi) {
266  for (size_t theta = 0; theta < num_theta; ++theta) {
267  for (size_t r = 0; r < num_radial; ++r) {
268  const size_t i =
269  r + theta * num_radial + phi * num_theta * num_radial;
270  radii[i] = radii_1d[r];
271  thetas[i] = thetas_1d[theta];
272  phis[i] = phis_1d[phi];
273  }
274  }
275  }
276 
277  // Compute x/y/z coordinates
278  // Note: theta measured from +z axis, phi measured from +x axis
279  tnsr::I<DataVector, 3, Frame::Inertial> target_points(num_total);
280  get<0>(target_points) = radii * sin(thetas) * cos(phis);
281  get<1>(target_points) = radii * sin(thetas) * sin(phis);
282  get<2>(target_points) = radii * cos(thetas);
283 
284  return target_points;
285  }
286 
287  template <typename Metavariables, typename DbTags, typename TemporalId>
288  static tnsr::I<DataVector, 3, Frame::Inertial> points(
289  const db::DataBox<DbTags>& box, const tmpl::type_<Metavariables>& meta,
290  const TemporalId& /*temporal_id*/) noexcept {
291  return points(box, meta);
292  }
293 };
