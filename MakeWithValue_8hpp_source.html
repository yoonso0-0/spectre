20 /// Defines a method for determining the number of points represented
21 /// by an object. This allows the object to appear as the input to
22 /// make_with_value.
23 ///
24 /// The MakeWithValueImpls::number_of_points convenience wrapper is
25 /// provided to simplify calling this.
26 template <typename T, typename = std::nullptr_t>
27 struct NumberOfPoints {
28  /// The default implementation will produce a compile-time error.
29  [[noreturn]] static SPECTRE_ALWAYS_INLINE size_t
30  apply(const T& /*input*/) noexcept {
31  static_assert(typename tmpl::has_type<T, std::false_type>::type{},
32  "Do not know how to obtain a size from this type. Either "
33  "implement NumberOfPoints or specialize MakeWithValueImpl "
34  "for the type you are trying to create.");
35  }
36 };
37 
38 /// The number of points represented by an object.
39 template <typename T>
40 size_t number_of_points(const T& input) noexcept {
41  return NumberOfPoints<T>::apply(input);
42 }

44 /// Defines a method for producing an object representing a given
45 /// number of points.
46 ///
47 /// Do not call these functions directly. Use make_with_value
48 /// instead, which can take a size as its first argument.
49 template <typename R, typename = std::nullptr_t>
50 struct MakeWithSize {
51  /// The default implementation will produce a compile-time error.
52  /// In specializations, the \p value parameter need not be a template.
53  template <typename T>
54  [[noreturn]] static SPECTRE_ALWAYS_INLINE R
55  apply(const size_t /*size*/, const T& /*value*/) noexcept {
56  static_assert(typename tmpl::has_type<R, std::false_type>::type{},
57  "Do not know how to create a sized object of this type. "
58  "Either implement MakeWithSize or specialize "
59  "MakeWithValueImpl for the type you are trying to create.");
60  }
61 };

63 template <typename R, typename T, typename = std::nullptr_t>
64 struct MakeWithValueImpl {
65  /// The default implementation uses \ref number_of_points and MakeWithSize.
66  template <typename ValueType>
67  static SPECTRE_ALWAYS_INLINE R apply(const T& input,
68  const ValueType value) noexcept {
69  return MakeWithSize<R>::apply(number_of_points(input), value);
70  }

74 /// \ingroup DataStructuresGroup
75 /// \brief Given an object of type `T`, create an object of type `R` whose
76 /// elements are initialized to `value`.
77 ///
78 /// \details This function is useful in function templates in order to
79 /// initialize the return type of a function template with `value` for functions
80 /// that can be called either at a single grid-point or to fill a data structure
81 /// at the same set of grid-points as the `input`
82 
83 /// \tparam ValueType The type of `value`. For most containers, this will be
84 /// `double`.
85 ///
86 /// \see MakeWithValueImpls
87 template <typename R, typename T, typename ValueType>
88 SPECTRE_ALWAYS_INLINE std::remove_const_t<R> make_with_value(
89  const T& input, const ValueType& value) noexcept {
90  return MakeWithValueImpls::MakeWithValueImpl<std::remove_const_t<R>,
91  T>::apply(input, value);
92 }

119 /// \brief Makes a `std::array`; each element of the `std::array`
120 /// must be `make_with_value`-creatable from a `InputType`.
121 template <size_t Size, typename T, typename InputType>
122 struct MakeWithValueImpl<std::array<T, Size>, InputType> {
123  template <typename ValueType>
124  static SPECTRE_ALWAYS_INLINE std::array<T, Size> apply(
125  const InputType& input, const ValueType value) noexcept {
126  return make_array<Size>(make_with_value<T>(input, value));
127  }
128 };
