18 /*!
19  * \ingroup TensorGroup
20  * \brief Compute the Euclidean magnitude of a rank-1 tensor
21  *
22  * \details
23  * Computes the square root of the sum of the squares of the components of
24  * the rank-1 tensor.
25  */
26 template <typename DataType, typename Index>
27 Scalar<DataType> magnitude(
28  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector) noexcept {
29  return Scalar<DataType>{sqrt(get(dot_product(vector, vector)))};
30 }

32 template <typename DataType, typename Index>
33 void magnitude(
34  const gsl::not_null<Scalar<DataType>*> magnitude,
35  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector) noexcept {
36  destructive_resize_components(magnitude, get_size(get<0>(vector)));
37  dot_product(magnitude, vector, vector);
38  get(*magnitude) = sqrt(get(*magnitude));
39 }

43 /*!
44  * \ingroup TensorGroup
45  * \brief Compute the magnitude of a rank-1 tensor
46  *
47  * \details
48  * Returns the square root of the input tensor contracted twice with the given
49  * metric.
50  */
51 template <typename DataType, typename Index>
52 Scalar<DataType> magnitude(
53  const Tensor<DataType, Symmetry<1>, index_list<Index>>& vector,
54  const Tensor<DataType, Symmetry<1, 1>,
55  index_list<change_index_up_lo<Index>,
56  change_index_up_lo<Index>>>& metric) noexcept {
57  Scalar<DataType> local_magnitude{get_size(get<0>(vector))};
58  magnitude(make_not_null(&local_magnitude), vector, metric);
59  return local_magnitude;
60 }

142 /// \ingroup DataBoxTagsGroup
143 /// \ingroup DataStructuresGroup
144 /// Normalizes the (co)vector represented by Tag
145 ///
146 /// This tag inherits from `Tags::Normalized<Tag>`
147 template <typename Tag>
148 struct NormalizedCompute : Normalized<Tag>, db::ComputeTag {
149  using base = Normalized<Tag>;
150  using return_type = typename base::type;
151  static void function(
152  const gsl::not_null<return_type*> normalized_vector,
153  const typename Tag::type& vector_in,
154  const typename Magnitude<Tag>::type& magnitude) noexcept {
155  destructive_resize_components(normalized_vector, get_size(get(magnitude)));
156  *normalized_vector = vector_in;
157  for (size_t d = 0; d < normalized_vector->index_dim(0); ++d) {
158  normalized_vector->get(d) /= get(magnitude);
159  }
160  }
161  using argument_tags = tmpl::list<Tag, Magnitude<Tag>>;
162 };
