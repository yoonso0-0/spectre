65 namespace detail {
66 template <typename DerivativeKind>
67 inline constexpr int derivative_spin_weight_impl() noexcept {
68  if (std::is_same_v<DerivativeKind, Eth> or
69  std::is_same_v<DerivativeKind, InverseEthbar>) {
70  return 1;
71  } else if (std::is_same_v<DerivativeKind, Ethbar> or
72  std::is_same_v<DerivativeKind, InverseEth>) {
73  return -1;
74  } else if (std::is_same_v<DerivativeKind, EthEth>) {
75  return 2;
76  } else if (std::is_same_v<DerivativeKind, EthbarEthbar>) {
77  return -2;
78  }
79  return 0;
80 }
81 } // namespace detail

90 // The below tags are used to find the new type represented by the spin-weighted
91 // derivative of a spin-weighted quantity. The derivatives alter the spin
92 // weights, and so the utility `adjust_spin_weight_t<Tag, DerivativeKind>` is a
93 // metafunction that determines the correct spin-weighted type for the
94 // spin-weighted derivative `DerivativeKind` of `Tag`.
95 //
96 // if there is no `Tag::type::spin`, but the internal type is a compatible
97 // complex vector type, the spin associated with `Tag` is assumed to be 0.
98 template <typename DataType, typename DerivativeKind>
99 struct adjust_spin_weight {
100  using type =
101  Scalar<SpinWeighted<DataType, derivative_spin_weight<DerivativeKind>>>;
102 };
103 
104 // case for if there is a `Tag::type::spin`
105 template <typename DataType, int Spin, typename DerivativeKind>
106 struct adjust_spin_weight<SpinWeighted<DataType, Spin>, DerivativeKind> {
107  using type = Scalar<
108  SpinWeighted<DataType, Spin + derivative_spin_weight<DerivativeKind>>>;
109 };
110 
111 template <typename Tag, typename DerivativeKind>
112 using adjust_spin_weight_t =
113  typename adjust_spin_weight<typename Tag::type::type, DerivativeKind>::type;

128 /// \ingroup SwshGroup
129 /// \brief Prefix tag representing the spin-weighted derivative of a
130 /// spin-weighted scalar.
131 ///
132 /// Template Parameters:
133 /// - `Tag`: The tag to prefix
134 /// - `DerivativeKind`: The type of spin-weighted derivative. This may be any of
135 /// the labeling structs: `Eth`, `Ethbar`, `EthEth`, `EthbarEth`, `EthEthbar`,
136 /// `EthbarEthbar`, or `NoDerivative`.
137 ///
138 /// Type Aliases and static values:
139 /// - `type`: Always a `SpinWeighted<Scalar<ComplexDataVector>, Spin>`, where
140 /// `Spin` is the spin weight after the derivative `DerivativeKind` has been
141 /// applied.
142 /// - `tag`: An alias to the wrapped tag `Tag`. Provided for applicability to
143 /// general `db::PrefixTag` functionality.
144 /// - `derivative_of`: Another alias to the wrapped tag `Tag`. Provided so that
145 /// utilities that use this prefix for taking derivatives can have a more
146 /// expressive code style.
147 /// - `derivative_kind`: Type alias to `DerivativeKind`, represents the kind of
148 /// spin-weighted derivative applied to `Tag`
149 /// - `spin`: The spin weight of the scalar after the derivative has been
150 /// applied.
151 template <typename Tag, typename DerivativeKind>
152 struct Derivative : db::PrefixTag, db::SimpleTag {
153  using type = detail::adjust_spin_weight_t<Tag, DerivativeKind>;
154  using tag = Tag;
155  // derivative_of is provided as a second alias so that utilities that assume a
156  // derivative prefix tag fail earlier during compilation
157  using derivative_of = Tag;
158  using derivative_kind = DerivativeKind;
159  const static int spin = type::type::spin;
160  static std::string name() noexcept {
161  return detail::compose_spin_weighted_derivative_name<DerivativeKind>(
162  db::tag_name<Tag>());
163  }
164 };

166 /// \ingroup SwshGroup
167 /// \brief Prefix tag representing the spin-weighted spherical harmonic
168 /// transform of a spin-weighted scalar.
169 ///
170 /// Template Parameters:
171 /// - `Tag`: The tag to prefix.
172 ///
173 /// Type aliases and static values:
174 /// - `type`: Always a `SpinWeighted<Scalar<ComplexModalVector>, Spin>`, where
175 /// `Spin` is the same spin weight of the pre-transformed `Tag`, and 0 if
176 /// provided a `Tag` with a `Type` that is not `SpinWeighted`
177 /// - `tag`: An alias to the wrapped tag `Tag`. Provided for applicability to
178 /// general `db::PrefixTag` functionality
179 /// - `transform_of`: Another alias to the wrapped tag `Tag`. Provided so that
180 /// utilities that use this prefix for taking transforms can have a more
181 /// expressive code style.
182 template <typename Tag>
183 struct SwshTransform : db::PrefixTag, db::SimpleTag {
184  using type = Scalar<SpinWeighted<
185  ComplexModalVector,
186  detail::adjust_spin_weight_t<Tag, NoDerivative>::type::spin>>;
187  using tag = Tag;
188  // transform_of is provided as a second alias so that utilities that assume a
189  // derivative prefix tag fail earlier during compilation
190  using transform_of = Tag;
191  const static int spin = type::type::spin;
192 };
