343 /// \ingroup DataBoxTagsGroup
344 /// \ingroup ComputationalDomainGroup
345 /// \brief Tag which is either a SimpleTag for quantities on an
346 /// interface, base tag to a compute item which acts on tags on an interface, or
347 /// base tag to a compute item which slices a tag from the volume to an
348 /// interface.
349 ///
350 /// The contained object will be a map from ::Direction to the item type of
351 /// `Tag`, with the set of directions being those produced by `DirectionsTag`.
352 ///
353 /// If a SimpleTag is desired on the interface, then this tag can be added to
354 /// the DataBox directly. If a ComputeTag which acts on Tags on the interface is
355 /// desired, then the tag should be added using `InterfaceCompute`. If a
356 /// ComputeTag which slices a TensorTag or a VariablesTag in the volume to an
357 /// Interface is desired, then it should be added using `Slice`. In all cases,
358 /// the tag can then be retrieved using `Tags::Interface<DirectionsTag, Tag>`.
359 ///
360 /// \tparam DirectionsTag the item of directions
361 /// \tparam Tag the tag labeling the item
362 ///
363 /// \see InterfaceCompute, Slice
364 template <typename DirectionsTag, typename Tag>
365 struct Interface : db::SimpleTag {
366  static_assert(db::is_simple_tag_v<DirectionsTag>);
367  static_assert(db::is_simple_tag_v<Tag>);
368  static std::string name() noexcept {
369  return "Interface<" + db::tag_name<DirectionsTag>() + ", " +
370  db::tag_name<Tag>() + ">";
371  };
372  using tag = Tag;
373  using type = std::unordered_map<::Direction<DirectionsTag::volume_dim>,
374  typename Tag::type>;
375 };

398  template <typename Subtag>
399  static void create_item(
400  const gsl::not_null<typename tag::type*> parent_value,
401  const gsl::not_null<typename Subtag::type*> sub_value) noexcept {
402  sub_value->clear();
403  for (auto& direction_vars : *parent_value) {
404  const auto& direction = direction_vars.first;
405  auto& parent_vars = get<typename Subtag::tag>(direction_vars.second);
406  auto& sub_var = (*sub_value)[direction];
407  for (auto vars_it = parent_vars.begin(), sub_var_it = sub_var.begin();
408  vars_it != parent_vars.end(); ++vars_it, ++sub_var_it) {
409  sub_var_it->set_data_ref(&*vars_it);
410  }
411  }
412  }
413 
414  // The `return_type` can be anything for Subitems because the DataBox figures
415  // out the correct return type, we just use the `return_type` type alias to
416  // signal to the DataBox we want mutating behavior.
417  using return_type = NoSuchType;
418 
419  template <typename Subtag>
420  static void create_compute_item(
421  const gsl::not_null<typename Subtag::type*> sub_value,
422  const typename tag::type& parent_value) noexcept {
423  for (const auto& direction_vars : parent_value) {
424  const auto& direction = direction_vars.first;
425  const auto& parent_vars =
426  get<typename Subtag::tag>(direction_vars.second);
427  auto& sub_var = (*sub_value)[direction];
428  auto sub_var_it = sub_var.begin();
429  for (auto vars_it = parent_vars.begin();
430  vars_it != parent_vars.end(); ++vars_it, ++sub_var_it) {
431  // clang-tidy: do not use const_cast
432  // The DataBox will only give out a const reference to the
433  // result of a compute item. Here, that is a reference to a
434  // const map to Tensors of DataVectors. There is no (publicly
435  // visible) indirection there, so having the map const will
436  // allow only allow const access to the contained DataVectors,
437  // so no modification through the pointer cast here is
438  // possible.
439  sub_var_it->set_data_ref(const_cast<DataVector*>(&*vars_it)); // NOLINT
440  }
441  }
442  }
443 };
