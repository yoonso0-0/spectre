81 /*!
82  * \ingroup DiscontinuousGalerkinGroup
83  * \brief %Observe volume tensor fields.
84  *
85  * A class that writes volume quantities to an h5 file during the simulation.
86  * The observed quantitites are:
87  * - `InertialCoordinates`
88  * - Tensors listed in `Tensors` template parameter
89  * - `Error(*)` = errors in `AnalyticSolutionTensors` =
90  * \f$\text{value} - \text{analytic solution}\f$
91  *
92  * The user may specify an `interpolation_mesh` to which the
93  * data is interpolated.
94  */
95 template <size_t VolumeDim, typename ObservationValueTag, typename... Tensors,
96  typename... AnalyticSolutionTensors, typename EventRegistrars,
97  typename... NonSolutionTensors>
98 class ObserveFields<VolumeDim, ObservationValueTag, tmpl::list<Tensors...>,
99  tmpl::list<AnalyticSolutionTensors...>, EventRegistrars,
100  tmpl::list<NonSolutionTensors...>>
101  : public Event<EventRegistrars> {
102  private:
103  static_assert(
104  std::is_same_v<
105  tmpl::list_difference<tmpl::list<AnalyticSolutionTensors...>,
106  tmpl::list<Tensors...>>,
107  tmpl::list<>>,
108  "All AnalyticSolutionTensors must be listed in Tensors.");
109  using coordinates_tag = domain::Tags::Coordinates<VolumeDim, Frame::Inertial>;

161  explicit ObserveFields(const std::string& subfile_name,
162  const std::vector<std::string>& variables_to_observe,
163  std::optional<Mesh<VolumeDim>> interpolation_mesh = {},
164  const Options::Context& context = {})
165  : subfile_path_("/" + subfile_name),
166  variables_to_observe_(variables_to_observe.begin(),
167  variables_to_observe.end()),
168  interpolation_mesh_(interpolation_mesh) {
169  using ::operator<<;
170  const std::unordered_set<std::string> valid_tensors{
171  db::tag_name<Tensors>()...};
172  for (const auto& name : variables_to_observe_) {
173  if (valid_tensors.count(name) != 1) {
174  PARSE_ERROR(
175  context,
176  name << " is not an available variable. Available variables:\n"
177  << (std::vector<std::string>{db::tag_name<Tensors>()...}));
178  }
179  if (alg::count(variables_to_observe, name) != 1) {
180  PARSE_ERROR(context, name << " specified multiple times");
181  }
182  }
183  variables_to_observe_.insert(coordinates_tag::name());
184  }

192  template <typename Metavariables, typename ParallelComponent>
193  void operator()(
194  const typename ObservationValueTag::type& observation_value,
195  const Mesh<VolumeDim>& mesh,
196  const tnsr::I<DataVector, VolumeDim, Frame::Inertial>&
197  inertial_coordinates,
198  const typename AnalyticSolutionTensors::
199  type&... analytic_solution_tensors,
200  const typename ::Tags::Analytic<
201  AnalyticSolutionTensors>::type&... analytic_solutions,
202  const typename NonSolutionTensors::type&... non_solution_tensors,
203  Parallel::GlobalCache<Metavariables>& cache,
204  const ElementId<VolumeDim>& array_index,
205  const ParallelComponent* const /*meta*/) const noexcept {
206  const std::string element_name =
207  MakeString{} << ElementId<VolumeDim>(array_index) << '/';
208 
209  // if no interpolation_mesh is provided, the interpolation is essentially
210  // ignored by the RegularGridInterpolant except for a single copy.
211  const intrp::RegularGrid interpolant(mesh,
212  interpolation_mesh_.value_or(mesh));
213 
214  // Remove tensor types, only storing individual components.
215  std::vector<TensorComponent> components;
216  // This is larger than we need if we are only observing some
217  // tensors, but that's not a big deal and calculating the correct
218  // size is nontrivial.
219  components.reserve(alg::accumulate(
220  std::initializer_list<size_t>{
221  inertial_coordinates.size(),
222  2 * AnalyticSolutionTensors::type::size()...,
223  NonSolutionTensors::type::size()...},
224  0_st));
225 
226  const auto record_tensor_components = [this, &components, &element_name,
227  &interpolant](
228  const auto tensor_tag_v,
229  const auto& tensor) noexcept {
230  using tensor_tag = tmpl::type_from<decltype(tensor_tag_v)>;
231  if (variables_to_observe_.count(db::tag_name<tensor_tag>()) == 1) {
232  for (size_t i = 0; i < tensor.size(); ++i) {
233  components.emplace_back(element_name + db::tag_name<tensor_tag>() +
234  tensor.component_suffix(i),
235  interpolant.interpolate(tensor[i]));
236  }
237  }
238  };
239  record_tensor_components(tmpl::type_<coordinates_tag>{},
240  inertial_coordinates);
241  EXPAND_PACK_LEFT_TO_RIGHT(record_tensor_components(
242  tmpl::type_<AnalyticSolutionTensors>{}, analytic_solution_tensors));
243  EXPAND_PACK_LEFT_TO_RIGHT(record_tensor_components(
244  tmpl::type_<NonSolutionTensors>{}, non_solution_tensors));
245 
246  const auto record_errors = [this, &components, &element_name, &interpolant](
247  const auto tensor_tag_v, const auto& tensor,
248  const auto& analytic_tensor) noexcept {
249  using tensor_tag = tmpl::type_from<decltype(tensor_tag_v)>;
250  if (variables_to_observe_.count(db::tag_name<tensor_tag>()) == 1) {
251  for (size_t i = 0; i < tensor.size(); ++i) {
252  DataVector error = tensor[i] - analytic_tensor[i];
253  components.emplace_back(element_name + "Error(" +
254  db::tag_name<tensor_tag>() + ")" +
255  tensor.component_suffix(i),
256  interpolant.interpolate(error));
257  }
258  }
259  };
260  EXPAND_PACK_LEFT_TO_RIGHT(
261  record_errors(tmpl::type_<AnalyticSolutionTensors>{},
262  analytic_solution_tensors, analytic_solutions));
263 
264  (void)(record_errors); // Silence GCC warning about unused variable
265 
266  // Send data to volume observer
267  auto& local_observer =
268  *Parallel::get_parallel_component<observers::Observer<Metavariables>>(
269  cache)
270  .ckLocalBranch();
271  Parallel::simple_action<observers::Actions::ContributeVolumeData>(
272  local_observer,
273  observers::ObservationId(observation_value, subfile_path_ + ".vol"),
274  subfile_path_,
275  observers::ArrayComponentId(
276  std::add_pointer_t<ParallelComponent>{nullptr},
277  Parallel::ArrayIndex<ElementId<VolumeDim>>(array_index)),
278  std::move(components), interpolation_mesh_.value_or(mesh).extents(),
279  interpolation_mesh_.value_or(mesh).basis(),
280  interpolation_mesh_.value_or(mesh).quadrature());
281  }
282

284  std::pair<observers::TypeOfObservation, observers::ObservationKey>
285  get_observation_type_and_key_for_registration() const noexcept {
286  return {observers::TypeOfObservation::Volume,
287  observers::ObservationKey(subfile_path_ + ".vol")};
288  }
