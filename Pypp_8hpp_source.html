341 template <typename ReturnType>
342 ReturnType call_work(PyObject* python_module, PyObject* func, PyObject* args) {
343  PyObject* value = PyObject_CallObject(func, args);
344  Py_DECREF(args); // NOLINT
345  if (value == nullptr) {
346  Py_DECREF(func); // NOLINT
347  Py_DECREF(python_module); // NOLINT
348  PyErr_Print();
349  throw std::runtime_error{"Function returned null"};
350  }
351 
352  auto ret = from_py_object<ReturnType>(value);
353  Py_DECREF(value); // NOLINT
354  return ret;
355 }

357 template <typename ReturnType, typename ConversionClassList,
358  typename = std::nullptr_t>
359 struct CallImpl {
360  template <typename... Args>
361  static ReturnType call(const std::string& module_name,
362  const std::string& function_name, const Args&... t) {
363  PyObject* python_module = PyImport_ImportModule(module_name.c_str());
364  if (python_module == nullptr) {
365  PyErr_Print();
366  throw std::runtime_error{std::string("Could not find python module.\n") +
367  module_name};
368  }
369  PyObject* func =
370  PyObject_GetAttrString(python_module, function_name.c_str());
371  if (func == nullptr or not PyCallable_Check(func)) {
372  if (PyErr_Occurred()) {
373  PyErr_Print();
374  }
375  throw std::runtime_error{"Could not find python function in module.\n"};
376  }
377  PyObject* args = pypp::make_py_tuple(t...);
378  auto ret = call_work<ReturnType>(python_module, func, args);
379  Py_DECREF(func); // NOLINT
380  Py_DECREF(python_module); // NOLINT
381  return ret;
382  }
383 };

385 template <typename ReturnType, typename ConversionClassList>
386 struct CallImpl<
387  ReturnType, ConversionClassList,
388  Requires<
389  (tt::is_a_v<Tensor, ReturnType> or tt::is_std_array_v<ReturnType>)and(
390  std::is_same_v<typename ContainerPackAndUnpack<
391  ReturnType, ConversionClassList>::packed_type,
392  DataVector> or
393  std::is_same_v<typename ContainerPackAndUnpack<
394  ReturnType, ConversionClassList>::packed_type,
395  ComplexDataVector>)>> {
396  template <typename... Args>
397  static ReturnType call(const std::string& module_name,
398  const std::string& function_name, const Args&... t) {
399  static_assert(sizeof...(Args) > 0,
400  "Call to python which returns a Tensor of DataVectors must "
401  "pass at least one argument");
402 
403  PyObject* python_module = PyImport_ImportModule(module_name.c_str());
404  if (python_module == nullptr) {
405  PyErr_Print();
406  throw std::runtime_error{std::string("Could not find python module.\n") +
407  module_name};
408  }
409  PyObject* func =
410  PyObject_GetAttrString(python_module, function_name.c_str());
411  if (func == nullptr or not PyCallable_Check(func)) {
412  if (PyErr_Occurred()) {
413  PyErr_Print();
414  }
415  throw std::runtime_error{"Could not find python function in module.\n"};
416  }
417 
418  const std::array<size_t, sizeof...(Args)> arg_sizes{
419  {ContainerPackAndUnpack<Args, ConversionClassList>::get_size(t)...}};
420  const size_t npts = *std::max_element(arg_sizes.begin(), arg_sizes.end());
421  for (size_t i = 0; i < arg_sizes.size(); ++i) {
422  if (gsl::at(arg_sizes, i) != 1 and gsl::at(arg_sizes, i) != npts) {
423  ERROR(
424  "Each argument must return size 1 or "
425  << npts
426  << " (the number of points in the DataVector), but argument number "
427  << i << " has size " << gsl::at(arg_sizes, i));
428  }
429  }
430  auto return_container = make_with_value<typename ContainerPackAndUnpack<
431  ReturnType, ConversionClassList>::packed_container>(npts, 0.0);
432 
433  for (size_t s = 0; s < npts; ++s) {
434  PyObject* args = pypp::make_py_tuple(
435  ContainerPackAndUnpack<Args, ConversionClassList>::unpack(t, s)...);
436  auto ret = call_work<typename ContainerPackAndUnpack<
437  ReturnType, ConversionClassList>::unpacked_container>(python_module,
438  func, args);
439  ContainerPackAndUnpack<ReturnType, ConversionClassList>::pack(
440  make_not_null(&return_container), ret, s);
441  }
442  Py_DECREF(func); // NOLINT
443  Py_DECREF(python_module); // NOLINT
444  return return_container;
445  }

449 /// Calls a Python function from a module/file with given parameters
450 ///
451 /// \param module_name name of module the function is in
452 /// \param function_name name of Python function in module
453 /// \param t the arguments to be passed to the Python function
454 /// \return the object returned by the Python function converted to a C++ type
455 ///
456 /// Custom conversion from containers to basic types that can be converted to
457 /// python objects via the `pypp::ToPyObject` class can be added by passing a
458 /// typelist of conversion class as the second template parameter to
459 /// `pypp::call`. This is generally used for converting classes like
460 /// `Tensor<DataVector, ...>` to a `Tensor<double, ...>` to support using
461 /// `numpy.einsum` in the python code. However, this can also be used to convert
462 /// a complicated class such as an equation of state to a bunch of numbers that
463 /// the python code for the particular test can use to compute the required data
464 /// without having to fully implement python bindings. The conversion classes
465 /// must have the following:
466 /// - a type alias `unpacked_container` that is the type at a single grid point.
467 /// - a type alias `packed_container` the type used when converting the result
468 /// from the python call back into the C++ code.
469 /// - a type alias `packed_type` which corresponds to the packed type that the
470 /// high-level container holds. For example, a `Scalar<DataVector>` would have
471 /// `packed_type` being `DataVector`, a `std::vector<Scalar<DataVector>>`
472 /// would have `packed_type` being `DataVector`, and a
473 /// `std::vector<Scalar<double>>` would have `packed_type` being `double`.
474 /// - an `unpack` function that takes as its arguments the `packed_container`
475 /// and the `grid_point_index`, and returns an `unpacked_container` object
476 /// - a `pack` function that takes as its arguments a
477 /// `gsl::not_null<packed_container*>`, an `const unpacked_container&`, and a
478 /// `size_t` corresponding to which grid point to pack
479 /// - a `get_size` function that takes as its only argument an object of
480 /// `packed_container` and returns the number of elements in the
481 /// `packed_type`
482 ///
483 /// Examples of conversion classes can be found in the specializations of
484 /// `ContainerPackAndUnpack` in the `Pypp.hpp` file. Below is an example of a
485 /// conversion class that takes a class (`ClassForConversionTest`) and returns
486 /// the `a_` member variable.
487 ///
488 /// \snippet Test_Pypp.cpp convert_arbitrary_a
489 ///
490 /// Here is the call to `pypp::call`:
491 ///
492 /// \snippet Test_Pypp.cpp convert_arbitrary_a_call
493 ///
494 /// A conversion class for retrieving the member variables `b_` can also be
495 /// written as follows:
496 ///
497 /// \snippet Test_Pypp.cpp convert_arbitrary_b
498 ///
499 /// \note In order to setup the python interpreter and add the local directories
500 /// to the path, a SetupLocalPythonEnvironment object needs to be constructed
501 /// in the local scope.
502 ///
503 /// \example
504 /// The following example calls the function `test_numeric` from the module
505 /// `pypp_py_tests` which multiplies two integers.
506 /// \snippet Test_Pypp.cpp pypp_int_test
507 /// Alternatively, this examples calls `test_vector` from `pypp_py_tests` which
508 /// converts two vectors to python lists and multiplies them pointwise.
509 /// \snippet Test_Pypp.cpp pypp_vector_test
510 ///
511 /// Pypp can also be used to take a function that performs manipulations of
512 /// NumPy arrays and apply it to either a Tensor of doubles or a Tensor of
513 /// DataVectors. This is useful for testing functions which act on Tensors
514 /// pointwise. For example, let's say we wanted to call the NumPy function which
515 /// performs \f$ v_i = A B^a C_{ia} + D^{ab} E_{iab} \f$, which is implemented
516 /// in python as
517 ///
518 /// \code{.py} def test_einsum(scalar, t_A, t_ia, t_AA, t_iaa):
519 /// return scalar * np.einsum("a,ia->i", t_A, t_ia) +
520 /// np.einsum("ab, iab->i", t_AA, t_iaa)
521 /// \endcode
522 ///
523 /// where \f$ v_i \f$ is the return tensor and
524 /// \f$ A, B^a, C_{ia},D^{ab}, E_{iab} \f$ are the input tensors respectively.
525 /// We call this function through C++ as:
526 /// \snippet Test_Pypp.cpp einsum_example
527 /// for type `T` either a `double` or `DataVector`.
528 ///
529 /// Pypp will also support testing of functions which return and operate on
530 /// `std::array`s of `DataVectors`s. To return a `std::array` of DataVectors,
531 /// the python function should return a python list of doubles.
532 ///
533 /// \note In order to return a
534 /// `Tensor<DataVector...>` from `pypp::call`, at least one
535 /// `Tensor<DataVector...>` must be taken as an argument, as the size of the
536 /// returned tensor needs to be deduced.
537 template <typename ReturnType, typename ConversionClassList = tmpl::list<>,
538  typename... Args>
539 ReturnType call(const std::string& module_name,
540  const std::string& function_name, const Args&... t) {
541  return detail::CallImpl<ReturnType, ConversionClassList>::call(
542  module_name, function_name, t...);
543 }
