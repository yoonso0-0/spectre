23 /// \ingroup ConservativeGroup
24 /// \brief Calculate \f$\partial u/\partial t\f$ for a conservative system.
25 ///
26 /// The time evolution of the variables \f$u\f$ of any conservative
27 /// system is given by \f$\partial_t u = - \partial_i F^i + S\f$,
28 /// where \f$F^i\f$ are the fluxes and \f$S\f$ are the sources.
29 ///
30 /// Source terms are only added for variables in the
31 /// `System::sourced_variables` type list.
32 template <typename System, ::dg::Formulation DgFormulation>
33 struct ConservativeDuDt {
34  static constexpr size_t volume_dim = System::volume_dim;
35  using frame = Frame::Inertial;
36 
37  template <template <class> class StepPrefix>
38  using return_tags = tmpl::list<
39  db::add_tag_prefix<StepPrefix, typename System::variables_tag>>;
40 
41  using argument_tags = tmpl::list<
42  domain::Tags::Mesh<volume_dim>,
43  domain::Tags::InverseJacobian<volume_dim, Frame::Logical,
44  Frame::Inertial>,
45  db::add_tag_prefix<::Tags::Flux, typename System::variables_tag,
46  tmpl::size_t<volume_dim>, frame>,
47  db::add_tag_prefix<::Tags::Source, typename System::variables_tag>>;
48 
49  template <template <class> class StepPrefix, typename... VarsTags, size_t Dim>
50  static void apply(
51  const gsl::not_null<Variables<tmpl::list<StepPrefix<VarsTags>...>>*>
52  dt_vars,
53  const Mesh<Dim> mesh,
54  const InverseJacobian<DataVector, Dim, Frame::Logical, Frame::Inertial>&
55  inverse_jacobian,
56  const Variables<tmpl::list<::Tags::Flux<VarsTags, tmpl::size_t<Dim>,
57  Frame::Inertial>...>>& fluxes,
58  const Variables<tmpl::list<::Tags::Source<VarsTags>...>>&
59  sources) noexcept {
60  static_assert(DgFormulation == ::dg::Formulation::StrongInertial,
61  "Curently only support StrongInertial DG formulation in "
62  "ConservativeDuDt.");
63 
64  const Variables<tmpl::list<::Tags::div<
65  ::Tags::Flux<VarsTags, tmpl::size_t<Dim>, Frame::Inertial>>...>>
66  div_fluxes = divergence(fluxes, mesh, inverse_jacobian);
67 
68  ASSERT(dt_vars->size() == div_fluxes.size(),
69  "The time derivative and the flux divergence have different sizes. "
70  "Time derivative: "
71  << dt_vars->size() << " div fluxes: " << div_fluxes.size() << " "
72  << dt_vars->number_of_grid_points() << " "
73  << div_fluxes.number_of_grid_points());
74  *dt_vars = -div_fluxes;
75  tmpl::for_each<typename System::sourced_variables>(
76  [&dt_vars, &sources](auto tag_v) noexcept {
77  using tag = typename decltype(tag_v)::type;
78  const auto& source = get<::Tags::Source<tag>>(sources);
79  auto& dt_var = get<StepPrefix<tag>>(*dt_vars);
80  ASSERT(source.size() == dt_var.size(),
81  "The source and the time derivative variable must have the "
82  "same number of independent components. Time derivative: "
83  << dt_var.size() << " source: " << source.size());
84  for (size_t i = 0; i < source.size(); ++i) {
85  dt_var[i] += source[i];
86  }
87  });
88  }
89 };
90 } // namespace dg
91 } // namespace evolution
