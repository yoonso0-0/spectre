52 /// \ingroup InitializationGroup
53 /// \brief Allocate variables needed for evolution of conservative systems
54 ///
55 /// Uses:
56 /// - DataBox:
57 /// * `Tags::Mesh<Dim>`
58 ///
59 /// DataBox changes:
60 /// - Adds:
61 /// * System::variables_tag
62 /// * db::add_tag_prefix<Tags::Flux, System::variables_tag>
63 /// * db::add_tag_prefix<Tags::Source, System::variables_tag>
64 ///
65 /// - Removes: nothing
66 /// - Modifies: nothing
67 ///
68 /// \note This action relies on the `SetupDataBox` aggregated initialization
69 /// mechanism, so `Actions::SetupDataBox` must be present in the
70 /// `Initialization` phase action list prior to this action.
71 template <typename System, typename EquationOfStateTag = NoSuchType>
72 struct ConservativeSystem {
73  private:
74  static_assert(System::is_in_flux_conservative_form,
75  "System is not in flux conservative form");
76 
77  static constexpr size_t dim = System::volume_dim;
78 
79  using variables_tag = typename System::variables_tag;
80  using fluxes_tag = db::add_tag_prefix<::Tags::Flux, variables_tag,
81  tmpl::size_t<dim>, Frame::Inertial>;
82  using sources_tag = db::add_tag_prefix<::Tags::Source, variables_tag>;
83 
84  template <typename LocalSystem,
85  bool = LocalSystem::has_primitive_and_conservative_vars>
86  struct simple_tags_impl {
87  using type = tmpl::list<variables_tag, fluxes_tag, sources_tag>;
88  };
89 
90  template <typename LocalSystem>
91  struct simple_tags_impl<LocalSystem, true> {
92  using type = tmpl::list<variables_tag, fluxes_tag, sources_tag,
93  typename System::primitive_variables_tag,
94  EquationOfStateTag>;
95  };
96 
97  public:
98  using simple_tags = typename simple_tags_impl<System>::type;
99 
100  using compute_tags = db::AddComputeTags<>;
101 
102  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
103  typename ArrayIndex, typename ActionList,
104  typename ParallelComponent>
105  static auto apply(db::DataBox<DbTagsList>& box,
106  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
107  const Parallel::GlobalCache<Metavariables>& /*cache*/,
108  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
109  const ParallelComponent* const /*meta*/) noexcept {
110  const size_t num_grid_points =
111  db::get<domain::Tags::Mesh<dim>>(box).number_of_grid_points();
112  typename variables_tag::type vars(num_grid_points);
113  typename fluxes_tag::type fluxes(num_grid_points);
114  typename sources_tag::type sources(num_grid_points);
115 
116  if constexpr (System::has_primitive_and_conservative_vars) {
117  using PrimitiveVars = typename System::primitive_variables_tag::type;
118 
119  PrimitiveVars primitive_vars{
120  db::get<domain::Tags::Mesh<dim>>(box).number_of_grid_points()};
121  auto equation_of_state =
122  db::get<::Tags::AnalyticSolutionOrData>(box).equation_of_state();
123  Initialization::mutate_assign<simple_tags>(
124  make_not_null(&box), std::move(vars), std::move(fluxes),
125  std::move(sources), std::move(primitive_vars),
126  std::move(equation_of_state));
127  } else {
128  Initialization::mutate_assign<simple_tags>(
129  make_not_null(&box), std::move(vars), std::move(fluxes),
130  std::move(sources));
131  }
132  return std::make_tuple(std::move(box));
133  }
134 };
135 } // namespace Actions
136 } // namespace Initialization
