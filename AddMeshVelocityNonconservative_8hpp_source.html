31 /*!
32  * \ingroup ActionsGroup
33  * \brief Compute and add the advective term for nonconservative systems on
34  * moving meshes
35  *
36  * Adds the following term to the time derivative:
37  *
38  * \f{align}{
39  * v^i_g \partial_i u_\alpha,
40  * \f}
41  *
42  * where \f$u_\alpha\f$ are the evolved variables and \f$v^i_g\f$ is the
43  * velocity of the mesh.
44  *
45  * \note The term is always added in the `Frame::Inertial` frame, and the plus
46  * sign arises because we add it to the time derivative.
47  *
48  * Uses:
49  * - DataBox:
50  * - `Tags::deriv<system::variables_tags, Dim, Frame::Inertial>`
51  * - `domain::Tags::MeshVelocity<Dim>`
52  *
53  * DataBox changes:
54  * - Adds: nothing
55  * - Removes: nothing
56  * - Modifies: `Tags::dt<system::variable_tags>`
57  */
58 struct AddMeshVelocityNonconservative {
59  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
60  typename ArrayIndex, typename ActionList,
61  typename ParallelComponent>
62  static std::tuple<db::DataBox<DbTagsList>&&> apply(
63  db::DataBox<DbTagsList>& box,
64  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
65  const Parallel::GlobalCache<Metavariables>& /*cache*/,
66  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
67  const ParallelComponent* const /*meta*/) noexcept { // NOLINT const
68  using variables_tags =
69  typename Metavariables::system::variables_tag::tags_list;
70  const auto& mesh_velocity =
71  db::get<::domain::Tags::MeshVelocity<Metavariables::volume_dim>>(box);
72 
73  if (static_cast<bool>(mesh_velocity)) {
74  tmpl::for_each<variables_tags>([&box, &mesh_velocity](
75  auto variables_tag_v) noexcept {
76  using variable_tag = typename decltype(variables_tag_v)::type;
77  using dt_variable_tag = ::Tags::dt<variable_tag>;
78  using deriv_tag =
79  ::Tags::deriv<variable_tag, tmpl::size_t<Metavariables::volume_dim>,
80  Frame::Inertial>;
81 
82  db::mutate<dt_variable_tag>(
83  make_not_null(&box),
84  [](const auto dt_var_ptr, const auto& deriv_tensor,
85  const boost::optional<tnsr::I<
86  DataVector, Metavariables::volume_dim, Frame::Inertial>>&
87  grid_velocity) noexcept {
88  for (size_t storage_index = 0;
89  storage_index < deriv_tensor.size(); ++storage_index) {
90  // We grab the `deriv_tensor_index`, which would be e.g.
91  // `(i, a, b)`, so `(0, 2, 3)`
92  const auto deriv_tensor_index =
93  deriv_tensor.get_tensor_index(storage_index);
94  // Then we drop the derivative index (the first entry) to get
95  // `(a, b)` (or `(2, 3)`)
96  const auto tensor_index =
97  all_but_specified_element_of(deriv_tensor_index, 0);
98  // Set `deriv_index` to `i` (or `0` in the example)
99  const size_t deriv_index = gsl::at(deriv_tensor_index, 0);
100  dt_var_ptr->get(tensor_index) +=
101  grid_velocity->get(deriv_index) *
102  deriv_tensor[storage_index];
103  }
104  },
105  db::get<deriv_tag>(box), mesh_velocity);
106  });
107  }
108  return std::forward_as_tuple(std::move(box));
109  }
110 };
111 } // namespace Actions
112 } // namespace evolution
