46 /*!
47  * \ingroup UtilitiesGroup
48  * The dot product of two vectors.
49  *
50  * \param N the length of the vectors.
51  * \param X a pointer to the first element of the first vector.
52  * \param INCX the stride for the elements of the first vector.
53  * \param Y a pointer to the first element of the second vector.
54  * \param INCY the stride for the elements of the second vector.
55  * \return the dot product of the given vectors.
56  */
57 inline double ddot_(const size_t& N, const double* X, const size_t& INCX,
58  const double* Y, const size_t& INCY) {
59  // INCX and INCY are allowed to be negative by BLAS, but we never
60  // use them that way. If needed, they can be changed here, but then
61  // code providing values will also have to be changed to int to
62  // avoid warnings.
63  return blas_detail::ddot_(gsl::narrow_cast<int>(N), X,
64  gsl::narrow_cast<int>(INCX), Y,
65  gsl::narrow_cast<int>(INCY));
66 }

69 /*!
70  * \ingroup UtilitiesGroup
71  * \brief Perform a matrix-matrix multiplication
72  *
73  * Perform the matrix-matrix multiplication
74  * \f[
75  * C = \alpha \mathrm{op}(A) \mathrm{op}(B) + \beta \mathrm{op}(C)
76  * \f]
77  *
78  * where \f$\mathrm{op}(A)\f$ represents either \f$A\f$ or \f$A^{T}\f$
79  * (transpose of \f$A\f$).
80  *
81  * LIBXSMM, which is much faster than BLAS for small matrices, can be called
82  * instead of BLAS by passing the template parameter `true`.
83  *
84  * \param TRANSA either 'N', 'T' or 'C', transposition of matrix A
85  * \param TRANSB either 'N', 'T' or 'C', transposition of matrix B
86  * \param M Number of rows in \f$\mathrm{op}(A)\f$
87  * \param N Number of columns in \f$\mathrm{op}(B)\f$ and \f$\mathrm{op}(C)\f$
88  * \param K Number of columns in \f$\mathrm{op}(A)\f$
89  * \param ALPHA specifies \f$\alpha\f$
90  * \param A Matrix \f$A\f$
91  * \param LDA Specifies first dimension of \f$\mathrm{op}(A)\f$
92  * \param B Matrix \f$B\f$
93  * \param LDB Specifies first dimension of \f$\mathrm{op}(B)\f$
94  * \param BETA specifies \f$\beta\f$
95  * \param C Matrix \f$C\f$
96  * \param LDC Specifies first dimension of \f$\mathrm{op}(C)\f$
97  * \tparam UseLibXsmm if `true` then use LIBXSMM
98  */
99 template <bool UseLibXsmm = false>
100 inline void dgemm_(const char& TRANSA, const char& TRANSB, const size_t& M,
101  const size_t& N, const size_t& K, const double& ALPHA,
102  const double* A, const size_t& LDA, const double* B,
103  const size_t& LDB, const double& BETA, double* C,
104  const size_t& LDC) {
105  ASSERT('N' == TRANSA or 'n' == TRANSA or 'T' == TRANSA or 't' == TRANSA or
106  'C' == TRANSA or 'c' == TRANSA,
107  "TRANSA must be upper or lower case N, T, or C. See the BLAS "
108  "documentation for help.");
109  ASSERT('N' == TRANSB or 'n' == TRANSB or 'T' == TRANSB or 't' == TRANSB or
110  'C' == TRANSB or 'c' == TRANSB,
111  "TRANSB must be upper or lower case N, T, or C. See the BLAS "
112  "documentation for help.");
113  blas_detail::dgemm_(
114  TRANSA, TRANSB, gsl::narrow_cast<int>(M), gsl::narrow_cast<int>(N),
115  gsl::narrow_cast<int>(K), ALPHA, A, gsl::narrow_cast<int>(LDA), B,
116  gsl::narrow_cast<int>(LDB), BETA, C, gsl::narrow_cast<int>(LDC));
117 }
