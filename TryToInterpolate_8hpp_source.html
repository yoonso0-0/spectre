34 // Interpolates data onto a set of points desired by an InterpolationTarget.
35 template <typename InterpolationTargetTag, typename Metavariables,
36  typename DbTags>
37 void interpolate_data(
38  const gsl::not_null<db::DataBox<DbTags>*> box,
39  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
40  db::mutate_apply<tmpl::list<Tags::InterpolatedVarsHolders<Metavariables>>,
41  tmpl::list<Tags::VolumeVarsInfo<Metavariables>>>(
42  [&temporal_id](
43  const gsl::not_null<
44  typename Tags::InterpolatedVarsHolders<Metavariables>::type*>
45  holders,
46  const typename Tags::VolumeVarsInfo<Metavariables>::type&
47  volume_vars_info) noexcept {
48  auto& interp_info =
49  get<Vars::HolderTag<InterpolationTargetTag, Metavariables>>(
50  *holders)
51  .infos.at(temporal_id);
52 
53  for (const auto& volume_info_outer : volume_vars_info) {
54  // Are we at the right time?
55  if (volume_info_outer.first != temporal_id) {
56  continue;
57  }
58 
59  // Get list of ElementIds that have the correct temporal_id and that
60  // have not yet been interpolated.
61  std::vector<ElementId<Metavariables::volume_dim>> element_ids;
62 
63  for (const auto& volume_info_inner : volume_info_outer.second) {
64  // Have we interpolated this element before?
65  if (interp_info.interpolation_is_done_for_these_elements.find(
66  volume_info_inner.first) ==
67  interp_info.interpolation_is_done_for_these_elements.end()) {
68  interp_info.interpolation_is_done_for_these_elements.emplace(
69  volume_info_inner.first);
70  element_ids.push_back(volume_info_inner.first);
71  }
72  }
73 
74  // Get element logical coordinates.
75  const auto element_coord_holders = element_logical_coordinates(
76  element_ids, interp_info.block_coord_holders);
77 
78  // Construct local vars and interpolate.
79  for (const auto& element_coord_pair : element_coord_holders) {
80  const auto& element_id = element_coord_pair.first;
81  const auto& element_coord_holder = element_coord_pair.second;
82  const auto& volume_info = volume_info_outer.second.at(element_id);
83 
84  // Construct local_vars which is some set of variables
85  // derived from volume_info.vars plus an arbitrary set
86  // of compute items in
87  // InterpolationTargetTag::compute_items_on_source.
88 
89  auto new_box = db::create<
90  db::AddSimpleTags<::Tags::Variables<
91  typename Metavariables::interpolator_source_vars>>,
92  db::AddComputeTags<
93  typename InterpolationTargetTag::compute_items_on_source>>(
94  volume_info.vars);
95 
96  Variables<
97  typename InterpolationTargetTag::vars_to_interpolate_to_target>
98  local_vars(volume_info.mesh.number_of_grid_points());
99 
100  tmpl::for_each<
101  typename InterpolationTargetTag::vars_to_interpolate_to_target>(
102  [&new_box, &local_vars](auto x) noexcept {
103  using tag = typename decltype(x)::type;
104  get<tag>(local_vars) = db::get<tag>(new_box);
105  });
106 
107  // Now interpolate.
108  intrp::Irregular<Metavariables::volume_dim> interpolator(
109  volume_info.mesh, element_coord_holder.element_logical_coords);
110  interp_info.vars.emplace_back(interpolator.interpolate(local_vars));
111  interp_info.global_offsets.emplace_back(
112  element_coord_holder.offsets);
113  }
114  }
115  },
116  box);
117 }
118 } // namespace interpolator_detail

120 /// Check if we have enough information to interpolate. If so, do the
121 /// interpolation and send data to the InterpolationTarget.
122 template <typename InterpolationTargetTag, typename Metavariables,
123  typename DbTags>
124 void try_to_interpolate(
125  const gsl::not_null<db::DataBox<DbTags>*> box,
126  const gsl::not_null<Parallel::GlobalCache<Metavariables>*> cache,
127  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
128  const auto& holders =
129  db::get<Tags::InterpolatedVarsHolders<Metavariables>>(*box);
130  const auto& vars_infos =
131  get<Vars::HolderTag<InterpolationTargetTag, Metavariables>>(holders)
132  .infos;
133 
134  // If we don't yet have any points for this InterpolationTarget at
135  // this temporal_id, we should exit (we can't interpolate anyway).
136  if (vars_infos.count(temporal_id) == 0) {
137  return;
138  }
139 
140  interpolator_detail::interpolate_data<InterpolationTargetTag, Metavariables>(
141  box, temporal_id);
142 
143  // Send interpolated data only if interpolation has been done on all
144  // of the local elements.
145  const auto& num_elements = db::get<Tags::NumberOfElements>(*box);
146  if (vars_infos.at(temporal_id)
147  .interpolation_is_done_for_these_elements.size() == num_elements) {
148  // Send data to InterpolationTarget, but only if the list of points is
149  // non-empty.
150  if (not vars_infos.at(temporal_id).global_offsets.empty()) {
151  const auto& info = vars_infos.at(temporal_id);
152  auto& receiver_proxy = Parallel::get_parallel_component<
153  InterpolationTarget<Metavariables, InterpolationTargetTag>>(*cache);
154  Parallel::simple_action<
155  Actions::InterpolationTargetReceiveVars<InterpolationTargetTag>>(
156  receiver_proxy, info.vars, info.global_offsets, temporal_id);
157  }
158 
159  // Clear interpolated data, since we don't need it anymore.
160  db::mutate<Tags::InterpolatedVarsHolders<Metavariables>>(
161  box, [&temporal_id](
162  const gsl::not_null<typename Tags::InterpolatedVarsHolders<
163  Metavariables>::type*>
164  holders_l) noexcept {
165  get<Vars::HolderTag<InterpolationTargetTag, Metavariables>>(
166  *holders_l)
167  .infos.erase(temporal_id);
168  });
169  }
170 }
