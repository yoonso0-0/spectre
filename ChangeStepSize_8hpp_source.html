29 /// \ingroup ActionsGroup
30 /// \ingroup TimeGroup
31 /// \brief Adjust the step size for local time stepping
32 ///
33 /// Uses:
34 /// - GlobalCache:
35 /// - Tags::StepChoosers<StepChooserRegistrars>
36 /// - Tags::StepController
37 /// - DataBox:
38 /// - Tags::HistoryEvolvedVariables
39 /// - Tags::TimeStep
40 /// - Tags::TimeStepId
41 /// - Tags::TimeStepper<>
42 ///
43 /// DataBox changes:
44 /// - Adds: nothing
45 /// - Removes: nothing
46 /// - Modifies: Tags::Next<Tags::TimeStepId>, Tags::TimeStep
47 template <typename StepChooserRegistrars>
48 struct ChangeStepSize {
49  using step_choosers_tag = Tags::StepChoosers<StepChooserRegistrars>;
50  using const_global_cache_tags =
51  tmpl::list<step_choosers_tag, Tags::StepController>;
52 
53  template <typename DbTags, typename... InboxTags, typename Metavariables,
54  typename ArrayIndex, typename ActionList,
55  typename ParallelComponent>
56  static std::tuple<db::DataBox<DbTags>&&> apply(
57  db::DataBox<DbTags>& box, tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
58  const Parallel::GlobalCache<Metavariables>& cache,
59  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
60  const ParallelComponent* const /*meta*/) noexcept {
61  static_assert(Metavariables::local_time_stepping,
62  "ChangeStepSize can only be used with local time-stepping.");
63 
64  const LtsTimeStepper& time_stepper = db::get<Tags::TimeStepper<>>(box);
65  const auto& step_choosers = Parallel::get<step_choosers_tag>(cache);
66  const auto& step_controller = Parallel::get<Tags::StepController>(cache);
67 
68  const auto& time_id = db::get<Tags::TimeStepId>(box);
69  const auto& history = db::get<Tags::HistoryEvolvedVariables<>>(box);
70 
71  if (not time_stepper.can_change_step_size(time_id, history)) {
72  return std::forward_as_tuple(std::move(box));
73  }
74 
75  const auto& current_step = db::get<Tags::TimeStep>(box);
76 
77  const double last_step_size =
78  history.size() > 0 ? abs(time_id.step_time() -
79  (history.end() - 1).time_step_id().step_time())
80  .value()
81  : std::numeric_limits<double>::infinity();
82 
83  // The step choosers return the magnitude of the desired step, so
84  // we always want the minimum requirement, but we have to negate
85  // the final answer if time is running backwards.
86  double desired_step = std::numeric_limits<double>::infinity();
87  for (const auto& step_chooser : step_choosers) {
88  desired_step = std::min(
89  desired_step, step_chooser->desired_step(last_step_size, box, cache));
90  }
91  if (not current_step.is_positive()) {
92  desired_step = -desired_step;
93  }
94 
95  const auto new_step =
96  step_controller.choose_step(time_id.step_time(), desired_step);
97  if (new_step != current_step) {
98  const auto new_next_time_id =
99  time_stepper.next_time_id(time_id, new_step);
100 
101  db::mutate<Tags::Next<Tags::TimeStepId>, Tags::TimeStep>(
102  make_not_null(&box),
103  [&new_next_time_id, &new_step](
104  const gsl::not_null<TimeStepId*> next_time_id,
105  const gsl::not_null<TimeDelta*> time_step) noexcept {
106  *next_time_id = new_next_time_id;
107  *time_step = new_step;
108  });
109  }
110 
111  return std::forward_as_tuple(std::move(box));
112  }
113 };
114 } // namespace Actions
