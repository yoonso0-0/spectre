22 /*!
23  * \ingroup ComputationalDomainGroup
24  * \brief A SegmentId labels a segment of the interval [-1,1] and is used to
25  * identify the bounds of an Element in a Block in each dimension.
26  *
27  * In \f$d\f$ dimensions, \f$d\f$ SegmentId%s are used to identify an Element.
28  * In each dimension, a segment spans the subinterval \f$[-1 + 2 \frac{i}{N}, -1
29  * + 2 \frac{i+1}{N}]\f$ of the logical coordinates of a Block, where \f$i \f$=
30  * `index` and \f$N = 2^L\f$ where \f$L\f$ = `refinement_level`.
31  *
32  * \image html SegmentId.svg "SegmentIds"
33  *
34  * In the figure, The `index` of segments increase from the `lower side` to the
35  * `upper side` in each dimension of a Block, while the `refinement level`
36  * increases as the segments are subdivided. For example, let the segment
37  * labeled `self` be on `refinement level` \f$L\f$, with `index` \f$i\f$. Its
38  * `parent` segment is on `refinement level` \f$L-1\f$ with `index`
39  * \f$\frac{i-1}{2}\f$. The `children` of `self` are on `refinement level`
40  * \f$L+1\f$, and have `index` \f$2i\f$ and \f$2i+1\f$ for the lower and upper
41  * child respectively. Also labeled on the figure are the `sibling` and
42  * `abutting nibling` (child of sibling) of `self`. These relationships between
43  * segments are important for h-refinement, since in each dimension an Element
44  * can be flagged to split into its two `children` segments, or join with its
45  * `sibling` segment to form its `parent` segment. As refinement levels of
46  * neighboring elements are kept within one, in the direction of its `sibling`,
47  * a segment can only abut its `sibling` or `abutting nibling`, while on the
48  * opposite side, it can abut a segment on its level, the next-lower, or the
49  * next-higher level.
50  *
51  * \details
52  * Because `ElementId` is built up of `VolumeDim` `SegmentId`s, and `ElementId`
53  * is used as a Charm++ index, `ElementId` has the following restrictions:
54  * - `ElementId` must satisfy `std::is_pod`
55  * - `ElementId` must not be larger than the size of three `int`s, i.e.
56  * `sizeof(ElementId) <= 3 * sizeof(int)`
57  * which means `SegmentId` must be the size of an `int` and satisfy
58  * `std::is_pod`. In order to satisfy the size requirement, we use bitfields
59  * internally in `SegmentId` with `ASSERT`s in the constructors to check for
60  * potential overflows.
61  */

152 inline SegmentId SegmentId::id_of_parent() const noexcept {
153  ASSERT(0 != refinement_level_,
154  "Cannot call id_of_parent() on root refinement level!");
155  // The parent has half as many segments as the child.
156  return {refinement_level() - 1, index() / 2};
157 }
158 
159 inline SegmentId SegmentId::id_of_child(Side side) const noexcept {
160  // We cannot ASSERT on the maximum level because it's only known at runtime
161  // and only known elsewhere in the code, not by SegmentId. I.e. SegmentId is
162  // too low-level to know about this.
163  // The child has twice as many segments as the parent, so for a particular
164  // parent segment, there is both an upper and lower child segment.
165  if (Side::Lower == side) {
166  return {refinement_level() + 1, index() * 2};
167  }
168  return {refinement_level() + 1, 1 + index() * 2};
169 }

171 inline SegmentId SegmentId::id_of_sibling() const noexcept {
172  ASSERT(0 != refinement_level(),
173  "The segment on the root refinement level has no sibling");
174  return {refinement_level(), (0 == index() % 2 ? index() + 1 : index() - 1)};
175 }
176 
177 inline SegmentId SegmentId::id_of_abutting_nibling() const noexcept {
178  ASSERT(0 != refinement_level(),
179  "The segment on the root refinement level has no abutting nibling");
180  return {refinement_level() + 1,
181  (0 == index() % 2 ? 2 * index() + 2 : 2 * index() - 1)};
182 }

184 inline Side SegmentId::side_of_sibling() const noexcept {
185  ASSERT(0 != refinement_level(),
186  "The segment on the root refinement level has no sibling");
187  return 0 == index() % 2 ? Side::Upper : Side::Lower;
188 }

190 inline SegmentId SegmentId::id_if_flipped() const noexcept {
191  return {refinement_level(), two_to_the(refinement_level()) - 1 - index()};
192 }

194 inline double SegmentId::endpoint(Side side) const noexcept {
195  if (Side::Lower == side) {
196  return -1.0 + (2.0 * index()) / two_to_the(refinement_level());
197  }
198  return -1.0 + (2.0 * index() + 2.0) / two_to_the(refinement_level());
199 }

201 inline bool SegmentId::overlaps(const SegmentId& other) const noexcept {
202  const size_t this_denom = two_to_the(refinement_level());
203  const size_t other_denom = two_to_the(other.refinement_level());
204  return index() * other_denom < (other.index() + 1) * this_denom and
205  other.index() * this_denom < (index() + 1) * other_denom;
206 }
