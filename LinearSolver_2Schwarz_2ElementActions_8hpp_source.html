205 // Once the residual data is available on all overlaps, solve the restricted
206 // problem for this element-centered subdomain. Apply the weighted solution on
207 // this element directly and send the solution on overlap regions to the
208 // neighbors that they overlap with.
209 template <typename FieldsTag, typename OptionsGroup, typename SubdomainOperator>
210 struct SolveSubdomain {
211  private:
212  using fields_tag = FieldsTag;
213  using residual_tag =
214  db::add_tag_prefix<LinearSolver::Tags::Residual, fields_tag>;
215  static constexpr size_t Dim = SubdomainOperator::volume_dim;
216  using SubdomainData =
217  ElementCenteredSubdomainData<Dim, typename residual_tag::tags_list>;
218  using OverlapData = typename SubdomainData::OverlapData;
219  using overlap_solution_inbox_tag =
220  OverlapSolutionInboxTag<Dim, OptionsGroup, OverlapData>;
221 
222  public:
223  using const_global_cache_tags =
224  tmpl::list<Tags::MaxOverlap<OptionsGroup>,
225  logging::Tags::Verbosity<OptionsGroup>>;
226 
227  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
228  typename ActionList, typename ParallelComponent>
229  static std::tuple<db::DataBox<DbTagsList>&&> apply(
230  db::DataBox<DbTagsList>& box,
231  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
232  Parallel::GlobalCache<Metavariables>& cache,
233  const ElementId<Dim>& element_id, const ActionList /*meta*/,
234  const ParallelComponent* const /*meta*/) noexcept {
235  const size_t iteration_id =
236  get<Convergence::Tags::IterationId<OptionsGroup>>(box);
237 
238  // Do some logging
239  if (UNLIKELY(get<logging::Tags::Verbosity<OptionsGroup>>(box) >=
240  ::Verbosity::Debug)) {
241  Parallel::printf(
242  "%s " + Options::name<OptionsGroup>() + "(%zu): Solve subdomain\n",
243  element_id, iteration_id);
244  }
245 
246  const auto& element = db::get<domain::Tags::Element<Dim>>(box);
247  const size_t max_overlap = db::get<Tags::MaxOverlap<OptionsGroup>>(box);
248 
249  // Assemble the subdomain data from the data on the element and the
250  // communicated overlap data
251  db::mutate<SubdomainDataBufferTag<SubdomainData, OptionsGroup>,
252  Tags::Overlaps<residual_tag, Dim, OptionsGroup>>(
253  make_not_null(&box),
254  [max_overlap, &element](
255  const gsl::not_null<SubdomainData*> subdomain_data,
256  const auto overlap_residuals, const auto& residual) noexcept {
257  subdomain_data->element_data = residual;
258  // Nothing was communicated if the overlaps are empty
259  if (LIKELY(max_overlap > 0 and element.number_of_neighbors() > 0)) {
260  subdomain_data->overlap_data = std::move(*overlap_residuals);
261  }
262  },
263  db::get<residual_tag>(box));
264  const auto& subdomain_residual =
265  db::get<SubdomainDataBufferTag<SubdomainData, OptionsGroup>>(box);
266 
267  // Allocate workspace memory for repeatedly applying the subdomain operator
268  const size_t num_points =
269  db::get<domain::Tags::Mesh<Dim>>(box).number_of_grid_points();
270  SubdomainOperator subdomain_operator{num_points};
271 
272  // Construct the subdomain operator
273  const auto apply_subdomain_operator =
274  [&box, &subdomain_operator](const gsl::not_null<SubdomainData*> result,
275  const SubdomainData& operand) noexcept {
276  // The subdomain operator can retrieve any information on the subdomain
277  // geometry that is available through the DataBox. The user is responsible
278  // for communicating this information across neighbors if necessary.
279  db::apply<typename SubdomainOperator::element_operator>(
280  box, operand, result, make_not_null(&subdomain_operator));
281  tmpl::for_each<tmpl::list<domain::Tags::InternalDirections<Dim>,
282  domain::Tags::BoundaryDirectionsInterior<Dim>>>(
283  [&box, &operand, &result,
284  &subdomain_operator](auto directions_v) noexcept {
285  using directions = tmpl::type_from<decltype(directions_v)>;
286  using face_operator =
287  typename SubdomainOperator::template face_operator<directions>;
288  interface_apply<directions, face_operator>(
289  box, operand, result, make_not_null(&subdomain_operator));
290  });
291  };
292 
293  // Solve the subdomain problem
294  const auto& subdomain_solver =
295  get<Tags::SubdomainSolverBase<OptionsGroup>>(box);
296  auto subdomain_solve_initial_guess_in_solution_out =
297  make_with_value<SubdomainData>(subdomain_residual, 0.);
298  const auto subdomain_solve_has_converged = subdomain_solver.solve(
299  make_not_null(&subdomain_solve_initial_guess_in_solution_out),
300  apply_subdomain_operator, subdomain_residual);
301  // Re-naming the solution buffer for the code below
302  auto& subdomain_solution = subdomain_solve_initial_guess_in_solution_out;
303 
304  // Do some logging and observing
305  if (UNLIKELY(get<logging::Tags::Verbosity<OptionsGroup>>(box) >=
306  ::Verbosity::Quiet)) {
307  if (not subdomain_solve_has_converged or
308  subdomain_solve_has_converged.reason() ==
309  Convergence::Reason::MaxIterations) {
310  Parallel::printf(
311  "%s WARNING: Subdomain solver did not converge in %zu iterations: "
312  "%e -> %e\n",
313  element_id, subdomain_solve_has_converged.num_iterations(),
314  subdomain_solve_has_converged.initial_residual_magnitude(),
315  subdomain_solve_has_converged.residual_magnitude());
316  } else if (UNLIKELY(get<logging::Tags::Verbosity<OptionsGroup>>(box) >=
317  ::Verbosity::Debug)) {
318  Parallel::printf(
319  "%s Subdomain solver converged in %zu iterations (%s): %e -> %e\n",
320  element_id, subdomain_solve_has_converged.num_iterations(),
321  subdomain_solve_has_converged.reason(),
322  subdomain_solve_has_converged.initial_residual_magnitude(),
323  subdomain_solve_has_converged.residual_magnitude());
324  }
325  }
326  contribute_to_subdomain_stats_observation<OptionsGroup, ParallelComponent>(
327  iteration_id + 1, subdomain_solve_has_converged.num_iterations(), cache,
328  element_id);
329 
330  // Apply weighting
331  if (LIKELY(max_overlap > 0)) {
332  subdomain_solution.element_data *=
333  get(db::get<Tags::Weight<OptionsGroup>>(box));
334  }
335 
336  // Apply solution to central element
337  db::mutate<fields_tag>(make_not_null(&box),
338  [&subdomain_solution](const auto fields) noexcept {
339  *fields += subdomain_solution.element_data;
340  });
341 
342  // Send overlap solutions back to the neighbors that they are on
343  if (LIKELY(max_overlap > 0)) {
344  auto& receiver_proxy =
345  Parallel::get_parallel_component<ParallelComponent>(cache);
346  for (auto& [overlap_id, overlap_solution] :
347  subdomain_solution.overlap_data) {
348  const auto& direction = overlap_id.first;
349  const auto& neighbor_id = overlap_id.second;
350  const auto& orientation =
351  element.neighbors().at(direction).orientation();
352  const auto direction_from_neighbor = orientation(direction.opposite());
353  Parallel::receive_data<overlap_solution_inbox_tag>(
354  receiver_proxy[neighbor_id], iteration_id,
355  std::make_pair(
356  OverlapId<Dim>{direction_from_neighbor, element.id()},
357  std::move(overlap_solution)));
358  }
359  }
360  return {std::move(box)};
361  }
362 };

365 // with neighboring element-centered subdomains. Combine the solutions as a

367 template <typename FieldsTag, typename OptionsGroup, typename SubdomainOperator>
368 struct ReceiveOverlapSolution {
369  private:
370  using fields_tag = FieldsTag;
371  using residual_tag =
372  db::add_tag_prefix<LinearSolver::Tags::Residual, fields_tag>;
373  static constexpr size_t Dim = SubdomainOperator::volume_dim;
374  using SubdomainData =
375  ElementCenteredSubdomainData<Dim, typename residual_tag::tags_list>;
376  using OverlapSolution = typename SubdomainData::OverlapData;
377  using overlap_solution_inbox_tag =
378  OverlapSolutionInboxTag<Dim, OptionsGroup, OverlapSolution>;
379 
380  public:
381  using const_global_cache_tags = tmpl::list<Tags::MaxOverlap<OptionsGroup>>;
382  using inbox_tags = tmpl::list<overlap_solution_inbox_tag>;
383 
384  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
385  size_t Dim>
386  static bool is_ready(const db::DataBox<DbTagsList>& box,
387  const tuples::TaggedTuple<InboxTags...>& inboxes,
388  const Parallel::GlobalCache<Metavariables>& /*cache*/,
389  const ElementId<Dim>& /*element_id*/) noexcept {
390  if (UNLIKELY(db::get<Tags::MaxOverlap<OptionsGroup>>(box) == 0)) {
391  return true;
392  }
393  return dg::has_received_from_all_mortars<overlap_solution_inbox_tag>(
394  get<Convergence::Tags::IterationId<OptionsGroup>>(box),
395  get<domain::Tags::Element<Dim>>(box), inboxes);
396  }

398  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
399  typename ActionList, typename ParallelComponent>
400  static std::tuple<db::DataBox<DbTagsList>&&> apply(
401  db::DataBox<DbTagsList>& box, tuples::TaggedTuple<InboxTags...>& inboxes,
402  const Parallel::GlobalCache<Metavariables>& /*cache*/,
403  const ElementId<Dim>& element_id, const ActionList /*meta*/,
404  const ParallelComponent* const /*meta*/) noexcept {
405  const size_t iteration_id =
406  get<Convergence::Tags::IterationId<OptionsGroup>>(box);
407  const auto& element = db::get<domain::Tags::Element<Dim>>(box);
408 
409  // Nothing to do if overlap is empty
410  if (UNLIKELY(db::get<Tags::MaxOverlap<OptionsGroup>>(box) == 0 or
411  element.number_of_neighbors() == 0)) {
412  return {std::move(box)};
413  }
414 
415  // Do some logging
416  if (UNLIKELY(get<logging::Tags::Verbosity<OptionsGroup>>(box) >=
417  ::Verbosity::Debug)) {
418  Parallel::printf("%s " + Options::name<OptionsGroup>() +
419  "(%zu): Receive overlap solution\n",
420  element_id, iteration_id);
421  }
422 
423  // Add solutions on overlaps to this element's solution in a weighted sum
424  const auto received_overlap_solutions =
425  std::move(tuples::get<overlap_solution_inbox_tag>(inboxes)
426  .extract(iteration_id)
427  .mapped());
428  db::mutate<fields_tag>(
429  make_not_null(&box),
430  [&received_overlap_solutions](
431  const auto fields, const Index<Dim>& full_extents,
432  const std::array<size_t, Dim>& all_intruding_extents,
433  const std::unordered_map<Direction<Dim>, Scalar<DataVector>>&
434  all_intruding_overlap_weights) noexcept {
435  for (const auto& [overlap_id, overlap_solution] :
436  received_overlap_solutions) {
437  const auto& direction = overlap_id.first;
438  const auto& intruding_extents =
439  gsl::at(all_intruding_extents, direction.dimension());
440  const auto& overlap_weight =
441  all_intruding_overlap_weights.at(direction);
442  LinearSolver::Schwarz::add_overlap_data(
443  fields, overlap_solution * get(overlap_weight), full_extents,
444  intruding_extents, direction);
445  }
446  },
447  db::get<domain::Tags::Mesh<Dim>>(box).extents(),
448  db::get<Tags::IntrudingExtents<Dim, OptionsGroup>>(box),
449  db::get<domain::Tags::Interface<domain::Tags::InternalDirections<Dim>,
450  Tags::Weight<OptionsGroup>>>(box));
451  return {std::move(box)};
452  }
453 };
