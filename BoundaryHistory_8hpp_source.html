25 /// \ingroup TimeSteppersGroup
26 /// History data used by a TimeStepper for boundary integration.
27 /// \tparam LocalVars local variables passed to the boundary coupling
28 /// \tparam RemoteVars remote variables passed to the boundary coupling
29 /// \tparam CouplingResult result of the coupling function
30 template <typename LocalVars, typename RemoteVars, typename CouplingResult>
31 class BoundaryHistory {
32  template <typename Vars>
33  using IteratorType = boost::transform_iterator<
34  const Time& (*)(const std::tuple<Time, Vars>&),
35  typename std::deque<std::tuple<Time, Vars>>::const_iterator>;
36  public:
37  using local_iterator = IteratorType<LocalVars>;
38  using remote_iterator = IteratorType<RemoteVars>;
39 
40  // No copying because of the pointers in the cache. Moving is fine
41  // because we also move the container being pointed into and maps
42  // guarantee that this doesn't invalidate pointers.
43  BoundaryHistory() = default;
44  BoundaryHistory(const BoundaryHistory&) = delete;
45  BoundaryHistory(BoundaryHistory&&) = default;
46  BoundaryHistory& operator=(const BoundaryHistory&) = delete;
47  BoundaryHistory& operator=(BoundaryHistory&&) = default;
48  ~BoundaryHistory() = default;
49 
50  /// Add a new value to the end of the history of the indicated side.
51  //@{
52  void local_insert(const TimeStepId& time_id, LocalVars vars) noexcept {
53  local_data_.emplace_back(time_id.substep_time(), std::move(vars));
54  }
55  void remote_insert(const TimeStepId& time_id, RemoteVars vars) noexcept {
56  remote_data_.emplace_back(time_id.substep_time(), std::move(vars));
57  }
58  //@}
59 
60  /// Add a new value to the front of the history of the indicated
61  /// side. This is often convenient for setting initial data.
62  //@{
63  void local_insert_initial(const TimeStepId& time_id,
64  LocalVars vars) noexcept {
65  local_data_.emplace_front(time_id.substep_time(), std::move(vars));
66  }
67  void remote_insert_initial(const TimeStepId& time_id,
68  RemoteVars vars) noexcept {
69  remote_data_.emplace_front(time_id.substep_time(), std::move(vars));
70  }
71  //@}
72 
73  /// Mark all data before the passed point in history on the
74  /// indicated side as unneeded so it can be removed. Calling this
75  /// directly should not often be necessary, as it is handled
76  /// internally by the time steppers.
77  //@{
78  void local_mark_unneeded(const local_iterator& first_needed) noexcept {
79  mark_unneeded<0>(make_not_null(&local_data_), first_needed);
80  }
81  void remote_mark_unneeded(const remote_iterator& first_needed) noexcept {
82  mark_unneeded<1>(make_not_null(&remote_data_), first_needed);
83  }
84  //@}
85 
86  /// Access to the sequence of times on the indicated side.
87  //@{
88  local_iterator local_begin() const noexcept {
89  return local_iterator(local_data_.begin(), std::get<0>);
90  }
91  local_iterator local_end() const noexcept {
92  return local_iterator(local_data_.end(), std::get<0>);
93  }
94 
95  remote_iterator remote_begin() const noexcept {
96  return remote_iterator(remote_data_.begin(), std::get<0>);
97  }
98  remote_iterator remote_end() const noexcept {
99  return remote_iterator(remote_data_.end(), std::get<0>);
100  }
101 
102  size_t local_size() const noexcept { return local_data_.size(); }
103  size_t remote_size() const noexcept { return remote_data_.size(); }
104  //@}
105 
106  /// Look up the stored local data at the `time_id`. It is an error to request
107  /// data at a `time_id` that has not been inserted yet.
108  const LocalVars& local_data(const TimeStepId& time_id) const noexcept {
109  const Time& time = time_id.substep_time();
110  // Look up the data for this time, starting at the end of the `std::deque`,
111  // i.e. the most-recently inserted data.
112  for (auto it = local_data_.rbegin(); it != local_data_.rend(); it++) {
113  if (std::get<0>(*it) == time) {
114  return std::get<1>(*it);
115  }
116  }
117  ERROR("No local data was found at time " << time << ".");
118  }
119 
120  /// Evaluate the coupling function at the given local and remote
121  /// history entries. The coupling function will be passed the local
122  /// and remote FluxVars and should return a CouplingResult. Values are
123  /// cached internally, so callers should ensure that the coupling
124  /// functions provided on separate calls produce the same result.
125  template <typename Coupling>
126  const CouplingResult& coupling(Coupling&& c, const local_iterator& local,
127  const remote_iterator& remote) const noexcept;
128 
129  // clang-tidy: google-runtime-references
130  void pup(PUP::er& p) noexcept; // NOLINT
131 
132  private:
133  template <size_t Side, typename DataType, typename Iterator>
134  void mark_unneeded(gsl::not_null<DataType*> data,
135  const Iterator& first_needed) noexcept;
136 
137  std::deque<std::tuple<Time, LocalVars>> local_data_;
138  std::deque<std::tuple<Time, RemoteVars>> remote_data_;
139  // We use pointers instead of iterators because deque invalidates
140  // iterators when elements are inserted or removed at the ends, but
141  // not pointers.
142  mutable std::map<std::pair<const Time*, const Time*>, CouplingResult>
143  coupling_cache_;
144 };

146 template <typename LocalVars, typename RemoteVars, typename CouplingResult>
147 template <size_t Side, typename DataType, typename Iterator>
148 void BoundaryHistory<LocalVars, RemoteVars, CouplingResult>::mark_unneeded(
149  gsl::not_null<DataType*> data, const Iterator& first_needed) noexcept {
150  for (auto it = data->begin(); it != first_needed.base(); ++it) {
151  // Clean out cache entries referring to the entry we are removing.
152  for (auto cache_entry = coupling_cache_.begin();
153  cache_entry != coupling_cache_.end();) {
154  if (std::get<Side>(cache_entry->first) == &std::get<0>(*it)) {
155  cache_entry = coupling_cache_.erase(cache_entry);
156  } else {
157  ++cache_entry;
158  }
159  }
160  }
161  data->erase(data->begin(), first_needed.base());
162 }

164 template <typename LocalVars, typename RemoteVars, typename CouplingResult>
165 template <typename Coupling>
166 const CouplingResult&
167 BoundaryHistory<LocalVars, RemoteVars, CouplingResult>::coupling(
168  Coupling&& c, const local_iterator& local,
169  const remote_iterator& remote) const noexcept {
170  const auto insert_result = coupling_cache_.insert(
171  std::make_pair(std::make_pair(&*local, &*remote), CouplingResult{}));
172  CouplingResult& inserted_value = insert_result.first->second;
173  const bool is_new_value = insert_result.second;
174  if (is_new_value) {
175  inserted_value =
176  std::forward<Coupling>(c)(std::as_const(std::get<1>(*local.base())),
177  std::as_const(std::get<1>(*remote.base())));
178  }
179  return inserted_value;
180 }

182 template <typename LocalVars, typename RemoteVars, typename CouplingResult>
183 void BoundaryHistory<LocalVars, RemoteVars, CouplingResult>::pup(
184  PUP::er& p) noexcept {
185  p | local_data_;
186  p | remote_data_;
187 
188  const size_t cache_size = PUP_stl_container_size(p, coupling_cache_);
189  if (p.isUnpacking()) {
190  for (size_t entry_num = 0; entry_num < cache_size; ++entry_num) {
191  size_t local_index = 0;
192  size_t remote_index = 0;
193  CouplingResult cache_value;
194  p | local_index;
195  p | remote_index;
196  p | cache_value;
197  const auto cache_key =
198  std::make_pair(&std::get<0>(local_data_[local_index]),
199  &std::get<0>(remote_data_[remote_index]));
200  coupling_cache_.insert(std::make_pair(cache_key, cache_value));
201  }
202  } else {
203  for (auto& cache_entry : coupling_cache_) {
204  // clang-tidy: modernize-use-auto - Ensuring the correct type is
205  // important here to prevent undefined behavior in charm. I
206  // want to be explicit.
207  size_t local_index = static_cast<size_t>( // NOLINT
208  std::find_if(
209  local_data_.begin(), local_data_.end(),
210  [goal = cache_entry.first.first](const auto& entry) noexcept {
211  return &std::get<0>(entry) == goal;
212  }) -
213  local_data_.begin());
214  ASSERT(local_index < local_data_.size(),
215  "Failed to find local history entry for cache entry");
216 
217  // clang-tidy: modernize-use-auto - Ensuring the correct type is
218  // important here to prevent undefined behavior in charm. I
219  // want to be explicit.
220  size_t remote_index = static_cast<size_t>( // NOLINT
221  std::find_if(
222  remote_data_.begin(), remote_data_.end(),
223  [goal = cache_entry.first.second](const auto& entry) noexcept {
224  return &std::get<0>(entry) == goal;
225  }) -
226  remote_data_.begin());
227  ASSERT(remote_index < remote_data_.size(),
228  "Failed to find remote history entry for cache entry");
229 
230  p | local_index;
231  p | remote_index;
232  p | cache_entry.second;
233  }
234  }
235 }
