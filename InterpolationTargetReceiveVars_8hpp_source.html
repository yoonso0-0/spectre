39 /// \ingroup ActionsGroup
40 /// \brief Receives interpolated variables from an `Interpolator` on a subset
41 /// of the target points.
42 ///
43 /// If interpolated variables for all target points have been received, then
44 /// - Calls `InterpolationTargetTag::post_interpolation_callback`
45 /// - Tells `Interpolator`s that the interpolation is complete
46 /// (by calling
47 /// `Actions::CleanUpInterpolator<InterpolationTargetTag>`)
48 /// - Removes the first `temporal_id` from `Tags::TemporalIds<TemporalId>`
49 /// - If there are more `temporal_id`s, begins interpolation at the next
50 /// `temporal_id` (by calling `InterpolationTargetTag::compute_target_points`)
51 ///
52 /// Uses:
53 /// - DataBox:
54 /// - `Tags::TemporalIds<TemporalId>`
55 /// - `Tags::IndicesOfFilledInterpPoints<TemporalId>`
56 /// - `Tags::InterpolatedVars<InterpolationTargetTag,TemporalId>`
57 ///
58 /// DataBox changes:
59 /// - Adds: nothing
60 /// - Removes: nothing
61 /// - Modifies:
62 /// - `Tags::TemporalIds<TemporalId>`
63 /// - `Tags::CompletedTemporalIds<TemporalId>`
64 /// - `Tags::IndicesOfFilledInterpPoints<TemporalId>`
65 /// - `Tags::InterpolatedVars<InterpolationTargetTag,TemporalId>`
66 /// - `::Tags::Variables<typename
67 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
68 ///
69 /// For requirements on InterpolationTargetTag, see InterpolationTarget
70 template <typename InterpolationTargetTag>
71 struct InterpolationTargetReceiveVars {
72  /// For requirements on Metavariables, see InterpolationTarget
73  template <
74  typename ParallelComponent, typename DbTags, typename Metavariables,
75  typename ArrayIndex, typename TemporalId,
76  Requires<tmpl::list_contains_v<DbTags, Tags::TemporalIds<TemporalId>>> =
77  nullptr>
78  static void apply(
79  db::DataBox<DbTags>& box,
80  Parallel::GlobalCache<Metavariables>& cache,
81  const ArrayIndex& /*array_index*/,
82  const std::vector<Variables<
83  typename InterpolationTargetTag::vars_to_interpolate_to_target>>&
84  vars_src,
85  const std::vector<std::vector<size_t>>& global_offsets,
86  const TemporalId& temporal_id) noexcept {
87  InterpolationTarget_detail::add_received_variables<InterpolationTargetTag>(
88  make_not_null(&box), vars_src, global_offsets, temporal_id);
89  if (InterpolationTarget_detail::have_data_at_all_points<
90  InterpolationTargetTag>(box, temporal_id)) {
91  // All the valid points have been interpolated.
92  if (InterpolationTarget_detail::call_callback<InterpolationTargetTag>(
93  make_not_null(&box), make_not_null(&cache), temporal_id)) {
94  InterpolationTarget_detail::clean_up_interpolation_target<
95  InterpolationTargetTag>(make_not_null(&box), temporal_id);
96  auto& interpolator_proxy =
97  Parallel::get_parallel_component<Interpolator<Metavariables>>(
98  cache);
99  Parallel::simple_action<
100  Actions::CleanUpInterpolator<InterpolationTargetTag>>(
101  interpolator_proxy, temporal_id);
102 
103  // If we have a sequential target, and there are further
104  // temporal_ids, begin interpolation for the next one.
105  if (InterpolationTargetTag::compute_target_points::is_sequential::
106  value) {
107  const auto& temporal_ids =
108  db::get<Tags::TemporalIds<TemporalId>>(box);
109  if (not temporal_ids.empty()) {
110  auto& my_proxy = Parallel::get_parallel_component<
111  InterpolationTarget<Metavariables, InterpolationTargetTag>>(
112  cache);
113  Parallel::simple_action<
114  SendPointsToInterpolator<InterpolationTargetTag>>(
115  my_proxy, temporal_ids.front());
116  }
117  }
118  }
119  }
120  }
121 };
122 } // namespace Actions
123 } // namespace intrp
