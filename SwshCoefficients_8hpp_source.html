24 /// \ingroup SwshGroup
25 /// \brief Convenience function for determining the number of spin-weighted
26 /// spherical harmonics coefficients that are stored for a given `l_max`
27 ///
28 /// \details This includes the factor of 2 associated with needing
29 /// to store both the transform of the real and imaginary parts, so is the
30 /// full size of the result of a libsharp swsh transform.
31 ///
32 /// \note Assumes the triangular libsharp representation is used.
33 constexpr SPECTRE_ALWAYS_INLINE size_t
34 size_of_libsharp_coefficient_vector(const size_t l_max) noexcept {
35  return (l_max + 1) * (l_max + 2); // "triangular" representation
36 }

38 /*!
39  * \ingroup SwshGroup
40  * \brief Compute the relative sign change necessary to convert between the
41  * libsharp basis for spin weight `from_spin_weight` to the basis for spin
42  * weight `to_spin_weight`, for the real component coefficients if `real` is
43  * true, otherwise for the imaginary component coefficients. The sign change for
44  * a given coefficient is equivalent to the product of
45  * `sharp_swsh_sign(from_spin, m, real) * sharp_swsh_sign(to_spin, m,
46  * real)`. Due to the form of the signs, it does not end up depending on m (the
47  * m's in the power of \f$-1\f$'s cancel).
48  * For full details of the libsharp sign conventions, see the documentation for
49  * TransformJob.
50  *
51  * \details The sign change is obtained by the
52  * difference between the libsharp convention and the convention which uses:
53  *
54  * \f[
55  * {}_s Y_{\ell m} (\theta, \phi) = (-1)^m \sqrt{\frac{2 l + 1}{4 \pi}}
56  * D^{\ell}{}_{-m s}(\phi, \theta, 0).
57  * \f]
58  *
59  * See \cite Goldberg1966uu
60  */
61 constexpr SPECTRE_ALWAYS_INLINE double sharp_swsh_sign_change(
62  const int from_spin_weight, const int to_spin_weight,
63  const bool real) noexcept {
64  if (real) {
65  return (from_spin_weight == 0 ? -1.0 : 1.0) *
66  (from_spin_weight >= 0 ? -1.0 : 1.0) *
67  ((from_spin_weight < 0 and (from_spin_weight % 2 == 0)) ? -1.0
68  : 1.0) *
69  (to_spin_weight == 0 ? -1.0 : 1.0) *
70  (to_spin_weight >= 0 ? -1.0 : 1.0) *
71  ((to_spin_weight < 0 and (to_spin_weight % 2 == 0)) ? -1.0 : 1.0);
72  }
73  return (from_spin_weight == 0 ? -1.0 : 1.0) *
74  ((from_spin_weight < 0 and (from_spin_weight % 2 == 0)) ? 1.0 : -1.0) *
75  (to_spin_weight == 0 ? -1.0 : 1.0) *
76  ((to_spin_weight < 0 and (to_spin_weight % 2 == 0)) ? 1.0 : -1.0);
77 }

79 /*!
80  * \ingroup SwshGroup
81  * \brief Compute the sign change between the libsharp convention and the set
82  * of spin-weighted spherical harmonics given by the relation to the Wigner
83  * rotation matrices.
84  *
85  * \details The sign change is obtained via the difference between the libsharp
86  * convention and the convention which uses:
87  *
88  * \f[
89  * {}_s Y_{\ell m} (\theta, \phi) = (-1)^m \sqrt{\frac{2 l + 1}{4 \pi}}
90  * D^{\ell}{}_{-m s}(\phi, \theta, 0).
91  * \f]
92  *
93  * See \cite Goldberg1966uu.
94  * The sign change is computed for the real component coefficients if `real` is
95  * true, otherwise for the imaginary component coefficients. For full details on
96  * the sign convention used in libsharp, see the documentation for TransformJob.
97  * This function outputs the \f$\mathrm{sign}(m, s, \mathrm{real})\f$ necessary
98  * to produce the conversion between Goldberg moments and libsharp moments:
99  *
100  * \f{align*}{
101  * {}_s Y_{\ell m}^{\mathrm{real}, \mathrm{sharp}} =& \mathrm{sign}(m, s,
102  * \mathrm{real=true}) {}_s Y_{\ell m}^{\mathrm{Goldberg}}\\
103  * {}_s Y_{\ell m}^{\mathrm{imag}, \mathrm{sharp}} =&
104  * \mathrm{sign}(m, s, \mathrm{real=false}) {}_s Y_{\ell
105  * m}^{\mathrm{Goldberg}}.
106  * \f}
107  *
108  * Note that in this equation, the "real" and "imag" superscripts refer to the
109  * set of basis functions used for the decomposition of the real and imaginary
110  * part of the spin-weighted collocation points, not real or imaginary parts of
111  * the basis functions themselves.
112  */
113 constexpr SPECTRE_ALWAYS_INLINE double sharp_swsh_sign(
114  const int spin_weight, const int m, const bool real) noexcept {
115  if (real) {
116  if (m >= 0) {
117  return (spin_weight > 0 ? -1.0 : 1.0) *
118  ((spin_weight < 0 and (spin_weight % 2 == 0)) ? -1.0 : 1.0);
119  }
120  return (spin_weight == 0 ? -1.0 : 1.0) *
121  ((spin_weight >= 0 and (m % 2 == 0)) ? -1.0 : 1.0) *
122  (spin_weight < 0 and (m + spin_weight) % 2 == 0 ? -1.0 : 1.0);
123  }
124  if (m >= 0) {
125  return (spin_weight == 0 ? -1.0 : 1.0) *
126  ((spin_weight < 0 and (spin_weight % 2 == 0)) ? 1.0 : -1.0);
127  }
128  return (spin_weight == 0 ? -1.0 : 1.0) *
129  ((spin_weight >= 0 and (m % 2 == 0)) ? -1.0 : 1.0) *
130  ((spin_weight < 0 and (m + spin_weight) % 2 != 0) ? -1.0 : 1.0);
131 }

151 /*!
152  * \ingroup SwshGroup
153  * \brief A container for libsharp metadata for the spin-weighted spherical
154  * harmonics modal representation.
155  *
156  * \details
157  * The CoefficientsMetadata class acts as a memory-safe container for a
158  * `sharp_alm_info*`, required for use of libsharp transform utilities.
159  * The libsharp utilities are currently constructed to only provide user
160  * functions with collocation data for spin-weighted functions and
161  * derivatives. This class also provides an iterator for
162  * easily traversing a libsharp-compatible modal representation.
163  *
164  * \note The libsharp representation of coefficients is altered from the
165  * standard mathematical definitions in a nontrivial way. There are a number of
166  * important features to the data storage of the coefficients.
167  * - they are stored as a set of complex values, but each vector of complex
168  * values is the transform of only the real or imaginary part of the
169  * collocation data
170  * - because each vector of complex coefficients is related to the transform of
171  * a set of doubles, only (about) half of the m's are stored (m >= 0), because
172  * the remaining m modes are determinable by conjugation from the positive m
173  * modes, given that they represent the transform of a purely real or purely
174  * imaginary collocation quantity
175  * - they are stored in an l-varies-fastest, triangular representation. To be
176  * concrete, for an l_max=2, the order of coefficient storage is (l, m):
177  * [(0, 0), (1, 0), (2, 0), (1, 1), (2, 1), (2, 2)]
178  * - due to the restriction of representing only the transform of real
179  * quantities, the m=0 modes always have vanishing imaginary component.
180  */
181 class CoefficientsMetadata {
182  public:
183  /// An iterator for easily traversing a libsharp-compatible spin-weighted
184  /// spherical harmonic modal representation.
185  /// The `operator*()` returns a `LibsharpCoefficientInfo`, which contains two
186  /// offsets, `transform_of_real_part_offset` and
187  /// `transform_of_imag_part_offset`, and the `l_max`, `l` and `m` associated
188  /// with the values at those offsets.
189  ///
190  /// \note this currently assumes, as do many of the utilities in this file,
191  /// that the libsharp representation is chosen to be the triangular
192  /// coefficient representation. If alternative representations are desired,
193  /// alterations will be needed.
194  class CoefficientsIndexIterator {
195  public:
196  explicit CoefficientsIndexIterator(const size_t l_max,
197  const size_t start_l = 0,
198  const size_t start_m = 0) noexcept
199  : m_{start_m}, l_{start_l}, l_max_{l_max} {}
200 
201  LibsharpCoefficientInfo operator*() const noexcept {
202  // permit dereferencing the iterator only if the return represents a
203  // viable location in the coefficient vector
204  ASSERT(l_ <= l_max_ && m_ <= l_max_, "coefficients iterator overflow");
205  size_t offset = ((3 + 2 * l_max_ - m_) * m_) / 2 + l_ - m_;
206  return LibsharpCoefficientInfo{
207  offset,
208  offset +
209  Spectral::Swsh::size_of_libsharp_coefficient_vector(l_max_) / 2,
210  l_max_, l_, m_};
211  }
212 
213  /// advance the iterator by one position (prefix)
214  CoefficientsIndexIterator& operator++() noexcept {
215  // permit altering the iterator only if the new value is between the
216  // anticipated begin and end, inclusive
217  ASSERT(l_ <= l_max_ && m_ <= l_max_, "coefficients iterator overflow");
218  if (l_ == l_max_) {
219  ++m_;
220  l_ = m_;
221  } else {
222  ++l_;
223  }
224  return *this;
225  };
226 
227  /// advance the iterator by one position (postfix)
228  // NOLINTNEXTLINE(readability-const-return-type)
229  const CoefficientsIndexIterator operator++(int) noexcept {
230  auto pre_increment = *this;
231  ++*this;
232  return pre_increment;
233  }
234 
235  /// retreat the iterator by one position (prefix)
236  CoefficientsIndexIterator& operator--() noexcept {
237  // permit altering the iterator only if the new value is between the
238  // anticipated begin and end, inclusive
239  ASSERT(l_ <= l_max_ + 1 && m_ <= l_max_ + 1,
240  "coefficients iterator overflow");
241  if (l_ == m_) {
242  --m_;
243  l_ = l_max_;
244  } else {
245  --l_;
246  }
247  return *this;
248  }
249 
250  /// retreat the iterator by one position (postfix)
251  // NOLINTNEXTLINE(readability-const-return-type)
252  const CoefficientsIndexIterator operator--(int) noexcept {
253  auto pre_decrement = *this;
254  --*this;
255  return pre_decrement;
256  }
257 
258  // @{
259  /// (In)Equivalence checks the object as well as the l and m current
260  /// position.
261  bool operator==(const CoefficientsIndexIterator& rhs) const noexcept {
262  return m_ == rhs.m_ and l_ == rhs.l_ and l_max_ == rhs.l_max_;
263  }
264  bool operator!=(const CoefficientsIndexIterator& rhs) const noexcept {
265  return not(*this == rhs);
266  }
267  // @}
268 
269  private:
270  size_t m_;
271  size_t l_;
272  size_t l_max_;
273  };

279  CoefficientsMetadata(const CoefficientsMetadata&) = delete;
280  CoefficientsMetadata(CoefficientsMetadata&&) = default;

282  CoefficientsMetadata& operator=(CoefficientsMetadata&&) = default;
283  sharp_alm_info* get_sharp_alm_info() const noexcept {
284  return alm_info_.get();
285  }

289  /// returns the number of (complex) entries in a libsharp-compatible
290  /// coefficients vector. This includes the factor of 2 associated with needing
291  /// to store both the transform of the real and imaginary parts, so is the
292  /// full size of the result of a libsharp swsh transform.
293  size_t size() const noexcept {
294  return size_of_libsharp_coefficient_vector(l_max_);
295  }

299  CoefficientsMetadata::CoefficientsIndexIterator begin() const noexcept {
300  return CoefficientsIndexIterator(l_max_, 0, 0);
301  }
302  CoefficientsMetadata::CoefficientsIndexIterator cbegin() const noexcept {
303  return begin();
304  }
305  // @}
306

309  CoefficientsMetadata::CoefficientsIndexIterator end() const noexcept {
310  return CoefficientsIndexIterator(l_max_, l_max_ + 1, l_max_ + 1);
311  }
312  CoefficientsMetadata::CoefficientsIndexIterator cend() const noexcept {
313  return end();
314  }
315  // @}

317  private:
318  std::unique_ptr<sharp_alm_info, detail::DestroySharpAlm> alm_info_;
319  size_t l_max_ = 0;
320 };

492 /*!
493  * \ingroup SwshGroup
494  * \brief Returns the index into a vector of modes consistent with
495  * \cite Goldberg1966uu.
496  *
497  * \details `radial_offset` is used to index into three-dimensional data in
498  * which concentric spherical shells are stored as consecutive blocks of angular
499  * modes. Goldberg modes are stored in an m varies fastest, and ascending
500  * \f$m\f$ and \f$l\f$ values. So, the first several modes are (l, m):
501  *
502  * [(0, 0), (1, -1), (1, 0), (1, 1), (2, -2), ...]
503  */
504 constexpr SPECTRE_ALWAYS_INLINE size_t
505 goldberg_mode_index(const size_t l_max, const size_t l, const int m,
506  const size_t radial_offset = 0) noexcept {
507  return static_cast<size_t>(
508  static_cast<int>(square(l_max + 1) * radial_offset + square(l) + l) + m);
509 }
