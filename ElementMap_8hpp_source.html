21 /*!
22  * \ingroup ComputationalDomainGroup
23  * \brief The CoordinateMap for the Element from the Logical frame to the
24  * `TargetFrame`
25  *
26  * An ElementMap takes a CoordinateMap for a Block and an ElementId as input,
27  * and then "prepends" the correct affine map to the CoordinateMap so that the
28  * map corresponds to the coordinate map for the Element rather than the Block.
29  * This allows DomainCreators to only specify the maps for the Blocks without
30  * worrying about how the domain may be decomposed beyond that.
31  */
32 template <size_t Dim, typename TargetFrame>
33 class ElementMap {
34  public:
35  static constexpr size_t dim = Dim;
36  using source_frame = Frame::Logical;
37  using target_frame = TargetFrame;
38 
39  /// \cond HIDDEN_SYMBOLS
40  ElementMap() = default;
41  /// \endcond
42 
43  ElementMap(ElementId<Dim> element_id,
44  std::unique_ptr<
45  domain::CoordinateMapBase<Frame::Logical, TargetFrame, Dim>>
46  block_map) noexcept;
47 
48  const domain::CoordinateMapBase<Frame::Logical, TargetFrame, Dim>& block_map()
49  const noexcept {
50  return *block_map_;
51  }
52 
53  const ElementId<Dim>& element_id() const noexcept { return element_id_; }
54 
55  template <typename T>
56  tnsr::I<T, Dim, TargetFrame> operator()(
57  tnsr::I<T, Dim, Frame::Logical> source_point) const noexcept {
58  apply_affine_transformation_to_point(source_point);
59  return block_map_->operator()(std::move(source_point));
60  }
61 
62  template <typename T>
63  tnsr::I<T, Dim, Frame::Logical> inverse(
64  tnsr::I<T, Dim, TargetFrame> target_point) const noexcept {
65  auto source_point{
66  block_map_->inverse(std::move(target_point)).value()};
67  // Apply the affine map to the points
68  for (size_t d = 0; d < Dim; ++d) {
69  source_point.get(d) =
70  source_point.get(d) * gsl::at(map_inverse_slope_, d) +
71  gsl::at(map_inverse_offset_, d);
72  }
73  return source_point;
74  }
75 
76  template <typename T>
77  InverseJacobian<T, Dim, Frame::Logical, TargetFrame> inv_jacobian(
78  tnsr::I<T, Dim, Frame::Logical> source_point) const noexcept {
79  apply_affine_transformation_to_point(source_point);
80  auto inv_jac = block_map_->inv_jacobian(std::move(source_point));
81  for (size_t d = 0; d < Dim; ++d) {
82  for (size_t i = 0; i < Dim; ++i) {
83  inv_jac.get(d, i) *= gsl::at(inverse_jacobian_, d);
84  }
85  }
86  return inv_jac;
87  }
88 
89  template <typename T>
90  Jacobian<T, Dim, Frame::Logical, TargetFrame> jacobian(
91  tnsr::I<T, Dim, Frame::Logical> source_point) const noexcept {
92  apply_affine_transformation_to_point(source_point);
93  auto jac = block_map_->jacobian(std::move(source_point));
94  for (size_t d = 0; d < Dim; ++d) {
95  for (size_t i = 0; i < Dim; ++i) {
96  jac.get(i, d) *= gsl::at(jacobian_, d);
97  }
98  }
99  return jac;
100  }
101 
102  // clang-tidy: do not use references
103  void pup(PUP::er& p) noexcept; // NOLINT
104 
105  private:
106  template <typename T>
107  void apply_affine_transformation_to_point(
108  tnsr::I<T, Dim, Frame::Logical>& source_point) const noexcept {
109  for (size_t d = 0; d < Dim; ++d) {
110  source_point.get(d) = source_point.get(d) * gsl::at(map_slope_, d) +
111  gsl::at(map_offset_, d);
112  }
113  }
114 
115  std::unique_ptr<domain::CoordinateMapBase<Frame::Logical, TargetFrame, Dim>>
116  block_map_{nullptr};
117  ElementId<Dim> element_id_{};
118  // map_slope_[i] = 0.5 * (segment_ids[i].endpoint(Side::Upper) -
119  // segment_ids[i].endpoint(Side::Lower))
120  std::array<double, Dim> map_slope_{
121  make_array<Dim>(std::numeric_limits<double>::signaling_NaN())};
122  // map_offset_[i] = 0.5 * (segment_ids[i].endpoint(Side::Upper) +
123  // segment_ids[i].endpoint(Side::Lower))
124  std::array<double, Dim> map_offset_{
125  make_array<Dim>(std::numeric_limits<double>::signaling_NaN())};
126  // map_inverse_slope_[i] = 1.0 / map_slope_[i]
127  std::array<double, Dim> map_inverse_slope_{
128  make_array<Dim>(std::numeric_limits<double>::signaling_NaN())};
129  // map_inverse_offset_[i] = -map_offset_[i] / map_slope_[i]
130  std::array<double, Dim> map_inverse_offset_{
131  make_array<Dim>(std::numeric_limits<double>::signaling_NaN())};
132  // Note: The Jacobian is diagonal
133  std::array<double, Dim> jacobian_{map_slope_};
134  std::array<double, Dim> inverse_jacobian_{map_inverse_slope_};
135 };
