10 /// \ingroup UtilitiesGroup
11 /// \brief Given a map of `std::unique_ptr` returns a copy of the map by
12 /// invoking `get_clone()` on each element of the input map.
13 template <typename KeyType, typename T>
14 std::unordered_map<KeyType, std::unique_ptr<T>> clone_unique_ptrs(
15  const std::unordered_map<KeyType, std::unique_ptr<T>>& map) noexcept {
16  std::unordered_map<KeyType, std::unique_ptr<T>> result{};
17  for (const auto& kv : map) {
18  result[kv.first] = kv.second->get_clone();
19  }
20  return result;
21 }
