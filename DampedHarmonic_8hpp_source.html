27 /*!
28  * \brief Damped harmonic gauge source function and its spacetime derivative.
29  *
30  * \details The gauge condition has been taken from \cite Szilagyi2009qz and
31  * \cite Deppe2018uye. We provide both a "rollon" version
32  * (`damped_harmonic_rollon`), and a "non-rollon" version (`damped_harmonic`).
33  * In the non-rollon version the rollon function \f$R(t)=1\f$.
34  *
35  * \warning Only the non-rollon version can be used with a moving mesh.
36  *
37  * The covariant form of the source function \f$H_a\f$ is written as:
38  *
39  * \f{align*}
40  * H_a := [1 - R(t)] H_a^\mathrm{init} +
41  * [\mu_{L1} \mathrm{log}(\sqrt{g}/N) + \mu_{L2} \mathrm{log}(1/N)] t_a
42  * - \mu_S g_{ai} N^i / N
43  * \f}
44  *
45  * where \f$N, N^k\f$ are the lapse and shift respectively, \f$t_a\f$ is the
46  * unit normal one-form to the spatial slice, and \f$g_{ab}\f$ is
47  * the spatial metric (obtained by projecting the spacetime metric onto the
48  * 3-slice, i.e. \f$g_{ab} = \psi_{ab} + t_a t_b\f$). The prefactors are:
49  *
50  * \f{align*}
51  * \mu_{L1} &= A_{L1} R(t) W(x^i) \mathrm{log}(\sqrt{g}/N)^{e_{L1}}, \\
52  * \mu_{L2} &= A_{L2} R(t) W(x^i) \mathrm{log}(1/N)^{e_{L2}}, \\
53  * \mu_{S} &= A_{S} R(t) W(x^i) \mathrm{log}(\sqrt{g}/N)^{e_{S}},
54  * \f}
55  *
56  * temporal roll-on function \f$ R(t)\f$ is:
57  *
58  * \f{align*}
59  * \begin{array}{ll}
60  * R(t) & = 0, & t< t_{0} \\
61  * & = 1 - \exp[-((t - t_{0})/ \sigma_t)^4], & t\geq t_{0} \\
62  * \end{array}
63  * \f}
64  *
65  * and the spatial weight function is:
66  *
67  * \f{align*}
68  * W(x^i) = \exp[-(r/\sigma_r)^2].
69  * \f}
70  *
71  * This weight function can be written with multiple constant factors in the
72  * exponent in literature \cite Deppe2018uye, but we absorb them all into
73  * \f$ \sigma_r\f$ here. The coordinate \f$ r\f$ is the Euclidean radius
74  * in Inertial coordinates.
75  *
76  * Note that for the last three terms in \f$H_a\f$ (with \f$ X = \{L1, L2, S\}
77  * \f$):
78  * - Amplitude factors \f$ A_{X} \f$ are taken as input here as `amp_coef_X`
79  * - Exponents \f$ e_X\f$ are taken as input here as `exp_X`.
80  * - Spatial weight function \f$W\f$ is specified completely by
81  * \f$\sigma_r\f$, which is taken as input here as `sigma_r`.
82  *
83  * Also computes spacetime derivatives, i.e. \f$\partial_a H_b\f$, of the damped
84  * harmonic source function H. Using notation from damped_harmonic_h(), we
85  * rewrite the same as:
86  *
87  * \f{align*}
88  * \partial_a H_b =& \partial_a T_1 + \partial_a T_2 + \partial_a T_3, \\
89  * H_a =& T_1 + T_2 + T_3,
90  * \f}
91  *
92  * where:
93  *
94  * \f{align*}
95  * T_1 =& [1 - R(t)] H_a^\mathrm{init}, \\
96  * T_2 =& [\mu_{L1} \mathrm{log}(\sqrt{g}/N) + \mu_{L2} \mathrm{log}(1/N)] t_a,
97  * \\
98  * T_3 =& - \mu_S g_{ai} N^i / N.
99  * \f}
100  *
101  * Derivation:
102  *
103  * \f$\blacksquare\f$ For \f$ T_1 \f$, the derivatives are:
104  * \f{align*}
105  * \partial_a T_1 = (1 - R(t))
106  * \partial_a H_b^\mathrm{init}
107  * - H_b^\mathrm{init} \partial_a R.
108  * \f}
109  *
110  * \f$\blacksquare\f$ Write \f$ T_2 \equiv (\mu_1 + \mu_2) t_b \f$. Then:
111  *
112  * \f{align*}
113  * \partial_a T_2 =& (\partial_a \mu_1 + \partial_a \mu_2) t_b \\
114  * +& (\mu_1 + \mu_2) \partial_a t_b,
115  * \f}
116  *
117  * where
118  *
119  * \f{align*}
120  * \partial_a t_b =& \left(-\partial_a N, 0, 0, 0\right) \\
121  *
122  * \partial_a \mu_1
123  * =& \partial_a [A_{L1} R(t) W(x^i) \mathrm{log}(\sqrt{g}/N)^{e_{L1} +
124  * 1}], \\
125  * =& A_{L1} R(t) W(x^i) \partial_a [\mathrm{log}(\sqrt{g}/N)^{e_{L1} +
126  * 1}] \\
127  * +& A_{L1} \mathrm{log}(\sqrt{g}/N)^{e_{L1} + 1} \partial_a [R(t)
128  * W(x^i)],\\
129  *
130  * \partial_a \mu_2
131  * =& \partial_a [A_{L2} R(t) W(x^i) \mathrm{log}(1/N)^{e_{L2} + 1}], \\
132  * =& A_{L2} R(t) W(x^i) \partial_a [\mathrm{log}(1/N)^{e_{L2} + 1}] \\
133  * +& A_{L2} \mathrm{log}(1/N)^{e_{L2} + 1} \partial_a [R(t) W(x^i)],
134  * \f}
135  *
136  * where \f$\partial_a [R W] = \left(\partial_0 R(t), \partial_i
137  * W(x^j)\right)\f$.
138  *
139  * \f$\blacksquare\f$ Finally, the derivatives of \f$ T_3 \f$ are:
140  *
141  * \f[
142  * \partial_a T_3 = -\partial_a(\mu_S/N) g_{bi} N^i
143  * -(\mu_S/N) \partial_a(g_{bi}) N^i
144  * -(\mu_S/N) g_{bi}\partial_a N^i,
145  * \f]
146  *
147  * where
148  *
149  * \f{align*}
150  * \partial_a(\mu_S / N) =& (1/N)\partial_a \mu_S
151  * - \frac{\mu_S}{N^2}\partial_a N, \,\,\mathrm{and}\\
152  * \partial_a \mu_S =& \partial_a [A_S R(t) W(x^i)
153  * \mathrm{log}(\sqrt{g}/N)^{e_S}], \\
154  * =& A_S R(t) W(x^i) \partial_a
155  * [\mathrm{log}(\sqrt{g}/N)^{e_S}] \\
156  * +& A_S \mathrm{log}(\sqrt{g} / N)^{e_S} \partial_a [R(t)
157  * W(x^i)].
158  * \f}
159  */
