76 template <typename System, typename InitialGuess, typename BoundaryConditions>
77 struct Metavariables {
78  using system = System;
79  static constexpr size_t volume_dim = system::volume_dim;
80  using initial_guess = InitialGuess;
81  using boundary_conditions = BoundaryConditions;
82 
83  static constexpr Options::String help{
84  "Find the solution to a linear elasticity problem."};
85 
86  using fluxes_computer_tag =
87  elliptic::Tags::FluxesComputer<typename system::fluxes>;
88 
89  // Only Dirichlet boundary conditions are currently supported, and they are
90  // are all imposed by analytic solutions right now.
91  // We will add support for Neumann boundary conditions ASAP.
92  using analytic_solution = boundary_conditions;
93  using analytic_solution_tag = Tags::AnalyticSolution<analytic_solution>;
94 
95  // We retrieve the constitutive relation from the analytic solution
96  using constitutive_relation_type =
97  typename analytic_solution::constitutive_relation_type;
98  using constitutive_relation_provider_option_tag =
99  OptionTags::AnalyticSolution<analytic_solution>;
100 
101  // The linear solver algorithm. We must use GMRES since the operator is
102  // not positive-definite for the first-order system.
103  using linear_solver =
104  LinearSolver::gmres::Gmres<Metavariables, typename system::fields_tag,
105  SolveElasticityProblem::OptionTags::GmresGroup,
106  false>;
107  using linear_solver_iteration_id =
108  Convergence::Tags::IterationId<typename linear_solver::options_group>;
109  // For the GMRES linear solver we need to apply the DG operator to its
110  // internal "operand" in every iteration of the algorithm.
111  using linear_operand_tag = db::add_tag_prefix<LinearSolver::Tags::Operand,
112  typename system::fields_tag>;
113  using primal_variables = db::wrap_tags_in<LinearSolver::Tags::Operand,
114  typename system::primal_fields>;
115  using auxiliary_variables =
116  db::wrap_tags_in<LinearSolver::Tags::Operand,
117  typename system::auxiliary_fields>;
118 
119  // Parse numerical flux parameters from the input file to store in the cache.
120  using normal_dot_numerical_flux = Tags::NumericalFlux<
121  elliptic::dg::NumericalFluxes::FirstOrderInternalPenalty<
122  volume_dim, fluxes_computer_tag, primal_variables,
123  auxiliary_variables>>;
124  // Specify the DG boundary scheme. We use the strong first-order scheme here
125  // that only requires us to compute normals dotted into the first-order
126  // fluxes.
127  using boundary_scheme = dg::FirstOrderScheme::FirstOrderScheme<
128  volume_dim, linear_operand_tag,
129  db::add_tag_prefix<LinearSolver::Tags::OperatorAppliedTo,
130  linear_operand_tag>,
131  normal_dot_numerical_flux, linear_solver_iteration_id>;
132 
133  // Collect events and triggers
134  // (public for use by the Charm++ registration code)
135  using observe_fields = typename system::fields_tag::tags_list;
136  using analytic_solution_fields = observe_fields;
137  using events = tmpl::list<
138  dg::Events::Registrars::ObserveFields<
139  volume_dim, linear_solver_iteration_id, observe_fields,
140  analytic_solution_fields>,
141  dg::Events::Registrars::ObserveErrorNorms<linear_solver_iteration_id,
142  analytic_solution_fields>,
143  dg::Events::Registrars::ObserveVolumeIntegrals<
144  volume_dim, linear_solver_iteration_id,
145  tmpl::list<Elasticity::Tags::PotentialEnergyDensity<volume_dim>>>>;
146  using triggers = tmpl::list<elliptic::Triggers::Registrars::EveryNIterations<
147  linear_solver_iteration_id>>;
148 
149  // Collect all items to store in the cache.
150  using const_global_cache_tags = tmpl::list<
151  analytic_solution_tag, fluxes_computer_tag, normal_dot_numerical_flux,
152  Elasticity::Tags::ConstitutiveRelation<constitutive_relation_type>,
153  Tags::EventsAndTriggers<events, triggers>>;
154 
155  // Collect all reduction tags for observers
156  using observed_reduction_data_tags =
157  observers::collect_reduction_data_tags<tmpl::flatten<tmpl::list<
158  typename Event<events>::creatable_classes, linear_solver>>>;
159 
160  // Specify all global synchronization points.
161  enum class Phase { Initialization, RegisterWithObserver, Solve, Exit };
162 
163  using initialization_actions = tmpl::list<
164  Actions::SetupDataBox, dg::Actions::InitializeDomain<volume_dim>,
165  dg::Actions::InitializeInterfaces<
166  system, dg::Initialization::slice_tags_to_face<>,
167  dg::Initialization::slice_tags_to_exterior<>,
168  dg::Initialization::face_compute_tags<
169  domain::Tags::BoundaryCoordinates<volume_dim>>,
170  dg::Initialization::exterior_compute_tags<>, false, false>,
171  typename linear_solver::initialize_element,
172  elliptic::Actions::InitializeSystem<system>,
173  Initialization::Actions::AddComputeTags<tmpl::list<
174  Elasticity::Tags::PotentialEnergyDensityCompute<volume_dim>>>,
175  elliptic::Actions::InitializeAnalyticSolution<analytic_solution_tag,
176  analytic_solution_fields>,
177  elliptic::dg::Actions::ImposeInhomogeneousBoundaryConditionsOnSource<
178  Metavariables>,
179  dg::Actions::InitializeMortars<boundary_scheme>,
180  elliptic::dg::Actions::InitializeFirstOrderOperator<
181  volume_dim, typename system::fluxes, typename system::sources,
182  linear_operand_tag, primal_variables, auxiliary_variables>,
183  Initialization::Actions::RemoveOptionsAndTerminatePhase>;
184 
185  using build_linear_operator_actions = tmpl::list<
186  dg::Actions::CollectDataForFluxes<
187  boundary_scheme, domain::Tags::InternalDirections<volume_dim>>,
188  dg::Actions::SendDataForFluxes<boundary_scheme>,
189  Actions::MutateApply<elliptic::FirstOrderOperator<
190  volume_dim, LinearSolver::Tags::OperatorAppliedTo,
191  linear_operand_tag>>,
192  elliptic::dg::Actions::ImposeHomogeneousDirichletBoundaryConditions<
193  linear_operand_tag, primal_variables>,
194  dg::Actions::CollectDataForFluxes<
195  boundary_scheme,
196  domain::Tags::BoundaryDirectionsInterior<volume_dim>>,
197  dg::Actions::ReceiveDataForFluxes<boundary_scheme>,
198  Actions::MutateApply<boundary_scheme>>;
199 
200  using register_actions =
201  tmpl::list<observers::Actions::RegisterEventsWithObservers,
202  Parallel::Actions::TerminatePhase>;
203 
204  using solve_actions = tmpl::list<
205  typename linear_solver::template solve<tmpl::list<
206  Actions::RunEventsAndTriggers, build_linear_operator_actions>>,
207  Actions::RunEventsAndTriggers, Parallel::Actions::TerminatePhase>;
208 
209  using dg_element_array = elliptic::DgElementArray<
210  Metavariables,
211  tmpl::list<Parallel::PhaseActions<Phase, Phase::Initialization,
212  initialization_actions>,
213  Parallel::PhaseActions<Phase, Phase::RegisterWithObserver,
214  register_actions>,
215  Parallel::PhaseActions<Phase, Phase::Solve, solve_actions>>>;
216 
217  // Specify all parallel components that will execute actions at some point.
218  using component_list = tmpl::flatten<
219  tmpl::list<dg_element_array, typename linear_solver::component_list,
220  observers::Observer<Metavariables>,
221  observers::ObserverWriter<Metavariables>>>;
222 
223  // Specify the transitions between phases.
224  static Phase determine_next_phase(
225  const Phase& current_phase,
226  const Parallel::CProxy_GlobalCache<
227  Metavariables>& /*cache_proxy*/) noexcept {
228  switch (current_phase) {
229  case Phase::Initialization:
230  return Phase::RegisterWithObserver;
231  case Phase::RegisterWithObserver:
232  return Phase::Solve;
233  case Phase::Solve:
234  return Phase::Exit;
235  case Phase::Exit:
236  ERROR(
237  "Should never call determine_next_phase with the current phase "
238  "being 'Exit'");
239  default:
240  ERROR(
241  "Unknown type of phase. Did you static_cast<Phase> an integral "
242  "value?");
243  }
244  }
245 };
