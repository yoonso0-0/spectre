26 /// \ingroup DataStructuresGroup
27 /// An integer multi-index.
28 ///
29 /// \tparam Dim the number of integers in the Index.
30 template <size_t Dim>
31 class Index {
32  public:
33  /// Construct with each element set to the same value.
34  explicit Index(const size_t i0 = std::numeric_limits<size_t>::max()) noexcept
35  : indices_(make_array<Dim>(i0)) {}
36 
37  /// Construct specifying value in each dimension
38  template <typename... I, Requires<(sizeof...(I) > 1)> = nullptr>
39  explicit Index(I... i) noexcept
40  : indices_(make_array(static_cast<size_t>(i)...)) {
41  static_assert(std::conjunction_v<tt::is_integer<I>...>,
42  "You must pass in a set of size_t's to Index.");
43  static_assert(Dim == sizeof...(I),
44  "The number of indices given to Index must be the same as "
45  "the dimensionality of the Index.");
46  }
47 
48  explicit Index(std::array<size_t, Dim> i) noexcept : indices_(std::move(i)) {}
49 
50  size_t operator[](const size_t d) const noexcept {
51  return gsl::at(indices_, d);
52  }
53  size_t& operator[](const size_t d) noexcept { return gsl::at(indices_, d); }
54 
55  typename std::array<size_t, Dim>::iterator begin() {
56  return indices_.begin();
57  }
58  typename std::array<size_t, Dim>::const_iterator begin() const {
59  return indices_.begin();
60  }
61 
62  typename std::array<size_t, Dim>::iterator end() { return indices_.end(); }
63  typename std::array<size_t, Dim>::const_iterator end() const {
64  return indices_.end();
65  }
66 
67  size_t size() const noexcept { return Dim; }
68 
69  /// The product of the indices.
70  /// If Dim = 0, the product is defined as 1.
71  template <int N = Dim, Requires<(N > 0)> = nullptr>
72  constexpr size_t product() const noexcept {
73  return indices_[N - 1] * product<N - 1>();
74  }
75  /// \cond
76  // Specialization for N = 0 to stop recursion
77  template <int N = Dim, Requires<(N == 0)> = nullptr>
78  constexpr size_t product() const noexcept {
79  return 1;
80  }
81  /// \endcond

83  /// Return a smaller Index with the d-th element removed.
84  ///
85  /// \param d the element to remove.
86  template <size_t N = Dim, Requires<(N > 0)> = nullptr>
87  Index<Dim - 1> slice_away(const size_t d) const noexcept {
88  ASSERT(d < Dim,
89  "Can't slice dimension " << d << " from an Index<" << Dim << ">");
90  std::array<size_t, Dim - 1> t{};
91  for (size_t i = 0; i < Dim; ++i) {
92  if (i < d) {
93  gsl::at(t, i) = gsl::at(indices_, i);
94  } else if (i > d) {
95  gsl::at(t, i - 1) = gsl::at(indices_, i);
96  }
97  }
98  return Index<Dim - 1>(t);
99  }
