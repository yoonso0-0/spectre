82 /// \brief Create a Domain consisting of multiple aligned Blocks arrayed in a
83 /// lattice.
84 ///
85 /// This is useful for setting up problems with piecewise smooth initial data,
86 /// problems that specify different boundary conditions on distinct parts of
87 /// the boundary, or problems that need different length scales initially.
88 ///
89 /// \note Adaptive mesh refinement can never join Block%s, so use the fewest
90 /// number of Block%s that your problem needs. More initial Element%s can be
91 /// created by specifying a larger `InitialRefinement`.
92 template <size_t VolumeDim>
93 class AlignedLattice : public DomainCreator<VolumeDim> {
94  public:
95  using maps_list = tmpl::list<
96  domain::CoordinateMap<Frame::Logical, Frame::Inertial,
97  CoordinateMaps::Affine>,
98  domain::CoordinateMap<
99  Frame::Logical, Frame::Inertial,
100  CoordinateMaps::ProductOf2Maps<CoordinateMaps::Affine,
101  CoordinateMaps::Affine>>,
102  domain::CoordinateMap<Frame::Logical, Frame::Inertial,
103  CoordinateMaps::ProductOf3Maps<
104  CoordinateMaps::Affine, CoordinateMaps::Affine,
105  CoordinateMaps::Affine>>>;
