29 /*!
30  * \ingroup ActionsGroup
31  * \brief Compute the volume fluxes of the evolved variables
32  *
33  * The mesh velocity is added to the flux automatically if the mesh is moving.
34  * That is,
35  *
36  * \f{align}{
37  * F^i_{\alpha}\to F^i_{\alpha}-v^i_{g} u_{\alpha}
38  * \f}
39  *
40  * where \f$F^i_{\alpha}\f$ are the fluxes when the mesh isn't moving,
41  * \f$v^i_g\f$ is the velocity of the mesh, and \f$u_{\alpha}\f$ are the evolved
42  * variables.
43  *
44  * Uses:
45  * - DataBox:
46  * - Items in system::volume_fluxes::argument_tags
47  * - `domain::Tags::MeshVelocity<Metavariables::volume_dim>`
48  * - `Metavariables::system::variables_tag`
49  *
50  * DataBox changes:
51  * - Adds: nothing
52  * - Removes: nothing
53  * - Modifies:
54  * db::add_tag_prefix<Tags::Flux, variables_tag,
55  * tmpl::size_t<system::volume_dim>, Frame::Inertial>
56  */
57 struct ComputeVolumeFluxes {
58  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
59  typename ArrayIndex, typename ActionList,
60  typename ParallelComponent,
61  Requires<tmpl::size<DbTagsList>::value != 0> = nullptr>
62  static std::tuple<db::DataBox<DbTagsList>&&> apply(
63  db::DataBox<DbTagsList>& box,
64  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
65  const Parallel::GlobalCache<Metavariables>& /*cache*/,
66  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
67  const ParallelComponent* const /*meta*/) noexcept { // NOLINT const
68  db::mutate_apply<typename Metavariables::system::volume_fluxes>(
69  make_not_null(&box));
70  const auto use_moving_mesh = static_cast<bool>(
71  db::get<domain::Tags::MeshVelocity<Metavariables::volume_dim>>(box));
72  if (use_moving_mesh) {
73  using variables_tag = typename Metavariables::system::variables_tag;
74  using fluxes_tag =
75  db::add_tag_prefix<::Tags::Flux, variables_tag,
76  tmpl::size_t<Metavariables::volume_dim>,
77  Frame::Inertial>;
78  db::mutate<fluxes_tag>(
79  make_not_null(&box),
80  [](const auto fluxes_ptr, const auto& evolved_vars,
81  const tnsr::I<DataVector, Metavariables::volume_dim,
82  Frame::Inertial>& grid_velocity) noexcept {
83  tmpl::for_each<typename fluxes_tag::tags_list>(
84  [&evolved_vars, &fluxes_ptr, &
85  grid_velocity ](auto tag_v) noexcept {
86  using ::get;
87  using flux_tag = typename decltype(tag_v)::type;
88  using var_tag = db::remove_tag_prefix<flux_tag>;
89  auto& flux_tensor = get<flux_tag>(*fluxes_ptr);
90  for (size_t storage_index = 0;
91  storage_index < flux_tensor.size(); ++storage_index) {
92  const auto flux_tensor_index =
93  flux_tensor.get_tensor_index(storage_index);
94  const auto tensor_index =
95  all_but_specified_element_of(flux_tensor_index, 0);
96  const size_t flux_index = gsl::at(flux_tensor_index, 0);
97 
98  flux_tensor[storage_index] -=
99  get<var_tag>(evolved_vars).get(tensor_index) *
100  grid_velocity.get(flux_index);
101  }
102  });
103  },
104  db::get<variables_tag>(box),
105  *db::get<domain::Tags::MeshVelocity<Metavariables::volume_dim>>(box));
106  }
107  return std::forward_as_tuple(std::move(box));
108  }
109 };
110 } // namespace Actions
