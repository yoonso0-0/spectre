33 template <size_t VolumeDim, typename Fr, class... MemberArgs, class T>
34 void check_impl(
35  const std::unique_ptr<
36  ::GeneralizedHarmonic::ConstraintDamping::DampingFunction<
37  VolumeDim, Fr>>& in_gh_damping_function,
38  const std::string& python_function_prefix, const T& used_for_size,
39  const std::array<std::pair<double, double>, 1> random_value_bounds,
40  const MemberArgs&... member_args) noexcept {
41  using GhDampingFunc =
42  ::GeneralizedHarmonic::ConstraintDamping::DampingFunction<VolumeDim, Fr>;
43 
44  const auto member_args_tuple = std::make_tuple(member_args...);
45  const auto helper =
46  [&python_function_prefix, &random_value_bounds, &member_args_tuple,
47  &used_for_size](
48  const std::unique_ptr<GhDampingFunc>& gh_damping_function) noexcept {
49  INFO("Testing call operator...")
50 
51  // Make a lambda that calls the damping function's call operator
52  // with a hard-coded FunctionsOfTime, since check_with_random_values
53  // cannot convert a FunctionsOfTime into a python type.
54  // The FunctionsOfTime contains a single FunctionOfTime
55  // \f$f(t) = a_0 + a_1 t + a_2 t^2 + a_3 t^3\f$, where
56  // \f$a_0 = 1.0\f$, \f$a_1 = 0.2\f$, \f$a_3 = 0.03,\f$ and
57  // \f$a_4 = 0.004\f$. The corresponding python function should use
58  // the same hard-coded coefficients to evaluate \f$f(t)\f$.
59  // Note that the FunctionOfTime never expires.
60  const auto damping_function_call_operator_helper =
61  [&gh_damping_function](const tnsr::I<T, VolumeDim, Fr>& coordinates,
62  const double time) {
63  std::unordered_map<
64  std::string,
65  std::unique_ptr<::domain::FunctionsOfTime::FunctionOfTime>>
66  functions_of_time{};
67  functions_of_time["ExpansionFactor"] = std::make_unique<
68  ::domain::FunctionsOfTime::PiecewisePolynomial<3>>(
69  0.0,
70  std::array<DataVector, 4>{{{1.0}, {0.2}, {0.03}, {0.004}}},
71  std::numeric_limits<double>::max());
72  // Default-construct the scalar, to test that the damping
73  // function's call operator correctly resizes it
74  // (in the case T is a DataVector) with
75  // destructive_resize_components()
76  Scalar<T> value_at_coordinates{};
77  gh_damping_function->operator()(
78  make_not_null(&value_at_coordinates), coordinates, time,
79  functions_of_time);
80  return value_at_coordinates;
81  };
82 
83  pypp::check_with_random_values<1>(
84  &decltype(damping_function_call_operator_helper)::operator(),
85  damping_function_call_operator_helper, "TestFunctions",
86  python_function_prefix + "_call_operator", random_value_bounds,
87  member_args_tuple, used_for_size);
88  INFO("Done testing call operator...")
89  INFO("Done\n\n")
90  };
91 
92  helper(in_gh_damping_function);
93  helper(serialize_and_deserialize(in_gh_damping_function));
94 }
95 } // namespace detail

97 /*!
98  * \ingroup TestingFrameworkGroup
99  * \brief Test a DampingFunction by comparing to python functions
100  *
101  * The python functions must be added to TestFunctions.py in
102  * tests/Unit/Evolution/Systems/GeneralizedHarmonic/ConstraintDamping/Python.
103  * Each python function for a corresponding DampingFunction should begin
104  * with a prefix `python_function_prefix`. The prefix for each class of
105  * DampingFunction is arbitrary, but should generally be descriptive (e.g.
106  * 'gaussian_plus_constant') of the DampingFunction.
107  *
108  * If a DampingFunction class has member variables set by its constructor, then
109  * these member variables must be passed in as the last arguments to the `check`
110  * function`. Each python function must take these same arguments as the
111  * trailing arguments.
112  */
113 template <class DampingFunctionType, class T, class... MemberArgs>
114 void check(std::unique_ptr<DampingFunctionType> in_gh_damping_function,
115  const std::string& python_function_prefix, const T& used_for_size,
116  const std::array<std::pair<double, double>, 1>& random_value_bounds,
117  const MemberArgs&... member_args) noexcept {
118  detail::check_impl(
119  std::unique_ptr<
120  ::GeneralizedHarmonic::ConstraintDamping::DampingFunction<
121  DampingFunctionType::volume_dim,
122  typename DampingFunctionType::frame>>(
123  std::move(in_gh_damping_function)),
124  python_function_prefix, used_for_size, random_value_bounds,
125  member_args...);
126 }
