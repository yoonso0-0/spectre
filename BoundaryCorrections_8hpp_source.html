68 template <typename BoundaryCorrection, typename... PackageTags,
69  typename... FaceTags, typename... VolumeTags, size_t Dim>
70 void call_dg_package_data(
71  const gsl::not_null<Variables<tmpl::list<PackageTags...>>*> package_data,
72  const BoundaryCorrection& correction,
73  const Variables<tmpl::list<FaceTags...>>& face_variables,
74  const tuples::TaggedTuple<VolumeTags...>& volume_data,
75  const tnsr::i<DataVector, Dim, Frame::Inertial>& unit_normal_covector,
76  const boost::optional<tnsr::I<DataVector, Dim, Frame::Inertial>>&
77  mesh_velocity) {
78  boost::optional<Scalar<DataVector>> normal_dot_mesh_velocity{};
79  if (static_cast<bool>(mesh_velocity)) {
80  normal_dot_mesh_velocity =
81  dot_product(*mesh_velocity, unit_normal_covector);
82  }
83  correction.dg_package_data(
84  make_not_null(&get<PackageTags>(*package_data))...,
85  get<FaceTags>(face_variables)..., unit_normal_covector, mesh_velocity,
86  normal_dot_mesh_velocity, get<VolumeTags>(volume_data)...);
87 }

104 template <typename System, typename BoundaryCorrection, size_t FaceDim,
105  typename... VolumeTags>
106 void test_boundary_correction_impl(
107  const BoundaryCorrection& correction, const Mesh<FaceDim>& face_mesh,
108  const tuples::TaggedTuple<VolumeTags...>& volume_data,
109  const bool use_moving_mesh, const ::dg::Formulation dg_formulation,
110  const ZeroOnSmoothSolution zero_on_smooth_solution) {
111  CAPTURE(use_moving_mesh);
112  CAPTURE(dg_formulation);
113  CAPTURE(FaceDim);
114  using variables_tags = typename System::variables_tag::tags_list;
115  using primitive_variables_tags = detail::get_system_primitive_vars<
116  System::has_primitive_and_conservative_vars, System>;
117  using flux_variables = typename System::flux_variables;
118  using flux_tags =
119  db::wrap_tags_in<::Tags::Flux, flux_variables, tmpl::size_t<FaceDim + 1>,
120  Frame::Inertial>;
121  using temporary_tags =
122  typename System::compute_volume_time_derivative_terms::temporary_tags;
123  using dt_variables_tags = db::wrap_tags_in<::Tags::dt, variables_tags>;
124 
125  using dg_package_field_tags =
126  typename BoundaryCorrection::dg_package_field_tags;
127  using package_temporary_tags =
128  typename BoundaryCorrection::dg_package_data_temporary_tags;
129  using package_primitive_tags = detail::get_correction_primitive_vars<
130  System::has_primitive_and_conservative_vars, BoundaryCorrection>;
131 
132  // Check that the temporary tags needed on the boundary
133  // (package_temporary_tags) are listed as temporary tags for the volume time
134  // derivative computation (temporary_tags).
135  static_assert(
136  std::is_same_v<
137  tmpl::list_difference<package_temporary_tags, temporary_tags>,
138  tmpl::list<>>,
139  "There are temporary tags needed by the boundary correction that are not "
140  "computed as temporary tags by the system");
141 
142  // Check that the primitive tags needed on the boundary
143  // (package_primitive_tags) are listed as the primitive tags in
144  // the system (primitive_variables_tags).
145  static_assert(
146  std::is_same_v<tmpl::list_difference<package_primitive_tags,
147  primitive_variables_tags>,
148  tmpl::list<>>,
149  "There are primitive tags needed by the boundary correction that are not "
150  "listed in the system as being primitive variables");
151 
152  MAKE_GENERATOR(gen);
153  std::uniform_real_distribution<> dist(0.0, 1.0);
154  std::uniform_real_distribution<> pos_neg_dist(-1.0, 1.0);
155  DataVector used_for_size{face_mesh.number_of_grid_points()};
156 
157  boost::optional<tnsr::I<DataVector, FaceDim + 1, Frame::Inertial>>
158  mesh_velocity{};
159  if (use_moving_mesh) {
160  mesh_velocity = make_with_random_values<
161  tnsr::I<DataVector, FaceDim + 1, Frame::Inertial>>(
162  make_not_null(&gen), make_not_null(&dist), used_for_size);
163  }
164 
165  Variables<dg_package_field_tags> interior_package_data{used_for_size.size()};
166  const auto interior_fields_on_face = make_with_random_values<
167  Variables<tmpl::append<variables_tags, flux_tags, package_temporary_tags,
168  package_primitive_tags>>>(
169  make_not_null(&gen), make_not_null(&dist), used_for_size);
170 
171  Variables<dg_package_field_tags> exterior_package_data{used_for_size.size()};
172  const auto exterior_fields_on_face = make_with_random_values<
173  Variables<tmpl::append<variables_tags, flux_tags, package_temporary_tags,
174  package_primitive_tags>>>(
175  make_not_null(&gen), make_not_null(&dist), used_for_size);
176 
177  // Compute the interior and exterior normal vectors so they are pointing in
178  // opposite directions.
179  auto interior_unit_normal_covector = make_with_random_values<
180  tnsr::i<DataVector, FaceDim + 1, Frame::Inertial>>(
181  make_not_null(&gen), make_not_null(&pos_neg_dist), used_for_size);
182  const Scalar<DataVector> interior_normal_magnitude =
183  magnitude(interior_unit_normal_covector);
184  for (auto& t : interior_unit_normal_covector) {
185  t /= get(interior_normal_magnitude);
186  }
187  auto exterior_unit_normal_covector = interior_unit_normal_covector;
188  for (auto& t : exterior_unit_normal_covector) {
189  t *= -1.0;
190  }
191 
192  call_dg_package_data(make_not_null(&interior_package_data), correction,
193  interior_fields_on_face, volume_data,
194  interior_unit_normal_covector, mesh_velocity);
195  call_dg_package_data(make_not_null(&exterior_package_data), correction,
196  exterior_fields_on_face, volume_data,
197  exterior_unit_normal_covector, mesh_velocity);
198 
199  Variables<dt_variables_tags> boundary_corrections{
200  face_mesh.number_of_grid_points()};
201  call_dg_boundary_terms(make_not_null(&boundary_corrections), correction,
202  interior_package_data, exterior_package_data,
203  dg_formulation);
204 
205  if (dg_formulation == ::dg::Formulation::StrongInertial) {
206  // The strong form should be (WeakForm - (n_i F^i)_{interior}).
207  // Since we also test conservation for the weak form we just need to test
208  // that the strong form satisfies the above definition.
209 
210  Variables<dt_variables_tags> expected_boundary_corrections{
211  face_mesh.number_of_grid_points()};
212  call_dg_boundary_terms(make_not_null(&expected_boundary_corrections),
213  correction, interior_package_data,
214  exterior_package_data,
215  ::dg::Formulation::WeakInertial);
216 
217  tmpl::for_each<flux_variables>([&interior_package_data,
218  &expected_boundary_corrections](
219  auto flux_variable_tag_v) noexcept {
220  using flux_variable_tag = tmpl::type_from<decltype(flux_variable_tag_v)>;
221  using normal_dot_flux_tag = ::Tags::NormalDotFlux<flux_variable_tag>;
222  using dt_tag = ::Tags::dt<flux_variable_tag>;
223  const auto& normal_dot_flux =
224  get<normal_dot_flux_tag>(interior_package_data);
225  auto& expected_boundary_correction =
226  get<dt_tag>(expected_boundary_corrections);
227  for (size_t tensor_index = 0;
228  tensor_index < expected_boundary_correction.size(); ++tensor_index) {
229  expected_boundary_correction[tensor_index] -=
230  normal_dot_flux[tensor_index];
231  }
232  });
233  {
234  INFO("Check weak and strong boundary terms match.");
235  CHECK(boundary_corrections == expected_boundary_corrections);
236  }
237 
238  if (zero_on_smooth_solution == ZeroOnSmoothSolution::Yes) {
239  INFO(
240  "Testing that if the solution is the same on both sides the "
241  "StrongInertial correction is identically zero.");
242  Variables<dg_package_field_tags> interior_package_data_opposite_signs{
243  used_for_size.size()};
244  call_dg_package_data(make_not_null(&interior_package_data_opposite_signs),
245  correction, interior_fields_on_face, volume_data,
246  exterior_unit_normal_covector, mesh_velocity);
247  Variables<dt_variables_tags> zero_boundary_correction{
248  face_mesh.number_of_grid_points()};
249  call_dg_boundary_terms(make_not_null(&zero_boundary_correction),
250  correction, interior_package_data,
251  interior_package_data_opposite_signs,
252  ::dg::Formulation::StrongInertial);
253  Variables<dt_variables_tags> expected_zero_boundary_correction{
254  face_mesh.number_of_grid_points(), 0.0};
255  tmpl::for_each<dt_variables_tags>([&expected_zero_boundary_correction,
256  &zero_boundary_correction](
257  auto dt_variables_tag_v) noexcept {
258  using dt_variables_tag = tmpl::type_from<decltype(dt_variables_tag_v)>;
259  const std::string tag_name = db::tag_name<dt_variables_tag>();
260  CAPTURE(tag_name);
261  Approx custom_approx = Approx::custom().epsilon(1.e-12).scale(1.0);
262  CHECK_ITERABLE_CUSTOM_APPROX(
263  get<dt_variables_tag>(zero_boundary_correction),
264  get<dt_variables_tag>(expected_zero_boundary_correction),
265  custom_approx);
266  });
267  }
268  } else if (dg_formulation == ::dg::Formulation::WeakInertial) {
269  INFO(
270  "Checking that swapping the two sides results in an overall minus "
271  "sign.");
272  Variables<dt_variables_tags> reverse_side_boundary_corrections{
273  face_mesh.number_of_grid_points()};
274  call_dg_boundary_terms(make_not_null(&reverse_side_boundary_corrections),
275  correction, exterior_package_data,
276  interior_package_data, dg_formulation);
277  // Check that the flux leaving one element equals the flux entering its
278  // neighbor, i.e., F*(interior, exterior) == -F*(exterior, interior)
279  reverse_side_boundary_corrections *= -1.0;
280  tmpl::for_each<flux_variables>([&boundary_corrections,
281  &reverse_side_boundary_corrections](
282  auto flux_variable_tag_v) {
283  using flux_variable_tag = tmpl::type_from<decltype(flux_variable_tag_v)>;
284  const std::string tag_name = db::tag_name<flux_variable_tag>();
285  CAPTURE(tag_name);
286  CHECK_ITERABLE_APPROX(
287  get<::Tags::dt<flux_variable_tag>>(reverse_side_boundary_corrections),
288  get<::Tags::dt<flux_variable_tag>>(boundary_corrections));
289  });
290  } else {
291  ERROR("DG formulation must be StrongInertial or WeakInertial, not "
292  << dg_formulation);
293  }
294 }

297 /*!
298  * \ingroup TestingFrameworkGroup
299  * \brief Checks that the boundary correction is conservative and that for
300  * smooth solutions the strong-form correction is zero.
301  */
302 template <typename System, typename BoundaryCorrection, size_t FaceDim,
303  typename... VolumeTags>
304 void test_boundary_correction_conservation(
305  const BoundaryCorrection& correction, const Mesh<FaceDim>& face_mesh,
306  const tuples::TaggedTuple<VolumeTags...>& volume_data,
307  const ZeroOnSmoothSolution zero_on_smooth_solution =
308  ZeroOnSmoothSolution::Yes) {
309  for (const auto use_moving_mesh : {true, false}) {
310  for (const auto& dg_formulation :
311  {::dg::Formulation::StrongInertial, ::dg::Formulation::WeakInertial}) {
312  detail::test_boundary_correction_impl<System>(
313  correction, face_mesh, volume_data, use_moving_mesh, dg_formulation,
314  zero_on_smooth_solution);
315  }
316  }
