37 // compute the time derivative using a centered sixth-order stencil
38 template <typename Solution>
39 Variables<valencia_tags> numerical_dt(
40  const Solution& solution, const tnsr::I<DataVector, 3, Frame::Inertial>& x,
41  const double time, const double delta_time) {
42  std::array<double, 6> six_times{
43  {time - 3.0 * delta_time, time - 2.0 * delta_time, time - delta_time,
44  time + delta_time, time + 2.0 * delta_time, time + 3.0 * delta_time}};
45  const size_t number_of_points = get<0>(x).size();
46  auto solution_at_six_times =
47  make_array<6>(Variables<valencia_tags>(number_of_points));
48 
49  using solution_tags =
50  tmpl::list<hydro::Tags::RestMassDensity<DataVector>,
51  hydro::Tags::SpecificInternalEnergy<DataVector>,
52  hydro::Tags::SpatialVelocity<DataVector, 3>,
53  hydro::Tags::MagneticField<DataVector, 3>,
54  hydro::Tags::DivergenceCleaningField<DataVector>,
55  hydro::Tags::LorentzFactor<DataVector>,
56  hydro::Tags::Pressure<DataVector>,
57  hydro::Tags::SpecificEnthalpy<DataVector>,
58  gr::Tags::SpatialMetric<3, Frame::Inertial, DataVector>,
59  gr::Tags::SqrtDetSpatialMetric<DataVector>>;
60 
61  for (size_t i = 0; i < 6; ++i) {
62  const auto vars =
63  solution.variables(x, gsl::at(six_times, i), solution_tags{});
64 
65  const auto& rest_mass_density =
66  get<hydro::Tags::RestMassDensity<DataVector>>(vars);
67  const auto& specific_internal_energy =
68  get<hydro::Tags::SpecificInternalEnergy<DataVector>>(vars);
69  const auto& spatial_velocity =
70  get<hydro::Tags::SpatialVelocity<DataVector, 3>>(vars);
71  const auto& magnetic_field =
72  get<hydro::Tags::MagneticField<DataVector, 3>>(vars);
73  const auto& divergence_cleaning_field =
74  get<hydro::Tags::DivergenceCleaningField<DataVector>>(vars);
75  const auto& lorentz_factor =
76  get<hydro::Tags::LorentzFactor<DataVector>>(vars);
77  const auto& pressure = get<hydro::Tags::Pressure<DataVector>>(vars);
78  const auto& specific_enthalpy =
79  get<hydro::Tags::SpecificEnthalpy<DataVector>>(vars);
80  const auto& spatial_metric =
81  get<gr::Tags::SpatialMetric<3, Frame::Inertial, DataVector>>(vars);
82  const auto& sqrt_det_spatial_metric =
83  get<gr::Tags::SqrtDetSpatialMetric<DataVector>>(vars);
84 
85  grmhd::ValenciaDivClean::ConservativeFromPrimitive::apply(
86  make_not_null(&get<grmhd::ValenciaDivClean::Tags::TildeD>(
87  gsl::at(solution_at_six_times, i))),
88  make_not_null(&get<grmhd::ValenciaDivClean::Tags::TildeTau>(
89  gsl::at(solution_at_six_times, i))),
90  make_not_null(&get<grmhd::ValenciaDivClean::Tags::TildeS<>>(
91  gsl::at(solution_at_six_times, i))),
92  make_not_null(&get<grmhd::ValenciaDivClean::Tags::TildeB<>>(
93  gsl::at(solution_at_six_times, i))),
94  make_not_null(&get<grmhd::ValenciaDivClean::Tags::TildePhi>(
95  gsl::at(solution_at_six_times, i))),
96  rest_mass_density, specific_internal_energy, specific_enthalpy,
97  pressure, spatial_velocity, lorentz_factor, magnetic_field,
98  sqrt_det_spatial_metric, spatial_metric, divergence_cleaning_field);
99  }
100 
101  return (-1.0 / (60.0 * delta_time)) * solution_at_six_times[0] +
102  (3.0 / (20.0 * delta_time)) * solution_at_six_times[1] +
103  (-0.75 / delta_time) * solution_at_six_times[2] +
104  (0.75 / delta_time) * solution_at_six_times[3] +
105  (-3.0 / (20.0 * delta_time)) * solution_at_six_times[4] +
106  (1.0 / (60.0 * delta_time)) * solution_at_six_times[5];
107 }

110 /// \ingroup TestingFrameworkGroup
111 /// \brief Determines if the given `solution` is a solution of the GRMHD
112 /// equations.
113 ///
114 /// Uses numerical derivatives to compute the solution, on the given `mesh` of
115 /// the root Element of the given `block` at the given `time` using a
116 /// sixth-order derivative in time for the given `delta_time`. The maximum
117 /// residual of the GRMHD equations must be zero within `error_tolerance`
118 template <typename Solution>
119 void verify_grmhd_solution(const Solution& solution, const Block<3>& block,
120  const Mesh<3>& mesh, const double error_tolerance,
121  const double time,
122  const double delta_time) noexcept {
123  static_assert(evolution::is_analytic_solution_v<Solution>,
124  "Solution was not derived from AnalyticSolution");
125  // Set up coordinates
126  const auto x_logical = logical_coordinates(mesh);
127  if (block.is_time_dependent()) {
128  ERROR("The block must be time-independent");
129  }
130  const auto x = block.stationary_map()(x_logical);
131 
132  // Evaluate analytic solution
133  using solution_tags = tmpl::list<
134  hydro::Tags::RestMassDensity<DataVector>,
135  hydro::Tags::SpecificInternalEnergy<DataVector>,
136  hydro::Tags::SpatialVelocity<DataVector, 3>,
137  hydro::Tags::MagneticField<DataVector, 3>,
138  hydro::Tags::DivergenceCleaningField<DataVector>,
139  hydro::Tags::LorentzFactor<DataVector>, hydro::Tags::Pressure<DataVector>,
140  hydro::Tags::SpecificEnthalpy<DataVector>, gr::Tags::Lapse<DataVector>,
141  gr::Tags::Shift<3, Frame::Inertial, DataVector>,
142  gr::Tags::SpatialMetric<3, Frame::Inertial, DataVector>,
143  gr::Tags::InverseSpatialMetric<3, Frame::Inertial, DataVector>,
144  gr::Tags::SqrtDetSpatialMetric<DataVector>,
145  ::Tags::deriv<gr::Tags::Lapse<DataVector>, tmpl::size_t<3>,
146  Frame::Inertial>,
147  ::Tags::deriv<gr::Tags::Shift<3, Frame::Inertial, DataVector>,
148  tmpl::size_t<3>, Frame::Inertial>,
149  ::Tags::deriv<gr::Tags::SpatialMetric<3, Frame::Inertial, DataVector>,
150  tmpl::size_t<3>, Frame::Inertial>,
151  gr::Tags::ExtrinsicCurvature<3, Frame::Inertial, DataVector>>;
152  const auto vars = solution.variables(x, time, solution_tags{});
153 
154  const auto& rest_mass_density =
155  get<hydro::Tags::RestMassDensity<DataVector>>(vars);
156  const auto& specific_internal_energy =
157  get<hydro::Tags::SpecificInternalEnergy<DataVector>>(vars);
158  const auto& spatial_velocity =
159  get<hydro::Tags::SpatialVelocity<DataVector, 3>>(vars);
160  const auto& magnetic_field =
161  get<hydro::Tags::MagneticField<DataVector, 3>>(vars);
162  const auto& divergence_cleaning_field =
163  get<hydro::Tags::DivergenceCleaningField<DataVector>>(vars);
164  const auto& lorentz_factor =
165  get<hydro::Tags::LorentzFactor<DataVector>>(vars);
166  const auto& pressure = get<hydro::Tags::Pressure<DataVector>>(vars);
167  const auto& specific_enthalpy =
168  get<hydro::Tags::SpecificEnthalpy<DataVector>>(vars);
169  const auto& lapse = get<gr::Tags::Lapse<DataVector>>(vars);
170  const auto& d_lapse =
171  get<::Tags::deriv<gr::Tags::Lapse<DataVector>, tmpl::size_t<3>,
172  Frame::Inertial>>(vars);
173  const auto& shift =
174  get<gr::Tags::Shift<3, Frame::Inertial, DataVector>>(vars);
175  const auto& d_shift =
176  get<::Tags::deriv<gr::Tags::Shift<3, Frame::Inertial, DataVector>,
177  tmpl::size_t<3>, Frame::Inertial>>(vars);
178  const auto& spatial_metric =
179  get<gr::Tags::SpatialMetric<3, Frame::Inertial, DataVector>>(vars);
180  const auto& d_spatial_metric =
181  get<::Tags::deriv<gr::Tags::SpatialMetric<3, Frame::Inertial, DataVector>,
182  tmpl::size_t<3>, Frame::Inertial>>(vars);
183  const auto& inv_spatial_metric =
184  get<gr::Tags::InverseSpatialMetric<3, Frame::Inertial, DataVector>>(vars);
185  const auto& sqrt_det_spatial_metric =
186  get<gr::Tags::SqrtDetSpatialMetric<DataVector>>(vars);
187  const auto& extrinsic_curvature =
188  get<gr::Tags::ExtrinsicCurvature<3, Frame::Inertial, DataVector>>(vars);
189 
190  const size_t number_of_points = mesh.number_of_grid_points();
191  Scalar<DataVector> tilde_d(number_of_points);
192  Scalar<DataVector> tilde_tau(number_of_points);
193  tnsr::i<DataVector, 3> tilde_s(number_of_points);
194  tnsr::I<DataVector, 3> tilde_b(number_of_points);
195  Scalar<DataVector> tilde_phi(number_of_points);
196 
197  grmhd::ValenciaDivClean::ConservativeFromPrimitive::apply(
198  make_not_null(&tilde_d), make_not_null(&tilde_tau),
199  make_not_null(&tilde_s), make_not_null(&tilde_b),
200  make_not_null(&tilde_phi), rest_mass_density, specific_internal_energy,
201  specific_enthalpy, pressure, spatial_velocity, lorentz_factor,
202  magnetic_field, sqrt_det_spatial_metric, spatial_metric,
203  divergence_cleaning_field);
204 
205  using flux_tags =
206  tmpl::list<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeD,
207  tmpl::size_t<3>, Frame::Inertial>,
208  Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeTau,
209  tmpl::size_t<3>, Frame::Inertial>,
210  Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeS<>,
211  tmpl::size_t<3>, Frame::Inertial>,
212  Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeB<>,
213  tmpl::size_t<3>, Frame::Inertial>,
214  Tags::Flux<grmhd::ValenciaDivClean::Tags::TildePhi,
215  tmpl::size_t<3>, Frame::Inertial>>;
216  Variables<flux_tags> fluxes(number_of_points);
217  auto& flux_tilde_d =
218  get<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeD, tmpl::size_t<3>,
219  Frame::Inertial>>(fluxes);
220  auto& flux_tilde_tau =
221  get<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeTau, tmpl::size_t<3>,
222  Frame::Inertial>>(fluxes);
223  auto& flux_tilde_s =
224  get<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeS<>, tmpl::size_t<3>,
225  Frame::Inertial>>(fluxes);
226  auto& flux_tilde_b =
227  get<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeB<>, tmpl::size_t<3>,
228  Frame::Inertial>>(fluxes);
229  auto& flux_tilde_phi =
230  get<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildePhi, tmpl::size_t<3>,
231  Frame::Inertial>>(fluxes);
232 
233  grmhd::ValenciaDivClean::ComputeFluxes::apply(
234  make_not_null(&flux_tilde_d), make_not_null(&flux_tilde_tau),
235  make_not_null(&flux_tilde_s), make_not_null(&flux_tilde_b),
236  make_not_null(&flux_tilde_phi), tilde_d, tilde_tau, tilde_s, tilde_b,
237  tilde_phi, lapse, shift, sqrt_det_spatial_metric, spatial_metric,
238  inv_spatial_metric, pressure, spatial_velocity, lorentz_factor,
239  magnetic_field);
240 
241  if (block.is_time_dependent()) {
242  ERROR("The block must be time-independent");
243  }
244  const auto div_of_fluxes = divergence<flux_tags, 3, Frame::Inertial>(
245  fluxes, mesh, block.stationary_map().inv_jacobian(x_logical));
246 
247  const auto& div_flux_tilde_d =
248  get<Tags::div<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeD,
249  tmpl::size_t<3>, Frame::Inertial>>>(
250  div_of_fluxes);
251  const auto& div_flux_tilde_tau =
252  get<Tags::div<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeTau,
253  tmpl::size_t<3>, Frame::Inertial>>>(
254  div_of_fluxes);
255  const auto& div_flux_tilde_s =
256  get<Tags::div<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeS<>,
257  tmpl::size_t<3>, Frame::Inertial>>>(
258  div_of_fluxes);
259  const auto& div_flux_tilde_b =
260  get<Tags::div<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildeB<>,
261  tmpl::size_t<3>, Frame::Inertial>>>(
262  div_of_fluxes);
263  const auto& div_flux_tilde_phi =
264  get<Tags::div<Tags::Flux<grmhd::ValenciaDivClean::Tags::TildePhi,
265  tmpl::size_t<3>, Frame::Inertial>>>(
266  div_of_fluxes);
267 
268  Scalar<DataVector> source_tilde_tau(number_of_points);
269  tnsr::i<DataVector, 3> source_tilde_s(number_of_points);
270  tnsr::I<DataVector, 3> source_tilde_b(number_of_points);
271  Scalar<DataVector> source_tilde_phi(number_of_points);
272  grmhd::ValenciaDivClean::ComputeSources::apply(
273  make_not_null(&source_tilde_tau), make_not_null(&source_tilde_s),
274  make_not_null(&source_tilde_b), make_not_null(&source_tilde_phi), tilde_d,
275  tilde_tau, tilde_s, tilde_b, tilde_phi, spatial_velocity, magnetic_field,
276  rest_mass_density, specific_enthalpy, lorentz_factor, pressure, lapse,
277  d_lapse, d_shift, spatial_metric, d_spatial_metric, inv_spatial_metric,
278  sqrt_det_spatial_metric, extrinsic_curvature, 0.0);
279 
280  Scalar<DataVector> residual_tilde_d(number_of_points, 0.);
281  Scalar<DataVector> residual_tilde_tau(number_of_points, 0.);
282  tnsr::i<DataVector, 3> residual_tilde_s(number_of_points, 0.);
283  tnsr::I<DataVector, 3> residual_tilde_b(number_of_points, 0.);
284  Scalar<DataVector> residual_tilde_phi(number_of_points, 0.);
285 
286  Variables<VerifyGrMhdSolution_detail::valencia_tags> dt_solution =
287  VerifyGrMhdSolution_detail::numerical_dt(solution, x, time, delta_time);
288 
289  get(residual_tilde_d) =
290  get(get<grmhd::ValenciaDivClean::Tags::TildeD>(dt_solution)) +
291  get(div_flux_tilde_d);
292  get(residual_tilde_tau) =
293  get(get<grmhd::ValenciaDivClean::Tags::TildeTau>(dt_solution)) +
294  get(div_flux_tilde_tau) - get(source_tilde_tau);
295  get<0>(residual_tilde_s) =
296  get<0>(get<grmhd::ValenciaDivClean::Tags::TildeS<>>(dt_solution)) +
297  get<0>(div_flux_tilde_s) - get<0>(source_tilde_s);
298  get<1>(residual_tilde_s) =
299  get<1>(get<grmhd::ValenciaDivClean::Tags::TildeS<>>(dt_solution)) +
300  get<1>(div_flux_tilde_s) - get<1>(source_tilde_s);
301  get<2>(residual_tilde_s) =
302  get<2>(get<grmhd::ValenciaDivClean::Tags::TildeS<>>(dt_solution)) +
303  get<2>(div_flux_tilde_s) - get<2>(source_tilde_s);
304  get<0>(residual_tilde_b) =
305  get<0>(get<grmhd::ValenciaDivClean::Tags::TildeB<>>(dt_solution)) +
306  get<0>(div_flux_tilde_b) - get<0>(source_tilde_b);
307  get<1>(residual_tilde_b) =
308  get<1>(get<grmhd::ValenciaDivClean::Tags::TildeB<>>(dt_solution)) +
309  get<1>(div_flux_tilde_b) - get<1>(source_tilde_b);
310  get<2>(residual_tilde_b) =
311  get<2>(get<grmhd::ValenciaDivClean::Tags::TildeB<>>(dt_solution)) +
312  get<2>(div_flux_tilde_b) - get<2>(source_tilde_b);
313  get(residual_tilde_phi) +=
314  get(get<grmhd::ValenciaDivClean::Tags::TildePhi>(dt_solution)) +
315  get(div_flux_tilde_phi) - get(source_tilde_phi);
316 
317  Approx numerical_approx =
318  Approx::custom().epsilon(error_tolerance).scale(1.0);
319  CHECK(max(abs(get(residual_tilde_d))) == numerical_approx(0.0));
320  CHECK(max(abs(get(residual_tilde_tau))) == numerical_approx(0.0));
321  CHECK(max(abs(get<0>(residual_tilde_s))) == numerical_approx(0.0));
322  CHECK(max(abs(get<1>(residual_tilde_s))) == numerical_approx(0.0));
323  CHECK(max(abs(get<2>(residual_tilde_s))) == numerical_approx(0.0));
324  CHECK(max(abs(get<0>(residual_tilde_b))) == numerical_approx(0.0));
325  CHECK(max(abs(get<1>(residual_tilde_b))) == numerical_approx(0.0));
326  CHECK(max(abs(get<2>(residual_tilde_b))) == numerical_approx(0.0));
327  CHECK(max(abs(get(residual_tilde_phi))) == numerical_approx(0.0));
328 }
