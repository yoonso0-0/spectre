24 /*!
25  * \brief Initializes the DataBox tags related to the elliptic system
26  *
27  * The system fields are initially set to zero.
28  *
29  * \note Currently the sources are always retrieved from an analytic solution.
30  *
31  * Uses:
32  * - Metavariables:
33  * - `analytic_solution_tag`
34  * - System:
35  * - `fields_tag`
36  * - `primal_fields`
37  * - DataBox:
38  * - `Tags::Mesh<Dim>`
39  * - `Tags::Coordinates<Dim, Frame::Inertial>`
40  *
41  * DataBox:
42  * - Adds:
43  * - `fields_tag`
44  * - `db::add_tag_prefix<::Tags::FixedSource, fields_tag>`
45  *
46  * \note This action relies on the `SetupDataBox` aggregated initialization
47  * mechanism, so `Actions::SetupDataBox` must be present in the `Initialization`
48  * phase action list prior to this action.
49  */
50 template <typename System>
51 struct InitializeSystem {
52  using fields_tag = typename System::fields_tag;
53  using fixed_sources_tag =
54  db::add_tag_prefix<::Tags::FixedSource, fields_tag>;
55 
56  using simple_tags = tmpl::list<fields_tag, fixed_sources_tag>;
57  using compute_tags = tmpl::list<>;
58 
59  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
60  size_t Dim, typename ActionList, typename ParallelComponent>
61  static auto apply(db::DataBox<DbTagsList>& box,
62  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
63  const Parallel::GlobalCache<Metavariables>& cache,
64  const ElementId<Dim>& /*array_index*/,
65  const ActionList /*meta*/,
66  const ParallelComponent* const /*meta*/) noexcept {
67  using linear_operator_applied_to_fields_tag =
68  db::add_tag_prefix<LinearSolver::Tags::OperatorAppliedTo, fields_tag>;
69 
70  const auto& mesh = db::get<domain::Tags::Mesh<Dim>>(box);
71  const size_t num_grid_points = mesh.number_of_grid_points();
72  const auto& inertial_coords =
73  get<domain::Tags::Coordinates<Dim, Frame::Inertial>>(box);
74 
75  // Set initial data to zero (for now).
76  typename fields_tag::type fields{num_grid_points, 0.};
77  // Since the initial data is zero we don't need to apply the DG operator but
78  // may just set it to zero as well. Once this condition is relaxed we will
79  // have to add a communication step to the initialization that computes the
80  // DG operator to the initial data.
81  db::mutate<linear_operator_applied_to_fields_tag>(
82  make_not_null(&box),
83  [&num_grid_points](
84  const gsl::not_null<
85  typename linear_operator_applied_to_fields_tag::type*>
86  linear_operator_applied_to_fields) noexcept {
87  *linear_operator_applied_to_fields =
88  typename linear_operator_applied_to_fields_tag::type{
89  num_grid_points, 0.};
90  });
91 
92  // Retrieve the sources of the elliptic system from the analytic solution,
93  // which defines the problem we want to solve.
94  // We need only retrieve sources for the primal fields, since the auxiliary
95  // fields will never be sourced.
96  typename fixed_sources_tag::type fixed_sources{num_grid_points, 0.};
97  fixed_sources.assign_subset(
98  Parallel::get<typename Metavariables::analytic_solution_tag>(cache)
99  .variables(inertial_coords,
100  db::wrap_tags_in<::Tags::FixedSource,
101  typename System::primal_fields>{}));
102  Initialization::mutate_assign<simple_tags>(
103  make_not_null(&box), std::move(fields), std::move(fixed_sources));
104  return std::make_tuple(std::move(box));
105  }
106 };
