94 /// \ingroup ActionsGroup
95 /// \brief Initializes an InterpolationTarget
96 ///
97 /// Uses: nothing
98 ///
99 /// DataBox changes:
100 /// - Adds:
101 /// - `Tags::IndicesOfFilledInterpPoints<TemporalId>`
102 /// - `Tags::IndicesOfInvalidInterpPoints<TemporalId>`
103 /// - `Tags::TemporalIds<TemporalId>`
104 /// - `Tags::CompletedTemporalIds<TemporalId>`
105 /// - `Tags::InterpolatedVars<InterpolationTargetTag,TemporalId>`
106 /// - `::Tags::Variables<typename
107 /// InterpolationTargetTag::vars_to_interpolate_to_target>`
108 /// - Removes: nothing
109 /// - Modifies: nothing
110 ///
111 /// For requirements on InterpolationTargetTag, see InterpolationTarget
112 ///
113 /// \note This action relies on the `SetupDataBox` aggregated initialization
114 /// mechanism, so `Actions::SetupDataBox` must be present in the
115 /// `Initialization` phase action list prior to this action.
116 template <typename Metavariables, typename InterpolationTargetTag>
117 struct InitializeInterpolationTarget {
118  using TemporalId = typename Metavariables::temporal_id::type;
119  using return_tag_list_initial = tmpl::list<
120  Tags::IndicesOfFilledInterpPoints<TemporalId>,
121  Tags::IndicesOfInvalidInterpPoints<TemporalId>,
122  Tags::TemporalIds<TemporalId>,
123  Tags::CompletedTemporalIds<TemporalId>,
124  Tags::InterpolatedVars<InterpolationTargetTag, TemporalId>,
125  ::Tags::Variables<
126  typename InterpolationTargetTag::vars_to_interpolate_to_target>>;
127  using return_tag_list =
128  tmpl::append<return_tag_list_initial,
129  typename initialize_interpolation_target_detail::
130  initialization_tags<InterpolationTargetTag>::type>;
131 
132  using simple_tags = tmpl::append<
133  return_tag_list_initial,
134  typename initialize_interpolation_target_detail::
135  compute_target_points_tags<InterpolationTargetTag>::simple_tags>;
136  using compute_tags = tmpl::append<
137  typename initialize_interpolation_target_detail::
138  compute_target_points_tags<InterpolationTargetTag>::compute_tags,
139  typename InterpolationTargetTag::compute_items_on_target>;
140 
141  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
142  typename ActionList, typename ParallelComponent>
143  static auto apply(db::DataBox<DbTagsList>& box,
144  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
145  const Parallel::GlobalCache<Metavariables>& cache,
146  const ArrayIndex& /*array_index*/,
147  const ActionList /*meta*/,
148  const ParallelComponent* const /*meta*/) noexcept {
149  initialize_interpolation_target_detail::mutate_initial_box<
150  InterpolationTargetTag>(make_not_null(&box), cache);
151  return std::make_tuple(std::move(box));
152  }
153 };
