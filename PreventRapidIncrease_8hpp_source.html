35 /// Avoids instabilities due to rapid increases in the step size by
36 /// preventing the step size from increasing unless all steps in the
37 /// time-stepper history are the same size. If there have been recent
38 /// step size changes the new size bound is the size of the most
39 /// recent step, otherwise it is infinite (no restriction is imposed).
40 ///
41 /// Changes in step size resulting from a slab size change are not
42 /// taken into account. In practice, this should not be an issue as
43 /// long as there are many steps between slab size changes.
44 template <typename StepChooserRegistrars =
45  tmpl::list<Registrars::PreventRapidIncrease>>
46 class PreventRapidIncrease : public StepChooser<StepChooserRegistrars> {
47  public:
48  /// \cond
49  PreventRapidIncrease() = default;
50  explicit PreventRapidIncrease(CkMigrateMessage* /*unused*/) noexcept {}
51  using PUP::able::register_constructor;
52  WRAPPED_PUPable_decl_template(PreventRapidIncrease); // NOLINT
53  /// \endcond
54 
55  static constexpr Options::String help{
56  "Prevents rapid increases in time step that can cause integrator \n"
57  "instabilities."};
58  using options = tmpl::list<>;
59 
60  using argument_tags = tmpl::list<Tags::HistoryEvolvedVariables<>>;
61 
62  template <typename Metavariables, typename History>
63  double operator()(const History& history, const double last_step_magnitude,
64  const Parallel::GlobalCache<Metavariables>& /*cache*/)
65  const noexcept {
66  if (history.size() < 2) {
67  return std::numeric_limits<double>::infinity();
68  }
69 
70  const double sloppiness = slab_rounding_error(history[0]);
71  for (auto step = history.begin(); step != history.end() - 1; ++step) {
72  // Potential roundoff error comes from the inability to make
73  // slabs exactly the same length.
74  if (abs(abs(*(step + 1) - *step).value() - last_step_magnitude) >
75  sloppiness) {
76  return last_step_magnitude;
77  }
78  }
79  // Request that the step size be at most infinity. This imposes
80  // no restriction on the chosen step.
81  return std::numeric_limits<double>::infinity();
82  }
83 };

89 /// \endcond
90 } // namespace StepChoosers
