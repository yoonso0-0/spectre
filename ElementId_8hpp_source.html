27 /*!
28  * \ingroup ComputationalDomainGroup
29  * \brief An ElementId uniquely labels an Element.
30  *
31  * It is constructed from the BlockId of the Block to which the Element belongs
32  * and the VolumeDim SegmentIds that label the segments of the Block that the
33  * Element covers.
34  *
35  * \details
36  * The `ElementId` serves as an index that is compatible with Charm++ and
37  * therefore must adhere to the restrictions imposed by Charm++. These are:
38  * - `ElementId` must satisfy `std::is_pod`
39  * - `ElementId` must not be larger than the size of three `int`s, i.e.
40  * `sizeof(ElementId) <= 3 * sizeof(int)`
41  *
42  * The latter restriction can be relaxed with a special compilation flag to
43  * Charm++, but we have not yet needed more elements than can be accounted for
44  * by densely packing bits together. `SegmentId` is responsible for handling the
45  * low-level bit manipulations to create an index that satisfies the size
46  * constraints.
47  */
48 template <size_t VolumeDim>
49 class ElementId {
50  public:
51  static constexpr size_t volume_dim = VolumeDim;
52

72  size_t block_id() const noexcept {
73  ASSERT(
74  alg::all_of(
75  segment_ids_,
76  [this](const SegmentId& current_id) noexcept {
77  return current_id.block_id() == segment_ids_[0].block_id();
78  }),
79  "Not all of the `SegmentId`s inside `ElementId` have same `BlockId`.");
80  return segment_ids_[0].block_id();
81  }
