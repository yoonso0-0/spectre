21 /// \ingroup DiscontinuousGalerkinGroup
22 /// \brief Lifts the flux contribution from an interface to the volume.
23 ///
24 /// The lifting operation takes the (d-1)-dimensional flux term at the
25 /// interface and computes the corresponding d-dimensional term in the
26 /// volume. SpECTRE implements an efficient DG method in which each
27 /// interface grid point contributes only to that same grid point of the
28 /// volume.
29 ///
30 /// \details
31 /// SpECTRE implements a DG method with a diagonalized mass matrix (also
32 /// known as a mass-lumping scheme). This choice gives a large
33 /// reduction in the computational cost of the lifting operation, however,
34 /// the scheme is slightly less accurate, especially when the grid is
35 /// deformed by non-trivial Jacobians. For more details on the
36 /// diagonalization of the mass matrix and its implications,
37 /// \cite Teukolsky2015ega, especially Section 3.
38 ///
39 /// \note The result is still provided only on the boundary grid. The
40 /// values away from the boundary are zero and are not stored.
41 template <typename... BoundaryCorrectionTags>
42 void lift_flux(
43  const gsl::not_null<Variables<tmpl::list<BoundaryCorrectionTags...>>*>
44  boundary_correction_terms,
45  const size_t extent_perpendicular_to_boundary,
46  Scalar<DataVector> magnitude_of_face_normal) noexcept {
47  DataVector lift_factor = std::move(get(magnitude_of_face_normal));
48  // For an Nth degree basis (i.e., one with N+1 basis functions), the LGL
49  // weights are:
50  // w_i = 2 / ((N + 1) * N * (P_{N}(xi_i))^2)
51  // and so at the end points (xi = +/- 1) we get:
52  // w_0 = 2 / ((N + 1) * N)
53  // w_N = 2 / ((N + 1) * N)
54  // The negative sign comes from bringing the boundary correction over to the
55  // RHS of the equal sign (e.g. `du/dt=Source - div Flux - boundary corr`),
56  // while the magnitude of the normal vector above accounts for the ratios of
57  // spatial metrics and Jacobians.
58  lift_factor *=
59  -0.5 * static_cast<double>((extent_perpendicular_to_boundary *
60  (extent_perpendicular_to_boundary - 1)));
61 
62  *boundary_correction_terms *= lift_factor;
63 }
64 
65 template <typename... FluxTags>
66 auto lift_flux(Variables<tmpl::list<FluxTags...>> flux,
67  const size_t extent_perpendicular_to_boundary,
68  Scalar<DataVector> magnitude_of_face_normal) noexcept
69  -> Variables<tmpl::list<db::remove_tag_prefix<FluxTags>...>> {
70  Variables<tmpl::list<db::remove_tag_prefix<FluxTags>...>> lifted_data(
71  std::move(flux));
72  lift_flux(make_not_null(&lifted_data), extent_perpendicular_to_boundary,
73  std::move(magnitude_of_face_normal));
74  return lifted_data;
75 }
