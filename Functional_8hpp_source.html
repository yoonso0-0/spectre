19 /*!
20  * \ingroup UtilitiesGroup
21  * \brief Higher order function objects similar to `std::plus`, etc.
22  *
23  * \details
24  * These chaining function objects can be used to represent highly general
25  * mathematical operations
26  * 1. as types, which can be passed around in template arguments, and
27  * 2. such that any time they can be evaluated at compile time, they will be.
28  *
29  * As an illustrative example, consider the definition of a general sinusoid
30  * function object type :
31  * \snippet Utilities/Test_Functional.cpp using_sinusoid
32  * which then gives a type which when instantiated and evaluated will give the
33  * answer \f$ a\times\sin(b + c \times d)\f$ from calling `Sinusoid{}(a,b,c,d)`
34  *
35  * As a more creative example, we can take advantage of literals to make, for
36  * instance, distributions. Let's make a Gaussian with mean at 5.0 and unity
37  * variance
38  * \snippet Utilities/Test_Functional.cpp using_gaussian
39  *
40  * This gives us a function object whose call operator takes one argument that
41  * gives the value of the desired Gaussian distribution \f$ e^{-(x - 5.0)^2}
42  * \f$
43  */

49 /// \cond
50 template <size_t Arity>
51 struct Functional {
52  static constexpr size_t arity = Arity;
53 
54  protected:
55  template <class C, size_t Offset, class... Ts, size_t... Is>
56  static constexpr decltype(auto) helper(
57  const std::tuple<Ts...>& t,
58  std::index_sequence<Is...> /*meta*/) noexcept {
59  return C{}(std::get<Offset + Is>(t)...);
60  }
61 };

66 /// Functional that asserts that the function object `C` applied to the first
67 /// and second arguments are equal and returns the function object C applied to
68 /// the first argument
69 template <class C = Identity>
70 struct AssertEqual : Functional<2> {
71  template <class T>
72  const T& operator()(const T& t0, const T& t1) noexcept {
73  DEBUG_STATIC_ASSERT(
74  C::arity == 1,
75  "The arity of the functional passed to AssertEqual must be 1");
76  ASSERT(C{}(t0) == C{}(t1), "Values are not equal in funcl::AssertEqual "
77  << C{}(t0) << " and " << C{}(t1));
78  return C{}(t0);
79  }
80 };

82 #define MAKE_BINARY_FUNCTIONAL(NAME, OPERATOR) \
83  /** Functional for computing `OPERATOR` from two objects */ \
84  template <class C0 = Identity, class C1 = C0> \
85  struct NAME : Functional<C0::arity + C1::arity> { \
86  using base = Functional<C0::arity + C1::arity>; \
87  template <class... Ts> \
88  constexpr auto operator()(const Ts&... ts) noexcept { \
89  return OPERATOR( \
90  base::template helper<C0, 0>(std::tuple<const Ts&...>(ts...), \
91  std::make_index_sequence<C0::arity>{}), \
92  base::template helper<C1, C0::arity>( \
93  std::tuple<const Ts&...>(ts...), \
94  std::make_index_sequence<C1::arity>{})); \
95  } \
96  }; \
97  /** \cond */ \
98  template <class C1> \
99  struct NAME<Identity, C1> : Functional<1 + C1::arity> { \
100  template <class T0, class... Ts> \
101  constexpr auto operator()(const T0& t0, const Ts&... ts) noexcept { \
102  return OPERATOR(t0, C1{}(ts...)); \
103  } \
104  }; \
105  template <> \
106  struct NAME<Identity, Identity> : Functional<2> { \
107  template <class T0, class T1> \
108  constexpr auto operator()(const T0& t0, const T1& t1) noexcept { \
109  return OPERATOR(t0, t1); \
110  } \
111  } /** \endcond */

205 /// Functional to retrieve the `ArgumentIndex`th argument
206 template <size_t Arity, size_t ArgumentIndex = 0, class C = Identity>
207 struct GetArgument : Functional<Arity> {
208  template <class... Ts>
209  constexpr decltype(auto) operator()(const Ts&... ts) noexcept {
210  static_assert(Arity == sizeof...(Ts),
211  "The arity passed to GetArgument must be the same as the "
212  "actually arity of the function.");
213  return C{}(std::get<ArgumentIndex>(std::tuple<const Ts&...>(ts...)));
214  }
215 };

311 /// Function for adding two `std::vector`s of `double` component-wise
312 struct VectorPlus {
313  std::vector<double> operator()(const std::vector<double>& lhs,
314  const std::vector<double>& rhs) const
315  noexcept {
316  ASSERT(lhs.size() == rhs.size(),
317  "Vector sizes in `funcl::VectorPlus` operator do not match. First "
318  "argument size: "
319  << lhs.size() << ". Second argument size: " << rhs.size()
320  << ".");
321  std::vector<double> result(lhs.size());
322  for (size_t i = 0; i < lhs.size(); ++i) {
323  result[i] = lhs[i] + rhs[i];
324  }
325  return result;
326  }
327 };
