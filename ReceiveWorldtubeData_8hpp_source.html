21 /*!
22  * \ingroup ActionsGroup
23  * \brief Takes the boundary data needed to perform the CCE linear solves as
24  * arguments and puts them in the \ref DataBoxGroup, updating the
25  * `Cce::Tags::BoundaryTime` accordingly.
26  *
27  * \details The boundary data is computed by a separate component, and packaged
28  * into a `Variables<tmpl::list<BoundaryTags...>>` which is sent in the argument
29  * of the simple action invocation. The `TimeStepId` is also provided to confirm
30  * the time associated with the passed boundary data.
31  *
32  * \ref DataBoxGroup changes:
33  * - Adds: nothing
34  * - Removes: nothing
35  * - Modifies:
36  * - All tags in `BoundaryTags`
37  * - `Cce::Tags::BoundaryTime`
38  */
39 template <typename Metavariables>
40 struct ReceiveWorldtubeData {
41  using inbox_tags = tmpl::list<Cce::ReceiveTags::BoundaryData<
42  typename Metavariables::cce_boundary_communication_tags>>;
43 
44  template <
45  typename DbTags, typename... InboxTags, typename ArrayIndex,
46  typename ActionList, typename ParallelComponent,
47  Requires<
48  tmpl::list_contains_v<
49  tmpl::list<InboxTags...>,
50  Cce::ReceiveTags::BoundaryData<
51  typename Metavariables::cce_boundary_communication_tags>> and
52  tmpl::list_contains_v<DbTags, ::Tags::TimeStepId>> = nullptr>
53  static auto apply(db::DataBox<DbTags>& box,
54  tuples::TaggedTuple<InboxTags...>& inboxes,
55  const Parallel::GlobalCache<Metavariables>& /*cache*/,
56  const ArrayIndex& /*array_index*/,
57  const ActionList /*meta*/,
58  const ParallelComponent* const /*meta*/) noexcept {
59  auto& inbox = tuples::get<Cce::ReceiveTags::BoundaryData<
60  typename Metavariables::cce_boundary_communication_tags>>(inboxes);
61  tmpl::for_each<typename Metavariables::cce_boundary_communication_tags>(
62  [&inbox, &box ](auto tag_v) noexcept {
63  using tag = typename decltype(tag_v)::type;
64  db::mutate<tag>(
65  make_not_null(&box),
66  [&inbox](const gsl::not_null<typename tag::type*> destination,
67  const TimeStepId& time) noexcept {
68  *destination = get<tag>(inbox[time]);
69  },
70  db::get<::Tags::TimeStepId>(box));
71  });
72  inbox.erase(db::get<::Tags::TimeStepId>(box));
73  return std::forward_as_tuple(std::move(box));
74  }
75  template <
76  typename DbTags, typename... InboxTags, typename ArrayIndex,
77  typename ActionList, typename ParallelComponent,
78  Requires<
79  not tmpl::list_contains_v<
80  tmpl::list<InboxTags...>,
81  Cce::ReceiveTags::BoundaryData<
82  typename Metavariables::cce_boundary_communication_tags>> or
83  not tmpl::list_contains_v<DbTags, ::Tags::TimeStepId>> = nullptr>
84  static auto apply(db::DataBox<DbTags>& box,
85  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
86  const Parallel::GlobalCache<Metavariables>& /*cache*/,
87  const ArrayIndex& /*array_index*/,
88  const ActionList /*meta*/,
89  const ParallelComponent* const /*meta*/) noexcept {
90  ERROR(
91  "Required tags not present in the inbox or databox to transfer the CCE "
92  "boundary data");
93  // provided for return type inference. This line will never be executed.
94  return std::forward_as_tuple(std::move(box));
95  }
96 
97  template <
98  typename DbTags, typename... InboxTags, typename ArrayIndex,
99  Requires<
100  tmpl::list_contains_v<
101  tmpl::list<InboxTags...>,
102  Cce::ReceiveTags::BoundaryData<
103  typename Metavariables::cce_boundary_communication_tags>> and
104  tmpl::list_contains_v<DbTags, ::Tags::TimeStepId>> = nullptr>
105  static bool is_ready(
106  const db::DataBox<DbTags>& box,
107  const tuples::TaggedTuple<InboxTags...>& inboxes,
108  const Parallel::GlobalCache<Metavariables>& /*cache*/,
109  const ArrayIndex& /*array_index*/) noexcept {
110  return tuples::get<Cce::ReceiveTags::BoundaryData<
111  typename Metavariables::cce_boundary_communication_tags>>(
112  inboxes)
113  .count(db::get<::Tags::TimeStepId>(box)) == 1;
114  }
115 
116  template <
117  typename DbTags, typename... InboxTags, typename ArrayIndex,
118  Requires<
119  not tmpl::list_contains_v<
120  tmpl::list<InboxTags...>,
121  Cce::ReceiveTags::BoundaryData<
122  typename Metavariables::cce_boundary_communication_tags>> or
123  not tmpl::list_contains_v<DbTags, ::Tags::TimeStepId>> = nullptr>
124  static bool is_ready(
125  const db::DataBox<DbTags>& /*box*/,
126  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
127  const Parallel::GlobalCache<Metavariables>& /*cache*/,
128  const ArrayIndex& /*array_index*/) noexcept {
129  return false;
130  }
131 };
