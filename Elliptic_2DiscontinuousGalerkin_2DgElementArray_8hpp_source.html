24 /*!
25  * \brief The parallel component responsible for managing the DG elements that
26  * compose the computational domain
27  *
28  * This parallel component will perform the actions specified by the
29  * `PhaseDepActionList`.
30  *
31  * \note This parallel component is nearly identical to
32  * `Evolution/DiscontinuousGalerkin/DgElementArray.hpp` right now, but will
33  * likely diverge in the future, for instance to support a multigrid domain.
34  *
35  */
36 template <class Metavariables, class PhaseDepActionList>
37 struct DgElementArray {
38  static constexpr size_t volume_dim = Metavariables::volume_dim;
39 
40  using chare_type = Parallel::Algorithms::Array;
41  using metavariables = Metavariables;
42  using phase_dependent_action_list = PhaseDepActionList;
43  using array_index = ElementId<volume_dim>;
44 
45  using const_global_cache_tags = tmpl::list<domain::Tags::Domain<volume_dim>>;
46 
47  using array_allocation_tags =
48  tmpl::list<domain::Tags::InitialRefinementLevels<volume_dim>>;
49 
50  using initialization_tags = Parallel::get_initialization_tags<
51  Parallel::get_initialization_actions_list<phase_dependent_action_list>,
52  array_allocation_tags>;
53 
54  static void allocate_array(
55  Parallel::CProxy_GlobalCache<Metavariables>& global_cache,
56  const tuples::tagged_tuple_from_typelist<initialization_tags>&
57  initialization_items) noexcept;
58 
59  static void execute_next_phase(
60  const typename Metavariables::Phase next_phase,
61  Parallel::CProxy_GlobalCache<Metavariables>& global_cache) noexcept {
62  auto& local_cache = *(global_cache.ckLocalBranch());
63  Parallel::get_parallel_component<DgElementArray>(local_cache)
64  .start_phase(next_phase);
65  }
66 };
67 
68 template <class Metavariables, class PhaseDepActionList>
69 void DgElementArray<Metavariables, PhaseDepActionList>::allocate_array(
70  Parallel::CProxy_GlobalCache<Metavariables>& global_cache,
71  const tuples::tagged_tuple_from_typelist<initialization_tags>&
72  initialization_items) noexcept {
73  auto& local_cache = *(global_cache.ckLocalBranch());
74  auto& dg_element_array =
75  Parallel::get_parallel_component<DgElementArray>(local_cache);
76  const auto& domain =
77  Parallel::get<domain::Tags::Domain<volume_dim>>(local_cache);
78  const auto& initial_refinement_levels =
79  get<domain::Tags::InitialRefinementLevels<volume_dim>>(
80  initialization_items);
81  for (const auto& block : domain.blocks()) {
82  const auto initial_ref_levs = initial_refinement_levels[block.id()];
83  const std::vector<ElementId<volume_dim>> element_ids =
84  initial_element_ids(block.id(), initial_ref_levs);
85  int which_proc = 0;
86  const int number_of_procs = Parallel::number_of_procs();
87  for (size_t i = 0; i < element_ids.size(); ++i) {
88  dg_element_array(ElementId<volume_dim>(element_ids[i]))
89  .insert(global_cache, initialization_items, which_proc);
90  which_proc = which_proc + 1 == number_of_procs ? 0 : which_proc + 1;
91  }
92  }
93  dg_element_array.doneInserting();
94 }
95 } // namespace elliptic
