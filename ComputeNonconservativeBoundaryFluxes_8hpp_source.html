26 /// \ingroup ActionsGroup
27 /// \ingroup DiscontinuousGalerkinGroup
28 /// \brief Compute \f$\hat{n} \cdot F\f$ on the boundaries for a
29 /// non-conservative system.
30 ///
31 /// Uses:
32 /// - System:
33 /// - `variables_tag`
34 /// - DataBox:
35 /// - `DirectionsTag`,
36 /// - Interface items as required by
37 /// `Metavariables::system::normal_dot_fluxes`
38 ///
39 /// DataBox changes:
40 /// - Adds: nothing
41 /// - Removes: nothing
42 /// - Modifies:
43 /// - Tags::Interface<
44 /// DirectionsTag, db::add_tag_prefix<Tags::NormalDotFlux, variables_tag>>
45 template <typename DirectionsTag>
46 struct ComputeNonconservativeBoundaryFluxes {
47  private:
48  template <typename Metavariables, typename... NormalDotFluxTags,
49  typename... Args>
50  static void apply_flux(
51  gsl::not_null<Variables<tmpl::list<NormalDotFluxTags...>>*> boundary_flux,
52  const Args&... boundary_variables) noexcept {
53  Metavariables::system::normal_dot_fluxes::apply(
54  make_not_null(&get<NormalDotFluxTags>(*boundary_flux))...,
55  boundary_variables...);
56  }
57 
58  public:
59  template <typename DbTags, typename... InboxTags, typename Metavariables,
60  typename ArrayIndex, typename ActionList,
61  typename ParallelComponent>
62  static std::tuple<db::DataBox<DbTags>&&> apply(
63  db::DataBox<DbTags>& box, tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
64  Parallel::GlobalCache<Metavariables>& /*cache*/,
65  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
66  const ParallelComponent* const /*meta*/) noexcept {
67  using system = typename Metavariables::system;
68  using variables_tag = typename system::variables_tag;
69 
70  using interface_normal_dot_fluxes_tag = domain::Tags::Interface<
71  DirectionsTag,
72  db::add_tag_prefix<::Tags::NormalDotFlux, variables_tag>>;
73 
74  db::mutate_apply<
75  tmpl::list<interface_normal_dot_fluxes_tag>,
76  tmpl::push_front<tmpl::transform<typename Metavariables::system::
77  normal_dot_fluxes::argument_tags,
78  tmpl::bind<domain::Tags::Interface,
79  DirectionsTag, tmpl::_1>>,
80  DirectionsTag>>(
81  [](const gsl::not_null<typename interface_normal_dot_fluxes_tag::type*>
82  boundary_fluxes,
83  const std::unordered_set<Direction<Metavariables::volume_dim>>&
84  internal_directions,
85  const auto&... tensors) noexcept {
86  for (const auto& direction : internal_directions) {
87  // Prepending the type name works around an issue with gcc-6
88  ComputeNonconservativeBoundaryFluxes::apply_flux<Metavariables>(
89  make_not_null(&boundary_fluxes->at(direction)),
90  tensors.at(direction)...);
91  }
92  return boundary_fluxes;
93  },
94  make_not_null(&box));
95 
96  return std::forward_as_tuple(std::move(box));
97  }
98 };
99 } // namespace Actions
100 } // namespace dg
