82 /// \ingroup DiscontinuousGalerkinGroup
83 /// Project variables from a face to a mortar.
84 template <typename Tags, size_t Dim>
85 Variables<Tags> project_to_mortar(const Variables<Tags>& vars,
86  const Mesh<Dim>& face_mesh,
87  const Mesh<Dim>& mortar_mesh,
88  const MortarSize<Dim>& mortar_size) noexcept {
89  const Matrix identity{};
90  auto projection_matrices = make_array<Dim>(std::cref(identity));
91 
92  const auto face_slice_meshes = face_mesh.slices();
93  const auto mortar_slice_meshes = mortar_mesh.slices();
94  for (size_t i = 0; i < Dim; ++i) {
95  const auto& face_slice_mesh = gsl::at(face_slice_meshes, i);
96  const auto& mortar_slice_mesh = gsl::at(mortar_slice_meshes, i);
97  const auto& slice_size = gsl::at(mortar_size, i);
98  if (slice_size != Spectral::MortarSize::Full or
99  face_slice_mesh != mortar_slice_mesh) {
100  gsl::at(projection_matrices, i) = projection_matrix_element_to_mortar(
101  slice_size, mortar_slice_mesh, face_slice_mesh);
102  }
103  }
104  return apply_matrices(projection_matrices, vars, face_mesh.extents());
105 }

107 /// \ingroup DiscontinuousGalerkinGroup
108 /// Project variables from a mortar to a face.
109 template <typename Tags, size_t Dim>
110 Variables<Tags> project_from_mortar(
111  const Variables<Tags>& vars, const Mesh<Dim>& face_mesh,
112  const Mesh<Dim>& mortar_mesh, const MortarSize<Dim>& mortar_size) noexcept {
113  ASSERT(face_mesh != mortar_mesh or
114  alg::any_of(mortar_size,
115  [](const Spectral::MortarSize& size) noexcept {
116  return size != Spectral::MortarSize::Full;
117  }),
118  "project_from_mortar should not be called if the interface mesh and "
119  "mortar mesh are identical. Please elide the copy instead.");
120 
121  const Matrix identity{};
122  auto projection_matrices = make_array<Dim>(std::cref(identity));
123 
124  const auto face_slice_meshes = face_mesh.slices();
125  const auto mortar_slice_meshes = mortar_mesh.slices();
126  for (size_t i = 0; i < Dim; ++i) {
127  const auto& face_slice_mesh = gsl::at(face_slice_meshes, i);
128  const auto& mortar_slice_mesh = gsl::at(mortar_slice_meshes, i);
129  const auto& slice_size = gsl::at(mortar_size, i);
130  if (slice_size != Spectral::MortarSize::Full or
131  face_slice_mesh != mortar_slice_mesh) {
132  gsl::at(projection_matrices, i) = projection_matrix_mortar_to_element(
133  slice_size, face_slice_mesh, mortar_slice_mesh);
134  }
135  }
136  return apply_matrices(projection_matrices, vars, mortar_mesh.extents());
137 }
