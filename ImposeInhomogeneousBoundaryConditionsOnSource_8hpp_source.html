34 /*!
35  * \brief Adds boundary contributions to the sources
36  *
37  * Imposes boundary conditions by adding contributions to the sources of the
38  * elliptic equations. With the source modified, we may then assume homogeneous
39  * (i.e. zero) Dirichlet boundary conditions throughout the elliptic solve.
40  *
41  * \note Only Dirichlet boundary conditions retrieved from the analytic solution
42  * are currently supported.
43  *
44  * With:
45  * - `fixed_sources_tag` = `db::add_tag_prefix<Tags::FixedSource, fields_tag>`
46  *
47  * Uses:
48  * - Metavariables:
49  * - `analytic_solution_tag`
50  * - `normal_dot_numerical_flux`
51  * - System:
52  * - `volume_dim`
53  * - `fields_tag`
54  * - `primal_fields`
55  * - DataBox:
56  * - `Tags::Mesh<volume_dim>`
57  * - `Tags::Coordinates<volume_dim, Frame::Inertial>`
58  * - `Tags::BoundaryDirectionsInterior<volume_dim>`
59  * - `Tags::Interface<Tags::BoundaryDirectionsExterior<volume_dim>,
60  * Tags::Coordinates<volume_dim, Frame::Inertial>>`
61  * - `Tags::Interface<Tags::BoundaryDirectionsInterior<volume_dim>,
62  * Tags::Normalized<Tags::UnnormalizedFaceNormal<volume_dim>>>`
63  * - `Tags::Interface<Tags::BoundaryDirectionsInterior<volume_dim>,
64  * Tags::Magnitude<Tags::UnnormalizedFaceNormal<volume_dim>>>`
65  *
66  * DataBox:
67  * - Modifies:
68  * - `fixed_sources_tag`
69  */
70 template <typename Metavariables>
71 struct ImposeInhomogeneousBoundaryConditionsOnSource {
72  using system = typename Metavariables::system;
73  static constexpr size_t volume_dim = system::volume_dim;
74  using FluxesType = typename system::fluxes;
75  using fluxes_computer_tag = elliptic::Tags::FluxesComputer<FluxesType>;
76 
77  using fixed_sources_tag =
78  db::add_tag_prefix<::Tags::FixedSource, typename system::fields_tag>;
79 
80  template <typename NormalDotNumericalFluxComputer,
81  typename... NumericalFluxTags, typename... BoundaryDataTags,
82  typename... FluxesArgs>
83  static void compute_dirichlet_boundary_normal_dot_numerical_flux(
84  const gsl::not_null<Variables<tmpl::list<NumericalFluxTags...>>*>
85  numerical_fluxes,
86  const NormalDotNumericalFluxComputer& normal_dot_numerical_flux_computer,
87  const Variables<tmpl::list<BoundaryDataTags...>>& boundary_data,
88  const Mesh<volume_dim>& volume_mesh,
89  const Direction<volume_dim>& direction,
90  const tnsr::i<DataVector, volume_dim, Frame::Inertial>&
91  normalized_face_normal,
92  const Scalar<DataVector>& face_normal_magnitude,
93  const FluxesType& fluxes_computer,
94  const FluxesArgs&... fluxes_args) noexcept {
95  normal_dot_numerical_flux_computer.compute_dirichlet_boundary(
96  make_not_null(&get<NumericalFluxTags>(*numerical_fluxes))...,
97  get<BoundaryDataTags>(boundary_data)..., volume_mesh, direction,
98  normalized_face_normal, face_normal_magnitude, fluxes_computer,
99  fluxes_args...);
100  }
101 
102  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
103  typename ActionList, typename ParallelComponent>
104  static std::tuple<db::DataBox<DbTagsList>&&> apply(
105  db::DataBox<DbTagsList>& box,
106  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
107  const Parallel::GlobalCache<Metavariables>& cache,
108  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
109  const ParallelComponent* const /*meta*/) noexcept {
110  const auto& analytic_solution =
111  Parallel::get<typename Metavariables::analytic_solution_tag>(cache);
112  const auto& normal_dot_numerical_flux_computer =
113  Parallel::get<typename Metavariables::normal_dot_numerical_flux>(cache);
114 
115  // Compute the contribution to the fixed sources from all Dirichlet
116  // boundaries
117  auto dirichlet_boundary_contributions = interface_apply<
118  domain::Tags::BoundaryDirectionsExterior<volume_dim>,
119  tmpl::push_front<
120  typename FluxesType::argument_tags, domain::Tags::Mesh<volume_dim>,
121  domain::Tags::Mesh<volume_dim - 1>,
122  domain::Tags::Direction<volume_dim>,
123  domain::Tags::Coordinates<volume_dim, Frame::Inertial>,
124  ::Tags::Normalized<
125  domain::Tags::UnnormalizedFaceNormal<volume_dim>>,
126  ::Tags::Magnitude<domain::Tags::UnnormalizedFaceNormal<volume_dim>>,
127  fluxes_computer_tag>,
128  tmpl::push_front<get_volume_tags<FluxesType>,
129  domain::Tags::Mesh<volume_dim>, fluxes_computer_tag>>(
130  [&analytic_solution, &normal_dot_numerical_flux_computer](
131  const Mesh<volume_dim>& volume_mesh,
132  const Mesh<volume_dim - 1>& face_mesh,
133  const Direction<volume_dim>& direction,
134  const tnsr::I<DataVector, volume_dim, Frame::Inertial>&
135  boundary_coordinates,
136  tnsr::i<DataVector, volume_dim> normalized_face_normal,
137  const Scalar<DataVector>& magnitude_of_face_normal,
138  const FluxesType& fluxes_computer, const auto&... fluxes_args) {
139  const size_t dimension = direction.dimension();
140  // We get the exterior face normal out of the box, so we flip its sign
141  // to obtain the interior face normal
142  for (size_t d = 0; d < volume_dim; d++) {
143  normalized_face_normal.get(d) *= -1.;
144  }
145  // Compute Dirichlet data on mortar
146  Variables<typename system::primal_fields> dirichlet_boundary_data{
147  face_mesh.number_of_grid_points(), 0.};
148  dirichlet_boundary_data.assign_subset(analytic_solution.variables(
149  boundary_coordinates, typename system::primal_fields{}));
150  // Compute the numerical flux contribution from the Dirichlet data
151  typename db::add_tag_prefix<::Tags::NormalDotNumericalFlux,
152  fixed_sources_tag>::type
153  boundary_normal_dot_numerical_fluxes{
154  face_mesh.number_of_grid_points(), 0.};
155  compute_dirichlet_boundary_normal_dot_numerical_flux(
156  make_not_null(&boundary_normal_dot_numerical_fluxes),
157  normal_dot_numerical_flux_computer,
158  std::move(dirichlet_boundary_data), volume_mesh, direction,
159  normalized_face_normal, magnitude_of_face_normal, fluxes_computer,
160  fluxes_args...);
161  // Flip sign of the boundary contributions, making them
162  // contributions to the source
163  return typename fixed_sources_tag::type{
164  -1. *
165  ::dg::lift_flux(std::move(boundary_normal_dot_numerical_fluxes),
166  volume_mesh.extents(dimension),
167  magnitude_of_face_normal)};
168  },
169  box);
170 
171  // Add the boundary contributions to the fixed sources
172  db::mutate<fixed_sources_tag>(
173  make_not_null(&box),
174  [&dirichlet_boundary_contributions](
175  const gsl::not_null<typename fixed_sources_tag::type*>
176  fixed_sources,
177  const Mesh<volume_dim>& mesh,
178  const std::unordered_set<::Direction<volume_dim>>&
179  directions) noexcept {
180  for (const auto& direction : directions) {
181  add_slice_to_data(
182  fixed_sources,
183  std::move(dirichlet_boundary_contributions.at(direction)),
184  mesh.extents(), direction.dimension(),
185  index_to_slice_at(mesh.extents(), direction));
186  }
187  },
188  get<domain::Tags::Mesh<volume_dim>>(box),
189  get<domain::Tags::BoundaryDirectionsInterior<volume_dim>>(box));
190 
191  return {std::move(box)};
192  }
193 };

195 } // namespace dg
