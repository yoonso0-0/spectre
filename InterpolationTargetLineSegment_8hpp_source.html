119 /// \brief Computes points on a line segment.
120 ///
121 /// For requirements on InterpolationTargetTag, see InterpolationTarget
122 template <typename InterpolationTargetTag, size_t VolumeDim>
123 struct LineSegment {
124  using const_global_cache_tags =
125  tmpl::list<Tags::LineSegment<InterpolationTargetTag, VolumeDim>>;
126  using is_sequential = std::false_type;
127 
128  template <typename Metavariables, typename DbTags>
129  static tnsr::I<DataVector, VolumeDim, Frame::Inertial> points(
130  const db::DataBox<DbTags>& box,
131  const tmpl::type_<Metavariables>& /*meta*/) noexcept {
132  const auto& options =
133  get<Tags::LineSegment<InterpolationTargetTag, VolumeDim>>(box);
134 
135  // Fill points on a line segment
136  const double fractional_distance = 1.0 / (options.number_of_points - 1);
137  tnsr::I<DataVector, VolumeDim, Frame::Inertial> target_points(
138  options.number_of_points);
139  for (size_t n = 0; n < options.number_of_points; ++n) {
140  for (size_t d = 0; d < VolumeDim; ++d) {
141  target_points.get(d)[n] =
142  gsl::at(options.begin, d) +
143  n * fractional_distance *
144  (gsl::at(options.end, d) - gsl::at(options.begin, d));
145  }
146  }
147  return target_points;
148  }
149 
150  template <typename Metavariables, typename DbTags, typename TemporalId>
151  static tnsr::I<DataVector, VolumeDim, Frame::Inertial> points(
152  const db::DataBox<DbTags>& box, const tmpl::type_<Metavariables>& meta,
153  const TemporalId& /*temporal_id*/) noexcept {
154  return points(box, meta);
155  }
156 };
