20 /// Data structures holding quantities that are interpolated by
21 /// `Interpolator` for use by `InterpolationTarget`s
22 namespace Vars {
23 /// \brief Holds a `Variables` interpolated onto a list of points, and
24 /// information about those points, for a local `Interpolator`.
25 ///
26 /// `TagList` is a `tmpl::list` of tags that go into the `Variables`.
27 template <size_t VolumeDim, typename TagList>
28 struct Info {
29  /// `block_coord_holders` holds the list of all points (in block
30  /// logical coordinates) that need to be interpolated onto for a
31  /// given `InterpolationTarget`.
32  ///
33  /// The number of interpolated points for which results are stored
34  /// in this `Info` (in `vars` and `global_offsets` below)
35  /// corresponds to only the subset of the points in
36  /// `block_coord_holders` that are contained in local `Element`s.
37  /// Moreover, the number of interpolated points stored in this
38  /// `Info` will change as more `Elements` send data to this
39  /// `Interpolator`, and will be less than or equal to the size of
40  /// `block_coord_holders` even after all `Element`s have sent their
41  /// data (this is because this `Info` lives only on a single core,
42  /// and this core will have access only to the local `Element`s).
43  std::vector<boost::optional<IdPair<
44  domain::BlockId, tnsr::I<double, VolumeDim, typename ::Frame::Logical>>>>
45  block_coord_holders;
46  /// `vars` holds the interpolated `Variables` on some subset of the
47  /// points in `block_coord_holders`. The grid points inside vars
48  /// are indexed according to `global_offsets` below. The size of
49  /// `vars` changes as more `Element`s send data to this `Interpolator`.
50  std::vector<Variables<TagList>> vars{};
51  /// `global_offsets[j][i]` is the index into `block_coord_holders` that
52  /// corresponds to the index `i` of the `DataVector` held in `vars[j]`.
53  /// The size of `global_offsets` changes as more `Element`s
54  /// send data to this `Interpolator`.
55  std::vector<std::vector<size_t>> global_offsets{};
56  /// Holds the `ElementId`s of `Element`s for which interpolation has
57  /// already been done for this `Info`.
58  std::unordered_set<ElementId<VolumeDim>>
59  interpolation_is_done_for_these_elements{};
60 };
61 
62 template <size_t VolumeDim, typename TagList>
63 void pup(PUP::er& p, Info<VolumeDim, TagList>& t) noexcept { // NOLINT
64  p | t.block_coord_holders;
65  p | t.vars;
66  p | t.global_offsets;
67  p | t.interpolation_is_done_for_these_elements;
68 }
69 
70 template <size_t VolumeDim, typename TagList>
71 void operator|(PUP::er& p, Info<VolumeDim, TagList>& t) noexcept { // NOLINT
72  pup(p, t);
73 }

75 /// Holds `Info`s at all `temporal_id`s for a given
76 /// `InterpolationTargetTag`. Also holds `temporal_id`s when data has
77 /// been interpolated; this is used for cleanup purposes. All
78 /// `Holder`s for all `InterpolationTargetTags` are held in a single
79 /// `TaggedTuple` that is in the `Interpolator`'s `DataBox` with the
80 /// tag `Tags::InterpolatedVarsHolders`.
81 template <typename Metavariables,
82  typename InterpolationTargetTag, typename TagList>
83 struct Holder {
84  std::unordered_map<typename Metavariables::temporal_id::type,
85  Info<Metavariables::volume_dim, TagList>>
86  infos;
87  std::unordered_set<typename Metavariables::temporal_id::type>
88  temporal_ids_when_data_has_been_interpolated;
89 };
90 
91 template <typename Metavariables, typename InterpolationTargetTag,
92  typename TagList>
93 void pup(PUP::er& p, // NOLINT
94  Holder<Metavariables, InterpolationTargetTag, TagList>& // NOLINT
95  t) noexcept { // NOLINT
96  p | t.infos;
97  p | t.temporal_ids_when_data_has_been_interpolated;
98 }
99 
100 template <typename Metavariables, typename InterpolationTargetTag,
101  typename TagList>
102 void operator|(PUP::er& p, // NOLINT
103  Holder<Metavariables, InterpolationTargetTag, TagList>&
104  t) noexcept { // NOLINT
105  pup(p, t);
106 }

108 /// Indexes a particular `Holder` in the `TaggedTuple` that is
109 /// accessed from the `Interpolator`'s `DataBox` with tag
110 /// `Tags::InterpolatedVarsHolders`.
111 template <typename InterpolationTargetTag, typename Metavariables>
112 struct HolderTag {
113  using type =
114  Holder<Metavariables, InterpolationTargetTag,
115  typename InterpolationTargetTag::vars_to_interpolate_to_target>;
116 };
