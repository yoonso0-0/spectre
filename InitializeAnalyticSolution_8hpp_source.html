32 /*!
33  * \brief Place the analytic solution of the system fields in the DataBox.
34  *
35  * Uses:
36  * - DataBox:
37  * - `AnalyticSolutionTag`
38  * - `Tags::Coordinates<Dim, Frame::Inertial>`
39  *
40  * DataBox:
41  * - Adds:
42  * - `db::wrap_tags_in<::Tags::Analytic, AnalyticSolutionFields>`
43  *
44  * \note This action relies on the `SetupDataBox` aggregated initialization
45  * mechanism, so `Actions::SetupDataBox` must be present in the `Initialization`
46  * phase action list prior to this action.
47  */
48 template <typename AnalyticSolutionTag, typename AnalyticSolutionFields>
49 struct InitializeAnalyticSolution {
50  using const_global_cache_tags = tmpl::list<AnalyticSolutionTag>;
51  using analytic_fields_tag =
52  db::add_tag_prefix<::Tags::Analytic,
53  ::Tags::Variables<AnalyticSolutionFields>>;
54 
55  using simple_tags = tmpl::list<analytic_fields_tag>;
56  using compute_tags = tmpl::list<>;
57 
58  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
59  size_t Dim, typename ActionList, typename ParallelComponent>
60  static auto apply(db::DataBox<DbTagsList>& box,
61  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
62  const Parallel::GlobalCache<Metavariables>& cache,
63  const ElementId<Dim>& /*array_index*/,
64  const ActionList /*meta*/,
65  const ParallelComponent* const /*meta*/) noexcept {
66 
67  const auto& inertial_coords =
68  get<domain::Tags::Coordinates<Dim, Frame::Inertial>>(box);
69  typename analytic_fields_tag::type analytic_fields{
70  variables_from_tagged_tuple(get<AnalyticSolutionTag>(cache).variables(
71  inertial_coords, AnalyticSolutionFields{}))};
72  Initialization::mutate_assign<simple_tags>(make_not_null(&box),
73  std::move(analytic_fields));
74 
75  return std::make_tuple(std::move(box));
76  }
77 };
