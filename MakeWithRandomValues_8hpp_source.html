153 /// \ingroup TestingFrameworkGroup
154 /// \brief Make a data structure and fill it with random values
155 ///
156 /// \details Given an object of type `T`, create an object of type `ReturnType`
157 /// whose elements are initialized to random values using the given random
158 /// number generator and random number distribution.
159 ///
160 /// \requires the type `ReturnType` to be creatable using
161 /// `make_with_value<ReturnType>(T)`
162 template <typename ReturnType, typename T, typename UniformRandomBitGenerator,
163  typename RandomNumberDistribution>
164 ReturnType make_with_random_values(
165  const gsl::not_null<UniformRandomBitGenerator*> generator,
166  const gsl::not_null<RandomNumberDistribution*> distribution,
167  const T& used_for_size) noexcept {
168  auto result = make_with_value<ReturnType>(
169  used_for_size,
170  std::numeric_limits<
171  tt::get_fundamental_type_t<ReturnType>>::signaling_NaN());
172  fill_with_random_values(make_not_null(&result), generator, distribution);
173  return result;
174 }
175 // distributions are sufficiently small to justify providing a convenience
176 // function that receives them by value, which is useful when obtaining pointers
177 // is inconvenient (e.g. for distributions that are obtained as
178 // rvalues). Generators should never be copied, as doing so will cause
179 // duplication of the pseudorandom numbers and performance hits due to the
180 // nontrivial size.
181 // clang-tidy: seems to erroneously believe this is a function declaration
182 // rather than a definition.
183 template <typename ReturnType, typename T, typename UniformRandomBitGenerator,
184  typename RandomNumberDistribution>
185 ReturnType make_with_random_values(
186  const gsl::not_null<UniformRandomBitGenerator*> generator, // NOLINT
187  RandomNumberDistribution distribution, const T& used_for_size) noexcept {
188  return make_with_random_values<ReturnType>(
189  generator, make_not_null(&distribution), used_for_size);
190 }
