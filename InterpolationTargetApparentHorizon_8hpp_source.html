131 /// \brief Computes points on a trial apparent horizon`.
132 ///
133 /// This differs from `KerrHorizon` in the following ways:
134 /// - It supplies points on a prolonged Strahlkorper, at a higher resolution
135 /// than the Strahlkorper in the DataBox, as needed for horizon finding.
136 /// - It uses a `FastFlow` in the DataBox.
137 /// - It has different options (including those for `FastFlow`).
138 ///
139 /// For requirements on InterpolationTargetTag, see InterpolationTarget
140 template <typename InterpolationTargetTag, typename Frame>
141 struct ApparentHorizon {
142  using const_global_cache_tags =
143  tmpl::list<Tags::ApparentHorizon<InterpolationTargetTag, Frame>>;
144  using initialization_tags =
145  tmpl::append<StrahlkorperTags::items_tags<Frame>,
146  tmpl::list<::ah::Tags::FastFlow,
147  logging::Tags::Verbosity<InterpolationTargetTag>>,
148  StrahlkorperTags::compute_items_tags<Frame>>;
149  using is_sequential = std::true_type;
150 
151  using simple_tags =
152  tmpl::push_back<StrahlkorperTags::items_tags<Frame>, ::ah::Tags::FastFlow,
153  logging::Tags::Verbosity<InterpolationTargetTag>>;
154  using compute_tags = typename StrahlkorperTags::compute_items_tags<Frame>;
155 
156  template <typename DbTags, typename Metavariables>
157  static void initialize(
158  const gsl::not_null<db::DataBox<DbTags>*> box,
159  const Parallel::GlobalCache<Metavariables>& cache) noexcept {
160  const auto& options =
161  Parallel::get<Tags::ApparentHorizon<InterpolationTargetTag, Frame>>(
162  cache);
163 
164  // Put Strahlkorper and its ComputeItems, FastFlow,
165  // and verbosity into a new DataBox.
166  Initialization::mutate_assign<simple_tags>(
167  box, options.initial_guess, options.fast_flow, options.verbosity);
168  }
169 
170  template <typename Metavariables, typename DbTags, typename TemporalId>
171  static tnsr::I<DataVector, 3, Frame> points(
172  const db::DataBox<DbTags>& box,
173  const tmpl::type_<Metavariables>& /*meta*/,
174  const TemporalId& /*temporal_id*/) noexcept {
175  const auto& fast_flow = db::get<::ah::Tags::FastFlow>(box);
176  const auto& strahlkorper =
177  db::get<StrahlkorperTags::Strahlkorper<Frame>>(box);
178 
179  const size_t L_mesh = fast_flow.current_l_mesh(strahlkorper);
180  const auto prolonged_strahlkorper =
181  Strahlkorper<Frame>(L_mesh, L_mesh, strahlkorper);
182 
183  Variables<tmpl::list<::Tags::Tempi<0, 2, ::Frame::Spherical<Frame>>,
184  ::Tags::Tempi<1, 3, Frame>, ::Tags::TempScalar<2>>>
185  temp_buffer(prolonged_strahlkorper.ylm_spherepack().physical_size());
186 
187  auto& theta_phi =
188  get<::Tags::Tempi<0, 2, ::Frame::Spherical<Frame>>>(temp_buffer);
189  auto& r_hat = get<::Tags::Tempi<1, 3, Frame>>(temp_buffer);
190  auto& radius = get(get<::Tags::TempScalar<2>>(temp_buffer));
191  StrahlkorperTags::ThetaPhiCompute<Frame>::function(
192  make_not_null(&theta_phi), prolonged_strahlkorper);
193  StrahlkorperTags::RhatCompute<Frame>::function(make_not_null(&r_hat),
194  theta_phi);
195  StrahlkorperTags::RadiusCompute<Frame>::function(make_not_null(&radius),
196  prolonged_strahlkorper);
197 
198  tnsr::I<DataVector, 3, Frame> prolonged_coords{};
199  StrahlkorperTags::CartesianCoordsCompute<Frame>::function(
200  make_not_null(&prolonged_coords), prolonged_strahlkorper, radius,
201  r_hat);
202 
203  return prolonged_coords;
204  }
205 };
