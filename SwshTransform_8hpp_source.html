38 // appends to a vector of pointers `collocation_data` the set of pointers
39 // associated with angular views of the provided collocation data `vector`. The
40 // associated views are appended into `collocation_views`. If `positive_spin` is
41 // false, this function will conjugate the views, and therefore potentially
42 // conjugate (depending on `Representation`) the input data `vectors`. This
43 // behavior is chosen to avoid frequent copies of the input data `vector` for
44 // optimization. The conjugation must be undone after the call to libsharp
45 // transforms using `conjugate_views` to restore the input data to its original
46 // state.
47 template <ComplexRepresentation Representation>
48 void append_libsharp_collocation_pointers(
49  gsl::not_null<std::vector<double*>*> collocation_data,
50  gsl::not_null<std::vector<ComplexDataView<Representation>>*>
51  collocation_views,
52  gsl::not_null<ComplexDataVector*> vector, size_t l_max,
53  bool positive_spin) noexcept;
54 
55 // Perform a conjugation on a vector of `ComplexDataView`s if `Spin` < 0. This
56 // is used for undoing the conjugation described in the code comment for
57 // `append_libsharp_collocation_pointers`
58 template <int Spin, ComplexRepresentation Representation>
59 SPECTRE_ALWAYS_INLINE void conjugate_views(
60  gsl::not_null<std::vector<ComplexDataView<Representation>>*>
61  collocation_views) noexcept {
62  if (Spin < 0) {
63  for (auto& view : *collocation_views) {
64  view.conjugate();
65  }
66  }
67 }

126 /*!
127  * \ingroup SwshGroup
128  * \brief Perform a forward libsharp spin-weighted spherical harmonic transform
129  * on any number of supplied `SpinWeighted<ComplexDataVector, Spin>`.
130  *
131  * \details This function places the result in one or more
132  * `SpinWeighted<ComplexModalVector, Spin>` returned via provided
133  * pointer. This is a simpler interface to the same functionality as the
134  * \ref DataBoxGroup mutation compatible `SwshTransform`.
135  *
136  * The following parameter ordering for the multiple input interface is enforced
137  * by the interior function calls, but is not obvious from the explicit
138  * parameter packs in this function signature:
139  *
140  * - `size_t l_max` : angular resolution for the transform
141  * - `size_t number_of_radial_points` : radial resolution (number of consecutive
142  * transforms in each of the vectors)
143  * - any number of `gsl::not_null<SpinWeighted<ComplexModalVector,
144  * Spin>*>`, all sharing the same `Spin` : The return-by-pointer containers for
145  * the transformed modal quantities
146  * - the same number of `const SpinWeighted<ComplexDataVector, Spin>&`, with the
147  * same `Spin` as the previous function argument set : The input containers of
148  * nodal spin-weighted spherical harmonic data.
149  *
150  * template parameters:
151  * - `Representation`: Either `ComplexRepresentation::Interleaved` or
152  * `ComplexRepresentation::RealsThenImags`, indicating the representation for
153  * intermediate steps of the transformation. The two representations will give
154  * identical results but may help or hurt performance depending on the task.
155  * If unspecified, defaults to `ComplexRepresentation::Interleaved`.
156  *
157  * The result is a set of libsharp-compatible coefficients.
158  * \see SwshTransform for more details on the mathematics of the libsharp
159  * data structures.
160  *
161  * \warning The collocation data is taken by const reference, but can be
162  * temporarily altered in-place during intermediate parts of the computation.
163  * The input data is guaranteed to return to its original state by the end of
164  * the function. In a setting in which multiple threads access the same data
165  * passed as input to this function, a lock must be used to prevent access
166  * during the execution of the transform.
167  */
168 template <
169  ComplexRepresentation Representation = ComplexRepresentation::Interleaved,
170  int Spin, typename... ModalThenNodalTypes>
171 void swsh_transform(
172  const size_t l_max, const size_t number_of_radial_points,
173  const gsl::not_null<SpinWeighted<ComplexModalVector, Spin>*>
174  first_coefficient,
175  const ModalThenNodalTypes&... coefficients_then_collocations) noexcept {
176  static_assert(
177  sizeof...(ModalThenNodalTypes) % 2 == 1,
178  "The number of modal arguments passed to swsh_transform must equal the "
179  "number of nodal arguments, so the total number of arguments must be "
180  "even.");
181  detail::swsh_transform_impl<Representation>(
182  l_max, number_of_radial_points,
183  std::make_index_sequence<(sizeof...(coefficients_then_collocations) + 1) /
184  2>{},
185  first_coefficient, coefficients_then_collocations...);
186 }

300 /*!
301  * \ingroup SwshGroup
302  * \brief A \ref DataBoxGroup mutate-compatible computational struct for
303  * performing several spin-weighted spherical harmonic transforms. Internally
304  * dispatches to libsharp.
305  *
306  * \details
307  * Template Parameters:
308  * - `Representation`: The element of the `ComplexRepresentation` enum which
309  * parameterizes how the data passed to libsharp will be represented. Two
310  * options are available:
311  * - `ComplexRepresentation:Interleaved`: indicates that the real and imaginary
312  * parts of the collocation values will be passed to libsharp as pointers
313  * into existing complex data structures, minimizing copies, but maintaining
314  * a stride of 2 for 'adjacent' real or imaginary values.
315  * - `ComplexRepresentation::RealsThenImags`: indicates that the real and
316  * imaginary parts of the collocation values will be passed to libsharp as
317  * separate contiguous blocks. At current, this introduces both allocations
318  * and copies. **optimization note** In the future most of the allocations
319  * can be aggregated to calling code, which would pass in buffers by
320  * `not_null` pointers.
321  *
322  * For performance-sensitive code, both options should be tested, as each
323  * strategy has trade-offs.
324  * - `TagList`: A `tmpl::list` of Tags to be forward transformed. The tags must
325  * represent the nodal data.
326  *
327  * \note The signs obtained from libsharp transformations must be handled
328  * carefully. (In particular, it does not use the sign convention you will find
329  * in [Wikipedia]
330  * (https://en.wikipedia.org/wiki/Spin-weighted_spherical_harmonics)).
331  * - libsharp deals only with the transformation of real values, so
332  * transformation of complex values must be done for real and imaginary parts
333  * separately.
334  * - due to only transforming real components, it stores only the positive
335  * \f$m\f$ modes (as the rest would be redundant). Therefore, the negative
336  * \f$m\f$ modes must be inferred from conjugation and further sign changes.
337  * - libsharp only has the capability of transforming positive spin weighted
338  * quantities. Therefore, additional steps are taken (involving further
339  * conjugation of the provided data) in order to use those utilities on
340  * negative spin-weighted quantities. The resulting modes have yet more sign
341  * changes that must be taken into account.
342  *
343  * The decomposition resulting from the libsharp transform for spin \f$s\f$ and
344  * complex spin-weighted \f${}_s f\f$ can be represented mathematically as:
345  *
346  * \f{align*}{
347  * {}_s f(\theta, \phi) = \sum_{\ell = 0}^{\ell_\mathrm{max}} \Bigg\{&
348  * \left(\sum_{m = 0}^{\ell} a^\mathrm{sharp, real}_{l m} {}_s Y_{\ell
349  * m}^\mathrm{sharp, real}\right) + \left(\sum_{m=1}^{\ell}
350  * \left(a^\mathrm{sharp, real}_{l m}{}\right)^*
351  * {}_s Y_{\ell -m}^\mathrm{sharp, real}\right) \notag\\
352  * &+ i \left[\left(\sum_{m = 0}^{\ell} a^\mathrm{sharp, imag}_{l m} {}_s
353  * Y_{\ell m}^\mathrm{sharp, imag}\right) + \left(\sum_{m=1}^{\ell}
354  * \left(a^\mathrm{sharp, imag}_{l m}{}\right)^* {}_s Y_{\ell -m}^\mathrm{sharp,
355  * imag} \right)\right] \Bigg\},
356  * \f}
357  *
358  * where
359  *
360  * \f{align*}{
361  * {}_s Y_{\ell m}^\mathrm{sharp, real} &=
362  * \begin{cases}
363  * (-1)^{s + 1} {}_s Y_{\ell m}, & \mathrm{for}\; s < 0, m \ge 0 \\
364  * {}_s Y_{\ell m}, & \mathrm{for}\; s = 0, m \ge 0 \\
365  * - {}_s Y_{\ell m}, & \mathrm{for}\; s > 0, m \ge 0 \\
366  * (-1)^{s + m + 1} {}_s Y_{\ell m}, & \mathrm{for}\; s < 0, m < 0 \\
367  * (-1)^{m} {}_s Y_{\ell m}, & \mathrm{for}\; s = 0, m < 0 \\
368  * (-1)^{m + 1} {}_s Y_{\ell m}, & \mathrm{for}\; s > 0, m < 0
369  * \end{cases} \\
370  * &\equiv {}_s S_{\ell m}^{\mathrm{real}} {}_s Y_{\ell m}\\
371  * {}_s Y_{\ell m}^\mathrm{sharp, imag} &=
372  * \begin{cases}
373  * (-1)^{s + 1} {}_s Y_{\ell m}, & \mathrm{for}\; s < 0, m \ge 0 \\
374  * -{}_s Y_{\ell m}, & \mathrm{for}\; s = 0, m \ge 0 \\
375  * {}_s Y_{\ell m}, & \mathrm{for}\; s > 0, m \ge 0 \\
376  * (-1)^{s + m} {}_s Y_{\ell m}, & \mathrm{for}\; s < 0, m < 0 \\
377  * (-1)^{m} {}_s Y_{\ell m}, & \mathrm{for}\; s = 0, m < 0 \\
378  * (-1)^{m + 1} {}_s Y_{\ell m}, & \mathrm{for}\; s > 0, m < 0
379  * \end{cases} \\
380  * &\equiv {}_s S_{\ell m}^{\mathrm{real}} {}_s Y_{\ell m},
381  * \f}
382  *
383  * where the unadorned \f${}_s Y_{\ell m}\f$ on the right-hand-sides follow the
384  * (older) convention of \cite Goldberg1966uu. Note the phase in these
385  * expressions is not completely standardized, so should be checked carefully
386  * whenever coefficient data is directly manipulated.
387  *
388  * For reference, we can compute the relation between Goldberg spin-weighted
389  * moments \f${}_s f_{\ell m}\f$, defined as
390  *
391  * \f[ {}_s f(\theta, \phi) = \sum_{\ell = 0}^{\ell_\mathrm{max}} \sum_{m =
392  * -\ell}^{\ell} {}_s f_{\ell m} {}_s Y_{\ell m}
393  * \f]
394  *
395  * so,
396  * \f[
397  * {}_s f_{\ell m} =
398  * \begin{cases}
399  * a_{\ell m}^{\mathrm{sharp}, \mathrm{real}} {}_s S_{\ell m}^{\mathrm{real}} +
400  * a_{\ell m}^{\mathrm{sharp}, \mathrm{imag}} {}_s S_{\ell m}^{\mathrm{imag}} &
401  * \mathrm{for} \; m \ge 0 \\
402  * \left(a_{\ell -m}^{\mathrm{sharp}, \mathrm{real}}\right)^* {}_s S_{\ell
403  * m}^{\mathrm{real}} + \left(a_{\ell -m}^{\mathrm{sharp},
404  * \mathrm{imag}}\right)^* {}_s S_{\ell m}^{\mathrm{imag}} &
405  * \mathrm{for} \; m < 0 \\
406  * \end{cases} \f]
407  *
408  *
409  * The resulting coefficients \f$a_{\ell m}\f$ are stored in a triangular,
410  * \f$\ell\f$-varies-fastest configuration. So, for instance, the first
411  * \f$\ell_\mathrm{max}\f$ entries contain the coefficients for \f$m=0\f$ and
412  * all \f$\ell\f$s, and the next \f$\ell_\mathrm{max} - 1\f$ entries contain the
413  * coefficients for \f$m=1\f$ and \f$1 \le \ell \le \ell_\mathrm{max} \f$, and
414  * so on.
415  */
416 template <typename TagList, ComplexRepresentation Representation =
417  ComplexRepresentation::Interleaved>
418 struct SwshTransform;

552 template <typename... TransformTags, ComplexRepresentation Representation>
553 void SwshTransform<tmpl::list<TransformTags...>, Representation>::
554  apply_to_vectors(const gsl::not_null<typename Tags::SwshTransform<
555  TransformTags>::type::type*>... coefficients,
556  const typename TransformTags::type::type&... collocations,
557  const size_t l_max,
558  const size_t number_of_radial_points) noexcept {
559  EXPAND_PACK_LEFT_TO_RIGHT(coefficients->destructive_resize(
560  size_of_libsharp_coefficient_vector(l_max) * number_of_radial_points));
561 
562  // assemble a list of pointers into the collocation point data. This is
563  // required because libsharp expects pointers to pointers.
564  std::vector<detail::ComplexDataView<Representation>> pre_transform_views;
565  pre_transform_views.reserve(number_of_radial_points *
566  sizeof...(TransformTags));
567  std::vector<double*> pre_transform_collocation_data;
568  pre_transform_collocation_data.reserve(2 * number_of_radial_points *
569  sizeof...(TransformTags));
570 
571  // clang-tidy: const-cast, object is temporarily modified and returned to
572  // original state
573  EXPAND_PACK_LEFT_TO_RIGHT(detail::append_libsharp_collocation_pointers(
574  make_not_null(&pre_transform_collocation_data),
575  make_not_null(&pre_transform_views),
576  make_not_null(&const_cast<typename TransformTags::type::type&>( // NOLINT
577  collocations)
578  .data()),
579  l_max, spin >= 0));
580 
581  std::vector<std::complex<double>*> post_transform_coefficient_data;
582  post_transform_coefficient_data.reserve(2 * number_of_radial_points *
583  sizeof...(TransformTags));
584 
585  EXPAND_PACK_LEFT_TO_RIGHT(detail::append_libsharp_coefficient_pointers(
586  make_not_null(&post_transform_coefficient_data),
587  make_not_null(&coefficients->data()), l_max));
588 
589  const size_t num_transforms =
590  (spin == 0 ? 2 : 1) * number_of_radial_points * sizeof...(TransformTags);
591  const auto* collocation_metadata =
592  &cached_collocation_metadata<Representation>(l_max);
593  const auto* alm_info =
594  cached_coefficients_metadata(l_max).get_sharp_alm_info();
595 
596  detail::execute_libsharp_transform_set(
597  SHARP_MAP2ALM, spin, make_not_null(&post_transform_coefficient_data),
598  make_not_null(&pre_transform_collocation_data),
599  make_not_null(collocation_metadata), alm_info, num_transforms);
600 
601  detail::conjugate_views<spin>(make_not_null(&pre_transform_views));
602 }

604 template <typename... TransformTags, ComplexRepresentation Representation>
605 void InverseSwshTransform<tmpl::list<TransformTags...>, Representation>::
606  apply_to_vectors(const gsl::not_null<
607  typename TransformTags::type::type*>... collocations,
608  const typename Tags::SwshTransform<
609  TransformTags>::type::type&... coefficients,
610  const size_t l_max,
611  const size_t number_of_radial_points) noexcept {
612  EXPAND_PACK_LEFT_TO_RIGHT(collocations->destructive_resize(
613  number_of_swsh_collocation_points(l_max) * number_of_radial_points));
614 
615  std::vector<std::complex<double>*> pre_transform_coefficient_data;
616  pre_transform_coefficient_data.reserve(2 * number_of_radial_points *
617  sizeof...(TransformTags));
618  // clang-tidy: const-cast, object is temporarily modified and returned to
619  // original state
620  EXPAND_PACK_LEFT_TO_RIGHT(detail::append_libsharp_coefficient_pointers(
621  make_not_null(&pre_transform_coefficient_data),
622  make_not_null(&const_cast<typename Tags::SwshTransform< // NOLINT
623  TransformTags>::type::type&>(coefficients)
624  .data()),
625  l_max));
626 
627  std::vector<detail::ComplexDataView<Representation>> post_transform_views;
628  post_transform_views.reserve(number_of_radial_points *
629  sizeof...(TransformTags));
630  std::vector<double*> post_transform_collocation_data;
631  post_transform_collocation_data.reserve(2 * number_of_radial_points *
632  sizeof...(TransformTags));
633 
634  EXPAND_PACK_LEFT_TO_RIGHT(detail::append_libsharp_collocation_pointers(
635  make_not_null(&post_transform_collocation_data),
636  make_not_null(&post_transform_views),
637  make_not_null(&collocations->data()), l_max, true));
638 
639  // libsharp considers two arrays per transform when spin is not zero.
640  const size_t num_transforms =
641  (spin == 0 ? 2 : 1) * number_of_radial_points * sizeof...(TransformTags);
642  const auto* collocation_metadata =
643  &cached_collocation_metadata<Representation>(l_max);
644  const auto* alm_info =
645  cached_coefficients_metadata(l_max).get_sharp_alm_info();
646 
647  detail::execute_libsharp_transform_set(
648  SHARP_ALM2MAP, spin, make_not_null(&pre_transform_coefficient_data),
649  make_not_null(&post_transform_collocation_data),
650  make_not_null(collocation_metadata), alm_info, num_transforms);
651 
652  detail::conjugate_views<spin>(make_not_null(&post_transform_views));
653 
654  // The inverse transformed collocation data has just been placed in the
655  // memory blocks controlled by the `ComplexDataView`s. Finally, that data
656  // must be flushed back to the Variables.
657  for (auto& view : post_transform_views) {
658  view.copy_back_to_source();
659  }
660 }
