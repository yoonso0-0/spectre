35 template <typename FieldsTag, typename OptionsGroup, typename BroadcastTarget>
36 struct CheckResidualMagnitude {
37  using fields_tag = FieldsTag;
38  using residual_tag =
39  db::add_tag_prefix<NonlinearSolver::Tags::Residual, fields_tag>;
40  using residual_magnitude_square_tag =
41  LinearSolver::Tags::MagnitudeSquare<residual_tag>;
42  using initial_residual_magnitude_tag =
43  ::Tags::Initial<LinearSolver::Tags::Magnitude<residual_tag>>;
44  using prev_residual_magnitude_square_tag =
45  NonlinearSolver::Tags::Globalization<residual_magnitude_square_tag>;
46 
47  template <typename ParallelComponent, typename DataBox,
48  typename Metavariables, typename ArrayIndex, typename... Args>
49  static void apply(DataBox& box, Parallel::GlobalCache<Metavariables>& cache,
50  const ArrayIndex& /*array_index*/,
51  Args&&... args) noexcept {
52  if constexpr (db::tag_is_retrievable_v<residual_magnitude_square_tag,
53  DataBox>) {
54  apply_impl(box, cache, std::forward<Args>(args)...);
55  } else {
56  ERROR(
57  "The residual monitor is not yet initialized. This is a bug, so "
58  "please file an issue.");
59  }
60  }
61 
62  private:
63  template <typename DbTagsList, typename Metavariables>
64  static void apply_impl(db::DataBox<DbTagsList>& box,
65  Parallel::GlobalCache<Metavariables>& cache,
66  const size_t iteration_id,
67  const size_t globalization_iteration_id,
68  const double next_residual_magnitude_square,
69  const double step_length) noexcept {
70  const double residual_magnitude = sqrt(next_residual_magnitude_square);
71 
72  NonlinearSolver::observe_detail::contribute_to_reduction_observer<
73  OptionsGroup>(iteration_id, globalization_iteration_id,
74  residual_magnitude, step_length, cache);
75 
76  if (UNLIKELY(iteration_id == 0)) {
77  db::mutate<initial_residual_magnitude_tag>(
78  make_not_null(&box),
79  [residual_magnitude](const gsl::not_null<double*>
80  initial_residual_magnitude) noexcept {
81  *initial_residual_magnitude = sqrt(residual_magnitude);
82  });
83  } else {
84  // Make sure we are converging. Far away from the solution the correction
85  // determined by the linear solve might be bad, so we employ a
86  // globalization strategy to guide the solver towards the solution when
87  // the residual doesn't decrease sufficiently. See the `NewtonRaphson`
88  // class documentation for details.
89  const double sufficient_decrease =
90  get<NonlinearSolver::Tags::SufficientDecrease<OptionsGroup>>(box);
91  const double residual_magnitude_square =
92  get<residual_magnitude_square_tag>(box);
93  const double initial_residual_magnitude =
94  get<initial_residual_magnitude_tag>(box);
95  const double abs_tolerance =
96  get<Convergence::Tags::Criteria<OptionsGroup>>(box).absolute_residual;
97  const double rel_tolerance =
98  get<Convergence::Tags::Criteria<OptionsGroup>>(box).relative_residual;
99  // This is the directional derivative of the residual magnitude square
100  // f(x) = |r(x)|^2 in the descent direction
101  const double residual_magnitude_square_slope =
102  -2. * residual_magnitude_square;
103  // Check the sufficient decrease condition. Also make sure the residual
104  // didn't hit the tolerance.
105  if (residual_magnitude > abs_tolerance and
106  residual_magnitude / initial_residual_magnitude > rel_tolerance and
107  next_residual_magnitude_square >
108  residual_magnitude_square + sufficient_decrease * step_length *
109  residual_magnitude_square_slope) {
110  // The residual didn't sufficiently decrease. Perform a globalization
111  // step.
112  if (globalization_iteration_id <
113  get<NonlinearSolver::Tags::MaxGlobalizationSteps<OptionsGroup>>(
114  box)) {
115  const double next_step_length = std::clamp(
116  NonlinearSolver::newton_raphson::next_step_length(
117  globalization_iteration_id, step_length,
118  get<NonlinearSolver::Tags::StepLength<OptionsGroup>>(box),
119  residual_magnitude_square, residual_magnitude_square_slope,
120  next_residual_magnitude_square,
121  get<prev_residual_magnitude_square_tag>(box)),
122  0.1 * step_length, 0.5 * step_length);
123  db::mutate<NonlinearSolver::Tags::StepLength<OptionsGroup>,
124  prev_residual_magnitude_square_tag>(
125  make_not_null(&box),
126  [step_length, next_residual_magnitude_square](
127  const gsl::not_null<double*> prev_step_length,
128  const gsl::not_null<double*>
129  prev_residual_magnitude_square) noexcept {
130  *prev_step_length = step_length;
131  *prev_residual_magnitude_square =
132  next_residual_magnitude_square;
133  });
134  // Do some logging
135  if (UNLIKELY(get<logging::Tags::Verbosity<OptionsGroup>>(box) >=
136  ::Verbosity::Verbose)) {
137  Parallel::printf(
138  "Step with length %s didn't sufficiently decrease the '" +
139  Options::name<OptionsGroup>() +
140  "' iteration %zu residual (possible overshoot). Residual: "
141  "%s. Next step length: %s.\n",
142  get_output(step_length), iteration_id,
143  get_output(residual_magnitude), get_output(next_step_length));
144  if (UNLIKELY(get<logging::Tags::Verbosity<OptionsGroup>>(box) >=
145  ::Verbosity::Debug)) {
146  Parallel::printf("Residual magnitude slope: %s\n",
147  get_output(residual_magnitude_square_slope));
148  }
149  }
150  // Broadcast back to the elements signaling that they should perform a
151  // globalization step, then return early.
152  Parallel::receive_data<Tags::GlobalizationResult<OptionsGroup>>(
153  Parallel::get_parallel_component<BroadcastTarget>(cache),
154  iteration_id,
155  std::variant<double, Convergence::HasConverged>{
156  next_step_length});
157  return;
158  } else if (UNLIKELY(get<logging::Tags::Verbosity<OptionsGroup>>(box) >=
159  ::Verbosity::Quiet)) {
160  Parallel::printf(
161  "WARNING: Failed to sufficiently decrease the '" +
162  Options::name<OptionsGroup>() +
163  "' iteration %zu residual in %zu globalization steps. This "
164  "is usually indicative of an ill-posed problem, for "
165  "example when the linearization of the nonlinear operator "
166  "is not computed correctly.",
167  iteration_id, globalization_iteration_id);
168  } // min_step_length
169  } // sufficient decrease condition
170  } // initial iteration
171 
172  db::mutate<residual_magnitude_square_tag>(
173  make_not_null(&box), [next_residual_magnitude_square](
174  const gsl::not_null<double*>
175  local_residual_magnitude_square) noexcept {
176  *local_residual_magnitude_square = next_residual_magnitude_square;
177  });
178 
179  // At this point, the iteration is complete. We proceed with logging and
180  // checking convergence before broadcasting back to the elements.
181 
182  // Determine whether the nonlinear solver has converged
183  Convergence::HasConverged has_converged{
184  get<Convergence::Tags::Criteria<OptionsGroup>>(box), iteration_id,
185  residual_magnitude, get<initial_residual_magnitude_tag>(box)};
186 
187  // Do some logging
188  if (UNLIKELY(get<logging::Tags::Verbosity<OptionsGroup>>(box) >=
189  ::Verbosity::Quiet)) {
190  if (UNLIKELY(iteration_id == 0)) {
191  Parallel::printf("Nonlinear solver '" + Options::name<OptionsGroup>() +
192  "' initialized with residual %e.\n",
193  residual_magnitude);
194  } else {
195  Parallel::printf("Nonlinear solver '" + Options::name<OptionsGroup>() +
196  "' iteration %zu done. Remaining residual: %e\n",
197  iteration_id, residual_magnitude);
198  }
199  }
200  if (UNLIKELY(has_converged and get<logging::Tags::Verbosity<OptionsGroup>>(
201  box) >= ::Verbosity::Quiet)) {
202  if (UNLIKELY(iteration_id == 0)) {
203  Parallel::printf("The nonlinear solver '" +
204  Options::name<OptionsGroup>() +
205  "' has converged without any iterations: %s\n",
206  has_converged);
207  } else {
208  Parallel::printf("The nonlinear solver '" +
209  Options::name<OptionsGroup>() +
210  "' has converged in %zu iterations: %s\n",
211  iteration_id, has_converged);
212  }
213  }
214 
215  Parallel::receive_data<Tags::GlobalizationResult<OptionsGroup>>(
216  Parallel::get_parallel_component<BroadcastTarget>(cache), iteration_id,
217  std::variant<double, Convergence::HasConverged>(
218  // NOLINTNEXTLINE(performance-move-const-arg)
219  std::move(has_converged)));
220  }
221 };
