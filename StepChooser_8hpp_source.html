33 /// \ingroup TimeSteppersGroup
34 ///
35 /// StepChoosers suggest upper bounds on step sizes. Concrete
36 /// StepChoosers should define operator() returning the magnitude of
37 /// the desired step (as a double).
38 ///
39 /// The step choosers valid for the integration being controlled are
40 /// specified by passing a `tmpl::list` of the corresponding
41 /// registrars.
42 template <typename StepChooserRegistrars>
43 class StepChooser : public PUP::able {
44  protected:
45  /// \cond HIDDEN_SYMBOLS
46  StepChooser() = default;
47  StepChooser(const StepChooser&) = default;
48  StepChooser(StepChooser&&) = default;
49  StepChooser& operator=(const StepChooser&) = default;
50  StepChooser& operator=(StepChooser&&) = default;
51  /// \endcond
52 
53  public:
54  ~StepChooser() override = default;
55 
56  WRAPPED_PUPable_abstract(StepChooser); // NOLINT
57 
58  using creatable_classes = Registration::registrants<StepChooserRegistrars>;
59 
60  /// The `last_step_magnitude` parameter describes the step size to be
61  /// adjusted. It may be the step size or the slab size, or may be
62  /// infinite if the appropriate size cannot be determined.
63  template <typename Metavariables, typename DbTags>
64  double desired_step(
65  const double last_step_magnitude, const db::DataBox<DbTags>& box,
66  const Parallel::GlobalCache<Metavariables>& cache) const noexcept {
67  ASSERT(last_step_magnitude > 0.,
68  "Passed non-positive step magnitude: " << last_step_magnitude);
69  const auto result = call_with_dynamic_type<double, creatable_classes>(
70  this,
71  [&last_step_magnitude, &box, &cache](
72  const auto* const chooser) noexcept {
73  return db::apply(*chooser, box, last_step_magnitude, cache);
74  });
75  ASSERT(
76  result > 0.,
77  "StepChoosers should always return positive values. Got " << result);
78  return result;
79  }
80 };
