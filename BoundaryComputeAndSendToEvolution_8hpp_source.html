62 /*!
63  * \ingroup ActionsGroup
64  * \brief Obtains the CCE boundary data at the specified `time`, and reports it
65  * to the `EvolutionComponent` via `Actions::ReceiveWorldtubeData`.
66  *
67  * \details This uses the `WorldtubeDataManager` to perform all of the work of
68  * managing the file buffer, interpolating to the desired time point, and
69  * compute the Bondi quantities on the boundary. Once readied, it sends each
70  * tensor from the the full `Variables<typename
71  * Metavariables::cce_boundary_communication_tags>` back to the
72  * `EvolutionComponent`
73  *
74  * Uses:
75  * - DataBox:
76  * - `Tags::H5WorldtubeBoundaryDataManager`
77  *
78  * \ref DataBoxGroup changes:
79  * - Adds: nothing
80  * - Removes: nothing
81  * - Modifies:
82  * - `Tags::Variables<typename
83  * Metavariables::cce_boundary_communication_tags>` (every tensor)
84  */
85 template <typename Metavariables, typename EvolutionComponent>
86 struct BoundaryComputeAndSendToEvolution<H5WorldtubeBoundary<Metavariables>,
87  EvolutionComponent> {
88  template <typename ParallelComponent, typename... DbTags, typename ArrayIndex,
89  Requires<tmpl2::flat_any_v<std::is_same_v<
90  ::Tags::Variables<
91  typename Metavariables::cce_boundary_communication_tags>,
92  DbTags>...>> = nullptr>
93  static void apply(db::DataBox<tmpl::list<DbTags...>>& box,
94  Parallel::GlobalCache<Metavariables>& cache,
95  const ArrayIndex& /*array_index*/,
96  const TimeStepId& time) noexcept {
97  bool successfully_populated = false;
98  db::mutate<Tags::H5WorldtubeBoundaryDataManager,
99  ::Tags::Variables<
100  typename Metavariables::cce_boundary_communication_tags>>(
101  make_not_null(&box),
102  [&successfully_populated, &time](
103  const gsl::not_null<std::unique_ptr<Cce::WorldtubeDataManager>*>
104  worldtube_data_manager,
105  const gsl::not_null<Variables<
106  typename Metavariables::cce_boundary_communication_tags>*>
107  boundary_variables) noexcept {
108  successfully_populated =
109  (*worldtube_data_manager)
110  ->populate_hypersurface_boundary_data(
111  boundary_variables, time.substep_time().value());
112  });
113  if (not successfully_populated) {
114  ERROR("Insufficient boundary data to proceed, exiting early at time " +
115  std::to_string(time.substep_time().value()));
116  }
117  Parallel::receive_data<Cce::ReceiveTags::BoundaryData<
118  typename Metavariables::cce_boundary_communication_tags>>(
119  Parallel::get_parallel_component<EvolutionComponent>(cache), time,
120  db::get<::Tags::Variables<
121  typename Metavariables::cce_boundary_communication_tags>>(box),
122  true);
123  }
124 };

126 /*!
127  * \ingroup ActionsGroup
128  * \brief Submits a request for CCE boundary data at the specified `time` to the
129  * `Cce::InterfaceManagers::GhInterfaceManager`, and sends the data to the
130  * `EvolutionComponent` (template argument) if it is ready.
131  *
132  * \details This uses the `Cce::InterfaceManagers::GhInterfaceManager` to
133  * perform all of the work of managing the buffer of data sent from the GH
134  * system and interpolating if necessary and supported. This dispatches then to
135  * `Cce::Actions::SendToEvolution<GhWorldtubeBoundary<Metavariables>,
136  * EvolutionComponent>` if the boundary data is ready, otherwise
137  * simply submits the request and waits for data to become available via
138  * `Cce::Actions::ReceiveGhWorldtubeData`, which will call
139  * `Cce::Actions::SendToEvolution<GhWorldtubeBoundary<Metavariables>,
140  * EvolutionComponent>` as soon as the data becomes available.
141  *
142  * \ref DataBoxGroup changes:
143  * - Adds: nothing
144  * - Removes: nothing
145  * - Modifies:
146  * - `Tags::GhInterfaceManager`
147  */
148 template <typename Metavariables, typename EvolutionComponent>
149 struct BoundaryComputeAndSendToEvolution<GhWorldtubeBoundary<Metavariables>,
150  EvolutionComponent> {
151  template <typename ParallelComponent, typename... DbTags, typename ArrayIndex,
152  Requires<tmpl2::flat_any_v<std::is_same_v<
153  ::Tags::Variables<
154  typename Metavariables::cce_boundary_communication_tags>,
155  DbTags>...>> = nullptr>
156  static void apply(db::DataBox<tmpl::list<DbTags...>>& box,
157  Parallel::GlobalCache<Metavariables>& cache,
158  const ArrayIndex& /*array_index*/,
159  const TimeStepId& time) noexcept {
160  db::mutate<Tags::GhInterfaceManager>(
161  make_not_null(&box),
162  [&time, &cache](const gsl::not_null<
163  std::unique_ptr<InterfaceManagers::GhInterfaceManager>*>
164  interface_manager) noexcept {
165  (*interface_manager)->request_gh_data(time);
166  const auto gh_data =
167  (*interface_manager)->retrieve_and_remove_first_ready_gh_data();
168  if (static_cast<bool>(gh_data)) {
169  Parallel::simple_action<Actions::SendToEvolution<
170  GhWorldtubeBoundary<Metavariables>, EvolutionComponent>>(
171  Parallel::get_parallel_component<
172  GhWorldtubeBoundary<Metavariables>>(cache),
173  get<0>(*gh_data), get<1>(*gh_data));
174  }
175  });
176  }
177 };
