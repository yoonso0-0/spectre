16 /// Register the classes in `ListOfClassesToRegister` with the serialization
17 /// framework.
18 template <typename ListOfClassesToRegister>
19 void register_classes_in_list() noexcept {
20  tmpl::for_each<ListOfClassesToRegister>([](auto class_v) noexcept {
21  using class_to_register = typename decltype(class_v)::type;
22  // We use PUPable_reg2 because this takes as a second argument the name of
23  // the class as a `const char*`, while PUPable_reg converts the argument
24  // verbatim to a string using the `#` preprocessor operator.
25  PUPable_reg2(class_to_register, typeid(class_to_register).name());
26  });
27 }

29 /// Register derived classes of the `Base` class
30 template <typename Base>
31 void register_derived_classes_with_charm() noexcept {
32  register_classes_in_list<typename Base::creatable_classes>();
33 }
