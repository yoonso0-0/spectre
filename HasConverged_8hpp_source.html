21 /*!
22  * \brief Determine whether the `criteria` are met.
23  *
24  * \note This function assumes the `iteration_id` is that of the latest
25  * completed step and that it is zero-indexed, where zero indicates the initial
26  * state of the algorithm. Therefore, the `MaxIteration` criterion will match if
27  * the `iteration_id` is equal or higher. For example, a `MaxIteration` of 0
28  * means the algorithm should run no iterations, so it matches if the
29  * `iteration_id` is 0 or higher since that's the initial state before any
30  * steps have been performed. A `MaxIteration` of 1 matches if the
31  * `iteration_id` is 1 or higher since one iteration is complete. At this point,
32  * also the `residual_magnitude` reflects the state of the algorithm after
33  * completion of the first iteration. The `initial_residual_magnitude` always
34  * refers to the state before the first iteration has begun, i.e. where the
35  * `iteration_id` is zero.
36  *
37  * \returns A `Convergence::Reason` if the criteria are met, or
38  * `std::nullopt` otherwise. The possible convergence reasons are:
39  *
40  * - `Convergence::Reason::AbsoluteResidual` if the `residual_magnitude`
41  * meets the convergence criteria's `absolute_residual`.
42  * - `Convergence::Reason::RelativeResidual` if `residual_magnitude /
43  * initial_residual_magnitude` meets the convergence criteria's
44  * `relative_residual`.
45  * - `Convergence::Reason::MaxIterations` if the `iteration_id` is the
46  * convergence_criteria's `max_iterations` or higher. This is often
47  * interpreted as an error because the algorithm did not converge in the
48  * alloted number of iterations.
49  */
50 std::optional<Reason> criteria_match(
51  const Criteria& criteria, size_t iteration_id, double residual_magnitude,
52  double initial_residual_magnitude) noexcept;

54 /*!
55  * \brief Signals convergence of the algorithm.
56  *
57  * \details Evaluates to `true` if the algorithm has converged and no
58  * further iterations should be performed. In this case, the `reason()` member
59  * function provides more information. If `false`, calling `reason()` is an
60  * error.
61  *
62  * The stream operator provides a human-readable description of the convergence
63  * status.
64  *
65  * This type default-constructs to a state that signals the algorithm has
66  * not yet converged.
67  */
68 struct HasConverged {
69  public:
70  HasConverged() = default;
71  /*!
72  * \brief Determine whether the \p criteria are met by means of
73  * `Convergence::criteria_match`.
74  */
75  HasConverged(const Criteria& criteria, size_t iteration_id,
76  double residual_magnitude,
77  double initial_residual_magnitude) noexcept;
78 
79  /// Construct at a state where `iteration_id` iterations of a total of
80  /// `num_iterations` have completed. Use when the algorithm is intended to run
81  /// for a fixed number of iterations. The convergence `reason()` will be
82  /// `Convergence::Reason::NumIterations`.
83  HasConverged(size_t num_iterations, size_t iteration_id) noexcept;
84 
85  explicit operator bool() const noexcept { return static_cast<bool>(reason_); }
86 
87  /*!
88  * \brief The reason the algorithm has converged.
89  *
90  * \warning Calling this function is an error if the algorithm has not yet
91  * converged.
92  */
93  Reason reason() const noexcept;
94 
95  /// The number of iterations the algorithm has completed
96  size_t num_iterations() const noexcept;
97 
98  /// The residual magnitude after the last iteration. NaN if no iteration has
99  /// completed yet.
100  double residual_magnitude() const noexcept;
101 
102  /// The residual magnitude before the first iteration. NaN if this information
103  /// is not available yet.
104  double initial_residual_magnitude() const noexcept;
105 
106  void pup(PUP::er& p) noexcept; // NOLINT

109  const HasConverged& rhs) noexcept;

111  const HasConverged& rhs) noexcept;
112 
113  friend std::ostream& operator<<(std::ostream& os,
114  const HasConverged& has_converged) noexcept;

116  private:
117  std::optional<Reason> reason_{};
118  Criteria criteria_{};
119  size_t iteration_id_{};
120  double residual_magnitude_{};
121  double initial_residual_magnitude_{};
122 };
