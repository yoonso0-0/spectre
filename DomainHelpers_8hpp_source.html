305 /// Iterates over the corners of a VolumeDim-dimensional cube.
306 template <size_t VolumeDim>
307 class VolumeCornerIterator {
308  public:
309  VolumeCornerIterator() noexcept { setup_from_local_corner_number(); }
310 
311  explicit VolumeCornerIterator(size_t initial_local_corner_number) noexcept
312  : local_corner_number_(initial_local_corner_number) {
313  setup_from_local_corner_number();
314  }
315  VolumeCornerIterator(
316  // The block index is also global corner
317  // index of the lowest corner of the block.
318  Index<VolumeDim> block_index,
319  Index<VolumeDim> global_corner_extents) noexcept
320  : global_corner_number_(
321  collapsed_index(block_index, global_corner_extents)),
322  global_corner_index_(block_index),
323  global_corner_extents_(global_corner_extents) {}
324 
325  void operator++() noexcept {
326  ++local_corner_number_;
327  setup_from_local_corner_number();
328  }
329 
330  explicit operator bool() const noexcept {
331  return local_corner_number_ < two_to_the(VolumeDim);
332  }
333 
334  size_t local_corner_number() const noexcept { return local_corner_number_; }
335 
336  size_t global_corner_number() const noexcept {
337  std::array<size_t, VolumeDim> new_indices{};
338  for (size_t i = 0; i < VolumeDim; i++) {
339  gsl::at(new_indices, i) =
340  global_corner_index_[i] +
341  (gsl::at(array_sides_, i) == Side::Upper ? 1 : 0);
342  }
343  const Index<VolumeDim> interior_multi_index(new_indices);
344  return collapsed_index(interior_multi_index, global_corner_extents_);
345  }
346 
347  const std::array<Side, VolumeDim>& operator()() const noexcept {
348  return array_sides_;
349  }
350 
351  const std::array<Side, VolumeDim>& operator*() const noexcept {
352  return array_sides_;
353  }
354 
355  const std::array<double, VolumeDim>& coords_of_corner() const noexcept {
356  return coords_of_corner_;
357  }
358 
359  const std::array<Direction<VolumeDim>, VolumeDim>& directions_of_corner()
360  const noexcept {
361  return array_directions_;
362  }
363 
364  void setup_from_local_corner_number() noexcept {
365  for (size_t i = 0; i < VolumeDim; i++) {
366  gsl::at(coords_of_corner_, i) =
367  2.0 * get_nth_bit(local_corner_number_, i) - 1.0;
368  gsl::at(array_sides_, i) =
369  2 * get_nth_bit(local_corner_number_, i) - 1 == 1 ? Side::Upper
370  : Side::Lower;
371  gsl::at(array_directions_, i) =
372  Direction<VolumeDim>(i, gsl::at(array_sides_, i));
373  }
374  }
375 
376  private:
377  size_t local_corner_number_ = 0;
378  size_t global_corner_number_{std::numeric_limits<size_t>::max()};
379  Index<VolumeDim> global_corner_index_{};
380  Index<VolumeDim> global_corner_extents_{};
381  std::array<Side, VolumeDim> array_sides_ = make_array<VolumeDim>(Side::Lower);
382  std::array<Direction<VolumeDim>, VolumeDim> array_directions_{};
383  std::array<double, VolumeDim> coords_of_corner_ = make_array<VolumeDim>(-1.0);
384 };

386 /// \ingroup ComputationalDomainGroup
387 /// Iterates over the 2^(VolumeDim-1) logical corners of the face of a
388 /// VolumeDim-dimensional cube in the given direction.
389 template <size_t VolumeDim>
390 class FaceCornerIterator {
391  public:
392  explicit FaceCornerIterator(Direction<VolumeDim> direction) noexcept;
393 
394  void operator++() noexcept {
395  face_index_++;
396  do {
397  index_++;
398  } while (get_nth_bit(index_, direction_.dimension()) ==
399  (direction_.side() == Side::Upper ? 0 : 1));
400  for (size_t i = 0; i < VolumeDim; ++i) {
401  corner_[i] = 2 * static_cast<int>(get_nth_bit(index_, i)) - 1;
402  }
403  }
404 
405  explicit operator bool() const noexcept {
406  return face_index_ < two_to_the(VolumeDim - 1);
407  }
