73 /*!
74  * \brief The dynamic part \f$\beta^i_\mathrm{excess}\f$ of the shift
75  * \f$\beta^i=\beta^i_\mathrm{background} + \beta^i_\mathrm{excess}\f$
76  *
77  * We commonly split off the part of the shift that diverges at large coordinate
78  * distances (the "background" shift \f$\beta^i_\mathrm{background}\f$) and
79  * solve only for the remainder (the "excess" shift
80  * \f$\beta^i_\mathrm{excess}\f$). For example, the background shift might be a
81  * uniform rotation \f$\beta^i_\mathrm{background}=(-\Omega y, \Omega x, 0)\f$
82  * with angular velocity \f$\Omega\f$ around the z-axis, given here in Cartesian
83  * coordinates.
84  *
85  * \see `Xcts::Tags::Background`
86  */
87 template <typename DataType, size_t Dim, typename Frame>
88 struct ShiftExcess : db::SimpleTag {
89  using type = tnsr::I<DataType, Dim, Frame>;
90 };

92 /*!
93  * \brief The symmetric "strain" of the shift vector
94  * \f$B_{ij} = \bar{\nabla}_{(i}\bar{\gamma}_{j)k}\beta^k =
95  * \left(\partial_{(i}\bar{\gamma}_{j)k} - \bar{\Gamma}_{kij}\right)\beta^k\f$
96  *
97  * This quantity is used in our formulations of the XCTS equations.
98  *
99  * Note that the shift is not a conformal quantity, so its index is generally
100  * raised and lowered with the spatial metric, not with the conformal metric.
101  * However, to compute this "strain" we use the conformal metric as defined
102  * above. The conformal longitudinal shift in terms of this quantity is then:
103  *
104  * \f{equation}
105  * (\bar{L}\beta)^{ij} = 2\left(\bar{\gamma}^{ik}\bar{\gamma}^{jl}
106  * - \frac{1}{3}\bar{\gamma}^{ij}\bar{\gamma}^{kl}\right) B_{kl}
107  * \f}
108  *
109  * Note that the conformal longitudinal shift is (minus) the "stress" quantity
110  * of a linear elasticity system in which the shift takes the role of the
111  * displacement vector and the definition of its "strain" remains the same. This
112  * auxiliary elasticity system is formulated on an isotropic constitutive
113  * relation based on the conformal metric with vanishing bulk modulus \f$K=0\f$
114  * (not to be confused with the extrinsic curvature trace \f$K\f$ in this
115  * context) and unit shear modulus \f$\mu=1\f$. See the
116  * `Elasticity::FirstOrderSystem` and the
117  * `Elasticity::ConstitutiveRelations::IsotropicHomogeneous` for details.
118  */
119 template <typename DataType, size_t Dim, typename Frame>
120 struct ShiftStrain : db::SimpleTag {
121  using type = tnsr::ii<DataType, Dim, Frame>;
122 };
