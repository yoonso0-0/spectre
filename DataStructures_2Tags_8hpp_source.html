20 /// Given the `Tag` holding a `Tensor<DataVector, ...>`, swap the
21 /// `DataVector` with a `double`.
22 template <typename Tag>
23 struct Mean : db::SimpleTag, db::PrefixTag {

43 /// Given a Tag with a `type` of `Tensor<VectorType, ...>`, acts as a new
44 /// version of the tag with `type` of `Tensor<SpinWeighted<VectorType,
45 /// SpinConstant::value>, ...>`, which is the preferred tensor type associated
46 /// with spin-weighted quantities. Here, `SpinConstant` must be a
47 /// `std::integral_constant` or similar type wrapper for a compile-time
48 /// constant, in order to work properly with DataBox utilities.
49 /// \note There are restrictions of which vector types can be wrapped
50 /// by SpinWeighted in a Tensor, so some `Tag`s that have valid `type`s may
51 /// give rise to compilation errors when used as `Tags::SpinWeighted<Tag,
52 /// SpinConstant>`. If you find such trouble, consult the whitelist of possible
53 /// Tensor storage types in `Tensor.hpp`.
54 template <typename Tag, typename SpinConstant>
55 struct SpinWeighted : db::PrefixTag, db::SimpleTag {
56  static_assert(not is_any_spin_weighted_v<typename Tag::type::type>,
57  "The SpinWeighted tag should only be used to create a "
58  "spin-weighted version of a non-spin-weighted tag. The "
59  "provided tag already has a spin-weighted type");
60  using type = TensorMetafunctions::swap_type<
61  ::SpinWeighted<typename Tag::type::type, SpinConstant::value>,
62  typename Tag::type>;
63  using tag = Tag;
64  static std::string name() noexcept {
65  return "SpinWeighted(" + db::tag_name<Tag>() + ", " +
66  std::to_string(SpinConstant::value) + ")";
67  }
68 };

77 /// A prefix tag representing the product of two other tags. Note that if
78 /// non-spin-weighted types are needed in this tag, the type alias
79 /// `detail::product_t` should be generalized to give a reasonable type for
80 /// those cases, or template specializations should be constructed for this tag.
81 template <typename LhsTag, typename RhsTag>
82 struct Multiplies : db::PrefixTag, db::SimpleTag {
83  using type = detail::product_t<typename LhsTag::type, typename RhsTag::type>;
84  using tag = LhsTag;
85  static std::string name() noexcept {
86  return "Multiplies(" + db::tag_name<LhsTag>() + ", " +
87  db::tag_name<RhsTag>() + ")";
88  }
