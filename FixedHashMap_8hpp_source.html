54 /*!
55  * \ingroup DataStructuresGroup
56  * \brief A hash table with a compile-time specified maximum size and ability to
57  * efficiently handle perfect hashes.
58  *
59  * There are a few requirements on the types passed to `FixedHashMap`. These
60  * are:
61  * - `Key` is CopyConstructible for copy and move semantics, as well as
62  * serialization
63  * - `ValueType` is MoveConstructible
64  * - If `Hash` is a perfect hash for some values of `MaxSize` then in order to
65  * use the perfect hashing improvements `Hash` must have a member variable
66  * template `is_perfect` that takes as its only template parameter a `size_t`
67  * equal to `MaxSize`. A perfect hash must map each `Key` to `[0, MaxSize)`
68  * uniquely.
69  * - Keys are distinct.
70  *
71  * The interface is similar to std::unordered_map, so see the documentation for
72  * that for details.
73  *
74  * #### Implementation Details
75  *
76  * - Uses linear probing for non-perfect hashes.
77  * - The hash `Hash` is assumed to be a perfect hash for `MaxSize` if
78  * `Hash::template is_perfect<MaxSize>` is true.
79  */

430 template <bool Assign, class M>
431 auto FixedHashMap<MaxSize, Key, ValueType, Hash,
432  KeyEqual>::insert_or_assign_impl(key_type&& key,
433  M&& obj) noexcept
434  -> std::pair<iterator, bool> {
435  auto data_it = get_data_entry<true>(key);
436  if (UNLIKELY(data_it == data_.end())) {
437  ERROR("Unable to insert element into FixedHashMap of maximum size "
438  << MaxSize << " because it is full. If the current size (" << size_
439  << ") is not equal to the maximum size then please file an issue "
440  "with the code you used to produce this error. "
441  << (hash_is_perfect ? " If a perfect hash is used and it hashes to "
442  "past the last element stored, then this "
443  "error may also be triggered."
444  : ""));
445  }
446  // data_it points to either the existing element or a new bucket to place the
447  // element.
448  const auto is_new_entry = not is_set(*data_it);
449  if (is_new_entry or Assign) {
450  if (is_new_entry) {
451  ++size_;
452  }
453  data_it->emplace(std::move(key), std::forward<M>(obj));
454  }
455  return std::make_pair(iterator{data_it, data_.end()}, is_new_entry);
456 }

458 template <size_t MaxSize, class Key, class ValueType, class Hash,
459  class KeyEqual>
460 auto FixedHashMap<MaxSize, Key, ValueType, Hash, KeyEqual>::erase(
461  const const_iterator& pos) noexcept -> iterator {
462  auto next_it = &(unconst(pos).get_optional());
463  --size_;
464  next_it->reset();
465  // pos now points to an unset entry, advance to next valid one
466  do {
467  ++next_it;
468  if (next_it == data_.end()) {
469  next_it = data_.begin();
470  }
471  if (is_set(*next_it)) {
472  return {next_it, data_.end()};
473  }
474  } while (next_it != &pos.get_optional());
475  return end();
476 }

540 template <size_t MaxSize, class Key, class ValueType, class Hash,
541  class KeyEqual>
542 template <bool IsInserting>
543 auto FixedHashMap<MaxSize, Key, ValueType, Hash, KeyEqual>::get_data_entry(
544  const Key& key) noexcept -> typename storage_type::iterator {
545  const auto hashed_key = hash(key);
546  auto it = data_.begin() + hashed_key;
547  if constexpr (not hash_is_perfect) {
548  // First search for an existing key.
549  while (not is_set(*it) or it->value().first != key) {
550  if (++it == data_.end()) {
551  it = data_.begin();
552  }
553  if (UNLIKELY(it == data_.begin() + hashed_key)) {
554  // not found, return end iterator to storage_type
555  it = data_.end();
556  break;
557  }
558  }
559  // If we are inserting an element but couldn't find an existing key, then
560  // find the first available bucket.
561  if (IsInserting and it == data_.end()) {
562  it = data_.begin() + hashed_key;
563  while (is_set(*it)) {
564  if (++it == data_.end()) {
565  it = data_.begin();
566  }
567  if (UNLIKELY(it == data_.begin() + hashed_key)) {
568  // could not find any open bucket
569  return data_.end();
570  }
571  }
572  }
573  }
574  return it;
575 }
