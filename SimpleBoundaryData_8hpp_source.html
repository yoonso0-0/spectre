31  * \brief Distinguishes between field data, which can be projected to a mortar,
32  * and extra data, which will not be projected.
33  */
34 template <typename FieldTags, typename ExtraDataTags = tmpl::list<>>
35 struct SimpleBoundaryData {
36  using field_tags = FieldTags;
37  using extra_data_tags = ExtraDataTags;
38 
39  /// Data projected to the mortar mesh
40  Variables<FieldTags> field_data;
41 
42  /// Data on the element face that needs no projection to the mortar mesh.
43  /// This is a `TaggedTuple` to support non-tensor quantities. It also helps
44  /// supporting an empty list of `ExtraDataTags`.
45  tuples::tagged_tuple_from_typelist<ExtraDataTags> extra_data;
46 
47  SimpleBoundaryData() = default;
48  SimpleBoundaryData(const SimpleBoundaryData&) noexcept = default;
49  SimpleBoundaryData& operator=(const SimpleBoundaryData&) noexcept = default;
50  SimpleBoundaryData(SimpleBoundaryData&&) noexcept = default;
51  SimpleBoundaryData& operator=(SimpleBoundaryData&&) noexcept = default;
52  ~SimpleBoundaryData() noexcept = default;
53 
54  explicit SimpleBoundaryData(const size_t num_points) noexcept
55  : field_data{num_points}, extra_data{} {}
56 
57  // NOLINTNEXTLINE(google-runtime-references)
58  void pup(PUP::er& p) noexcept {
59  p | field_data;
60  p | extra_data;
61  }

63  /// Project the `field_data` to the mortar
64  ///
65  /// \see `dg::project_to_mortar`
66  template <size_t MortarDim>
67  SimpleBoundaryData<FieldTags, ExtraDataTags> project_to_mortar(
68  const Mesh<MortarDim>& face_mesh, const Mesh<MortarDim>& mortar_mesh,
69  const std::array<Spectral::MortarSize, MortarDim>& mortar_size) const
70  noexcept {
71  SimpleBoundaryData<FieldTags, ExtraDataTags> projected_data{};
72  projected_data.field_data = dg::project_to_mortar(
73  this->field_data, face_mesh, mortar_mesh, mortar_size);
74  projected_data.extra_data = this->extra_data;
75  return projected_data;
76  }
77 
78  /// Orient the `field_data`

80  /// \see `orient_variables_on_slice`
81  template <size_t MortarDim>
82  void orient_on_slice(
83  const Index<MortarDim>& slice_extents, const size_t sliced_dim,

85  this->field_data = orient_variables_on_slice(
86  this->field_data, slice_extents, sliced_dim, orientation_of_neighbor);
87  }
