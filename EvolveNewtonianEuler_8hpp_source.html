296  static Phase determine_next_phase(
297  const Phase& current_phase,
298  const Parallel::CProxy_GlobalCache<
299  EvolutionMetavars>& /*cache_proxy*/) noexcept {
300  switch (current_phase) {
301  case Phase::Initialization:
302  return Phase::InitializeTimeStepperHistory;
303  case Phase::InitializeTimeStepperHistory:
304  return Phase::RegisterWithObserver;
305  case Phase::RegisterWithObserver:
306  return Phase::Evolve;
307  case Phase::Evolve:
308  return Phase::Exit;
309  case Phase::Exit:
310  ERROR(
311  "Should never call determine_next_phase with the current phase "
312  "being 'Exit'");
313  default:
314  ERROR(
315  "Unknown type of phase. Did you static_cast<Phase> to an integral "
316  "value?");
317  }
318  }
