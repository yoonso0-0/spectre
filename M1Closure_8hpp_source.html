46 /*!
47  * Compute the 2nd moment of the neutrino distribution function
48  * in the inertial frame (pressure tensor) from the 0th (energy density)
49  * and 1st (momentum density) moments. The M1 closure sets
50  * \f{align}{
51  * P_{ij} = d_{\rm thin} P_{ij,\rm thin} + d_{\rm thick} P_{ij,\rm thick}
52  * \f}
53  * with \f$P_{ij}\f$ the pressure tensor in the inertial frame, and
54  * \f$P_{ij,thin/thick}\f$ its value assuming an optically thick / optically
55  * thin medium.
56  *
57  * Following the algorithm described in Appendix A of \cite Foucart2015vpa,
58  * we choose the Minerbo closure (\cite Minerbo1978) which sets
59  * \f{align}{
60  * d_{\rm thin} = & 1.5\chi-0.5\\
61  * d_{\rm thick} = & 1.5-1.5\chi \\
62  * \chi = & \frac{1}{3} + \xi^2 \frac{6-2\xi+6\xi^2}{15} \\
63  * \xi = & H^aH_a/J^2
64  * \f}
65  * with \f$J\f$ the energy density in the
66  * frame comoving with the fluid, and \f$H^a\f$ the momentum density
67  * in that frame.
68  * The optically thick closure is
69  * \f{align}{
70  * K^{ab,\rm thick} = \frac{J_{\rm thick}}{3} (g^{ab}+u^a u^b)
71  * \f}
72  * with \f$ K^{ab,\rm thick} \f$ the pressure tensor in the frame comoving with
73  * the fluid, and \f$ u^a \f$ the fluid 4-velocity, while the optically thin
74  * closure is
75  * \f{align}{
76  * P^{ij,\rm thin} = \frac{S^i S^j}{S^k S_k} E
77  * \f}
78  * with \f$ E \f$, \f$ S^{i}
79  * \f$ the 0th and 1st moments in the inertial frame (see paper for the
80  * computation of \f$P_{ij,thick}\f$ from the other moments).
81  *
82  * The main step of this function is to solve the equation
83  * \f{align}{
84  * \frac{\xi^2 J^2 - H^aH_a}{E^2} = 0
85  * \f}
86  * for \f$\xi\f$, with \f$J\f$ and \f$H^a\f$ being functions
87  * of \f$\xi\f$ and of the input variables (the 0th and 1st moments
88  * in the inertial frame). This is done by separating \f$H^a\f$ as
89  * \f{align}{
90  * H_a = - H_t t_a - H_v v_a - H_f F_a
91  * \f}
92  * (with \f$v^a\f$ the fluid 3-velocity and \f$t^a\f$ the unit normal
93  * to the slice) and then writing each of the
94  * variables \f$J\f$, \f$H_n\f$, \f$H_v\f$, \f$H_f\f$ as
95  * \f{align}{
96  * X = X_0 + d_{\rm thin} X_{\rm thin} + d_{\rm thick} X_{\rm thick}
97  * \f}
98  * so that evaluating
99  * \f{align}{
100  * \frac{\xi^2 J^2 - H^aH_a}{E^2} = 0
101  * \f}
102  * for a given \f$\xi\f$ only requires recomputing \f$d_{\rm thin,thick}\f$
103  * and their derivatives with respect to \f$\xi\f$.
104  * We perform the root-finding using a Newton-Raphson algorithm, with the
105  * accuracy set by the variable root_find_number_of_digits (6 significant digits
106  * at the moment).
107  *
108  * The function returns the closure factors \f$\xi\f$ (to be used as initial
109  * guess for this function at the next step), the pressure tensor \f$P_{ij}\f$,
110  * and the neutrino moments in the frame comoving with the fluid.
111  * The momentum density in the frame comoving with the fluid
112  * is decomposed into its normal component \f$ H^a t_a\f$, and its spatial
113  * components \f$ \gamma_{ia} H^a\f$.
114  */
115 template <typename... NeutrinoSpecies>
