50 /*!
51  * \ingroup SpectralGroup
52  * \brief The choice of basis functions for computing collocation points and
53  * weights.
54  *
55  * \details Choose `Legendre` for a general-purpose DG mesh, unless you have a
56  * particular reason for choosing another basis.
57  *
58  * \warning The `FiniteDifference` "basis" is used to denote that only the
59  * collocation points are defined, but that differentiation, integration, and
60  * interpolation schemes are to be chosen locally wherever a `FiniteDifference`
61  * mesh is being used. The reason is that there isn't a requirement that the
62  * basis and collocation point locations are at all related to the
63  * differentiation, integration, or interpolation methods - it is merely a
64  * convenient choice in a lot of cases. For `FiniteDifference` we need to choose
65  * the order of the scheme (and hence the weights, differentiation matrix,
66  * integration weights, and interpolant) locally in space and time to handle
67  * discontinuous solutions, hence none of those are defined for the
68  * `FiniteDifference` basis.
69  */
70 enum class Basis { Chebyshev, Legendre, FiniteDifference };

76 /*!
77  * \ingroup SpectralGroup
78  * \brief The choice of quadrature method to compute integration weights.
79  *
80  * \details Integrals using \f$N\f$ collocation points with Gauss quadrature are
81  * exact to polynomial order \f$p=2N-1\f$. Gauss-Lobatto quadrature is exact
82  * only to polynomial order \f$p=2N-3\f$, but includes collocation points at the
83  * domain boundary.
84  *
85  * \warning `CellCentered` and `FaceCentered` are intended to be used with the
86  * `FiniteDifference` basis (though in principle they could be used with any
87  * basis), and thus do not implement differentiation matrices, integration
88  * weights, and interpolation matrices.
89  */
90 enum class Quadrature { Gauss, GaussLobatto, CellCentered, FaceCentered };

97 namespace detail {
98 constexpr size_t minimum_number_of_points(
99  const Basis /*basis*/, const Quadrature quadrature) noexcept {
100  // NOLINTNEXTLINE(bugprone-branch-clone)
101  if (quadrature == Quadrature::Gauss) {
102  return 1;
103  // NOLINTNEXTLINE(bugprone-branch-clone)
104  } else if (quadrature == Quadrature::GaussLobatto) {
105  return 2;
106  // NOLINTNEXTLINE(bugprone-branch-clone)
107  } else if (quadrature == Quadrature::CellCentered) {
108  return 1;
109  // NOLINTNEXTLINE(bugprone-branch-clone)
110  } else if (quadrature == Quadrature::FaceCentered) {
111  return 2;
112  }
113  return std::numeric_limits<size_t>::max();
114 }
115 } // namespace detail

117 /*!
118  * \brief Minimum number of possible collocation points for a quadrature type.
119  *
120  * \details Since Gauss-Lobatto quadrature has points on the domain boundaries
121  * it must have at least two collocation points. Gauss quadrature can have only
122  * one collocation point.
123  *
124  * \details For `CellCentered` the minimum number of points is 1, while for
125  * `FaceCentered` it is 2.
126  */

128 constexpr size_t minimum_number_of_points =
129  detail::minimum_number_of_points(basis, quadrature);

131 /*!
132  * \brief Maximum number of allowed collocation points.
133  *
134  * \details We choose a limit of 23 FD grid points because for DG-subcell the
135  * number of points in an element is `2 * (number_dg_points - 1)`. Because there
136  * is no way of generically retrieving the maximum number of grid points for
137  * a non-FD basis, we need to hard-code both values here. If the number of grid
138  * points is increased for the non-FD bases, it should also be increased for the
139  * FD basis. Note that for good task-based parallelization 23 grid points is
140  * already a fairly large number.
141  */
142 template <Basis basis>
143 constexpr size_t maximum_number_of_points =
144  basis == Basis::FiniteDifference ? 23 : 12;

198 /*!
199  * \brief Weights to compute definite integrals.
200  *
201  * \details These are the coefficients to contract with the nodal
202  * function values \f$f_k\f$ to approximate the definite integral \f$I[f]=\int
203  * f(x)\mathrm{d}x\f$.
204  *
205  * Note that the term _quadrature_ also often refers to the quantity
206  * \f$Q[f]=\int f(x)w(x)\mathrm{d}x\approx \sum_k f_k w_k\f$. Here, \f$w(x)\f$
207  * denotes the basis-specific weight function w.r.t. to which the basis
208  * functions \f$\Phi_k\f$ are orthogonal, i.e \f$\int\Phi_i(x)\Phi_j(x)w(x)=0\f$
209  * for \f$i\neq j\f$. The weights \f$w_k\f$ approximate this inner product. To
210  * approximate the definite integral \f$I[f]\f$ we must employ the
211  * coefficients \f$\frac{w_k}{w(\xi_k)}\f$ instead, where the \f$\xi_k\f$ are
212  * the collocation points. These are the coefficients this function returns.
213  * Only for a unit weight function \f$w(x)=1\f$, i.e. a Legendre basis, is
214  * \f$I[f]=Q[f]\f$ so this function returns the \f$w_k\f$ identically.
215  *
216  * \warning for a `FiniteDifference` basis or `CellCentered` and `FaceCentered`
217  * quadratures the weights are `NaN`.
218  *
219  * \param num_points The number of collocation points
220  */
221 template <Basis BasisType, Quadrature QuadratureType>
222 const DataVector& quadrature_weights(size_t num_points) noexcept;

271 /*!
272  * \brief %Matrix used to perform an indefinite integral of a function over the
273  * logical grid. The left boundary condition is such that the integral is 0 at
274  * \f$\xi=-1\f$
275  *
276  * Currently only Legendre and Chebyshev polynomials are implemented, but we
277  * provide a derivation for how to compute the indefinite integration matrix for
278  * general Jacobi polynomials.
279  *
280  * #### Legendre Polynomials
281  * The Legendre polynomials have the identity:
282  *
283  * \f{align*}{
284  * P_n(x) = \frac{1}{2n+1}\frac{d}{dx}\left(P_{n+1}(x) - P_{n-1}(x)\right)
285  * \f}
286  *
287  * The goal is to evaluate the integral of a function \f$u\f$ expanded in terms
288  * of Legendre polynomials as
289  *
290  * \f{align*}{
291  * u(x) = \sum_{i=0}^{N} c_i P_i(x)
292  * \f}
293  *
294  * We similarly expand the indefinite integral of \f$u\f$ as
295  *
296  * \f{align*}{
297  * \left.\int u(y) dy\right\rvert_{x}=&\sum_{i=0}^N \tilde{c}_i P_i(x) \\
298  * =&\left.\int\sum_{i=1}^{N}\frac{c_i}{2i+1}
299  * \left(P_{i+1}(y)-P_{i-1}(y)\right)dy\right\rvert_{x}
300  * + \tilde{c}_0 P_0(x) \\
301  * =&\sum_{i=1}^{N}\left(\frac{c_{i-1}}{2i-1} - \frac{c_{i+1}}{2i+3}\right)
302  * P_i(x) + \tilde{c}_0 P_0(x)
303  * \f}
304  *
305  * Thus we get that for \f$i>0\f$
306  *
307  * \f{align*}{
308  * \tilde{c}_i=\frac{c_{i-1}}{2i-1}-\frac{c_{i+1}}{2i+3}
309  * \f}
310  *
311  * and \f$\tilde{c}_0\f$ is a constant of integration, which we choose such that
312  * the integral is 0 at the left boundary of the domain (\f$x=-1\f$). The
313  * condition for this is:
314  *
315  * \f{align*}{
316  * \tilde{c}_0=\sum_{i=1}^{N}(-1)^{i+1}\tilde{c}_i
317  * \f}
318  *
319  * The matrix returned by this function is the product of the tridiagonal matrix
320  * for the \f$\tilde{c}_i\f$ and the matrix for the boundary condition.
321  *
322  * #### Chebyshev Polynomials
323  *
324  * A similar derivation leads to the relations:
325  *
326  * \f{align*}{
327  * \tilde{c}_i=&\frac{c_{i-1}-c_{i+1}}{2i},&\mathrm{if}\;i>1 \\
328  * \tilde{c}_1=&c_0 - \frac{c_2}{2},&\mathrm{if}\;i=1 \\
329  * \f}
330  *
331  * We again have:
332  *
333  * \f{align*}{
334  * \tilde{c}_0=\sum_{i=1}^N(-1)^{i+1}\tilde{c}_i
335  * \f}
336  *
337  * These are then used to define the indefinite integration matrix.
338  *
339  * #### Jacobi Polynomials
340  *
341  * For general Jacobi polynomials \f$P^{(\alpha,\beta)}_n(x)\f$ given by
342  *
343  * \f{align*}{
344  * (1-x)^\alpha(1+x)^\beta P^{(\alpha,\beta)}_n(x)=\frac{(-1)^n}{2^n n!}
345  * \frac{d^n}{dx^n}\left[(1-x)^{\alpha+n}(1+x)^{\beta+n}\right]
346  * \f}
347  *
348  * we have that
349  *
350  * \f{align*}{
351  * P^{(\alpha,\beta)}_n(x)=\frac{d}{dx}\left(
352  * b^{(\alpha,\beta)}_{n-1,n}P^{(\alpha,\beta)}_{n-1}(x) +
353  * b^{(\alpha,\beta)}_{n,n}P^{(\alpha,\beta)}_n(x) +
354  * b^{(\alpha,\beta)}_{n+1,n}P^{(\alpha,\beta)}_{n+1}(x)
355  * \right)
356  * \f}
357  *
358  * where
359  *
360  * \f{align*}{
361  * b^{(\alpha,\beta)}_{n-1,n}=&-\frac{1}{n+\alpha+\beta}
362  * a^{(\alpha,\beta)}_{n-1,n} \\
363  * b^{(\alpha,\beta)}_{n,n}=&-\frac{2}{\alpha+\beta}
364  * a^{(\alpha,\beta)}_{n,n} \\
365  * b^{(\alpha,\beta)}_{n+1,n}=&\frac{1}{n+1}
366  * a^{(\alpha,\beta)}_{n+1,n} \\
367  * a^{(\alpha,\beta)}_{n-1,n}=&\frac{2(n+\alpha)(n+\beta)}
368  * {(2n+\alpha+\beta+1)(2n+\alpha+\beta)} \\
369  * a^{(\alpha,\beta)}_{n,n}=&-\frac{\alpha^2-\beta^2}
370  * {(2n+\alpha+\beta+2)(2n+\alpha+\beta)} \\
371  * a^{(\alpha,\beta)}_{n-1,n}=&\frac{2(n+1)(n+\alpha+\beta+1)}
372  * {(2n+\alpha+\beta+2)(2n+\alpha+\beta+1)}
373  * \f}
374  *
375  * Following the same derivation we get that
376  *
377  * \f{align*}{
378  * \tilde{c}_i=c_{i+1}b^{(\alpha,\beta)}_{i,i+1}
379  * +c_i b^{(\alpha,\beta)}_{i,i}
380  * +c_{i-1}b^{(\alpha,\beta)}_{i,i-1}
381  * \f}
382  *
383  * and the boundary condition is
384  *
385  * \f{align*}{
386  * \tilde{c}_0=\sum_{i=1}^N(-1)^{i+1}
387  * \frac{\Gamma(i+\alpha+1)}{i!\Gamma(\alpha+1)} \tilde{c}_i
388  * \f}
389  *
390  * where \f$\Gamma(x)\f$ is the Gamma function.
391  */

471 /*!
472  * \brief %Matrix used to transform from the spectral coefficients (modes) of a
473  * function to its nodal coefficients. Also referred to as the _Vandermonde
474  * matrix_.
475  *
476  * \details The Vandermonde matrix is computed as
477  * \f$\mathcal{V}_{ij}=\Phi_j(\xi_i)\f$ where the \f$\Phi_j(x)\f$ are the
478  * spectral basis functions used in the modal expansion
479  * \f$u(x)=\sum_j \widetilde{u}_j\Phi_j(x)\f$, e.g. normalized Legendre
480  * polynomials, and the \f$\xi_i\f$ are the collocation points used to construct
481  * the interpolating Lagrange polynomials in the nodal expansion
482  * \f$u(x)=\sum_j u_j l_j(x)\f$. Then the Vandermonde matrix arises as
483  * \f$u(\xi_i)=u_i=\sum_j \widetilde{u}_j\Phi_j(\xi_i)=\sum_j
484  * \mathcal{V}_{ij}\widetilde{u}_j\f$.
485  *
486  * \param num_points The number of collocation points
487 
488  * \see nodal_to_modal_matrix(size_t)
489  */
490 template <Basis BasisType, Quadrature QuadratureType>
491 const Matrix& modal_to_nodal_matrix(size_t num_points) noexcept;

501 /*!
502  * \brief %Matrix used to transform from the nodal coefficients of a function to
503  * its spectral coefficients (modes). Also referred to as the inverse
504  * _Vandermonde matrix_.
505  *
506  * \details This is the inverse to the Vandermonde matrix \f$\mathcal{V}\f$
507  * computed in modal_to_nodal_matrix(size_t). It can be computed
508  * analytically for Gauss quadrature by evaluating
509  * \f$\sum_j\mathcal{V}^{-1}_{ij}u_j=\widetilde{u}_i=
510  * \frac{(u,\Phi_i)}{\gamma_i}\f$
511  * for a Lagrange basis function \f$u(x)=l_k(x)\f$ to find
512  * \f$\mathcal{V}^{-1}_{ij}=\mathcal{V}_{ji}\frac{w_j}{\gamma_i}\f$ where the
513  * \f$w_j\f$ are the Gauss quadrature weights and \f$\gamma_i\f$ is the norm
514  * square of the spectral basis function \f$\Phi_i\f$.
515  *
516  * \param num_points The number of collocation points
517  *
518  * \see modal_to_nodal_matrix(size_t)
519  */
520 template <Basis BasisType, Quadrature QuadratureType>
521 const Matrix& nodal_to_modal_matrix(size_t num_points) noexcept;
