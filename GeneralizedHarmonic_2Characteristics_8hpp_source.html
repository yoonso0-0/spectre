32 /*!
33  * \brief Compute the characteristic speeds for the generalized harmonic system.
34  *
35  * Computes the speeds as described in "A New Generalized Harmonic
36  * Evolution System" by Lindblom et. al \cite Lindblom2005qh
37  * [see text following Eq.(34)]. The characteristic fields' names used here
38  * differ from this paper:
39  *
40  * \f{align*}
41  * \mathrm{SpECTRE} && \mathrm{Lindblom} \\
42  * u^{\psi}_{ab} && u^\hat{0}_{ab} \\
43  * u^0_{iab} && u^\hat{2}_{iab} \\
44  * u^{\pm}_{ab} && u^{\hat{1}\pm}_{ab}
45  * \f}
46  *
47  * The corresponding characteristic speeds \f$v\f$ are given in the text between
48  * Eq.(34) and Eq.(35) of \cite Lindblom2005qh :
49  *
50  * \f{align*}
51  * v_{\psi} =& -(1 + \gamma_1) n_k N^k \\
52  * v_{0} =& -n_k N^k \\
53  * v_{\pm} =& -n_k N^k \pm N
54  * \f}
55  *
56  * where \f$N, N^k\f$ are the lapse and shift respectively, \f$\gamma_1\f$ is a
57  * constraint damping parameter, and \f$n_k\f$ is the unit normal to the
58  * surface.
59  */
60 template <size_t Dim, typename Frame>
61 std::array<DataVector, 4> characteristic_speeds(
62  const Scalar<DataVector>& gamma_1, const Scalar<DataVector>& lapse,
63  const tnsr::I<DataVector, Dim, Frame>& shift,
64  const tnsr::i<DataVector, Dim, Frame>& unit_normal_one_form) noexcept;

96 /*!
97  * \brief Computes characteristic fields from evolved fields
98  *
99  * \ref CharacteristicFieldsCompute and
100  * \ref EvolvedFieldsFromCharacteristicFieldsCompute convert between
101  * characteristic and evolved fields for the generalized harmonic system.
102  *
103  * \ref CharacteristicFieldsCompute computes
104  * characteristic fields as described in "A New Generalized Harmonic
105  * Evolution System" by Lindblom et. al \cite Lindblom2005qh .
106  * Their names used here differ from this paper:
107  *
108  * \f{align*}
109  * \mathrm{SpECTRE} && \mathrm{Lindblom} \\
110  * u^{\psi}_{ab} && u^\hat{0}_{ab} \\
111  * u^0_{iab} && u^\hat{2}_{iab} \\
112  * u^{\pm}_{ab} && u^{\hat{1}\pm}_{ab}
113  * \f}
114  *
115  * The characteristic fields \f$u\f$ are given in terms of the evolved fields by
116  * Eq.(32) - (34) of \cite Lindblom2005qh, respectively:
117  * \f{align*}
118  * u^{\psi}_{ab} =& \psi_{ab} \\
119  * u^0_{iab} =& (\delta^k_i - n_i n^k) \Phi_{kab} := P^k_i \Phi_{kab} \\
120  * u^{\pm}_{ab} =& \Pi_{ab} \pm n^i \Phi_{iab} - \gamma_2\psi_{ab}
121  * \f}
122  *
123  * where \f$\psi_{ab}\f$ is the spacetime metric, \f$\Pi_{ab}\f$ and
124  * \f$\Phi_{iab}\f$ are evolved generalized harmonic fields introduced by first
125  * derivatives of \f$\psi_{ab}\f$, \f$\gamma_2\f$ is a constraint damping
126  * parameter, and \f$n_k\f$ is the unit normal to the surface.
127  *
128  * \ref EvolvedFieldsFromCharacteristicFieldsCompute computes evolved fields
129  * \f$w\f$ in terms of the characteristic fields. This uses the inverse of
130  * above relations:
131  *
132  * \f{align*}
133  * \psi_{ab} =& u^{\psi}_{ab}, \\
134  * \Pi_{ab} =& \frac{1}{2}(u^{+}_{ab} + u^{-}_{ab}) + \gamma_2 u^{\psi}_{ab}, \\
135  * \Phi_{iab} =& \frac{1}{2}(u^{+}_{ab} - u^{-}_{ab}) n_i + u^0_{iab}.
136  * \f}
137  *
138  * The corresponding characteristic speeds \f$v\f$ are computed by
139  * \ref CharacteristicSpeedsCompute .
140  */
141 template <size_t Dim, typename Frame>
142 typename Tags::CharacteristicFields<Dim, Frame>::type characteristic_fields(
143  const Scalar<DataVector>& gamma_2,
144  const tnsr::II<DataVector, Dim, Frame>& inverse_spatial_metric,
