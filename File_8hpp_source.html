31 /*!
32  * \ingroup HDF5Group
33  * \brief Opens an HDF5 file for access and allows manipulation of data
34  *
35  * Opens an HDF5 file either in ReadOnly or ReadWrite mode depending on the
36  * template parameter `Access_t`. In ReadWrite mode h5::Object's can be inserted
37  * into the file, and objects can be retrieved to have their data manipulated.
38  * Example objects are dat files, text files, and volume data files. A single
39  * H5File can contain many different objects so that the number of files stored
40  * during a simulation is reduced.
41  *
42  * When an h5::object inside an H5File is opened or created the H5File object
43  * holds a copy of the h5::object. Only one object can be open at a time, which
44  * means if a reference to the object is kept around after the H5File's current
45  * object is closed there is a dangling reference.
46  *
47  * \example
48  * To open a file for read-write access:
49  * \snippet Test_H5.cpp h5file_readwrite_file
50  *
51  * \note The dangling reference issue could be fixed by having a function in
52  * addition to `get` that takes a lambda. The lambda takes exactly one parameter
53  * of the type of the h5::Object it will be operating on. While this approach is
54  * likely to be syntactically strange for many users it will most likely be more
55  * performant than the `shared_ptr` solution.
56  *
57  * @tparam Access_t either h5::AccessType::ReadWrite or h5::AccessType::ReadOnly
58  */

188 template <AccessType Access_t>
189 template <typename ObjectType, typename... Args,
190  typename std::enable_if_t<((void)sizeof(ObjectType),
191  Access_t == AccessType::ReadWrite)>*>
192 ObjectType& H5File<Access_t>::get(const std::string& path, Args&&... args) {
193  // Ensure we call the const version of the get function to avoid infinite
194  // recursion. The reason this is implemented in this manner is to avoid code
195  // duplication.
196  // clang-tidy: do not use const_cast
197  return const_cast<ObjectType&>( // NOLINT
198  static_cast<H5File<Access_t> const*>(this)->get<ObjectType>(
199  path, std::forward<Args>(args)...));
200 }

202 template <AccessType Access_t>
203 template <typename ObjectType, typename... Args>
204 const ObjectType& H5File<Access_t>::get(const std::string& path,
205  Args&&... args) const {
206  try {
207  current_object_ = nullptr;
208  // C++17: structured bindings
209  auto exists_group_name = check_if_object_exists<ObjectType>(path);
210  hid_t group_id = std::get<1>(exists_group_name).id();
211  if (not std::get<0>(exists_group_name)) {
212  current_object_ = nullptr;
213  CHECK_H5(H5Fclose(file_id_),
214  "Failed to close file: '" << file_name_ << "'");
215  ERROR("Cannot open the object '" << path + ObjectType::extension()
216  << "' because it does not exist.");
217  }
218  current_object_ = std::make_unique<ObjectType>(
219  std::get<0>(exists_group_name),
220  std::move(std::get<1>(exists_group_name)), group_id,
221  std::move(std::get<2>(exists_group_name)), std::forward<Args>(args)...);
222  return dynamic_cast<const ObjectType&>(*current_object_);
223  // LCOV_EXCL_START
224  } catch (const std::bad_cast& e) {
225  current_object_ = nullptr;
226  CHECK_H5(H5Fclose(file_id_),
227  "Failed to close file: '" << file_name_ << "'");
228  ERROR("Failed to cast " << path << " to "
229  << pretty_type::get_name<ObjectType>()
230  << ".\nCast error: " << e.what());
231  } catch (const std::exception& e) {
232  current_object_ = nullptr;
233  CHECK_H5(H5Fclose(file_id_),
234  "Failed to close file: '" << file_name_ << "'");
235  ERROR("Unknown exception caught: " << e.what());
236  // LCOV_EXCL_STOP
237  }
238 }

240 template <AccessType Access_t>
241 template <typename ObjectType, typename... Args>
242 ObjectType& H5File<Access_t>::insert(const std::string& path, Args&&... args) {
243  static_assert(AccessType::ReadWrite == Access_t,
244  "Can only insert into ReadWrite access H5 files.");
245  current_object_ = nullptr;
246  // C++17: structured bindings
247  auto exists_group_name = check_if_object_exists<ObjectType>(path);
248  if (std::get<0>(exists_group_name)) {
249  current_object_ = nullptr;
250  CHECK_H5(H5Fclose(file_id_),
251  "Failed to close file: '" << file_name_ << "'");
252  ERROR(
253  "Cannot insert an Object that already exists. Failed to add Object "
254  "named: "
255  << path);
256  }
257 
258  hid_t group_id = std::get<1>(exists_group_name).id();
259  return convert_to_derived<ObjectType>(
260  current_object_ = std::make_unique<ObjectType>(
261  std::get<0>(exists_group_name),
262  std::move(std::get<1>(exists_group_name)), group_id,
263  std::move(std::get<2>(exists_group_name)),
264  std::forward<Args>(args)...));
265 }

317 template <AccessType Access_t>
318 template <typename ObjectType>
319 std::tuple<bool, detail::OpenGroup, std::string>
320 H5File<Access_t>::check_if_object_exists(const std::string& path) const {
321  std::string name_only = "/";
322  if (path != "/") {
323  name_only = file_system::get_file_name(path);
324  }
325  const std::string name_with_extension = name_only + ObjectType::extension();
326  detail::OpenGroup group(file_id_, file_system::get_parent_path(path),
327  Access_t);
328 #pragma GCC diagnostic push
329 #pragma GCC diagnostic ignored "-Wold-style-cast"
330  const bool object_exists =
331  name_with_extension == "/" or
332  H5Lexists(group.id(), name_with_extension.c_str(), H5P_DEFAULT) or
333  H5Aexists(group.id(), name_with_extension.c_str());
334 #pragma GCC diagnostic pop
335  return std::make_tuple(object_exists, std::move(group), std::move(name_only));
336 }
