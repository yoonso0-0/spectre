23 template <size_t Dim, typename... PrimalFields, typename... AuxiliaryFields>
24 struct FirstOrderFluxesImpl<Dim, tmpl::list<PrimalFields...>,
25  tmpl::list<AuxiliaryFields...>> {
26  template <typename VarsTags, typename FluxesComputer, typename... FluxesArgs>
27  static constexpr void apply(
28  const gsl::not_null<Variables<db::wrap_tags_in<
29  ::Tags::Flux, VarsTags, tmpl::size_t<Dim>, Frame::Inertial>>*>
30  fluxes,
31  const Variables<VarsTags>& vars, const FluxesComputer& fluxes_computer,
32  const FluxesArgs&... fluxes_args) noexcept {
33  // Compute fluxes for primal fields
34  fluxes_computer.apply(
35  make_not_null(&get<::Tags::Flux<PrimalFields, tmpl::size_t<Dim>,
36  Frame::Inertial>>(*fluxes))...,
37  fluxes_args..., get<AuxiliaryFields>(vars)...);
38  // Compute fluxes for auxiliary fields
39  fluxes_computer.apply(
40  make_not_null(&get<::Tags::Flux<AuxiliaryFields, tmpl::size_t<Dim>,
41  Frame::Inertial>>(*fluxes))...,
42  fluxes_args..., get<PrimalFields>(vars)...);
43  }
44 };

46 template <size_t Dim, typename PrimalFields, typename AuxiliaryFields,
47  typename SourcesComputer>
48 struct FirstOrderSourcesImpl;
49 
50 template <size_t Dim, typename... PrimalFields, typename... AuxiliaryFields,
51  typename SourcesComputer>
52 struct FirstOrderSourcesImpl<Dim, tmpl::list<PrimalFields...>,
53  tmpl::list<AuxiliaryFields...>, SourcesComputer> {
54  template <typename VarsTags, typename... SourcesArgs>
55  static constexpr void apply(
56  const gsl::not_null<
57  Variables<db::wrap_tags_in<::Tags::Source, VarsTags>>*>
58  sources,
59  const Variables<VarsTags>& vars,
60  const Variables<db::wrap_tags_in<
61  ::Tags::Flux, VarsTags, tmpl::size_t<Dim>, Frame::Inertial>>& fluxes,
62  const SourcesArgs&... sources_args) noexcept {
63  // Set auxiliary field sources to the auxiliary field values to begin with.
64  // This is the standard choice, since the auxiliary equations define the
65  // auxiliary variables. Source computers can adjust or add to these sources.
66  tmpl::for_each<tmpl::list<AuxiliaryFields...>>(
67  [&sources, &vars](const auto auxiliary_field_tag_v) noexcept {
68  using auxiliary_field_tag =
69  tmpl::type_from<decltype(auxiliary_field_tag_v)>;
70  get<::Tags::Source<auxiliary_field_tag>>(*sources) =
71  get<auxiliary_field_tag>(vars);
72  });
73  // Call into the sources computer to set primal field sources and possibly
74  // adjust auxiliary field sources. The sources depend on the primal and the
75  // auxiliary variables. However, we pass the volume fluxes instead of the
76  // auxiliary variables to the source computer as an optimization so they
77  // don't have to be re-computed.
78  SourcesComputer::apply(
79  make_not_null(&get<::Tags::Source<PrimalFields>>(*sources))...,
80  make_not_null(&get<::Tags::Source<AuxiliaryFields>>(*sources))...,
81  sources_args..., get<PrimalFields>(vars)...,
82  get<::Tags::Flux<PrimalFields, tmpl::size_t<Dim>, Frame::Inertial>>(
83  fluxes)...);
84  }
85 };

147  first_order_operator_detail::FirstOrderSourcesImpl<
148  Dim, PrimalFields, AuxiliaryFields,

170 /*!
171  * \brief Compute the bulk contribution to the operator represented by the
172  * `OperatorTags`.
173  *
174  * This function computes \f$A(u)=-\partial_i F^i(u) + S(u)\f$, where \f$F^i\f$
175  * and \f$S\f$ are the fluxes and sources of the system of first-order PDEs,
176  * respectively. They are defined such that \f$A(u(x))=f(x)\f$ is the full
177  * system of equations, with \f$f(x)\f$ representing sources that are
178  * independent of the variables \f$u\f$. In a DG setting, boundary contributions
179  * can be added to \f$A(u)\f$ to build the full DG operator.
180  */
181 template <typename OperatorTags, typename DivFluxesTags, typename SourcesTags>
182 void first_order_operator(
183  const gsl::not_null<Variables<OperatorTags>*> operator_applied_to_vars,
184  const Variables<DivFluxesTags>& div_fluxes,
185  const Variables<SourcesTags>& sources) noexcept {
186  *operator_applied_to_vars = sources - div_fluxes;
187 }

189 /*!
190  * \brief Mutating DataBox invokable to compute the bulk contribution to the
191  * operator represented by the `OperatorTag` applied to the `VarsTag`
192  *
193  * \see `elliptic::first_order_operator`
194  *
195  * We generally build the operator \f$A(u)\f$ to perform elliptic solver
196  * iterations. This invokable can be used to build operators for both the linear
197  * solver and the nonlinear solver iterations by providing the appropriate
198  * `OperatorTag` and `VarsTag`. For example, the `OperatorTag` for the linear
199  * solver would typically be `LinearSolver::Tags::OperatorAppliedTo`. It is the
200  * elliptic equivalent to the time derivative in evolution schemes. Instead of
201  * using it to evolve the variables, the iterative linear solver uses it to
202  * compute a better approximation to the equation \f$A(u)=f\f$ (as detailed
203  * above).
204  *
205  * With:
206  * - `operator_tag` = `db::add_tag_prefix<OperatorTag, VarsTag>`
207  * - `fluxes_tag` = `db::add_tag_prefix<::Tags::Flux, VarsTag,
208  * tmpl::size_t<Dim>, Frame::Inertial>`
209  * - `div_fluxes_tag` = `db::add_tag_prefix<Tags::div, fluxes_tag>`
210  * - `sources_tag` = `db::add_tag_prefix<::Tags::Source, VarsTag>`
211  *
212  * Uses:
213  * - DataBox:
214  * - `div_fluxes_tag`
215  * - `sources_tag`
216  *
217  * DataBox changes:
218  * - Modifies:
219  * - `operator_tag`
220  */
221 template <size_t Dim, template <typename> class OperatorTag, typename VarsTag>
222 struct FirstOrderOperator {
223  private:
224  using operator_tag = db::add_tag_prefix<OperatorTag, VarsTag>;
225  using fluxes_tag = db::add_tag_prefix<::Tags::Flux, VarsTag,
226  tmpl::size_t<Dim>, Frame::Inertial>;
227  using div_fluxes_tag = db::add_tag_prefix<::Tags::div, fluxes_tag>;
228  using sources_tag = db::add_tag_prefix<::Tags::Source, VarsTag>;
229 
230  public:
231  using return_tags = tmpl::list<operator_tag>;
232  using argument_tags = tmpl::list<div_fluxes_tag, sources_tag>;
233  static constexpr auto apply =
234  &first_order_operator<typename operator_tag::tags_list,
235  typename div_fluxes_tag::tags_list,
236  typename sources_tag::tags_list>;
237 };
