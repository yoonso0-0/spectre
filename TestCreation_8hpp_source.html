16 template <typename Tag, typename = void>
17 struct AddGroups {
18  template <typename OptionTag = Tag>
19  static std::string apply(std::string construction_string) noexcept {
20  construction_string.insert(0, 2, ' ');
21  construction_string =
22  boost::algorithm::replace_all_copy(construction_string, "\n", "\n ");
23  construction_string.insert(0, Options::name<OptionTag>() + ":\n");
24  return construction_string;
25  }
26 };

28 template <typename Tag>
29 struct AddGroups<Tag, std::void_t<typename Tag::group>> {
30  static std::string apply(const std::string& construction_string) noexcept {
31  return AddGroups<typename Tag::group>::apply(
32  AddGroups<void>::template apply<Tag>(construction_string));
33  }

46 /// \ingroup TestingFrameworkGroup
47 /// Construct an object or enum from a given string.
48 ///
49 /// When creating a non-enum option the string must not contain the name of the
50 /// option (specifically, the struct being created from options, which is the
51 /// name of the struct by default if no `name()` function is present). For
52 /// example, to create a struct named `ClassWithoutMetavariables` with an option
53 /// tag named `SizeT` the following would be used:
54 ///
55 /// \snippet Test_TestCreation.cpp size_t_argument
56 ///
57 /// When creating an enum option the string must not contain the name of the
58 /// enum being constructed. The following is an example of an enum named `Color`
59 /// with a member `Purple` being created from options.
60 ///
61 /// \snippet Test_TestCreation.cpp enum_purple
62 ///
63 /// Option tags can be tested by passing them as the second template parameter.
64 /// If the metavariables are required to create the class then the metavariables
65 /// must be passed as the third template parameter. The default option tag is
66 /// `TestCreationOpt<T>`.
67 template <typename T, typename OptionTag = TestCreationOpt<T>,
68  typename Metavariables = NoSuchType>
69 T test_creation(const std::string& construction_string) noexcept {
70  Options::Parser<tmpl::list<OptionTag>> options("");
71  options.parse(
72  TestCreation_detail::AddGroups<OptionTag>::apply(construction_string));
73  return options.template get<OptionTag, Metavariables>();
74 }

76 /// \ingroup TestingFrameworkGroup
77 /// Construct a factory object from a given string.
78 ///
79 /// The string must contain the name of the option (specifically, the struct
80 /// being created from options, which is the name of the struct). Note that it
81 /// is not the name of the base class, but the name of the derived class that
82 /// must be given. For example, to create a `BaseClass*` pointing to a derived
83 /// class of type `size_t_argument_base` with an option tag name `SizeT` the
84 /// following would be used:
85 ///
86 /// \snippet Test_TestCreation.cpp size_t_argument_base
87 ///
88 /// Option tags can be tested by passing them as the second template parameter.
89 /// If the metavariables are required to create the class then the metavariables
90 /// must be passed as the third template parameter. The default option tag is
91 /// `TestCreationOpt<std::unique_ptr<BaseClass>>`.
92 template <typename BaseClass,
93  typename OptionTag = TestCreationOpt<std::unique_ptr<BaseClass>>,
94  typename Metavariables = NoSuchType>
95 std::unique_ptr<BaseClass> test_factory_creation(
96  const std::string& construction_string) noexcept {
97  return TestHelpers::test_creation<std::unique_ptr<BaseClass>, OptionTag,
98  Metavariables>(construction_string);
99 }
