101 /*!
102  * \brief Process the worldtube data from modal metric components and
103  * derivatives with incorrectly normalized radial derivatives from an old
104  * version of SpEC to desired Bondi quantities, placing the result in the passed
105  * \ref DataBoxGroup.
106  *
107  * \details
108  * The mathematics are a bit complicated for all of the coordinate
109  * transformations that are necessary to obtain the Bondi gauge quantities.
110  * For full mathematical details, see the documentation for functions in
111  * `BoundaryData.hpp` and \cite Barkett2019uae \cite Bishop1998uk.
112  *
113  * This function takes as input the full set of ADM metric data and its radial
114  * and time derivatives on a two-dimensional surface of constant \f$r\f$ and
115  * \f$t\f$ in numerical coordinates. This data must be provided as spherical
116  * harmonic coefficients in the libsharp format. This data is provided in nine
117  * `Tensor`s.
118  *
119  * Sufficient tags to provide full worldtube boundary data at a particular
120  * time are set in `bondi_boundary_data`. In particular, the set of tags in
121  * `Tags::characteristic_worldtube_boundary_tags` in the provided \ref
122  * DataBoxGroup are assigned to the worldtube boundary values associated with
123  * the input metric components.
124  *
125  * The majority of the mathematical transformations are implemented as a set of
126  * individual cascaded functions below. The details of the manipulations that
127  * are performed to the input data may be found in the individual functions
128  * themselves, which are called in the following order:
129  * - `trigonometric_functions_on_swsh_collocation()`
130  * - `cartesian_to_spherical_coordinates_and_jacobians()`
131  * - `cartesian_spatial_metric_and_derivatives_from_unnormalized_spec_modes()`
132  * - `cartesian_shift_and_derivatives_from_unnormalized_spec_modes()`
133  * - `cartesian_lapse_and_derivatives_from_unnormalized_spec_modes()`
134  * - `GeneralizedHarmonic::phi()`
135  * - `gr::time_derivative_of_spacetime_metric`
136  * - `gr::spacetime_metric`
137  * - `generalized_harmonic_quantities()`
138  * - `worldtube_normal_and_derivatives()`
139  * - `null_vector_l_and_derivatives()`
140  * - `null_metric_and_derivative()`
141  * - `dlambda_null_metric_and_inverse()`
142  * - `bondi_r()`
143  * - `d_bondi_r()`
144  * - `dyads()`
145  * - `beta_worldtube_data()`
146  * - `bondi_u_worldtube_data()`
147  * - `bondi_w_worldtube_data()`
148  * - `bondi_j_worldtube_data()`
149  * - `dr_bondi_j()`
150  * - `d2lambda_bondi_r()`
151  * - `bondi_q_worldtube_data()`
152  * - `bondi_h_worldtube_data()`
153  * - `du_j_worldtube_data()`
154  */
155 template <typename TagList>
156 void create_bondi_boundary_data_from_unnormalized_spec_modes(
157  const gsl::not_null<Variables<TagList>*> bondi_boundary_data,
158  const tnsr::ii<ComplexModalVector, 3>& spatial_metric_coefficients,
159  const tnsr::ii<ComplexModalVector, 3>& dt_spatial_metric_coefficients,
160  const tnsr::ii<ComplexModalVector, 3>& dr_spatial_metric_coefficients,
161  const tnsr::I<ComplexModalVector, 3>& shift_coefficients,
162  const tnsr::I<ComplexModalVector, 3>& dt_shift_coefficients,
163  const tnsr::I<ComplexModalVector, 3>& dr_shift_coefficients,
164  const Scalar<ComplexModalVector>& lapse_coefficients,
165  const Scalar<ComplexModalVector>& dt_lapse_coefficients,
166  const Scalar<ComplexModalVector>& dr_lapse_coefficients,
167  const double extraction_radius, const size_t l_max) noexcept {
168  const size_t size = Spectral::Swsh::number_of_swsh_collocation_points(l_max);
169 
170  // Most allocations required for the full boundary computation are merged into
171  // a single, large Variables allocation. There remain a handful of cases in
172  // the computational functions called where an intermediate quantity that is
173  // not re-used is allocated rather than taking a buffer. These cases are
174  // marked with code comments 'Allocation'; In future, allocations are
175  // identified as a point to optimize, those buffers may be allocated here and
176  // passed as function arguments
177  Variables<tmpl::list<
178  Tags::detail::CosPhi, Tags::detail::CosTheta, Tags::detail::SinPhi,
179  Tags::detail::SinTheta, Tags::detail::CartesianCoordinates,
180  Tags::detail::CartesianToSphericalJacobian,
181  Tags::detail::InverseCartesianToSphericalJacobian,
182  gr::Tags::SpatialMetric<3, ::Frame::Inertial, DataVector>,
183  gr::Tags::InverseSpatialMetric<3, ::Frame::Inertial, DataVector>,
184  ::Tags::deriv<gr::Tags::SpatialMetric<3, ::Frame::Inertial, DataVector>,
185  tmpl::size_t<3>, ::Frame::Inertial>,
186  ::Tags::dt<gr::Tags::SpatialMetric<3, ::Frame::Inertial, DataVector>>,
187  gr::Tags::Shift<3, ::Frame::Inertial, DataVector>,
188  ::Tags::deriv<gr::Tags::Shift<3, ::Frame::Inertial, DataVector>,
189  tmpl::size_t<3>, ::Frame::Inertial>,
190  ::Tags::dt<gr::Tags::Shift<3, ::Frame::Inertial, DataVector>>,
191  gr::Tags::Lapse<DataVector>,
192  ::Tags::deriv<gr::Tags::Lapse<DataVector>, tmpl::size_t<3>,
193  ::Frame::Inertial>,
194  ::Tags::dt<gr::Tags::Lapse<DataVector>>,
195  gr::Tags::SpacetimeMetric<3, ::Frame::Inertial, DataVector>,
196  ::Tags::dt<gr::Tags::SpacetimeMetric<3, ::Frame::Inertial, DataVector>>,
197  GeneralizedHarmonic::Tags::Phi<3, ::Frame::Inertial>,
198  Tags::detail::WorldtubeNormal, ::Tags::dt<Tags::detail::WorldtubeNormal>,
199  Tags::detail::NullL, ::Tags::dt<Tags::detail::NullL>,
200  // for the detail function called at the end
201  gr::Tags::SpacetimeMetric<3, Frame::RadialNull, DataVector>,
202  ::Tags::dt<gr::Tags::SpacetimeMetric<3, Frame::RadialNull, DataVector>>,
203  gr::Tags::InverseSpacetimeMetric<3, Frame::RadialNull, DataVector>,
204  Tags::detail::AngularDNullL,
205  Tags::detail::DLambda<
206  gr::Tags::SpacetimeMetric<3, Frame::RadialNull, DataVector>>,
207  Tags::detail::DLambda<
208  gr::Tags::InverseSpacetimeMetric<3, Frame::RadialNull, DataVector>>,
209  ::Tags::spacetime_deriv<Tags::detail::RealBondiR, tmpl::size_t<3>,
210  Frame::RadialNull>,
211  Tags::detail::DLambda<Tags::detail::DLambda<Tags::detail::RealBondiR>>,
212  ::Tags::deriv<Tags::detail::DLambda<Tags::detail::RealBondiR>,
213  tmpl::size_t<2>, Frame::RadialNull>,
214  ::Tags::TempScalar<0, DataVector>>>
215  computation_variables{size};
216 
217  Variables<
218  tmpl::list<::Tags::SpinWeighted<::Tags::TempScalar<0, ComplexDataVector>,
219  std::integral_constant<int, 0>>,
220  ::Tags::SpinWeighted<::Tags::TempScalar<0, ComplexDataVector>,
221  std::integral_constant<int, 1>>>>
222  derivative_buffers{size};
223  auto& cos_phi = get<Tags::detail::CosPhi>(computation_variables);
224  auto& cos_theta = get<Tags::detail::CosTheta>(computation_variables);
225  auto& sin_phi = get<Tags::detail::SinPhi>(computation_variables);
226  auto& sin_theta = get<Tags::detail::SinTheta>(computation_variables);
227  trigonometric_functions_on_swsh_collocation(
228  make_not_null(&cos_phi), make_not_null(&cos_theta),
229  make_not_null(&sin_phi), make_not_null(&sin_theta), l_max);
230 
231  // NOTE: to handle the singular values of polar coordinates, the phi
232  // components of all tensors are scaled according to their sin(theta)
233  // prefactors.
234  // so, any down-index component get<2>(A) represents 1/sin(theta) A_\phi,
235  // and any up-index component get<2>(A) represents sin(theta) A^\phi.
236  // This holds for Jacobians, and so direct application of the Jacobians
237  // brings the factors through.
238  auto& cartesian_coords =
239  get<Tags::detail::CartesianCoordinates>(computation_variables);
240  auto& cartesian_to_spherical_jacobian =
241  get<Tags::detail::CartesianToSphericalJacobian>(computation_variables);
242  auto& inverse_cartesian_to_spherical_jacobian =
243  get<Tags::detail::InverseCartesianToSphericalJacobian>(
244  computation_variables);
245  cartesian_to_spherical_coordinates_and_jacobians(
246  make_not_null(&cartesian_coords),
247  make_not_null(&cartesian_to_spherical_jacobian),
248  make_not_null(&inverse_cartesian_to_spherical_jacobian), cos_phi,
249  cos_theta, sin_phi, sin_theta, extraction_radius);
250 
251  auto& cartesian_spatial_metric =
252  get<gr::Tags::SpatialMetric<3, ::Frame::Inertial, DataVector>>(
253  computation_variables);
254  auto& inverse_spatial_metric =
255  get<gr::Tags::InverseSpatialMetric<3, ::Frame::Inertial, DataVector>>(
256  computation_variables);
257  auto& d_cartesian_spatial_metric = get<
258  ::Tags::deriv<gr::Tags::SpatialMetric<3, ::Frame::Inertial, DataVector>,
259  tmpl::size_t<3>, ::Frame::Inertial>>(computation_variables);
260  auto& dt_cartesian_spatial_metric = get<
261  ::Tags::dt<gr::Tags::SpatialMetric<3, ::Frame::Inertial, DataVector>>>(
262  computation_variables);
263  auto& interpolation_buffer =
264  get<::Tags::SpinWeighted<::Tags::TempScalar<0, ComplexDataVector>,
265  std::integral_constant<int, 0>>>(
266  derivative_buffers);
267  Scalar<SpinWeighted<ComplexModalVector, 0>> interpolation_modal_buffer{size};
268  auto& eth_buffer =
269  get<::Tags::SpinWeighted<::Tags::TempScalar<0, ComplexDataVector>,
270  std::integral_constant<int, 1>>>(
271  derivative_buffers);
272  auto& radial_correction_factor =
273  get<::Tags::TempScalar<0, DataVector>>(computation_variables);
274  cartesian_spatial_metric_and_derivatives_from_unnormalized_spec_modes(
275  make_not_null(&cartesian_spatial_metric),
276  make_not_null(&inverse_spatial_metric),
277  make_not_null(&d_cartesian_spatial_metric),
278  make_not_null(&dt_cartesian_spatial_metric),
279  make_not_null(&interpolation_modal_buffer),
280  make_not_null(&interpolation_buffer), make_not_null(&eth_buffer),
281  make_not_null(&radial_correction_factor), spatial_metric_coefficients,
282  dr_spatial_metric_coefficients, dt_spatial_metric_coefficients,
283  inverse_cartesian_to_spherical_jacobian, cartesian_coords, l_max);
284 
285  auto& cartesian_shift =
286  get<gr::Tags::Shift<3, ::Frame::Inertial, DataVector>>(
287  computation_variables);
288  auto& d_cartesian_shift =
289  get<::Tags::deriv<gr::Tags::Shift<3, ::Frame::Inertial, DataVector>,
290  tmpl::size_t<3>, ::Frame::Inertial>>(
291  computation_variables);
292  auto& dt_cartesian_shift =
293  get<::Tags::dt<gr::Tags::Shift<3, ::Frame::Inertial, DataVector>>>(
294  computation_variables);
295 
296  cartesian_shift_and_derivatives_from_unnormalized_spec_modes(
297  make_not_null(&cartesian_shift), make_not_null(&d_cartesian_shift),
298  make_not_null(&dt_cartesian_shift),
299  make_not_null(&interpolation_modal_buffer),
300  make_not_null(&interpolation_buffer), make_not_null(&eth_buffer),
301  shift_coefficients, dr_shift_coefficients, dt_shift_coefficients,
302  inverse_cartesian_to_spherical_jacobian, radial_correction_factor, l_max);
303 
304  auto& cartesian_lapse =
305  get<gr::Tags::Lapse<DataVector>>(computation_variables);
306  auto& d_cartesian_lapse =
307  get<::Tags::deriv<gr::Tags::Lapse<DataVector>, tmpl::size_t<3>,
308  ::Frame::Inertial>>(computation_variables);
309  auto& dt_cartesian_lapse =
310  get<::Tags::dt<gr::Tags::Lapse<DataVector>>>(computation_variables);
311  cartesian_lapse_and_derivatives_from_unnormalized_spec_modes(
312  make_not_null(&cartesian_lapse), make_not_null(&d_cartesian_lapse),
313  make_not_null(&dt_cartesian_lapse),
314  make_not_null(&interpolation_modal_buffer),
315  make_not_null(&interpolation_buffer), make_not_null(&eth_buffer),
316  lapse_coefficients, dr_lapse_coefficients, dt_lapse_coefficients,
317  inverse_cartesian_to_spherical_jacobian, radial_correction_factor, l_max);
318 
319  auto& phi = get<GeneralizedHarmonic::Tags::Phi<3, ::Frame::Inertial>>(
320  computation_variables);
321  auto& dt_spacetime_metric = get<
322  ::Tags::dt<gr::Tags::SpacetimeMetric<3, ::Frame::Inertial, DataVector>>>(
323  computation_variables);
324  auto& spacetime_metric =
325  get<gr::Tags::SpacetimeMetric<3, ::Frame::Inertial, DataVector>>(
326  computation_variables);
327  GeneralizedHarmonic::phi(
328  make_not_null(&phi), cartesian_lapse, d_cartesian_lapse, cartesian_shift,
329  d_cartesian_shift, cartesian_spatial_metric, d_cartesian_spatial_metric);
330  gr::time_derivative_of_spacetime_metric(
331  make_not_null(&dt_spacetime_metric), cartesian_lapse, dt_cartesian_lapse,
332  cartesian_shift, dt_cartesian_shift, cartesian_spatial_metric,
333  dt_cartesian_spatial_metric);
334  gr::spacetime_metric(make_not_null(&spacetime_metric), cartesian_lapse,
335  cartesian_shift, cartesian_spatial_metric);
336 
337  auto& dt_worldtube_normal =
338  get<::Tags::dt<Tags::detail::WorldtubeNormal>>(computation_variables);
339  auto& worldtube_normal =
340  get<Tags::detail::WorldtubeNormal>(computation_variables);
341  worldtube_normal_and_derivatives(
342  make_not_null(&worldtube_normal), make_not_null(&dt_worldtube_normal),
343  cos_phi, cos_theta, spacetime_metric, dt_spacetime_metric, sin_phi,
344  sin_theta, inverse_spatial_metric);
345 
346  auto& du_null_l = get<::Tags::dt<Tags::detail::NullL>>(computation_variables);
347  auto& null_l = get<Tags::detail::NullL>(computation_variables);
348  null_vector_l_and_derivatives(
349  make_not_null(&du_null_l), make_not_null(&null_l), dt_worldtube_normal,
350  dt_cartesian_lapse, dt_spacetime_metric, dt_cartesian_shift,
351  cartesian_lapse, spacetime_metric, cartesian_shift, worldtube_normal);
352 
353  // pass to the next step that is common between the 'modal' input and 'GH'
354  // input strategies
355  detail::create_bondi_boundary_data(
356  bondi_boundary_data, make_not_null(&computation_variables),
357  make_not_null(&derivative_buffers), dt_spacetime_metric, phi,
358  spacetime_metric, null_l, du_null_l, cartesian_to_spherical_jacobian,
359  l_max, extraction_radius);
