50 /*!
51  * \brief A uniform rotation about the \f$z\f$ axis:
52  * \f{eqnarray*}
53  * x &\to& x \cos \alpha(t) - y \sin \alpha(t)\text{,} \\
54  * y &\to& x \sin \alpha(t) + y \cos \alpha(t)\text{,}
55  * \f}
56  * where \f$\alpha(t)\f$ is a `domain::FunctionsOfTime::FunctionOfTime`. For 3
57  * spatial dimensions, \f$z \to z\f$, and the rotation is implemented as a
58  * product of the 2D rotation and an identity map. The rotation is undefined
59  * (and therefore unimplemented here) for 1 spatial dimension.
60  */
61 template <size_t MeshDim>
62 class UniformRotationAboutZAxis final : public TimeDependence<MeshDim> {
63  static_assert(
64  MeshDim > 1,
65  "UniformRotationAboutZAxis<MeshDim> undefined for MeshDim == 1");
66

69  using Rotation = domain::CoordinateMaps::TimeDependent::Rotation<2>;
70 
71  public:

74  domain::CoordinateMap<
75  Frame::Grid, Frame::Inertial,
76  CoordinateMaps::TimeDependent::ProductOf2Maps<Rotation, Identity>>>;
77 
78  static constexpr size_t mesh_dim = MeshDim;
