91 // Loop over all the phase dependent action list structs and for each one
92 // compute the list of DataBox types. The loop is initiated by taking the last
93 // DataBox of the previous phase. The results for each phase are stored in the
94 // `PhaseDependentDataBoxTypes` struct.
95 //
96 // The DataBox type that leaves one phase is what enters the next phase in the
97 // order that the PDAL's (phase dependent action lists) are specified in the
98 // parallel component. Switching from one phase to any other is also supported
99 // as long as the DataBox types match up correctly. While the code could be
100 // generalized to generically support switching from any phase to any other
101 // phase with different DataBox types, this would be a large tensor product at
102 // compile time leading to significantly more complex code and also longer
103 // compile times.
104 template <typename CumulativeDataboxTypes, typename InputDataBox,
105  typename InboxTagsList, typename Metavariables, typename ArrayIndex,
106  typename ParallelComponent, typename CurrentPhaseDepActionList,
107  typename... Rest>
108 struct build_databox_types<
109  CumulativeDataboxTypes, tmpl::list<CurrentPhaseDepActionList, Rest...>,
110  InputDataBox,
111 
112  InboxTagsList, Metavariables, ArrayIndex, ParallelComponent> {
113  using additional_args_list =
114  tmpl::list<tuples::tagged_tuple_from_typelist<InboxTagsList>,
115  Parallel::GlobalCache<Metavariables>, ArrayIndex,
116  typename CurrentPhaseDepActionList::action_list,
117  std::add_pointer_t<ParallelComponent>>;
118  using action_list_of_this_phase =
119  typename CurrentPhaseDepActionList::action_list;
120  using databox_types_for_this_phase =
121  typename build_action_return_types<action_list_of_this_phase>::template f<
122  InputDataBox, additional_args_list>;
123  using current_phase_dep_databox_types =
124  PhaseDependentDataBoxTypes<typename CurrentPhaseDepActionList::phase_type,
125  CurrentPhaseDepActionList::phase,
126  databox_types_for_this_phase>;
127  using cumulative_databox_types =
128  tmpl::push_back<CumulativeDataboxTypes, current_phase_dep_databox_types>;
129 
130  using type = typename build_databox_types<
131  cumulative_databox_types, tmpl::list<Rest...>,
132  tmpl::back<databox_types_for_this_phase>, InboxTagsList, Metavariables,
133  ArrayIndex, ParallelComponent>::type;
134 };
