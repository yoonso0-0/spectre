158 /// \ingroup DataBoxTagsGroup
159 /// \ingroup ComputationalDomainGroup
160 /// Specialisation of UnnormalizedFaceNormal for the external boundaries which
161 /// inverts the normals. Since ExternalBoundariesDirections are meant to
162 /// represent ghost elements, the normals should correspond to the normals in
163 /// said element, which are inverted with respect to the current element.
164 template <size_t VolumeDim, typename Frame>
165 struct InterfaceCompute<Tags::BoundaryDirectionsExterior<VolumeDim>,
166  UnnormalizedFaceNormalCompute<VolumeDim, Frame>>
167  : db::ComputeTag,
168  Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
169  Tags::UnnormalizedFaceNormal<VolumeDim, Frame>> {
170  using base = Tags::Interface<Tags::BoundaryDirectionsExterior<VolumeDim>,
171  Tags::UnnormalizedFaceNormal<VolumeDim, Frame>>;
172  using dirs = BoundaryDirectionsExterior<VolumeDim>;
173 
174  static std::string name() noexcept {
175  return "BoundaryDirectionsExterior<UnnormalizedFaceNormal>";
176  }
177  using return_type = std::unordered_map<::Direction<VolumeDim>,
178  tnsr::i<DataVector, VolumeDim, Frame>>;
179  static void function(const gsl::not_null<return_type*> normals,
180  const std::unordered_map<::Direction<VolumeDim>,
181  ::Mesh<VolumeDim - 1>>& meshes,
182  const ::ElementMap<VolumeDim, Frame>& map) noexcept {
183  for (const auto& direction_and_mesh : meshes) {
184  const auto& direction = direction_and_mesh.first;
185  const auto& mesh = direction_and_mesh.second;
186  auto internal_face_normal =
187  unnormalized_face_normal(mesh, map, direction);
188  std::transform(internal_face_normal.begin(), internal_face_normal.end(),
189  internal_face_normal.begin(), std::negate<>());
190  (*normals)[direction] = std::move(internal_face_normal);
191  }
192  }
193 
194  using argument_tags = tmpl::list<Tags::Interface<dirs, Mesh<VolumeDim - 1>>,
195  Tags::ElementMap<VolumeDim, Frame>>;
196 };
