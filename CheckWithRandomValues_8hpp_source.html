135 template <class F, class T, class TagsList, class Klass, class... ReturnTypes,
136  class... ArgumentTypes, class... MemberArgs, size_t... ResultIs,
137  size_t... ArgumentIs, size_t... MemberArgsIs>
138 void check_with_random_values_impl(
139  F&& f, const Klass& klass, const std::string& module_name,
140  const std::vector<std::string>& function_names, std::mt19937 generator,
141  std::array<std::uniform_real_distribution<>, sizeof...(ArgumentTypes)>
142  distributions,
143  const std::tuple<MemberArgs...>& member_args, const T& used_for_size,
144  tmpl::list<ReturnTypes...> /*return_types*/,
145  tmpl::list<ArgumentTypes...> /*argument_types*/,
146  std::index_sequence<ResultIs...> /*index_return_types*/,
147  std::index_sequence<ArgumentIs...> /*index_argument_types*/,
148  std::index_sequence<MemberArgsIs...> /*index_member_args*/,
149  // NOLINTNEXTLINE(readability-avoid-const-params-in-decls)
150  TagsList /*meta*/, const double epsilon = 1.0e-12) {
151  // Note: generator and distributions cannot be const.
152  std::tuple<ArgumentTypes...> args{
153  make_with_value<ArgumentTypes>(used_for_size, 0.0)...};
154  EXPAND_PACK_LEFT_TO_RIGHT(fill_with_random_values(
155  make_not_null(&std::get<ArgumentIs>(args)), make_not_null(&generator),
156  make_not_null(&(distributions[ArgumentIs]))));
157 
158  size_t count = 0;
159  tmpl::for_each<TagsList>([&f, &klass, &args, &used_for_size, &member_args,
160  &epsilon, &module_name, &function_names,
161  &count](auto tag) {
162  (void)member_args; // Avoid compiler warning
163  (void)used_for_size; // Avoid compiler warning
164  using Tag = tmpl::type_from<decltype(tag)>;
165  const auto result =
166  tuples::get<Tag>((klass.*f)(std::get<ArgumentIs>(args)...));
167  INFO("function: " << function_names[count]);
168  try {
169  CHECK_ITERABLE_CUSTOM_APPROX(
170  result,
171  (pypp::call<std::decay_t<decltype(result)>>(
172  module_name, function_names[count], std::get<ArgumentIs>(args)...,
173  forward_to_pypp<std::decay_t<decltype(result)>>(
174  std::get<MemberArgsIs>(member_args), used_for_size)...)),
175  Approx::custom().epsilon(epsilon).scale(1.0));
176  count++;
177  } catch (const std::exception& e) {
178  INFO("Python call failed: " << e.what());
179  CHECK(false);
180  }
181  });
182 }

184 template <class F, class T, class Klass, class... ArgumentTypes,
185  class... MemberArgs, size_t... ArgumentIs, size_t... MemberArgsIs>
186 void check_with_random_values_impl(
187  F&& f, const Klass& klass, const std::string& module_name,
188  const std::string& function_name, std::mt19937 generator,
189  std::array<std::uniform_real_distribution<>, sizeof...(ArgumentTypes)>
190  distributions,
191  const std::tuple<MemberArgs...>& member_args, const T& used_for_size,
192  tmpl::list<ArgumentTypes...> /*argument_types*/,
193  std::index_sequence<ArgumentIs...> /*index_argument_types*/,
194  std::index_sequence<MemberArgsIs...> /*index_member_args*/,
195  NoSuchType /*meta*/, const double epsilon = 1.0e-12) {
196  // Note: generator and distributions cannot be const.
197  using f_info = tt::function_info<cpp20::remove_cvref_t<F>>;
198  using ResultType = typename f_info::return_type;
199  std::tuple<ArgumentTypes...> args{
200  make_with_value<ArgumentTypes>(used_for_size, 0.0)...};
201  EXPAND_PACK_LEFT_TO_RIGHT(fill_with_random_values(
202  make_not_null(&std::get<ArgumentIs>(args)), make_not_null(&generator),
203  make_not_null(&(distributions[ArgumentIs]))));
204  const auto result = [&args, &f, &klass]() noexcept {
205  (void)args;
206  if constexpr (std::is_same_v<NoSuchType, std::decay_t<Klass>>) {
207  (void)klass;
208  return f(std::get<ArgumentIs>(args)...);
209  } else {
210  return (klass.*f)(std::get<ArgumentIs>(args)...);
211  }
212  }();
213  INFO("function: " << function_name);
214  try {
215  CHECK_ITERABLE_CUSTOM_APPROX(
216  result,
217  pypp::call<ResultType>(
218  module_name, function_name, std::get<ArgumentIs>(args)...,
219  forward_to_pypp<ResultType>(std::get<MemberArgsIs>(member_args),
220  used_for_size)...),
221  Approx::custom().epsilon(epsilon).scale(1.0));
222  } catch (const std::exception& e) {
223  INFO("Python call failed: " << e.what());
224  CHECK(false);
225  }
226 }

228 template <class F, class T, class Klass, class... ReturnTypes,
229  class... ArgumentTypes, class... MemberArgs, size_t... ResultIs,
230  size_t... ArgumentIs, size_t... MemberArgsIs>
231 void check_with_random_values_impl(
232  F&& f, const Klass& klass, const std::string& module_name,
233  const std::vector<std::string>& function_names, std::mt19937 generator,
234  std::array<std::uniform_real_distribution<>, sizeof...(ArgumentTypes)>
235  distributions,
236  const std::tuple<MemberArgs...>& member_args, const T& used_for_size,
237  tmpl::list<ReturnTypes...> /*return_types*/,
238  tmpl::list<ArgumentTypes...> /*argument_types*/,
239  std::index_sequence<ResultIs...> /*index_return_types*/,
240  std::index_sequence<ArgumentIs...> /*index_argument_types*/,
241  std::index_sequence<MemberArgsIs...> /*index_member_args*/,
242  NoSuchType /* meta */, const double epsilon = 1.0e-12) {
243  // Note: generator and distributions cannot be const.
244  std::tuple<ReturnTypes...> results{
245  make_with_value<ReturnTypes>(used_for_size, 0.0)...};
246  std::tuple<ArgumentTypes...> args{
247  make_with_value<ArgumentTypes>(used_for_size, 0.0)...};
248  EXPAND_PACK_LEFT_TO_RIGHT(fill_with_random_values(
249  make_not_null(&std::get<ArgumentIs>(args)), make_not_null(&generator),
250  make_not_null(&(distributions[ArgumentIs]))));
251  EXPAND_PACK_LEFT_TO_RIGHT(fill_with_random_values(
252  make_not_null(&std::get<ResultIs>(results)), make_not_null(&generator),
253  make_not_null(&(distributions[0]))));
254  if constexpr (std::is_same_v<NoSuchType, std::decay_t<Klass>>) {
255  f(make_not_null(&std::get<ResultIs>(results))...,
256  std::get<ArgumentIs>(args)...);
257  } else {
258  (klass.*f)(make_not_null(&std::get<ResultIs>(results))...,
259  std::get<ArgumentIs>(args)...);
260  }
261  const auto helper = [&module_name, &function_names, &args, &results, &epsilon,
262  &member_args, &used_for_size](auto result_i) {
263  (void)member_args; // avoid compiler warning
264  (void)used_for_size; // avoid compiler warning
265  constexpr size_t iter = decltype(result_i)::value;
266  INFO("function: " << function_names[iter]);
267  try {
268  CHECK_ITERABLE_CUSTOM_APPROX(
269  std::get<iter>(results),
270  (pypp::call<std::tuple_element_t<iter, std::tuple<ReturnTypes...>>>(
271  module_name, function_names[iter], std::get<ArgumentIs>(args)...,
272  forward_to_pypp<
273  std::tuple_element_t<iter, std::tuple<ReturnTypes...>>>(
274  std::get<MemberArgsIs>(member_args), used_for_size)...)),
275  Approx::custom().epsilon(epsilon).scale(1.0));
276  } catch (const std::exception& e) {
277  INFO("Python call failed: " << e.what());
278  CHECK(false);
279  }
280  };
281  EXPAND_PACK_LEFT_TO_RIGHT(helper(std::integral_constant<size_t, ResultIs>{}));
282 }

285 /*!
286  * \brief Tests a C++ function returning by value by comparing the result to a
287  * python function
288  *
289  * Tests the function `f` by comparing the result to that of the python function
290  * `function_name` in the file `module_name`. The function is tested by
291  * generated random values in the half-open range [`lower_bound`,
292  * `upper_bound`). The argument `used_for_size` is used for constructing the
293  * arguments of `f` by calling `make_with_value<ArgumentType>(used_for_size,
294  * 0.0)`.
295  *
296  * \note You must explicitly pass the number of bounds you will be passing as
297  * the first template parameter, the rest will be inferred.
298  *
299  * \note If you have a test fail you can replay the scenario by feeding in the
300  * seed that was printed out in the failed test as the last argument.
301  *
302  * \param f The C++ function to test
303  * \param module_name The python file relative to the directory used in
304  * `SetupLocalPythonEnvironment`
305  * \param function_name The name of the python function inside `module_name`
306  * \param lower_and_upper_bounds The lower and upper bounds for the randomly
307  * generated numbers. Must be either an array of a single pair, or of as many
308  * pairs as there are arguments to `f` that are not a `gsl::not_null`
309  * \param used_for_size The type `X` for the arguments of `f` of type
310  *`Tensor<X>`
311  * \param epsilon A double specifying the comparison tolerance
312  * (default 1.0e-12)
313  * \param seed The seed for the random number generator. This should only be
314  * specified when debugging a failure with a particular set of random numbers,
315  * in general it should be left to the default value.
316  */
317 template <size_t NumberOfBounds, class F, class T,
318  Requires<not std::is_same_v<
319  typename tt::function_info<cpp20::remove_cvref_t<F>>::return_type,
320  void>> = nullptr>
321 // The Requires is used so that we can call the std::vector<std::string> with
322 // braces and not have it be ambiguous.
323 void check_with_random_values(
324  F&& f, const std::string& module_name, const std::string& function_name,
325  const std::array<std::pair<double, double>, NumberOfBounds>&
326  lower_and_upper_bounds,
327  const T& used_for_size, const double epsilon = 1.0e-12,
328  const typename std::random_device::result_type seed =
329  std::random_device{}()) {
330  MAKE_GENERATOR(generator, seed);
331  using f_info = tt::function_info<cpp20::remove_cvref_t<F>>;
332  using number_of_not_null =
333  tmpl::count_if<typename f_info::argument_types,
334  tmpl::bind<TestHelpers_detail::is_not_null, tmpl::_1>>;
335  using argument_types = tmpl::transform<
336  tmpl::pop_front<typename f_info::argument_types, number_of_not_null>,
337  std::decay<tmpl::_1>>;
338 
339  static_assert(number_of_not_null::value == 0,
340  "Cannot return arguments by gsl::not_null if the python "
341  "function name is passed as a string. If the function only "
342  "returns one gsl::not_null then you must pass in a one element "
343  "vector<string>.");
344  static_assert(tmpl::size<argument_types>::value != 0,
345  "The function 'f' must take at least one argument.");
346  static_assert(NumberOfBounds == 1 or
347  NumberOfBounds == tmpl::size<argument_types>::value,
348  "The number of lower and upper bound pairs must be either 1 or "
349  "equal to the number of arguments taken by f that are not "
350  "gsl::not_null.");
351  std::array<std::uniform_real_distribution<>,
352  tmpl::size<argument_types>::value>
353  distributions;
354  for (size_t i = 0; i < tmpl::size<argument_types>::value; ++i) {
355  gsl::at(distributions, i) = std::uniform_real_distribution<>{
356  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).first,
357  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).second};
358  }
359  TestHelpers_detail::check_with_random_values_impl(
360  std::forward<F>(f), NoSuchType{}, module_name, function_name, generator,
361  std::move(distributions), std::tuple<>{}, used_for_size, argument_types{},
362  std::make_index_sequence<tmpl::size<argument_types>::value>{},
363  std::make_index_sequence<0>{}, NoSuchType{}, epsilon);
364 }

366 /*!
367  * \brief Tests a C++ function returning by `gsl::not_null` by comparing the
368  * result to a python function
369  *
370  * Tests the function `f` by comparing the result to that of the python function
371  * `function_name` in the file `module_name`. The function is tested by
372  * generated random values in the half-open range [`lower_bound`, `upper_bound`)
373  * for each argument. The argument `used_for_size` is used for constructing the
374  * arguments of `f` by calling `make_with_value<ArgumentType>(used_for_size,
375  * 0.0)`. For functions that return by `gsl::not_null`, the result will be
376  * initialized with random values rather than to signaling `NaN`s. This means
377  * functions do not need to support receiving a signaling `NaN` in their return
378  * argument to be tested using this function.
379  *
380  * \note You must explicitly pass the number of bounds you will be passing as
381  * the first template parameter, the rest will be inferred.
382  *
383  * \note If you have a test fail you can replay the scenario by feeding in the
384  * seed that was printed out in the failed test as the last argument.
385  *
386  * \param f The C++ function to test
387  * \param module_name The python file relative to the directory used in
388  * `SetupLocalPythonEnvironment`
389  * \param function_names The names of the python functions inside `module_name`
390  * in the order that they return the `gsl::not_null` results
391  * \param lower_and_upper_bounds The lower and upper bounds for the randomly
392  * generated numbers. Must be either an array of a single pair, or of as many
393  * pairs as there are arguments to `f` that are not a `gsl::not_null`
394  * \param used_for_size The type `X` for the arguments of `f` of type
395  * `Tensor<X>`
396  * \param epsilon A double specifying the comparison tolerance
397  * (default 1.0e-12)
398  * \param seed The seed for the random number generator. This should only be
399  * specified when debugging a failure with a particular set of random numbers,
400  * in general it should be left to the default value.
401  */
402 template <size_t NumberOfBounds, class F, class T>
403 void check_with_random_values(
404  F&& f, const std::string& module_name,
405  const std::vector<std::string>& function_names,
406  const std::array<std::pair<double, double>, NumberOfBounds>&
407  lower_and_upper_bounds,
408  const T& used_for_size, const double epsilon = 1.0e-12,
409  const typename std::random_device::result_type seed =
410  std::random_device{}()) {
411  MAKE_GENERATOR(generator, seed);
412  using f_info = tt::function_info<cpp20::remove_cvref_t<F>>;
413  using number_of_not_null =
414  tmpl::count_if<typename f_info::argument_types,
415  tmpl::bind<TestHelpers_detail::is_not_null, tmpl::_1>>;
416  using argument_types = tmpl::transform<
417  tmpl::pop_front<typename f_info::argument_types, number_of_not_null>,
418  std::decay<tmpl::_1>>;
419  using return_types =
420  tmpl::transform<tmpl::pop_back<typename f_info::argument_types,
421  tmpl::size<argument_types>>,
422  TestHelpers_detail::RemoveNotNull<tmpl::_1>>;
423 
424  static_assert(number_of_not_null::value != 0,
425  "You must return at least one argument by gsl::not_null when "
426  "passing the python function names as a vector<string>. If "
427  "your function returns by value do not pass the function name "
428  "as a vector<string> but just a string.");
429  static_assert(
430  std::is_same_v<typename f_info::return_type, void>,
431  "A function returning by gsl::not_null must have a void return type.");
432  static_assert(tmpl::size<argument_types>::value != 0,
433  "The function 'f' must take at least one argument.");
434  static_assert(NumberOfBounds == 1 or
435  NumberOfBounds == tmpl::size<argument_types>::value,
436  "The number of lower and upper bound pairs must be either 1 or "
437  "equal to the number of arguments taken by f that are not "
438  "gsl::not_null.");
439  if (function_names.size() != number_of_not_null::value) {
440  ERROR(
441  "The number of python functions passed must be the same as the number "
442  "of gsl::not_null arguments in the C++ function. The order of the "
443  "python functions must also be the same as the order of the "
444  "gsl::not_null arguments.");
445  }
446  std::array<std::uniform_real_distribution<>,
447  tmpl::size<argument_types>::value>
448  distributions;
449  for (size_t i = 0; i < tmpl::size<argument_types>::value; ++i) {
450  gsl::at(distributions, i) = std::uniform_real_distribution<>{
451  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).first,
452  gsl::at(lower_and_upper_bounds, NumberOfBounds == 1 ? 0 : i).second};
453  }
454  TestHelpers_detail::check_with_random_values_impl(
455  std::forward<F>(f), NoSuchType{}, module_name, function_names, generator,
456  std::move(distributions), std::tuple<>{}, used_for_size, return_types{},
457  argument_types{},
458  std::make_index_sequence<tmpl::size<return_types>::value>{},
459  std::make_index_sequence<tmpl::size<argument_types>::value>{},
460  std::make_index_sequence<0>{}, NoSuchType{}, epsilon);
461 }

463 /*!
464  * \brief Tests a member function of a class returning by value by comparing the
465  * result to a python function
466  *
467  * Tests the function `f` by comparing the result to that of the python function
468  * `function_name` in the file `module_name`. An instance of the class is passed
469  * in as the second argument and is the object on which the member function `f`
470  * will be invoked. The member function is invoked as `klass.function`, so
471  * passing in pointers is not supported. The function is tested by generated
472  * random values in the half-open range [`lower_bound`, `upper_bound`). The
473  * argument `used_for_size` is used for constructing the arguments of `f` by
474  * calling `make_with_value<ArgumentType>(used_for_size, 0.0)`.
475  *
476  * \note You must explicitly pass the number of bounds you will be passing
477  * as the first template parameter, the rest will be inferred.
478  *
479  * \note If you have a test fail you can replay the scenario by feeding in
480  * the seed that was printed out in the failed test as the last argument.
481  *
482  * \param f The member function to test
483  * \param klass the object on which to invoke `f`
484  * \param module_name The python file relative to the directory used in
485  * `SetupLocalPythonEnvironment`
486  * \param function_name The name of the python function inside `module_name`
