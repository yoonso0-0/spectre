51 /*!
52  * \brief Creates both the Jacobian and inverse Jacobian between Cartesian and
53  * spherical coordinates, and the coordinates themselves
54  *
55  * \details The `cartesian_to_spherical_jacobian` is
56  * \f$dx^i/d\tilde{x}^{\tilde j}\f$,
57  * where the Cartesian components are in order \f$x^i = \{x, y, z\}\f$
58  * and the spherical coordinates are
59  * \f$\tilde{x}^{\tilde j} = \{r, \theta, \phi\}\f$.
60  * The Cartesian coordinates given are the standard unit sphere coordinates:
61  *
62  * \f{align*}{
63  * x &= \cos(\phi) \sin(\theta)\\
64  * y &= \sin(\phi) \sin(\theta)\\
65  * z &= \cos(\theta)
66  * \f}
67  *
68  * \note These Jacobians are adjusted to improve regularity near the pole, in
69  * particular the \f$\partial \phi / \partial x^i\f$ components have been scaled
70  * by \f$\sin \theta\f$ (omitting a \f$1/\sin(\theta)\f$) and the
71  * \f$\partial x^i/\partial \phi\f$ components have been scaled by
72  * \f$1/\sin(\theta)\f$ (omitting a \f$\sin(\theta)\f$). The reason is that in
73  * most careful calculations, these problematic sin factors can actually be
74  * omitted because they cancel. In cases where they are actually required, they
75  * must be put in by hand.
76  */

78  gsl::not_null<tnsr::I<DataVector, 3>*> unit_cartesian_coords,

165 /*!
166  * \brief Computes the spacetime metric and its first derivative in the
167  * intermediate radial null coordinates
168  *
169  * \details These components are obtained by the steps in
170  * Section II-A of \cite Barkett2019uae, which is based on the computation from
171  * Section 4.3 of \cite Bishop1998uk. The most direct comparison is to be made
172  * with equation (31) of \cite Barkett2019uae, which gives the null metric
173  * components explicitly. The time derivative is then (using notation from
174  * equation (31) of \cite Barkett2019uae):
175  *
176  * \f{align}{
177  * \partial_{\bar u} g_{\bar u \bar \lambda} =
178  * \partial_{\bar u} g_{\bar \lambda \bar \lambda} =
179  * \partial_{\bar u} g_{\bar \lambda \bar A} &= 0 \\
180  * \partial_{\bar u} g_{\bar u \bar u} &=
181  * \partial_{\breve t} g_{\breve t \breve t} \\
182  * \partial_{\bar u} g_{\bar u \bar A} &=
183  * \frac{\partial \breve x^{\breve i}}{\partial \bar x^{\bar A}}\\
184  * g_{\breve i \breve t}
185  * \partial_{\bar u} g_{\bar A \bar B}
186  * &= \frac{\partial \breve x^{\breve i}}{\partial \bar x^{\bar A}}
187  * \frac{\partial \breve x^{\breve j}}{\partial \bar x^{\bar B}}
188  * g_{\breve i \breve j}
189  * \f}
190  */

459  * equations (56) and (57) of \cite Barkett2019uae)
460  *
461  * \f[
462  * \partial_\lambda U = - \left(\partial_\lambda g^{\lambda A}
463  * + \frac{\partial_A \partial_\lambda r}{\partial_\lambda r} g^{A B}
464  * + \frac{\partial_B r}{\partial_\lambda r} \partial_\lambda g^{A B}\right) q_A
465  * + 2 \partial_\lambda \beta (U + g^{\lambda A} q_A)
466  * \f]
467  *
468  * and
469  *
470  * \f[
471  * Q = r^2 (J \partial_\lambda \bar U + K \partial_\lambda U)
472  * \f]
473  *
474  * also provided is \f$\partial_r U\f$, which is separately useful to cache for
475  * other intermediate steps in the CCE computation.
476  */
477 void bondi_q_worldtube_data(
478  gsl::not_null<Scalar<SpinWeighted<ComplexDataVector, 1>>*> bondi_q,

492  * \brief Compute the Bondi metric contribution \f$(\partial_u J)_{y} \equiv
493  * H\f$ (the retarded time derivative evaluated at fixed $y$ coordinate) on the
494  * worldtube boundary.
495  *
496  * \details The numerical time derivative (along the worldtube, rather than
497  * along the surface of constant Bondi \f$r\f$) is computed by (see equation
498  * (48) of \cite Barkett2019uae)
499  *
500  * \f[
501  * (\partial_u J)_y = \frac{1}{2 r^2} q^A q^B \partial_u g_{A B}
502  * - \frac{2 \partial_u r}{r} J
503  * \f]
504  *
505  * \note There is the regrettable notation difference with the primary reference
506  * for these formulas \cite Barkett2019uae in that we denote with \f$H\f$ the
507  * time derivative at constant numerical radius, where \cite Barkett2019uae uses
508  * \f$H\f$ to denote the time derivative at constant Bondi radius.

592  *computation_variables);
593 
594  // the below scaling process is used to reduce accumulation of numerical
595  // error in the determinant evaluation
596 
597  // buffer reuse because the scaled null metric is only needed until the
598  // `determinant_and_inverse` call
599  auto& scaled_null_metric =
600  get<gr::Tags::InverseSpacetimeMetric<3, Frame::RadialNull, DataVector>>(
601  *computation_variables);
602  for (size_t i = 0; i < 4; ++i) {
603  for (size_t j = i; j < 4; ++j) {
604  if (i > 1 and j > 1) {
605  scaled_null_metric.get(i, j) =
606  null_metric.get(i, j) / square(extraction_radius);
607  } else if (i > 1 or j > 1) {
608  scaled_null_metric.get(i, j) =
609  null_metric.get(i, j) / extraction_radius;
610  } else {
611  scaled_null_metric.get(i, j) = null_metric.get(i, j);
612  }
613  }
614  }
615  // Allocation
616  const auto scaled_inverse_null_metric =
617  determinant_and_inverse(scaled_null_metric).second;
618  for (size_t i = 0; i < 4; ++i) {
619  for (size_t j = i; j < 4; ++j) {
620  if (i > 1 and j > 1) {
621  inverse_null_metric.get(i, j) =
622  scaled_inverse_null_metric.get(i, j) / square(extraction_radius);
623  } else if (i > 1 or j > 1) {
624  inverse_null_metric.get(i, j) =
625  scaled_inverse_null_metric.get(i, j) / extraction_radius;
626  } else {
627  inverse_null_metric.get(i, j) = scaled_inverse_null_metric.get(i, j);
628  }

638  auto& eth_buffer =
639  get(get<::Tags::SpinWeighted<::Tags::TempScalar<0, ComplexDataVector>,
640  std::integral_constant<int, 1>>>(
641  *derivative_buffers));
642  for (size_t a = 0; a < 4; ++a) {
643  buffer_for_derivatives.data() =
644  std::complex<double>(1.0, 0.0) * null_l.get(a);
645  Spectral::Swsh::angular_derivatives<tmpl::list<Spectral::Swsh::Tags::Eth>>(
646  l_max, 1, make_not_null(&eth_buffer), buffer_for_derivatives);

648  angular_d_null_l.get(1, a) = -imag(eth_buffer.data());
649  }
650 
651  auto& dlambda_null_metric = get<Tags::detail::DLambda<
652  gr::Tags::SpacetimeMetric<3, Frame::RadialNull, DataVector>>>(
653  *computation_variables);
654  auto& dlambda_inverse_null_metric = get<Tags::detail::DLambda<
655  gr::Tags::InverseSpacetimeMetric<3, Frame::RadialNull, DataVector>>>(
656  *computation_variables);
657  dlambda_null_metric_and_inverse(
658  make_not_null(&dlambda_null_metric),
659  make_not_null(&dlambda_inverse_null_metric), angular_d_null_l,

662 
663  auto& r = get<Tags::BoundaryValue<Tags::BondiR>>(*bondi_boundary_data);
664  bondi_r(make_not_null(&r), null_metric);
665 
666  auto& d_r =
667  get<::Tags::spacetime_deriv<Tags::detail::RealBondiR, tmpl::size_t<3>,
668  Frame::RadialNull>>(*computation_variables);
669  d_bondi_r(make_not_null(&d_r), r, dlambda_null_metric, du_null_metric,
670  inverse_null_metric, l_max);
671  get(get<Tags::BoundaryValue<Tags::DuRDividedByR>>(*bondi_boundary_data))
672  .data() =
673  std::complex<double>(1.0, 0.0) * get<0>(d_r) / get(r).data();
674  get(get<Tags::BoundaryValue<Tags::Du<Tags::BondiR>>>(*bondi_boundary_data))
675  .data() = std::complex<double>(1.0, 0.0) * get<0>(d_r);
676 
677  auto& down_dyad = get<Tags::detail::DownDyad>(dyad_variables);
678  auto& up_dyad = get<Tags::detail::UpDyad>(dyad_variables);
679  dyads(make_not_null(&down_dyad), make_not_null(&up_dyad));
680 
681  beta_worldtube_data(make_not_null(&get<Tags::BoundaryValue<Tags::BondiBeta>>(
682  *bondi_boundary_data)),
683  d_r);
684 
685  auto& bondi_u = get<Tags::BoundaryValue<Tags::BondiU>>(*bondi_boundary_data);
686  bondi_u_worldtube_data(make_not_null(&bondi_u), down_dyad, d_r,
687  inverse_null_metric);
688 
689  bondi_w_worldtube_data(make_not_null(&get<Tags::BoundaryValue<Tags::BondiW>>(

691  d_r, inverse_null_metric, r);
692

694  get<Tags::BoundaryValue<Tags::BondiJ>>(*bondi_boundary_data);
695  bondi_j_worldtube_data(make_not_null(&bondi_j), null_metric, r, up_dyad);
696 
697  auto& dr_j =
698  get<Tags::BoundaryValue<Tags::Dr<Tags::BondiJ>>>(*bondi_boundary_data);
699  auto& denominator_buffer =
700  get<::Tags::SpinWeighted<::Tags::TempScalar<0, ComplexDataVector>,
701  std::integral_constant<int, 0>>>(
702  *derivative_buffers);
703  dr_bondi_j(make_not_null(&get<Tags::BoundaryValue<Tags::Dr<Tags::BondiJ>>>(
704  *bondi_boundary_data)),
705  make_not_null(&denominator_buffer), dlambda_null_metric, d_r,
706  bondi_j, r, up_dyad);
707 
708  auto& d2lambda_r = get<
709  Tags::detail::DLambda<Tags::detail::DLambda<Tags::detail::RealBondiR>>>(
710  *computation_variables);
711  d2lambda_bondi_r(make_not_null(&d2lambda_r), d_r, dr_j, bondi_j, r);

716  *computation_variables);

719  l_max, 1, make_not_null(&eth_buffer), buffer_for_derivatives);

721  angular_d_dlambda_r.get(1) = -imag(eth_buffer.data());
722 
723  bondi_q_worldtube_data(
724  make_not_null(
725  &get<Tags::BoundaryValue<Tags::BondiQ>>(*bondi_boundary_data)),
726  make_not_null(&get<Tags::BoundaryValue<Tags::Dr<Tags::BondiU>>>(
727  *bondi_boundary_data)),
728  d2lambda_r, dlambda_inverse_null_metric, d_r, down_dyad,
729  angular_d_dlambda_r, inverse_null_metric, bondi_j, r, bondi_u);

742 /*!
743  * \brief Process the worldtube data from generalized harmonic quantities
744  * to desired Bondi quantities, placing the result in the passed
745  * `Variables`.
746  *
747  * \details
748  * The mathematics are a bit complicated for all of the coordinate
749  * transformations that are necessary to obtain the Bondi gauge quantities.
750  * For full mathematical details, see the documentation for functions in
751  * `BoundaryData.hpp` and \cite Barkett2019uae \cite Bishop1998uk.
752  *
753  * This function takes as input the full set of Generalized harmonic metric data
754  * on a two-dimensional surface of constant \f$r\f$ and \f$t\f$ in numerical
755  * coordinates.
756  *
757  * Sufficient tags to provide full worldtube boundary data at a particular
758  * time are set in `bondi_boundary_data`. In particular, the set of tags in
759  * `Tags::characteristic_worldtube_boundary_tags` in the provided `Variables`
760  * are assigned to the worldtube boundary values associated with the input
761  * metric components.
762  *
763  * The majority of the mathematical transformations are implemented as a set of
764  * individual cascaded functions below. The details of the manipulations that
765  * are performed to the input data may be found in the individual functions
766  * themselves, which are called in the following order:
767  * - `trigonometric_functions_on_swsh_collocation()`
768  * - `gr::shift()`
769  * - `gr::lapse()`
770  * - `worldtube_normal_and_derivatives()`
771  * - `gr::spacetime_normal_vector()`
772  * - `GeneralizedHarmonic::time_deriv_of_lapse()`
773  * - `GeneralizedHarmonic::time_deriv_of_shift()`
774  * - `null_vector_l_and_derivatives()`
775  * - `cartesian_to_spherical_coordinates_and_jacobians()`
776  * - `null_metric_and_derivative()`
777  * - `dlambda_null_metric_and_inverse()`
778  * - `bondi_r()`
779  * - `d_bondi_r()`
780  * - `dyads()`
781  * - `beta_worldtube_data()`
782  * - `bondi_u_worldtube_data()`
783  * - `bondi_w_worldtube_data()`
784  * - `bondi_j_worldtube_data()`
785  * - `dr_bondi_j()`
786  * - `d2lambda_bondi_r()`
787  * - `bondi_q_worldtube_data()`
788  * - `bondi_h_worldtube_data()`
789  * - `du_j_worldtube_data()`
790  */
791 template <typename BoundaryTagList>
792 void create_bondi_boundary_data(
793  const gsl::not_null<Variables<BoundaryTagList>*> bondi_boundary_data,
794  const tnsr::iaa<DataVector, 3>& phi, const tnsr::aa<DataVector, 3>& pi,
795  const tnsr::aa<DataVector, 3>& spacetime_metric,
796  const double extraction_radius, const size_t l_max) noexcept {
797  const size_t size = Spectral::Swsh::number_of_swsh_collocation_points(l_max);
798  // Most allocations required for the full boundary computation are merged into
799  // a single, large Variables allocation. There remain a handful of cases in
800  // the computational functions called where an intermediate quantity that is
801  // not re-used is allocated rather than taking a buffer. These cases are
802  // marked with code comments 'Allocation'; In the future, if allocations are
803  // identified as a point to optimize, those buffers may be allocated here and
804  // passed as function arguments
805  Variables<tmpl::list<
806  Tags::detail::CosPhi, Tags::detail::CosTheta, Tags::detail::SinPhi,

818  Tags::detail::NullL, ::Tags::dt<Tags::detail::NullL>,

820  gr::Tags::SpacetimeMetric<3, Frame::RadialNull, DataVector>,
821  ::Tags::dt<gr::Tags::SpacetimeMetric<3, Frame::RadialNull, DataVector>>,
822  gr::Tags::InverseSpacetimeMetric<3, Frame::RadialNull, DataVector>,
823  Tags::detail::AngularDNullL,
824  Tags::detail::DLambda<
825  gr::Tags::SpacetimeMetric<3, Frame::RadialNull, DataVector>>,

827  gr::Tags::InverseSpacetimeMetric<3, Frame::RadialNull, DataVector>>,

829  Frame::RadialNull>,
830  Tags::detail::DLambda<Tags::detail::DLambda<Tags::detail::RealBondiR>>,
831  ::Tags::deriv<Tags::detail::DLambda<Tags::detail::RealBondiR>,
832  tmpl::size_t<2>, Frame::RadialNull>>>

836  tmpl::list<::Tags::SpinWeighted<::Tags::TempScalar<0, ComplexDataVector>,
837  std::integral_constant<int, 0>>,
838  ::Tags::SpinWeighted<::Tags::TempScalar<0, ComplexDataVector>,
839  std::integral_constant<int, 1>>>>

851  // components of all tensors are scaled according to their sin(theta)
852  // prefactors.
853  // so, any down-index component get<2>(A) represents 1/sin(theta) A_\phi,
854  // and any up-index component get<2>(A) represents sin(theta) A^\phi.
855  // This holds for Jacobians, and so direct application of the Jacobians
856  // brings the factors through.

866  make_not_null(&cartesian_to_spherical_jacobian),

868  cos_theta, sin_phi, sin_theta, extraction_radius);

895  auto& dt_worldtube_normal =
896  get<::Tags::dt<Tags::detail::WorldtubeNormal>>(computation_variables);
897  auto& worldtube_normal =
898  get<Tags::detail::WorldtubeNormal>(computation_variables);
899  worldtube_normal_and_derivatives(
900  make_not_null(&worldtube_normal), make_not_null(&dt_worldtube_normal),
901  cos_phi, cos_theta, spacetime_metric, dt_spacetime_metric, sin_phi,
902  sin_theta, inverse_spatial_metric);
903  auto& spacetime_unit_normal =
904  get<gr::Tags::SpacetimeNormalVector<3, ::Frame::Inertial, DataVector>>(
905  computation_variables);
906  gr::spacetime_normal_vector(make_not_null(&spacetime_unit_normal), lapse,

937  * derivatives to desired Bondi quantities, placing the result in the passed
938  * `Variables`.
939  *
940  * \details
941  * The mathematics are a bit complicated for all of the coordinate
942  * transformations that are necessary to obtain the Bondi gauge quantities.
943  * For full mathematical details, see the documentation for functions in
944  * `BoundaryData.hpp` and \cite Barkett2019uae \cite Bishop1998uk.
945  *
946  * This function takes as input the full set of ADM metric data and its radial
947  * and time derivatives on a two-dimensional surface of constant \f$r\f$ and
948  * \f$t\f$ in numerical coordinates. This data must be provided as spherical
949  * harmonic coefficients in the libsharp format. This data is provided in nine
950  * `Tensor`s.
951  *
952  * Sufficient tags to provide full worldtube boundary data at a particular
953  * time are set in `bondi_boundary_data`. In particular, the set of tags in
954  * `Tags::characteristic_worldtube_boundary_tags` in the provided `Variables`
955  * are assigned to the worldtube boundary values associated with the input
956  * metric components.
957  *
958  * The majority of the mathematical transformations are implemented as a set of
959  * individual cascaded functions below. The details of the manipulations that
960  * are performed to the input data may be found in the individual functions
961  * themselves, which are called in the following order:
962  * - `trigonometric_functions_on_swsh_collocation()`
963  * - `cartesian_to_spherical_coordinates_and_jacobians()`
964  * - `cartesian_spatial_metric_and_derivatives_from_modes()`
965  * - `cartesian_shift_and_derivatives_from_modes()`
966  * - `cartesian_lapse_and_derivatives_from_modes()`
967  * - `GeneralizedHarmonic::phi()`
968  * - `gr::time_derivative_of_spacetime_metric`
969  * - `gr::spacetime_metric`
970  * - `generalized_harmonic_quantities()`
971  * - `worldtube_normal_and_derivatives()`
972  * - `null_vector_l_and_derivatives()`
973  * - `null_metric_and_derivative()`
974  * - `dlambda_null_metric_and_inverse()`
975  * - `bondi_r()`
976  * - `d_bondi_r()`
977  * - `dyads()`
978  * - `beta_worldtube_data()`
979  * - `bondi_u_worldtube_data()`
980  * - `bondi_w_worldtube_data()`
981  * - `bondi_j_worldtube_data()`
982  * - `dr_bondi_j()`
983  * - `d2lambda_bondi_r()`
984  * - `bondi_q_worldtube_data()`
985  * - `bondi_h_worldtube_data()`
986  * - `du_j_worldtube_data()`
