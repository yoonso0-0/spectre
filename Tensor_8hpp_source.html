48 /*!
49  * \ingroup TensorGroup
50  * \brief Represents an object with multiple components
51  *
52  * \details
53  * Tensor is a container that represents indexable geometric objects. Each index
54  * has its own dimension, valence, and frame and must be either spatial or
55  * spacetime. Note that the dimension passed to `SpatialIndex` and
56  * `SpacetimeIndex` is always the spatial dimension of the index. Tensors with
57  * symmetric indices are stored only once and must be of the same
58  * type. A list of available type aliases can be found in the ::tnsr namespace
59  * where the adopted conventions are:
60  *
61  * 1. Upper case for contravariant or upper indices, lower case for covariant or
62  * lower indices.
63  *
64  * 2. `a, b, c, d` are used for spacetime indices while `i, j, k, l` are used
65  * for spatial indices.
66  *
67  * 3. `::Scalar` is not inside the `::tnsr` namespace but is used to represent
68  * a scalar with no indices.
69  *
70  * \example
71  * \snippet Test_Tensor.cpp scalar
72  * \snippet Test_Tensor.cpp spatial_vector
73  * \snippet Test_Tensor.cpp spacetime_vector
74  * \snippet Test_Tensor.cpp rank_3_122
75  *
76  * \tparam X the type held
77  * \tparam Symm the ::Symmetry of the indices
78  * \tparam IndexList a typelist of \ref SpacetimeIndex "TensorIndexType"'s
79  */
80 template <typename X, typename Symm, template <typename...> class IndexList,
81  typename... Indices>
82 class Tensor<X, Symm, IndexList<Indices...>> {
83  static_assert(sizeof...(Indices) < 5,
84  "If you are sure you need rank 5 or higher Tensor's please "
85  "file an issue on GitHub or discuss with a core developer of "

92  is_spin_weighted_of_v<ComplexDataVector, X> or
93  is_spin_weighted_of_v<ComplexModalVector, X>,

103  ///
104  /// Each tensor index structure, e.g. \f$T_{ab}\f$, \f$T_a{}^b\f$ or
105  /// \f$T^{ab}\f$ has its own Tensor_detail::TensorStructure that holds
106  /// information about how the data is stored, what the multiplicity of the
107  /// stored indices are, the number of (independent) components, etc.
108  using structure = Tensor_detail::Structure<Symm, Indices...>;
