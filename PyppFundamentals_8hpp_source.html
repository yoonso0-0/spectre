67 /// Create a python tuple from Args
68 ///
69 /// \tparam Args the types of the arguments to be put in the tuple (deducible)
70 /// \param t the arguments to put into the tuple
71 /// \return PyObject* containing a Python tuple
72 template <typename... Args>
73 PyObject* make_py_tuple(const Args&... t) {
74  PyObject* py_tuple = PyTuple_New(sizeof...(Args));
75  int entry = 0;
76  const auto add_entry = [&entry, &py_tuple](const auto& arg) {
77  PyObject* value = to_py_object(arg);
78  PyTuple_SetItem(py_tuple, entry, value);
79  entry++;
80  return '0';
81  };
82  (void)add_entry; // GCC warns that add_entry is unused
83  (void)std::initializer_list<char>{add_entry(t)...};
84  return py_tuple;
85 }

490 template <typename T>
491 T tensor_conversion_impl(PyObject* p) {
492  if (p == nullptr) {
493  throw std::runtime_error{"Received null PyObject."};
494  }
495  // clang-tidy: c-style casts. (Expanded from macro)
496  if (not PyArray_CheckExact(p)) { // NOLINT
497  throw std::runtime_error{"Cannot convert non-array type to Tensor."};
498  }
499  // clang-tidy: reinterpret_cast
500  const auto npy_array = reinterpret_cast<PyArrayObject*>(p); // NOLINT
501  if (PyArray_TYPE(npy_array) != NPY_DOUBLE) {
502  throw std::runtime_error{
503  "Cannot convert array of non-double type to Tensor."};
504  } else if (PyArray_NDIM(npy_array) != static_cast<long>(T::rank())) {
505  throw std::runtime_error{"Mismatch between ndim of numpy ndarray (" +
506  std::to_string(PyArray_NDIM(npy_array)) +
507  ") and rank of Tensor (" +
508  std::to_string(T::rank()) + ")"};
509  }
510 
511  const auto* npy_array_dims = PyArray_DIMS(npy_array);
512  constexpr auto t_array_dims = T::index_dims();
513  for (size_t i = 0; i < T::rank(); ++i) {
514  // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)
515  if (npy_array_dims[i] != static_cast<long>(gsl::at(t_array_dims, i))) {
516  throw std::runtime_error{
517  "Mismatch between number of components of ndarray (" +
518  std::to_string(npy_array_dims[i]) // NOLINT
519  + ") and Tensor of rank " + std::to_string(T::rank()) + " in " +
520  std::to_string(i) + "\'th index with dimension " +
521  std::to_string(static_cast<long>(gsl::at(t_array_dims, i)))};
522  }
523  }
524  auto t = make_with_value<T>(
525  *get_ptr_to_elem<typename T::type>(npy_array, make_array<T::rank()>(0ul)),
526  static_cast<typename T::value_type>(0.0));
527  for (IndexIterator<T::rank()> index_it((Index<T::rank()>(T::index_dims())));
528  index_it; ++index_it) {
529  const auto tensor_idx = (*index_it).indices();
530  t.get(tensor_idx) =
531  *get_ptr_to_elem<typename T::type>(npy_array, tensor_idx);
532  }
533  return t;
534 }
