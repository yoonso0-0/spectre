21 /*!
22  * \brief Initializes the DataBox on the observer parallel component
23  *
24  * Uses:
25  * - Metavariables:
26  * - `observed_reduction_data_tags` (see ContributeReductionData)
27  *
28  * \note This action relies on the `SetupDataBox` aggregated initialization
29  * mechanism, so `Actions::SetupDataBox` must be present in the `Initialization`
30  * phase action list prior to this action.
31  */
32 template <class Metavariables>
33 struct Initialize {
34  using simple_tags = tmpl::append<
35  tmpl::list<Tags::ExpectedContributorsForObservations,
36  Tags::ContributorsOfReductionData,
37  Tags::ContributorsOfTensorData, Tags::TensorData>,
38  typename Metavariables::observed_reduction_data_tags,
39  tmpl::transform<
40  typename Metavariables::observed_reduction_data_tags,
41  tmpl::bind<detail::reduction_data_to_reduction_names, tmpl::_1>>>;
42  using compute_tags = tmpl::list<>;
43 
44  using return_tag_list = tmpl::append<simple_tags, compute_tags>;
45 
46  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
47  typename ActionList, typename ParallelComponent>
48  static auto apply(db::DataBox<DbTagsList>& box,
49  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
50  const Parallel::GlobalCache<Metavariables>& /*cache*/,
51  const ArrayIndex& /*array_index*/,
52  const ActionList /*meta*/,
53  const ParallelComponent* const /*meta*/) noexcept {
54  return std::make_tuple(std::move(box));
55  }
56 };
