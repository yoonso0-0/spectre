43 /*!
44  * \brief A plane wave solution to the Euclidean wave equation
45  *
46  * The solution is given by \f$\Psi(\vec{x},t) = F(u(\vec{x},t))\f$
47  * where the profile \f$F\f$ of the plane wave is an arbitrary one-dimensional
48  * function of \f$u = \vec{k} \cdot (\vec{x} - \vec{x_o}) - \omega t\f$
49  * with the wave vector \f$\vec{k}\f$, the frequency \f$\omega = ||\vec{k}||\f$
50  * and initial center of the profile \f$\vec{x_o}\f$.
51  *
52  * \tparam Dim the spatial dimension of the solution
53  */
54 template <size_t Dim>
55 class PlaneWave : public MarkAsAnalyticSolution {
