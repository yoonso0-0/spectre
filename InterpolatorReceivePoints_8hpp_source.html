42 /// \ingroup ActionsGroup
43 /// \brief Receives target points from an InterpolationTarget.
44 ///
45 /// After receiving the points, interpolates volume data onto them
46 /// if it already has all the volume data.
47 ///
48 /// Uses:
49 /// - Databox:
50 /// - `Tags::NumberOfElements`
51 /// - `Tags::InterpolatedVarsHolders<Metavariables>`
52 /// - `Tags::VolumeVarsInfo<Metavariables>`
53 ///
54 /// DataBox changes:
55 /// - Adds: nothing
56 /// - Removes: nothing
57 /// - Modifies:
58 /// - `Tags::InterpolatedVarsHolders<Metavariables>`
59 ///
60 /// For requirements on InterpolationTargetTag, see InterpolationTarget
61 template <typename InterpolationTargetTag>
62 struct ReceivePoints {
63  template <
64  typename ParallelComponent, typename DbTags, typename Metavariables,
65  typename ArrayIndex, size_t VolumeDim,
66  Requires<tmpl::list_contains_v<DbTags, ::intrp::Tags::NumberOfElements>> =
67  nullptr>
68  static void apply(
69  db::DataBox<DbTags>& box,
70  Parallel::GlobalCache<Metavariables>& cache,
71  const ArrayIndex& /*array_index*/,
72  const typename Metavariables::temporal_id::type& temporal_id,
73  std::vector<boost::optional<
74  IdPair<domain::BlockId,
75  tnsr::I<double, VolumeDim, typename ::Frame::Logical>>>>&&
76  block_logical_coords) noexcept {
77  db::mutate<intrp::Tags::InterpolatedVarsHolders<Metavariables>>(
78  make_not_null(&box),
79  [
80  &temporal_id, &block_logical_coords
81  ](const gsl::not_null<
82  typename intrp::Tags::InterpolatedVarsHolders<Metavariables>::type*>
83  vars_holders) noexcept {
84  auto& vars_infos =
85  get<intrp::Vars::HolderTag<InterpolationTargetTag,
86  Metavariables>>(*vars_holders)
87  .infos;
88 
89  // Add the target interpolation points at this temporal_id.
90  vars_infos.emplace(std::make_pair(
91  temporal_id,
92  intrp::Vars::Info<VolumeDim, typename InterpolationTargetTag::
93  vars_to_interpolate_to_target>{
94  std::move(block_logical_coords)}));
95  });
96 
97  try_to_interpolate<InterpolationTargetTag>(
98  make_not_null(&box), make_not_null(&cache), temporal_id);
99  }
100 };
