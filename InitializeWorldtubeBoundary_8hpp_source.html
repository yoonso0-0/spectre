29 /*!
30  * \ingroup ActionsGroup
31  * \brief Initializes a H5WorldtubeBoundary
32  *
33  * \details Uses:
34  * - initialization tag
35  * `Cce::InitializationTags::H5WorldtubeBoundaryDataManager`,
36  * - const global cache tag `Cce::Tags::LMax`.
37  *
38  * Databox changes:
39  * - Adds:
40  * - `Cce::Tags::H5WorldtubeBoundaryDataManager`
41  * - `Tags::Variables<typename
42  * Metavariables::cce_boundary_communication_tags>`
43  * - Removes: nothing
44  * - Modifies: nothing
45  *
46  * \note This action relies on the `SetupDataBox` aggregated initialization
47  * mechanism, so `Actions::SetupDataBox` must be present in the `Initialization`
48  * phase action list prior to this action.
49  */
50 template <typename BoundaryCommunicationTags>
51 struct InitializeH5WorldtubeBoundary {
52  using initialization_tags =
53  tmpl::list<Tags::H5WorldtubeBoundaryDataManager>;
54  using initialization_tags_to_keep =
55  tmpl::list<Tags::H5WorldtubeBoundaryDataManager>;
56  using const_global_cache_tags =
57  tmpl::list<Tags::LMax, Tags::EndTimeFromFile, Tags::StartTimeFromFile>;
58 
59  using simple_tags = tmpl::list<::Tags::Variables<BoundaryCommunicationTags>>;
60 
61  template <typename DbTags, typename... InboxTags, typename Metavariables,
62  typename ArrayIndex, typename ActionList,
63  typename ParallelComponent,
64  Requires<tmpl::list_contains_v<
65  DbTags, Tags::H5WorldtubeBoundaryDataManager>> = nullptr>
66  static auto apply(db::DataBox<DbTags>& box,
67  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
68  const Parallel::GlobalCache<Metavariables>& /*cache*/,
69  const ArrayIndex& /*array_index*/,
70  const ActionList /*meta*/,
71  const ParallelComponent* const /*meta*/) noexcept {
72  const size_t l_max = db::get<Tags::LMax>(box);
73  Variables<typename Metavariables::cce_boundary_communication_tags>
74  boundary_variables{
75  Spectral::Swsh::number_of_swsh_collocation_points(l_max)};
76 
77  Initialization::mutate_assign<simple_tags>(make_not_null(&box),
78  std::move(boundary_variables));
79 
80  return std::make_tuple(std::move(box));
81  }
82 
83  template <typename DbTags, typename... InboxTags, typename Metavariables,
84  typename ArrayIndex, typename ActionList,
85  typename ParallelComponent,
86  Requires<not tmpl::list_contains_v<
87  DbTags, Tags::H5WorldtubeBoundaryDataManager>> = nullptr>
88  static auto apply(db::DataBox<DbTags>& box,
89  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
90  const Parallel::GlobalCache<Metavariables>& /*cache*/,
91  const ArrayIndex& /*array_index*/,
92  const ActionList /*meta*/,
93  const ParallelComponent* const /*meta*/) noexcept {
94  return std::make_tuple(std::move(box));
95  }
96 };
