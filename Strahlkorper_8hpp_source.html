58  /// Construct a Strahlkorper from a DataVector containing the radius
59  /// at the collocation points.
60  ///
61  /// \note The collocation points of the constructed Strahlkorper
62  /// will not be exactly `radius_at_collocation_points`. Instead,
63  /// the constructed Strahlkorper will match the shape given by
64  /// `radius_at_collocation_points` only to order (`l_max`,`m_max`).
65  /// This is because the YlmSpherepack representation of the
66  /// Strahlkorper has more collocation points than spectral
67  /// coefficients. Specifically, `radius_at_collocation_points` has
68  /// \f$(l_{\rm max} + 1) (2 m_{\rm max} + 1)\f$ degrees of freedom,
69  /// but because there are only
70  /// \f$m_{\rm max}^2+(l_{\rm max}-m_{\rm max})(2m_{\rm max}+1)\f$
71  /// spectral coefficients, it is not possible to choose spectral
72  /// coefficients to exactly match all points in
73  /// `radius_at_collocation_points`.
74  Strahlkorper(size_t l_max, size_t m_max,
75  const DataVector& radius_at_collocation_points,
76  std::array<double, 3> center) noexcept;
77 
78  /// Prolong or restrict another surface to the given `l_max` and `m_max`.

97  /*!
98  * These coefficients are stored as SPHEREPACK coefficients.
99  * Suppose you represent a set of coefficients \f$F^{lm}\f$ in the expansion
100  * \f[
101  * f(\theta,\phi) =
102  * \sum_{l=0}^{l_{max}} \sum_{m=-l}^{l} F^{lm} Y^{lm}(\theta,\phi)
103  * \f]
104  * Here the \f$Y^{lm}(\theta,\phi)\f$ are the usual complex-valued scalar
105  * spherical harmonics, so \f$F^{lm}\f$ are also complex-valued.
106  * But here we assume that \f$f(\theta,\phi)\f$ is real, so therefore
107  * the \f$F^{lm}\f$ obey \f$F^{l-m} = (-1)^m (F^{lm})^\star\f$. So one
108  * does not need to store both real and imaginary parts for both positive
109  * and negative \f$m\f$, and the stored coefficients can all be real.
110  *
111  * So the stored coefficients are:
112  * \f{align}
113  * \text{coefficients()(l,m)} &= (-1)^m \sqrt{\frac{2}{\pi}}
114  * \Re(F^{lm}) \quad \text{for} \quad m\ge 0, \\
115  * \text{coefficients()(l,m)} &= (-1)^m \sqrt{\frac{2}{\pi}}
116  * \Im(F^{lm}) \quad \text{for} \quad m<0
117  * \f}
118  */
119  SPECTRE_ALWAYS_INLINE const DataVector& coefficients() const noexcept {
120  return strahlkorper_coefs_;
121  }
122  SPECTRE_ALWAYS_INLINE DataVector& coefficients() noexcept {
123  return strahlkorper_coefs_;
124  }
125 
126  /// Point about which the spectral basis of the Strahlkorper is expanded.
127  /// The center is given in the frame in which the Strahlkorper is defined.
128  /// This center must be somewhere inside the Strahlkorper, but in principle
129  /// it can be anywhere. See `physical_center()` for a different measure.
130  SPECTRE_ALWAYS_INLINE const std::array<double, 3>& center() const noexcept {
131  return center_;
132  }
133 
134  /// Approximate physical center (determined by \f$l=1\f$ coefficients)
135  /// Implementation of Eqs. (38)-(40) in \cite Hemberger2012jz
136  std::array<double, 3> physical_center() const noexcept;
137 
138  /// Average radius of the surface (determined by \f$Y_{00}\f$ coefficient)
139  double average_radius() const noexcept;
140 
141  /// Maximum \f$l\f$ in \f$Y_{lm}\f$ decomposition.
142  SPECTRE_ALWAYS_INLINE size_t l_max() const noexcept { return l_max_; }
143 
144  /// Maximum \f$m\f$ in \f$Y_{lm}\f$ decomposition.
145  SPECTRE_ALWAYS_INLINE size_t m_max() const noexcept { return m_max_; }
146 
147  /// Radius at a particular angle \f$(\theta,\phi)\f$.
148  /// This is inefficient if done at multiple points many times.
149  /// See YlmSpherepack for alternative ways of computing this.
150  double radius(double theta, double phi) const noexcept;
151 
152  /// Determine if a point `x` is contained inside the surface.
153  /// The point must be given in Cartesian coordinates in the frame in
154  /// which the Strahlkorper is defined.
155  /// This is inefficient if done at multiple points many times.
156  bool point_is_contained(const std::array<double, 3>& x) const noexcept;
157 
158  SPECTRE_ALWAYS_INLINE const YlmSpherepack& ylm_spherepack() const noexcept {
159  return ylm_;
160  }
161 
162  private:
163  size_t l_max_{2}, m_max_{2};
164  YlmSpherepack ylm_{2, 2};
165  std::array<double, 3> center_{{0.0, 0.0, 0.0}};
166  DataVector strahlkorper_coefs_ = DataVector(ylm_.spectral_size(), 0.0);
167 };
