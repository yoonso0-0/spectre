400  static Phase determine_next_phase(
401  const Phase& current_phase,
402  const Parallel::CProxy_GlobalCache<
403  EvolutionMetavars>& /*cache_proxy*/) noexcept {
404  switch (current_phase) {
405  case Phase::Initialization:
406  return evolution::is_numeric_initial_data_v<initial_data>
407  ? Phase::RegisterWithElementDataReader
408  : Phase::InitializeInitialDataDependentQuantities;
409  case Phase::RegisterWithElementDataReader:
410  return Phase::ImportInitialData;
411  case Phase::ImportInitialData:
412  return Phase::InitializeInitialDataDependentQuantities;
413  case Phase::InitializeInitialDataDependentQuantities:
414  return Phase::InitializeTimeStepperHistory;
415  case Phase::InitializeTimeStepperHistory:
416  return Phase::Register;
417  case Phase::Register:
418  return Phase::Evolve;
419  case Phase::Evolve:
420  return Phase::Exit;
421  case Phase::Exit:
422  ERROR(
423  "Should never call determine_next_phase with the current phase "
424  "being 'Exit'");
425  default:
426  ERROR(
427  "Unknown type of phase. Did you static_cast<Phase> an integral "
428  "value?");
429  }
430  }
