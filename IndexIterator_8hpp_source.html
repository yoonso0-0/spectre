19 /// Each integer of the Index will vary from 0 to extents[d] - 1,

35  IndexIterator<Dim>& operator=(IndexIterator<Dim>&&) = delete;
36  // @}
37  /// \endcond

40  explicit operator bool() const noexcept { return valid_; }
