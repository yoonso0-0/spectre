74 /// \brief post_interpolation_callback that outputs
75 /// a time series on a surface.
76 ///
77 /// Uses:
78 /// - Metavariables
79 /// - `temporal_id`
80 /// - DataBox:
81 /// - `TagsToObserve`
82 ///
83 /// `ObservationType` is a type that distinguishes this observation
84 /// from other things that call observers::ThreadedActions::ObserverWriter,
85 /// so that different observations do not collide.
86 ///
87 /// This is an InterpolationTargetTag::post_interpolation_callback;
88 /// see InterpolationTarget for a description of InterpolationTargetTag.
89 template <typename TagsToObserve, typename ObservationType,
90  typename InterpolationTargetTag>
91 struct ObserveTimeSeriesOnSurface {
92  using observed_reduction_data_tags = observers::make_reduction_data_tags<
93  tmpl::list<typename detail::reduction_data_type<TagsToObserve>::type>>;
94  using observation_types = tmpl::list<ObservationType>;
95 
96  static constexpr double fill_invalid_points_with =
97  std::numeric_limits<double>::quiet_NaN();
98 
99  template <typename DbTags, typename Metavariables>
100  static void apply(
101  const db::DataBox<DbTags>& box,
102  Parallel::GlobalCache<Metavariables>& cache,
103  const typename Metavariables::temporal_id::type& temporal_id) noexcept {
104  auto& proxy = Parallel::get_parallel_component<
105  observers::ObserverWriter<Metavariables>>(cache);
106 
107  // We call this on proxy[0] because the 0th element of a NodeGroup is
108  // always guaranteed to be present.
109  Parallel::threaded_action<observers::ThreadedActions::WriteReductionData>(
110  proxy[0],
111  observers::ObservationId(temporal_id.substep_time().value(),
112  pretty_type::get_name<ObservationType>()),
113  static_cast<size_t>(Parallel::my_node()),
114  std::string{"/" + pretty_type::short_name<InterpolationTargetTag>()},
115  detail::make_legend(TagsToObserve{}),
116  detail::make_reduction_data(box, temporal_id.substep_time().value(),
117  TagsToObserve{}));
118  }
119 };
120 } // namespace callbacks
