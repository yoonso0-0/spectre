84 template <typename T, typename Metavariables>
85 T Option::parse_as() const {
86  try {
87  // yaml-cpp's `as` method won't parse empty nodes, so we need to
88  // inline a bit of its logic.
89  Options_detail::wrap_create_types<T, Metavariables> result{};
90  if (YAML::convert<decltype(result)>::decode(node(), result)) {
91  return Options_detail::unwrap_create_types(std::move(result));
92  }
93  // clang-tidy: thrown exception is not nothrow copy constructible
94  throw YAML::BadConversion(node().Mark()); // NOLINT
95  } catch (const YAML::BadConversion& e) {
96  // This happens when trying to parse an empty value as a container
97  // with no entries.
98  if ((tt::is_a_v<std::vector, T> or tt::is_std_array_of_size_v<0, T> or
99  tt::is_maplike_v<T>) and node().IsNull()) {
100  return T{};
101  }
102  Context error_context = context();
103  error_context.line = e.mark.line;
104  error_context.column = e.mark.column;
105  std::ostringstream ss;
106  ss << "Failed to convert value to type "
107  << Options_detail::yaml_type<T>::value() << ":";
108 
109  const std::string value_text = YAML::Dump(node());
110  if (value_text.find('\n') == std::string::npos) {
111  ss << " " << value_text;
112  } else {
113  // Indent each line of the value by two spaces and start on a new line
114  ss << "\n ";
115  for (char c : value_text) {
116  ss << c;
117  if (c == '\n') {
118  ss << " ";
119  }
120  }
121  }
122 
123  if (tt::is_a_v<std::vector, T> or tt::is_std_array_v<T>) {
124  ss << "\n\nNote: For sequences this can happen because the length of the "
125  "sequence specified\nin the input file is not equal to the length "
126  "expected by the code. Sequences in\nfiles can be denoted either "
127  "as a bracket enclosed list ([foo, bar]) or with each\nentry on a "
128  "separate line, indented and preceeded by a dash ( - foo).";
129  }
130  PARSE_ERROR(error_context, ss.str());
131  } catch (const Options_detail::propagate_context& e) {
132  Context error_context = context();
133  // Avoid line numbers in the middle of the trace
134  error_context.line = -1;
135  error_context.column = -1;
136  PARSE_ERROR(error_context, e.message());
137  } catch (std::exception& e) {
138  ERROR("Unexpected exception: " << e.what());
139  }
140 }

147 /// \ingroup OptionParsingGroup
148 /// \brief Class that handles parsing an input file
149 ///
150 /// Options must be given YAML data to parse before output can be
151 /// extracted. This can be done either from a file (parse_file
152 /// method), from a string (parse method), or, in the case of
153 /// recursive parsing, from an Option (parse method). The options
154 /// can then be extracted using the get method.
155 ///
156 /// \example
157 /// \snippet Test_Options.cpp options_example_scalar_struct
158 /// \snippet Test_Options.cpp options_example_scalar_parse
159 ///
160 /// \see the \ref dev_guide_option_parsing tutorial
161 ///
162 /// \tparam OptionList the list of option structs to parse
163 /// \tparam Group the option group with a group hierarchy
164 template <typename OptionList, typename Group = NoSuchType>
165 class Parser {
166  public:
167  /// \param help_text an overall description of the options
168  explicit Parser(std::string help_text) noexcept;
169 
170  /// Parse a string to obtain options and their values.
171  ///
172  /// \param options the string holding the YAML formatted options
173  void parse(std::string options) noexcept;
174 
175  /// Parse an Option to obtain options and their values.
176  void parse(const Option& options);
177 
178  /// Parse a file containing options
179  ///
180  /// \param file_name the path to the file to parse
181  void parse_file(const std::string& file_name) noexcept;
182 
183  /// Parse a YAML node containing options
184  void parse(const YAML::Node& node);
185 
186  /// Get the value of the specified option
187  ///
188  /// \tparam T the option to retrieve
189  /// \return the value of the option
190  template <typename T, typename Metavariables = NoSuchType>
191  typename T::type get() const;
192 
193  /// Call a function with the specified options as arguments.
194  ///
195  /// \tparam TagList a typelist of options to pass
196  /// \return the result of the function call
197  template <typename TagList, typename Metavariables = NoSuchType, typename F>
198  decltype(auto) apply(F&& func) const;
199 
200  /// Get the help string
201  std::string help() const noexcept;
202 
203  private:
204  template <typename, typename, typename>
205  friend struct Options_detail::get_impl;
206 
207  static_assert(tt::is_a<tmpl::list, OptionList>::value,
208  "The OptionList template parameter to Options must be a "
209  "tmpl::list<...>.");
210 
211  /// All top-level options and top-level groups of options. Every option in
212  /// `OptionList` is either in this list or in the hierarchy of one of the
213  /// groups in this list.
214  using tags_and_subgroups_list = tmpl::remove_duplicates<tmpl::transform<
215  OptionList, Options_detail::find_subgroup<tmpl::_1, tmpl::pin<Group>>>>;
216 
217  // The maximum length of an option label.
218  static constexpr int max_label_size_ = 70;
219 
220  /// Check that the size is not smaller than the lower bound
221  ///
222  /// \tparam T the option struct
223  /// \param t the value of the read in option
224  template <typename T>
225  void check_lower_bound_on_size(const typename T::type& t,
226  const Context& context) const;
227 
228  /// Check that the size is not larger than the upper bound
229  ///
230  /// \tparam T the option struct
231  /// \param t the value of the read in option
232  template <typename T>
233  void check_upper_bound_on_size(const typename T::type& t,
234  const Context& context) const;
235 
236  /// If the options has a lower bound, check it is satisfied.
237  ///
238  /// Note: Lower bounds are >=, not just >.
239  /// \tparam T the option struct
240  /// \param t the value of the read in option
241  template <typename T>
242  void check_lower_bound(const typename T::type& t,
243  const Context& context) const;
244 
245  /// If the options has a upper bound, check it is satisfied.
246  ///
247  /// Note: Upper bounds are <=, not just <.
248  /// \tparam T the option struct
249  /// \param t the value of the read in option
250  template <typename T>
251  void check_upper_bound(const typename T::type& t,
252  const Context& context) const;
253 
254  /// Get the help string for parsing errors
255  std::string parsing_help(const YAML::Node& options) const noexcept;
256 
257  /// Error message when failed to parse an input file.
258  [[noreturn]] void parser_error(const YAML::Exception& e) const noexcept;
259 
260  std::string help_text_{};
261  Context context_{};
262  std::string input_source_{};
263  std::unordered_map<std::string, YAML::Node> parsed_options_{};
264 };

266 template <typename OptionList, typename Group>
267 Parser<OptionList, Group>::Parser(std::string help_text) noexcept
268  : help_text_(std::move(help_text)) {
269  tmpl::for_each<tags_and_subgroups_list>([](auto t) noexcept {
270  using T = typename decltype(t)::type;
271  const std::string label = name<T>();
272  ASSERT(label.size() <= max_label_size_,
273  "The option name " << label
274  << " is too long for nice formatting, "
275  "please shorten the name to "
276  << max_label_size_ << " characters or fewer");
277  ASSERT(std::strlen(T::help) > 0,
278  "You must supply a help string of non-zero length for " << label);
279  });
280 }

318 template <typename OptionList, typename Group>
319 void Parser<OptionList, Group>::parse(const YAML::Node& node) {
320  if (not(node.IsMap() or node.IsNull())) {
321  PARSE_ERROR(context_, "'" << node << "' does not look like options.\n"
322  << help());
323  }
324 
325  // Use an ordered container so the missing options are reported in
326  // the order they are given in the help string.
327  std::vector<std::string> valid_names;
328  valid_names.reserve(tmpl::size<tags_and_subgroups_list>{});
329  tmpl::for_each<tags_and_subgroups_list>([&valid_names](auto opt) noexcept {
330  using Opt = tmpl::type_from<decltype(opt)>;
331  const std::string label = name<Opt>();
332  ASSERT(alg::find(valid_names, label) == valid_names.end(),
333  "Duplicate option name: " << label);
334  valid_names.push_back(label);
335  });
336 
337  for (const auto& name_and_value : node) {
338  const auto& name = name_and_value.first.as<std::string>();
339  const auto& value = name_and_value.second;
340  auto context = context_;
341  context.line = name_and_value.first.Mark().line;
342  context.column = name_and_value.first.Mark().column;
343 
344  // Check for duplicate key
345  if (0 != parsed_options_.count(name)) {
346  PARSE_ERROR(context, "Option '" << name << "' specified twice.\n"
347  << parsing_help(node));
348  }
349 
350  // Check for invalid key
351  const auto name_it = alg::find(valid_names, name);
352  if (name_it == valid_names.end()) {
353  PARSE_ERROR(context, "Option '" << name << "' is not a valid option.\n"
354  << parsing_help(node));
355  }
356 
357  parsed_options_.emplace(name, value);
358  valid_names.erase(name_it);
359  }
360 
361  if (not valid_names.empty()) {
362  PARSE_ERROR(context_, "You did not specify the option"
363  << (valid_names.size() == 1 ? " " : "s ")
364  << (MakeString{} << valid_names) << "\n" << parsing_help(node));
365  }
366 
367  // Any actual warnings will be printed by later calls to get or
368  // apply, but it is not clear how to determine in those functions
369  // whether this message should be printed.
370  if (std::is_same_v<Group, NoSuchType> and context_.top_level) {
371  Parallel::printf_error(
372  "The following options differ from their suggested values:\n");
373  }
374 }

377 template <typename Tag, typename Metavariables, typename Subgroup>
378 struct get_impl {
379  template <typename OptionList, typename Group>
380  static typename Tag::type apply(const Parser<OptionList, Group>& opts) {
381  static_assert(
382  tmpl::list_contains_v<OptionList, Tag>,
383  "Could not find requested option in the list of options provided. Did "
384  "you forget to add the option tag to the OptionList?");
385  const std::string subgroup_label = name<Subgroup>();
386  Parser<options_in_group<OptionList, Subgroup>, Subgroup> subgroup_options(
387  Subgroup::help);
388  subgroup_options.context_ = opts.context_;
389  subgroup_options.context_.append("In group " + subgroup_label);
390  subgroup_options.parse(opts.parsed_options_.find(subgroup_label)->second);
391  return subgroup_options.template get<Tag, Metavariables>();
392  }
393 };

395 template <typename Tag, typename Metavariables>
396 struct get_impl<Tag, Metavariables, Tag> {
397  template <typename OptionList, typename Group>
398  static typename Tag::type apply(const Parser<OptionList, Group>& opts) {
399  static_assert(
400  tmpl::list_contains_v<OptionList, Tag>,
401  "Could not find requested option in the list of options provided. Did "
402  "you forget to add the option tag to the OptionList?");
403  const std::string label = name<Tag>();
404 
405  Option option(opts.parsed_options_.find(label)->second, opts.context_);
406  option.append_context("While parsing option " + label);
407 
408  auto t = option.parse_as<typename Tag::type, Metavariables>();
409 
410  if constexpr (Options_detail::has_suggested<Tag>::value) {
411  static_assert(
412  std::is_same_v<decltype(Tag::suggested_value()), typename Tag::type>,
413  "Suggested value is not of the same type as the option.");
414 
415  // This can be easily relaxed, but using it would require
416  // writing comparison operators for abstract base classes. If
417  // someone wants this enough to go though the effort of doing
418  // that, it would just require comparing the dereferenced
419  // pointers below to decide whether the suggestion was followed.
420  static_assert(not tt::is_a_v<std::unique_ptr, typename Tag::type>,
421  "Suggestions are not supported for pointer types.");
422 
423  const auto suggested_value = Tag::suggested_value();
424  {
425  Context context;
426  context.append("Checking SUGGESTED value for " + name<Tag>());
427  opts.template check_lower_bound_on_size<Tag>(suggested_value, context);
428  opts.template check_upper_bound_on_size<Tag>(suggested_value, context);
429  opts.template check_lower_bound<Tag>(suggested_value, context);
430  opts.template check_upper_bound<Tag>(suggested_value, context);
431  }
432 
433  if (t != suggested_value) {
434  Parallel::printf_error(
435  "%s, line %d:\n Specified: %s\n Suggested: %s\n",
436  label, option.context().line + 1,
437  (MakeString{} << std::boolalpha << t),
438  (MakeString{} << std::boolalpha << suggested_value));
439  }
440  }
441 
442  opts.template check_lower_bound_on_size<Tag>(t, option.context());
443  opts.template check_upper_bound_on_size<Tag>(t, option.context());
444  opts.template check_lower_bound<Tag>(t, option.context());
445  opts.template check_upper_bound<Tag>(t, option.context());
446  return t;
447  }
448 };

588 template <typename OptionList, typename Group>
589 [[noreturn]] void Parser<OptionList, Group>::parser_error(
590  const YAML::Exception& e) const noexcept {
591  auto context = context_;
592  context.line = e.mark.line;
593  context.column = e.mark.column;
594  // Inline the top_level branch of PARSE_ERROR to avoid warning that
595  // the other branch would call terminate. (Parser errors can only
596  // be generated at top level.)
597  ERROR(
598  "\n"
599  << context
