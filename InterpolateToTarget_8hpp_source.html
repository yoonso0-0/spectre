29 /// \ingroup ActionsGroup
30 /// \brief Interpolates and sends points to an InterpolationTarget.
31 ///
32 /// This is invoked on DgElementArray.
33 /// This action should be used only if the elements are using global
34 /// timestepping. An alternative strategy involving dense output must be devised
35 /// for interpolation during local timestepping
36 ///
37 /// Uses:
38 /// - DataBox:
39 /// - `intrp::Tags::InterpPointInfo<Metavariables>`
40 /// - `Tags::Mesh<Metavariables::volume_dim>`
41 /// - Variables tagged by
42 /// InterpolationTargetTag::vars_to_interpolate_to_target
43 /// - any box tags used by the `should_interpolate` function of
44 /// `InterpolationTargetTag`
45 ///
46 /// DataBox changes:
47 /// - Adds: nothing
48 /// - Removes: nothing
49 /// - Modifies: nothing
50 template <typename InterpolationTargetTag>
51 struct InterpolateToTarget {
52  template <typename DbTags, typename Metavariables, typename... InboxTags,
53  typename ArrayIndex, typename ActionList,
54  typename ParallelComponent>
55  static std::tuple<db::DataBox<DbTags>&&> apply(
56  db::DataBox<DbTags>& box,
57  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
58  Parallel::GlobalCache<Metavariables>& cache,
59  const ArrayIndex& array_index, const ActionList /*meta*/,
60  const ParallelComponent* const /*meta*/) noexcept {
61  // Exit early if we should not interpolate in the current state
62  if constexpr (InterpolationTarget_detail::is_should_interpolate_callable_v<
63  InterpolationTargetTag, db::DataBox<DbTags>&>) {
64  if (not InterpolationTargetTag::should_interpolate(box)) {
65  return std::forward_as_tuple(std::move(box));
66  }
67  }
68  static constexpr size_t dim = Metavariables::volume_dim;
69 
70  // Get element logical coordinates.
71  const auto& block_logical_coords =
72  get<Vars::PointInfoTag<InterpolationTargetTag, dim>>(
73  db::get<Tags::InterpPointInfo<Metavariables>>(box));
74  const std::vector<ElementId<dim>> element_ids{{array_index}};
75  const auto element_coord_holders =
76  element_logical_coordinates(element_ids, block_logical_coords);
77 
78  // There is exactly one element_id in the list of element_ids.
79  if (element_coord_holders.count(element_ids[0]) == 0) {
80  // There are no points in this element, so we don't need
81  // to do anything.
82  return std::forward_as_tuple(std::move(box));
83  }
84 
85  // There are points in this element, so interpolate to them and
86  // send the interpolated data to the target. This is done
87  // in several steps:
88  const auto& element_coord_holder = element_coord_holders.at(element_ids[0]);
89 
90  using compute_items_in_new_box = tmpl::list_difference<
91  typename InterpolationTargetTag::compute_items_on_source, DbTags>;
92 
93  // 1. Create a new DataBox that contains
94  // InterpolationTargetTag::compute_items_on_source
95  auto new_box =
96  db::create_from<db::RemoveTags<>, db::AddSimpleTags<>,
97  db::AddComputeTags<compute_items_in_new_box>>(
98  std::move(box));
99 
100  // 2. Set up local variables to hold vars_to_interpolate and fill it
101  const auto& mesh = db::get<domain::Tags::Mesh<dim>>(new_box);
102  Variables<typename InterpolationTargetTag::vars_to_interpolate_to_target>
103  local_vars(mesh.number_of_grid_points());
104 
105  tmpl::for_each<
106  typename InterpolationTargetTag::vars_to_interpolate_to_target>(
107  [&new_box, &local_vars ](auto tag_v) noexcept {
108  using tag = typename decltype(tag_v)::type;
109  get<tag>(local_vars) = db::get<tag>(new_box);
110  });
111 
112  // 3. Set up interpolator
113  intrp::Irregular<dim> interpolator(
114  mesh, element_coord_holder.element_logical_coords);
115 
116  // 4. Interpolate and send interpolated data to target
117  auto& receiver_proxy = Parallel::get_parallel_component<
118  InterpolationTarget<Metavariables, InterpolationTargetTag>>(cache);
119  Parallel::simple_action<
120  Actions::InterpolationTargetVarsFromElement<InterpolationTargetTag>>(
121  receiver_proxy,
122  std::vector<Variables<
123  typename InterpolationTargetTag::vars_to_interpolate_to_target>>(
124  {interpolator.interpolate(local_vars)}),
125  std::vector<std::vector<size_t>>({element_coord_holder.offsets}),
126  db::get<typename Metavariables::temporal_id>(new_box));
127 
128  // 5. Put back original DataBox.
129  box = db::create_from<db::RemoveTags<compute_items_in_new_box>,
130  db::AddSimpleTags<>>(std::move(new_box));
131 
132  return std::forward_as_tuple(std::move(box));
133  }
134 };
135 } // namespace Actions
136 } // namespace intrp
