22 /*!
23  * \ingroup DiscontinuousGalerkinGroup
24  * \brief A cached exponential filter.
25  *
26  * Applies an exponential filter in each logical direction to each component of
27  * the tensors `TagsToFilter`. The exponential filter rescales the 1d modal
28  * coefficients \f$c_i\f$ as:
29  *
30  * \f{align*}{
31  * c_i\to c_i \exp\left[-\alpha_{\mathrm{ef}}
32  * \left(\frac{i}{N}\right)^{2\beta_{\mathrm{ef}}}\right]
33  * \f}
34  *
35  * where \f$N\f$ is the basis degree (number of grid points per element per
36  * dimension minus one), \f$\alpha_{\mathrm{ef}}\f$ determines how much the
37  * coefficients are rescaled, and \f$\beta_{\mathrm{ef}}\f$ (given by the
38  * `HalfPower` option) determines how aggressive/broad the filter is (lower
39  * values means filtering more coefficients). Setting
40  * \f$\alpha_{\mathrm{ef}}=36\f$ results in effectively zeroing the highest
41  * coefficient (in practice it gets rescaled by machine epsilon). The same
42  * \f$\alpha_{\mathrm{ef}}\f$ and \f$\beta_{\mathrm{ef}}\f$ are used in each
43  * logical direction. For a discussion of filtering see section 5.3 of
44  * \cite HesthavenWarburton.
45  *
46  * #### Design decision:
47  *
48  * - The reason for the `size_t` template parameter is to allow for different
49  * `Alpha` and `HalfPower` parameters for different tensors while still being
50  * able to cache the matrices. If different `Alpha` or `HalfPower` parameters
51  * are desired for filtering different tensors, then multiple filters must be
52  * inserted into the GlobalCache with different `FilterIndex` values. In
53  * the input file these will be specified as `ExpFilterFILTER_INDEX`, e.g.
54  * - Filtering:
55  * - ExpFilter0:
56  * - Alpha: 12
57  * - HalfPower: 32
58  * - ExpFilter1:
59  * - Alpha: 36
60  * - HalfPower: 32
61  */
62 template <size_t FilterIndex>
63 class Exponential {
64  public:
65  /// \brief The value of `exp(-alpha)` is what the highest modal coefficient is
66  /// rescaled by.
67  struct Alpha {
68  using type = double;
69  static constexpr Options::String help =
70  "exp(-alpha) is rescaling of highest coefficient";
71  static type lower_bound() noexcept { return 0.0; }
72  };
73 
74  /*!
75  * \brief Half of the exponent in the exponential.
76  *
77  * \f{align*}{
78  * c_i\to c_i \exp\left[-\alpha \left(\frac{i}{N}\right)^{2m}\right]
79  * \f}
80  */
81  struct HalfPower {
82  using type = unsigned;
83  static constexpr Options::String help =
84  "Half of the exponent in the generalized Gaussian";
85  static type lower_bound() noexcept { return 1; }
86  };
87 
88  /// \brief Turn the filter off
89  ///
90  /// This option exists to temporarily disable the filter for debugging
91  /// purposes. For problems where filtering is not needed, the preferred
92  /// approach is to not compile the filter into the executable.
93  struct DisableForDebugging {
94  using type = bool;
95  static type suggested_value() noexcept { return false; }
96  static constexpr Options::String help = {"Disable the filter"};
97  };
98 
99  using options = tmpl::list<Alpha, HalfPower, DisableForDebugging>;
100  static constexpr Options::String help = {"An exponential filter."};
101  static std::string name() noexcept {
102  return "ExpFilter" + std::to_string(FilterIndex);
103  }
104 
105  Exponential() = default;
106 
107  Exponential(double alpha, unsigned half_power,
108  bool disable_for_debugging) noexcept;

111  const Matrix& filter_matrix(const Mesh<1>& mesh) const noexcept;
112 
113  bool disable_for_debugging() const noexcept { return disable_for_debugging_; }
114 
115  // NOLINTNEXTLINE(google-runtime-references)
116  void pup(PUP::er& p) noexcept;
117

119  template <size_t LocalFilterIndex>
120  // NOLINTNEXTLINE(readability-redundant-declaration)
121  friend bool operator==(const Exponential<LocalFilterIndex>& lhs,
122  const Exponential<LocalFilterIndex>& rhs) noexcept;
123 
124  double alpha_{36.0};
125  unsigned half_power_{16};
126  bool disable_for_debugging_{false};
127 };
