19 /// \ingroup ComputationalDomainGroup
20 /// \brief A function that has a piecewise-constant `MaxDeriv`th derivative.
21 template <size_t MaxDeriv>
22 class PiecewisePolynomial : public FunctionOfTime {

39  // clang-tidy: google-runtime-references

41  WRAPPED_PUPable_decl_template(PiecewisePolynomial<MaxDeriv>); // NOLINT

43  /// Returns the function at an arbitrary time `t`.
44  std::array<DataVector, 1> func(double t) const noexcept override {
45  return func_and_derivs<0>(t);
46  }
47  /// Returns the function and its first derivative at an arbitrary time `t`.
48  std::array<DataVector, 2> func_and_deriv(double t) const noexcept override {
49  return func_and_derivs<1>(t);
50  }
51  /// Returns the function and the first two derivatives at an arbitrary time
52  /// `t`.
53  std::array<DataVector, 3> func_and_2_derivs(double t) const
54  noexcept override {
55  return func_and_derivs<2>(t);
56  }
57 
58  /// Updates the `MaxDeriv`th derivative of the function at the given time.
59  /// `updated_max_deriv` is a vector of the `MaxDeriv`ths for each component.
60  /// `next_expiration_time` is the next expiration time.
61  void update(double time_of_update, DataVector updated_max_deriv,
62  double next_expiration_time) noexcept;
63 
64  /// Used by `FunctionOfTimeUpdater` to reset the expiration time
65  /// when it decides not to do an update, but instead decides to
66  /// keep the current values valid for longer.
67  void reset_expiration_time(double next_expiration_time) noexcept;
68 
69  /// Returns the domain of validity of the function,
70  /// including the extrapolation region.
71  std::array<double, 2> time_bounds() const noexcept override {
72  return {{deriv_info_at_update_times_.front().time,
73  expiration_time_}};
74  }
75 
76  // NOLINTNEXTLINE(google-runtime-references)
77  void pup(PUP::er& p) override;
78 
79  private:
80  template <size_t LocalMaxDeriv>
81  friend bool operator==( // NOLINT(readability-redundant-declaration)
82  const PiecewisePolynomial<LocalMaxDeriv>& lhs,
83  const PiecewisePolynomial<LocalMaxDeriv>& rhs) noexcept;
84 
85  /// Returns the function and `MaxDerivReturned` derivatives at
86  /// an arbitrary time `t`.
87  /// The function has multiple components.
88  template <size_t MaxDerivReturned = MaxDeriv>
89  std::array<DataVector, MaxDerivReturned + 1> func_and_derivs(double t) const
90  noexcept;
91 
92  // There exists a DataVector for each deriv order that contains
93  // the values of that deriv order for all components.
94  using value_type = std::array<DataVector, MaxDeriv + 1>;
95 
96  // Holds information at single time at which the `MaxDeriv`th
97  // derivative has been updated.
98  struct DerivInfo {
99  double time{std::numeric_limits<double>::signaling_NaN()};
100  value_type derivs_coefs;
101 
102  DerivInfo() = default;
103 
104  // Constructor is needed for use of emplace_back of a vector
105  // (additionally, the constructor converts the supplied derivs to
106  // coefficients for simplified polynomial evaluation.)
107  DerivInfo(double t, value_type deriv) noexcept;
108 
109  // NOLINTNEXTLINE(google-runtime-references)
110  void pup(PUP::er& p) noexcept;
111 
112  bool operator==(const DerivInfo& rhs) const noexcept;
113  };
114 
115  /// Returns a DerivInfo corresponding to the closest element in the range of
116  /// DerivInfos with an update time that is less than or equal to `t`.
117  /// The function throws an error if `t` is less than all DerivInfo update
118  /// times. (unless `t` is just less than the earliest update time by roundoff,
119  /// in which case it returns the DerivInfo at the earliest update time.)
120  const DerivInfo& deriv_info_from_upper_bound(double t) const noexcept;
121 
122  std::vector<DerivInfo> deriv_info_at_update_times_;
123  double expiration_time_{std::numeric_limits<double>::lowest()};
124 };
