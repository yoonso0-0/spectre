20 /// \ingroup SwshGroup
21 /// \brief Convenience function for determining the number of spin-weighted
22 /// spherical harmonic collocation values that are stored for a given `l_max`
23 /// for a libsharp-compatible set of collocation points.
24 constexpr SPECTRE_ALWAYS_INLINE size_t
25 number_of_swsh_collocation_points(const size_t l_max) noexcept {
26  return (l_max + 1) * (2 * l_max + 1);
27 }

29 /// \ingroup SwshGroup
30 /// \brief Returns the number of spin-weighted spherical harmonic collocation
31 /// values in \f$\theta\f$ for a libsharp-compatible set of collocation
32 /// points.
33 ///
34 /// \details The full number of collocation points is the product of the number
35 /// of \f$\theta\f$ points and the number of \f$\phi\f$ points (a 'rectangular'
36 /// grid).
37 constexpr SPECTRE_ALWAYS_INLINE size_t
38 number_of_swsh_theta_collocation_points(const size_t l_max) noexcept {
39  return (l_max + 1);
40 }

42 /// \ingroup SwshGroup
43 /// \brief Returns the number of spin-weighted spherical harmonic collocation
44 /// values in \f$\phi\f$ for a libsharp-compatible set of collocation
45 /// points.
46 ///
47 /// \details The full number of collocation points is the product of the number
48 /// of \f$\theta\f$ points and the number of \f$\phi\f$ points (a 'rectangular'
49 /// grid).
50 constexpr SPECTRE_ALWAYS_INLINE size_t
51 number_of_swsh_phi_collocation_points(const size_t l_max) noexcept {
52  return (2 * l_max + 1);

55 /// \ingroup SwshGroup
56 /// \brief Obtain the three-dimensional mesh associated with a
57 /// libsharp-compatible sequence of spherical nodal shells.
58 /// \warning This is to be used only for operations in the radial direction! The
59 /// angular collocation points should not be regarded as having a useful product
60 /// representation for e.g. taking derivatives in just the theta direction.
61 /// Instead, use spin-weighted utilities for all angular operations.
62 SPECTRE_ALWAYS_INLINE Mesh<3> swsh_volume_mesh_for_radial_operations(
63  const size_t l_max, const size_t number_of_radial_points) {
64  return Mesh<3>{{{number_of_swsh_phi_collocation_points(l_max),
65  number_of_swsh_theta_collocation_points(l_max),
66  number_of_radial_points}},
67  Spectral::Basis::Legendre,
68  Spectral::Quadrature::GaussLobatto};
69 }

88 /// \brief A wrapper class for the spherical harmonic library collocation data
89 ///
90 /// \details The currently chosen library for spin-weighted spherical harmonic
91 /// transforms is libsharp. The `CollocationMetadata` class stores the
92 /// libsharp `sharp_geom_info` object, which contains data about
93 /// 1. The angular collocation points used in spin-weighted spherical harmonic
94 /// transforms
95 /// 2. The memory representation of double-type values at those collocation
96 /// points
97 ///
98 /// \tparam Representation the ComplexRepresentation, either
99 /// `ComplexRepresentation::Interleaved` or
100 /// `ComplexRepresentation::RealsThenImags` compatible with the generated
101 /// `CollocationMetadata` - this is necessary because the stored libsharp type
102 /// contains memory stride information.
103 template <ComplexRepresentation Representation>
104 class CollocationMetadata {
105  public:
106  class CollocationConstIterator {
107  public:
108  /// create a new iterator. defaults to the start of the supplied object
109  explicit CollocationConstIterator(
110  const gsl::not_null<const CollocationMetadata<Representation>*>
111  collocation,
112  const size_t start_index = 0) noexcept
113  : index_{start_index}, collocation_{collocation} {}
114 
115  /// recovers the data at the collocation point using a
116  /// `LibsharpCollocationPoint`, which stores the vector `offset` of the
117  /// location of the `theta`, `phi` point in libsharp compatible data
118  LibsharpCollocationPoint operator*() const noexcept {
119  return LibsharpCollocationPoint{index_, collocation_->theta(index_),
120  collocation_->phi(index_)};
121  }
122  /// advance the iterator by one position (prefix)
123  CollocationConstIterator& operator++() noexcept {
124  ++index_;
125  return *this;
126  };
127  /// advance the iterator by one position (postfix)
128  // clang-tidy wants this to return a const iterator
129  CollocationConstIterator operator++(int) noexcept { // NOLINT
130  return CollocationConstIterator(collocation_, index_++);
131  }
132 
133  /// retreat the iterator by one position (prefix)
134  CollocationConstIterator& operator--() noexcept {
135  --index_;
136  return *this;
137  };
138  /// retreat the iterator by one position (prefix)
139  // clang-tidy wants this to return a const iterator
140  CollocationConstIterator operator--(int) noexcept { // NOLINT
141  return CollocationConstIterator(collocation_, index_--);
142  }
143 
144  // @{
145  /// (In)Equivalence checks both the object and index for the iterator
146  bool operator==(const CollocationConstIterator& rhs) const noexcept {
147  return index_ == rhs.index_ and collocation_ == rhs.collocation_;
148  }
149  bool operator!=(const CollocationConstIterator& rhs) const noexcept {
150  return not(*this == rhs);
151  }
152  // @}
153 
154  private:
155  size_t index_;
156  const CollocationMetadata<Representation>* const collocation_;
157  };
158 
159  /// The representation of the block of complex values, which sets the stride
160  /// inside the libsharp type
161  static constexpr ComplexRepresentation complex_representation =
162  Representation;
163 
164  /// \brief Generates the libsharp collocation information and stores it in
165  /// `geom_info_`.
166  ///
167  /// \note If you will potentially use the same `l_max` collocation set more
168  /// than once, it is probably better to use the
169  /// `precomputed_spherical_harmonic_collocation` function
170  explicit CollocationMetadata(size_t l_max) noexcept;
171 
172  /// default constructor required for iterator use
173  ~CollocationMetadata() = default;
174  CollocationMetadata() = default;
175  CollocationMetadata(const CollocationMetadata&) = default;
176  CollocationMetadata(CollocationMetadata&&) = default;
177  CollocationMetadata& operator=(CollocationMetadata&) = default;
178  CollocationMetadata& operator=(CollocationMetadata&&) = default;
179 
180  /// retrieve the `sharp_geom_info*` stored. This should largely be used only
181  /// for passing to other libsharp functions. Otherwise, access elements
182  /// through iterator or access functions.
183  sharp_geom_info* get_sharp_geom_info() const noexcept {
184  return geom_info_.get();
185  }
186 
187  /// Retrieve the \f$\theta\f$ value for a given index in a libsharp-compatible
188  /// array
189  double theta(size_t offset) const noexcept;
190  /// Retrieve the \f$\phi\f$ value for a given index in a libsharp-compatible
191  /// array
192  double phi(size_t offset) const noexcept;
193 
194  constexpr size_t l_max() const noexcept { return l_max_; }
195 
196  /// Compute the number of entries the libsharp-compatible data structure
197  /// should have
198  constexpr size_t size() const noexcept {
199  return (l_max_ + 1) * (2 * l_max_ + 1);
200  }
201 
202  // @{
203  /// Get a bidirectional iterator to the start of the grid. `operator*` for
204  /// that iterator gives a `LibsharpCollocationPoint` with members `offset`,
205  /// `theta`, and `phi`
206  CollocationMetadata<Representation>::CollocationConstIterator begin() const
207  noexcept {
208  return CollocationConstIterator{make_not_null(this), 0};
209  }
210  CollocationMetadata<Representation>::CollocationConstIterator cbegin() const
211  noexcept {
212  return begin();
213  }
214  // @}
215  // @{
216  /// Get a bidirectional iterator to the end of the grid. `operator*` for
217  /// that iterator gives a `LibsharpCollocationPoint` with members `offset`,
218  /// `theta`, and `phi`
219  CollocationMetadata<Representation>::CollocationConstIterator end() const
220  noexcept {
221  return CollocationConstIterator{make_not_null(this), size()};
222  }
223  CollocationMetadata<Representation>::CollocationConstIterator cend() const
224  noexcept {
225  return end();
226  }
227  // @}
228 
229  private:
230  // an extra pointer layer is required for the peculiar way that libsharp
231  // constructs these values.
232  size_t l_max_ = 0;
233  std::unique_ptr<sharp_geom_info, detail::DestroySharpGeometry> geom_info_;
234 };
