88 template <typename InterfaceReturnType, typename DirectionsTag,
89  typename VolumeTagsList, typename... ArgumentTags>
90 struct DispatchInterfaceInvokable<true, InterfaceReturnType, DirectionsTag,
91  VolumeTagsList, ArgumentTags...> {
92  template <typename InterfaceInvokable, typename DbTagsList,
93  typename... ExtraArgs>
94  static constexpr auto apply(InterfaceInvokable&& /*interface_invokable*/,
95  const db::DataBox<DbTagsList>& box,
96  ExtraArgs&&... extra_args) noexcept {
97  DirectionMap<DirectionsTag::volume_dim, InterfaceReturnType> result{};
98  for (const auto& direction : get<DirectionsTag>(box)) {
99  auto interface_value = InterfaceInvokable::apply(
100  unmap_interface_args<
101  tmpl::list_contains_v<VolumeTagsList, ArgumentTags>>::
102  apply(direction,
103  get<tmpl::type_from<make_interface_tag_impl<
104  ArgumentTags, DirectionsTag, VolumeTagsList>>>(box))...,
105  extra_args...);
106  result.insert({direction, std::move(interface_value)});
107  }
108  return result;
109  }
110 };

231 // @{
232 /*!
233  * \brief Apply an invokable to the `box` on all interfaces given by the
234  * `DirectionsTag`.
235  *
236  * This function has an overload that takes an `interface_invokable` as its
237  * first argument, and one that takes an `InterfaceInvokable` class as its first
238  * template parameter instead. For the latter, the `InterfaceInvokable` class
239  * must have a static `apply` function. The `interface_invokable` or static
240  * `apply` function, respectively, is expected to be invokable with the types
241  * held by the `ArgumentTags`, followed by the `extra_args`. The `ArgumentTags`
242  * will be prefixed as `::Tags::Interface<DirectionsTag, ArgumentTag>` and thus
243  * taken from the interface, except for those specified in the `VolumeTags`.
244  *
245  * This function returns a `DirectionMap` that holds the value returned by
246  * the `interface_invokable` in every direction of the `DirectionsTag`.
247  *
248  * Here is an example how to use this function:
249  *
250  * \snippet Test_InterfaceHelpers.cpp interface_apply_example
251  *
252  * Here is another example how to use this function with a stateless invokable
253  * class:
254  *
255  * \snippet Test_InterfaceHelpers.cpp interface_apply_example_stateless
256  *
257  * This is the class that defines the invokable in the example above:
258  *
259  * \snippet Test_InterfaceHelpers.cpp interface_invokable_example
260  */
261 template <typename DirectionsTag, typename ArgumentTags, typename VolumeTags,
262  typename InterfaceInvokable, typename DbTagsList,
263  typename... ExtraArgs>
264 SPECTRE_ALWAYS_INLINE constexpr auto interface_apply(
265  InterfaceInvokable&& interface_invokable,
266  const db::DataBox<DbTagsList>& box, ExtraArgs&&... extra_args) noexcept {
267  return InterfaceHelpers_detail::
268  InterfaceApplyImpl<DirectionsTag, ArgumentTags, VolumeTags>::apply(
269  std::forward<InterfaceInvokable>(interface_invokable), box,
270  std::forward<ExtraArgs>(extra_args)...);
271 }
272 
273 // The `box` argument to this function overload is not constrained to
274 // `db::DataBox` to work around an issue with GCC <= 8.
275 //
276 // Details on the issue:
277 //
278 // GCC <= 8 tries to instantiate the `db::DataBox<DbTagsList>` template even
279 // when this function overload is _not_ SFINAE-selected. In that case the
280 // template parameter substitution for `DbTagsList` may contain base tags that
281 // causes errors with the `db::DataBox` template instantiation.
282 template <typename DirectionsTag, typename InterfaceInvokable,
283  typename DataBoxType, typename... ExtraArgs,
284  // Needed to disambiguate the overloads
285  typename ArgumentTags = typename InterfaceInvokable::argument_tags>
286 SPECTRE_ALWAYS_INLINE constexpr auto interface_apply(
287  const DataBoxType& box, ExtraArgs&&... extra_args) noexcept {
288  return interface_apply<DirectionsTag, ArgumentTags,
289  get_volume_tags<InterfaceInvokable>>(
290  InterfaceInvokable{}, box, std::forward<ExtraArgs>(extra_args)...);
291 }
