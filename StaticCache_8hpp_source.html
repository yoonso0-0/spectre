15 /// \ingroup UtilitiesGroup
16 /// Range of values for StaticCache indices. The `Start` is inclusive
17 /// and the `End` is exclusive. The range must not be empty.
18 template <size_t Start, size_t End>
19 struct CacheRange {
20  static_assert(Start < End, "CacheRange must include at least one value");
21  constexpr static size_t start = Start;
22  constexpr static size_t end = End;
23  constexpr static size_t size = end - start;
24  using value_type = size_t;
25 };

38 /// \ingroup UtilitiesGroup
39 /// A cache of objects intended to be stored in a static variable.
40 ///
41 /// Objects can be accessed via a combination of several `size_t` and `enum`
42 /// arguments. The range of each `size_t` argument is specified via a template
43 /// parameter of type `CacheRange<start, end>`, giving the first and
44 /// one-past-last values for the range. Each `enum` argument is specified by a
45 /// template parameter of type `CacheEnumeration<EnumerationType, Members...>`
46 /// giving the enumeration type and an explicit set of every enum member to be
47 /// cached.
48 ///
49 /// \example
50 /// A cache with only numeric indices:
51 /// \snippet Test_StaticCache.cpp static_cache
52 ///
53 /// \example
54 /// A cache with enumeration indices:
55 /// \snippet Test_StaticCache.cpp static_cache_with_enum
56 ///
57 /// \example
58 /// A cache with mixed numeric and enumeration indices:
59 /// \snippet Test_StaticCache.cpp static_cache_with_enum_and_numeric
60 ///
61 /// \example
62 /// A cache with no arguments at all (caching only a single object)
63 /// \snippet Test_StaticCache.cpp static_cache_no_args
64 ///
65 /// \see make_static_cache
66 ///
67 /// \tparam T type held in the cache
68 /// \tparam Ranges ranges of valid indices
69 template <typename Generator, typename T, typename... Ranges>
70 class StaticCache {
71  public:
72  template <typename Gen>
73  // NOLINTNEXTLINE(bugprone-forwarding-reference-overload)
74  explicit StaticCache(Gen&& generator) noexcept
75  : generator_{std::forward<Gen>(generator)} {}
76 
77  template <typename... Args>
78  const T& operator()(const Args... parameters) const noexcept {
79  static_assert(sizeof...(parameters) == sizeof...(Ranges),
80  "Number of arguments must match number of ranges.");
81  return unwrap_cache(generate_tuple<Ranges>(parameters)...);
82  }
83 
84  private:
85  template <typename Range, typename T1,
86  Requires<not std::is_enum<T1>::value> = nullptr>
87  auto generate_tuple(const T1 parameter) const noexcept {
88  static_assert(
89  tt::is_integer_v<std::decay_t<T1>>,
90  "The parameter passed for a CacheRange must be an integer type.");
91  return std::make_tuple(static_cast<size_t>(parameter),
92  std::integral_constant<size_t, Range::start>{},
93  std::make_index_sequence<Range::size>{});
94  }
95 
96  template <typename Range, typename T1,
97  Requires<std::is_enum<T1>::value> = nullptr>
98  std::tuple<std::decay_t<T1>, Range> generate_tuple(const T1 parameter) const
99  noexcept {
100  static_assert(
101  std::is_same<typename Range::value_type, std::decay_t<T1>>::value,
102  "Mismatched enum parameter type and cached type.");
103  return {parameter, Range{}};
104  }
105 
106  template <typename... IntegralConstantValues>
107  const T& unwrap_cache() const noexcept {
108  static const T cached_object = generator_(IntegralConstantValues::value...);
109  return cached_object;
110  }
111 
112  template <typename... IntegralConstantValues, size_t IndexOffset,
113  size_t... Is, typename... Args>
114  const T& unwrap_cache(
115  std::tuple<size_t, std::integral_constant<size_t, IndexOffset>,
116  std::index_sequence<Is...>>
117  parameter0,
118  Args... parameters) const noexcept {
119  if (UNLIKELY(IndexOffset > std::get<0>(parameter0) or
120  std::get<0>(parameter0) >= IndexOffset + sizeof...(Is))) {
121  ERROR("Index out of range: " << IndexOffset
122  << " <= " << std::get<0>(parameter0) << " < "
123  << IndexOffset + sizeof...(Is));
124  }
125  // note that the act of assigning to the specified function pointer type
126  // fixes the template arguments that need to be inferred.
127  static const std::array<
128  const T& (StaticCache<Generator, T, Ranges...>::*)(Args...) const,
129  sizeof...(Is)>
130  cache{{&StaticCache<Generator, T, Ranges...>::unwrap_cache<
131  IntegralConstantValues...,
132  std::integral_constant<size_t, Is + IndexOffset>>...}};
133  // The array `cache` holds pointers to member functions, so we dereference
134  // the pointer and invoke it on `this`.
135  return (this->*gsl::at(cache, std::get<0>(parameter0) - IndexOffset))(
136  parameters...);
137  }
138 
139  template <typename... IntegralConstantValues, typename EnumType,
140  EnumType... EnumValues, typename... Args>
141  const T& unwrap_cache(
142  std::tuple<EnumType, CacheEnumeration<EnumType, EnumValues...>>
143  parameter0,
144  Args... parameters) const noexcept {
145  size_t array_location = std::numeric_limits<size_t>::max();
146  static const std::array<EnumType, sizeof...(EnumValues)> values{
147  {EnumValues...}};
148  for (size_t i = 0; i < sizeof...(EnumValues); ++i) {
149  if (std::get<0>(parameter0) == gsl::at(values, i)) {
150  array_location = i;
151  break;
152  }
153  }
154  if (UNLIKELY(array_location == std::numeric_limits<size_t>::max())) {
155  ERROR("Uncached enumeration value: " << std::get<0>(parameter0));
156  }
157  // note that the act of assigning to the specified function pointer type
158  // fixes the template arguments that need to be inferred.
159  static const std::array<
160  const T& (StaticCache<Generator, T, Ranges...>::*)(Args...) const,
161  sizeof...(EnumValues)>
162  cache{{&StaticCache<Generator, T, Ranges...>::unwrap_cache<
163  IntegralConstantValues...,
164  std::integral_constant<EnumType, EnumValues>>...}};
165  // The array `cache` holds pointers to member functions, so we dereference
166  // the pointer and invoke it on `this`.
167  return (this->*gsl::at(cache, array_location))(parameters...);
168  }
169 
170  const Generator generator_;
171 };
