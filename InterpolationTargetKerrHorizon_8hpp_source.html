45 /// A surface that conforms to the horizon of a Kerr black hole in
46 /// Kerr-Schild coordinates.
47 ///
48 /// \details In addition to the parameters for the Kerr black hole, this holder
49 /// contains the `Lmax` which encodes the angular resolution of the spherical
50 /// harmonic basis and `ThetaVariesFastest` which encodes the collocation
51 /// ordering. For example, the apparent horizon finder relies on spherepack
52 /// routines that require `true` for `ThetaVariesFastest`, and using this
53 /// surface for a CCE worldtube requires `false` for `ThetaVariesFastest`.
54 struct KerrHorizon {
55  struct Lmax {
56  using type = size_t;
57  static constexpr Options::String help = {
58  "KerrHorizon is expanded in Ylms up to l=Lmax"};
59  };
60  struct Center {
61  using type = std::array<double, 3>;
62  static constexpr Options::String help = {"Center of black hole"};
63  };
64  struct Mass {
65  using type = double;
66  static constexpr Options::String help = {"Mass of black hole"};
67  };
68  struct DimensionlessSpin {
69  using type = std::array<double, 3>;
70  static constexpr Options::String help = {
71  "Dimensionless spin of black hole"};
72  };
73  struct ThetaVariesFastest {
74  using type = bool;
75  static constexpr Options::String help = {
76  "Chooses theta,phi ordering in 2d array"};
77  };
78  using options =
79  tmpl::list<Lmax, Center, Mass, DimensionlessSpin, ThetaVariesFastest>;
80  static constexpr Options::String help = {
81  "A Strahlkorper conforming to the horizon (in Kerr-Schild coordinates)"
82  " of a Kerr black hole with a specified center, mass, and spin."};
83 
84  KerrHorizon(size_t l_max_in, std::array<double, 3> center_in, double mass_in,
85  std::array<double, 3> dimensionless_spin_in,
86  bool theta_varies_fastest_in = true,
87  const Options::Context& context = {});

92  KerrHorizon(KerrHorizon&& /*rhs*/) noexcept = default;

139 /// \brief Computes points on a Kerr horizon.
140 ///
141 /// The points are such that they conform to the horizon of a Kerr
142 /// black hole (in Kerr-Schild coordinates) with given center, mass,
143 /// and dimensionless spin, as specified in the options.
144 ///
145 /// \note The returned points are not actually on the horizon;
146 /// instead, they are collocation points of a Strahlkorper whose
147 /// spectral representation matches the horizon shape up to order
148 /// Lmax, where Lmax is the spherical-harmonic order specified in the
149 /// options. As Lmax increases, the returned points converge to the
150 /// horizon.
151 ///
152 /// For requirements on InterpolationTargetTag, see InterpolationTarget
153 template <typename InterpolationTargetTag, typename Frame>
154 struct KerrHorizon {
155  using const_global_cache_tags =
156  tmpl::list<Tags::KerrHorizon<InterpolationTargetTag>>;
157  using initialization_tags =
158  tmpl::append<StrahlkorperTags::items_tags<Frame>,
159  StrahlkorperTags::compute_items_tags<Frame>>;
160  using is_sequential = std::false_type;
161 
162  using simple_tags = typename StrahlkorperTags::items_tags<Frame>;
163  using compute_tags = typename StrahlkorperTags::compute_items_tags<Frame>;
164 
165  template <typename DbTags, typename Metavariables>
166  static void initialize(
167  const gsl::not_null<db::DataBox<DbTags>*> box,
168  const Parallel::GlobalCache<Metavariables>& cache) noexcept {
169  const auto& kerr_horizon =
170  Parallel::get<Tags::KerrHorizon<InterpolationTargetTag>>(cache);
171 
172  // Make a Strahlkorper with the correct shape.
173  ::Strahlkorper<Frame> strahlkorper(
174  kerr_horizon.l_max, kerr_horizon.l_max,
175  get(gr::Solutions::kerr_horizon_radius(
176  ::YlmSpherepack(kerr_horizon.l_max, kerr_horizon.l_max)
177  .theta_phi_points(),
178  kerr_horizon.mass, kerr_horizon.dimensionless_spin)),
179  kerr_horizon.center);
180  Initialization::mutate_assign<simple_tags>(box, std::move(strahlkorper));
181  }

183  template <typename Metavariables, typename DbTags, typename TemporalId>
184  static tnsr::I<DataVector, 3, ::Frame::Inertial> points(
185  const db::DataBox<DbTags>& box,
186  const tmpl::type_<Metavariables>& /*meta*/,
187  const TemporalId& /*temporal_id*/) noexcept {
188  // In the future, when we add support for multiple Frames,
189  // the code that transforms coordinates from the Strahlkorper Frame
190  // to Frame::Inertial will go here. That transformation
191  // may depend on `temporal_id`.
192  const auto& kerr_horizon =
193  db::get<Tags::KerrHorizon<InterpolationTargetTag>>(box);
194  if (kerr_horizon.theta_varies_fastest_memory_layout) {
195  return db::get<StrahlkorperTags::CartesianCoords<::Frame::Inertial>>(box);
196  } else {
197  const auto& strahlkorper =
198  db::get<StrahlkorperTags::Strahlkorper<::Frame::Inertial>>(box);
199  const auto& coords =
200  db::get<StrahlkorperTags::CartesianCoords<::Frame::Inertial>>(box);
201  const auto physical_extents =
202  strahlkorper.ylm_spherepack().physical_extents();
203  auto transposed_coords =
204  tnsr::I<DataVector, 3, ::Frame::Inertial>(get<0>(coords).size());
205  for (size_t i = 0; i < 3; ++i) {
206  transpose(make_not_null(&transposed_coords.get(i)), coords.get(i),
207  physical_extents[0], physical_extents[1]);
208  }
209  return transposed_coords;
210  }
211  }
212  template <typename Metavariables, typename DbTags>
213  static tnsr::I<DataVector, 3, ::Frame::Inertial> points(
214  const db::DataBox<DbTags>& box,
215  const tmpl::type_<Metavariables>& /*meta*/) noexcept {
216  return db::get<StrahlkorperTags::CartesianCoords<::Frame::Inertial>>(box);
217  }
