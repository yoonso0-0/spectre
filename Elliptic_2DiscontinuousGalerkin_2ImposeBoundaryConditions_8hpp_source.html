27 /*!
28  * \brief Set the `exterior_vars` so that they represent homogeneous (zero)
29  * Dirichlet boundary conditions.
30  *
31  * To impose homogeneous Dirichlet boundary conditions we mirror the
32  * `interior_vars` and invert their sign. Variables that are not in the
33  * `DirichletTags` list are mirrored without changing their sign, so no boundary
34  * conditions are imposed on them.
35  */
36 template <typename DirichletTags, typename TagsList>
37 void homogeneous_dirichlet_boundary_conditions(
38  const gsl::not_null<Variables<TagsList>*> exterior_vars,
39  const Variables<TagsList>& interior_vars) noexcept {
40  // By default, use the variables on the external boundary for the
41  // exterior
42  *exterior_vars = interior_vars;
43  // For those variables where we have boundary conditions, impose
44  // zero Dirichlet b.c. here. The non-zero boundary conditions are
45  // handled as contributions to the source in InitializeElement.
46  // Imposing them here would not work because we are working with the
47  // linear solver operand.
48  tmpl::for_each<DirichletTags>(
49  [&exterior_vars](auto dirichlet_tag_val) noexcept {
50  using dirichlet_tag = tmpl::type_from<decltype(dirichlet_tag_val)>;
51  // Use mirror principle
52  auto& exterior_dirichlet_field = get<dirichlet_tag>(*exterior_vars);
53  for (size_t i = 0; i < exterior_dirichlet_field.size(); i++) {
54  exterior_dirichlet_field[i] *= -1.;
55  }
56  });
57 }

61 /*!
62  * \brief Set field data on external boundaries so that they represent
63  * homogeneous (zero) Dirichlet boundary conditions.
64  *
65  * This action imposes homogeneous boundary conditions on all `DirichletTags`
66  *
67  * \see `elliptic::dg::homogeneous_dirichlet_boundary_conditions`
68  *
69  * \note We cannot impose inhomogeneous boundary conditions here because it
70  * would break linearity of the DG operator: If in the system of equations
71  * \f$A(x)=b\f$ the DG operator \f$A\f$ had non-zero boundary contributions then
72  * \f$A(x=0)\neq 0\f$, which breaks linearity. Instead, inhomogeneous
73  * boundary conditions are handled as contributions to the source \f$b\f$
74  * during initialization.
75  *
76  * With:
77  * - `interior<Tag> =
78  * Tags::Interface<Tags::BoundaryDirectionsInterior<volume_dim>, Tag>`
79  * - `exterior<Tag> =
80  * Tags::Interface<Tags::BoundaryDirectionsExterior<volume_dim>, Tag>`
81  *
82  * Uses:
83  * - DataBox:
84  * - `interior<VariablesTag>`
85  *
86  * DataBox changes:
87  * - Modifies:
88  * - `exterior<VariablesTag>`
89  */
90 template <typename VariablesTag, typename DirichletTags>
91 struct ImposeHomogeneousDirichletBoundaryConditions {
92  template <typename DbTags, typename... InboxTags, typename Metavariables,
93  size_t Dim, typename ActionList, typename ParallelComponent>
94  static auto apply(db::DataBox<DbTags>& box,
95  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
96  const Parallel::GlobalCache<Metavariables>& /*cache*/,
97  const ElementId<Dim>& /*array_index*/,
98  const ActionList /*meta*/,
99  const ParallelComponent* const /*meta*/) noexcept {
100  // Set the data on exterior (ghost) faces to impose the boundary conditions
101  db::mutate<domain::Tags::Interface<
102  domain::Tags::BoundaryDirectionsExterior<Dim>, VariablesTag>>(
103  make_not_null(&box),
104  [](const auto exterior_boundary_vars,
105  const auto& interior_vars) noexcept {
106  for (auto& exterior_direction_and_vars : *exterior_boundary_vars) {
107  auto& direction = exterior_direction_and_vars.first;
108  homogeneous_dirichlet_boundary_conditions<DirichletTags>(
109  make_not_null(&exterior_direction_and_vars.second),
110  interior_vars.at(direction));
111  }
112  },
113  get<domain::Tags::Interface<
114  domain::Tags::BoundaryDirectionsInterior<Dim>, VariablesTag>>(box));
115 
116  return std::forward_as_tuple(std::move(box));
117  }
118 };
