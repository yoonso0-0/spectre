35 /// \brief ParallelComponent representing a set of points to be interpolated
36 /// to and a function to call upon interpolation to those points.
37 ///
38 /// Each InterpolationTarget will communicate with the `Interpolator`.
39 ///
40 /// `InterpolationTargetTag` must contain the following type aliases:
41 /// - vars_to_interpolate_to_target:
42 /// A `tmpl::list` of tags describing variables to interpolate.
43 /// Will be used to construct a `Variables`.
44 /// - compute_items_on_source:
45 /// A `tmpl::list` of compute items that uses
46 /// `Metavariables::interpolator_source_vars` as input and computes the
47 /// `Variables` defined by `vars_to_interpolate_to_target`.
48 /// - compute_items_on_target:
49 /// A `tmpl::list` of compute items that uses
50 /// `vars_to_interpolate_to_target` as input.
51 /// - compute_target_points:
52 /// A `simple_action` of `InterpolationTarget`
53 /// that computes the target points and sends them to `Interpolators`. It
54 /// takes a `temporal_id` as an extra argument. `compute_target_points`
55 /// can (optionally) have an additional function
56 ///```
57 /// static auto initialize(const gsl::not_null<db::DataBox<DbTags>*>,
58 /// const Parallel::GlobalCache<Metavariables>&)
59 /// noexcept;
60 ///```
61 /// that mutates arbitrary tags in the `DataBox` when the
62 /// `InterpolationTarget` is initialized. Additional simple tags to
63 /// include should be put in the typelist `simple_tags` and additional
64 /// compute tags in `compute_tags`. If `compute_target_points` has
65 /// an `initialize` function, it must also have a type alias
66 /// `initialization_tags` which is a `tmpl::list` of the tags that are
67 /// added by `initialize`.
68 /// - post_interpolation_callback:
69 /// A struct with a type alias `const_global_cache_tags` (listing tags that
70 /// should be read from option parsing), with a type alias
71 /// `observation_types` (listing any ObservationTypes that the callback
72 /// will use in constructing ObserverIds to call
73 /// observers::ThreadedActions::WriteReductionData), and with a function
74 ///```
75 /// void apply(const DataBox<DbTags>&,
76 /// const intrp::GlobalCache<Metavariables>&,
77 /// const Metavariables::temporal_id&) noexcept;
78 ///```
79 /// or
80 ///```
81 /// bool apply(const gsl::not_null<db::DataBox<DbTags>*>,
82 /// const gsl::not_null<intrp::GlobalCache<Metavariables>*>,
83 /// const Metavariables::temporal_id&) noexcept;
84 ///```
85 /// that will be called when interpolation is complete. `DbTags` includes
86 /// everything in `vars_to_interpolate_to_target`, plus everything in
87 /// `compute_items_on_target`. The second form of the `apply` function
88 /// should return false only if it calls another `intrp::Action` that still
89 /// needs the volume data at this temporal_id (such as another iteration of
90 /// the horizon finder).
91 ///
92 /// post_interpolation_callback can optionally have a static constexpr
93 /// double called `fill_invalid_points_with`. Any points outside the
94 /// Domain will be filled with this value. If this variable is not defined,
95 /// then the `apply` function must check for invalid points,
96 /// and should typically exit with an error message if it finds any.
97 /// - should_interpolate (optional):
98 /// A template function that takes as a single argument a const reference
99 /// to the \ref DataBoxGroup, and returns a `bool`: `true` if the
100 /// element should interpolate in the current state, `false` otherwise.
101 /// Often, this will examine the `temporal_id` to determine whether to
102 /// interpolate, but it may examine any part of the \ref DataBoxGroup.
103 /// If omitted, the behavior is as if the `should_interpolate` function
104 /// always returns true. This function is called using the DG element's
105 /// \ref DataBoxGroup and is only used if `InterpolateToTarget` is
106 /// in the element's action list.
107 /// - interpolating_component (used only if not `is_sequential`):
108 /// A type alias for the component that will be interpolating to the
109 /// interpolation target.
110 ///
111 /// `Metavariables` must contain the following type aliases:
112 /// - interpolator_source_vars:
113 /// A `tmpl::list` of tags that define a `Variables` sent from all
114 /// `Element`s to the local `Interpolator`.
115 /// - interpolation_target_tags:
116 /// A `tmpl::list` of all `InterpolationTargetTag`s.
117 /// - temporal_id:
118 /// The type held by ::intrp::Tags::TemporalIds.
119 ///
120 /// `Metavariables` must contain the following static constexpr members:
121 /// - size_t volume_dim:
122 /// The dimension of the Domain.
123 template <class Metavariables, typename InterpolationTargetTag>
124 struct InterpolationTarget {
125  struct RegistrationHelper {
126  template <typename ParallelComponent, typename DbTagsList,
127  typename ArrayIndex>
128  static std::pair<observers::TypeOfObservation, observers::ObservationKey>
129  register_info(const db::DataBox<DbTagsList>& /*box*/,
130  const ArrayIndex& /*array_index*/) noexcept {
131  return {observers::TypeOfObservation::Reduction,
132  observers::ObservationKey(
133  pretty_type::get_name<tmpl::front<
134  typename InterpolationTargetTag::
135  post_interpolation_callback::observation_types>>())};
136  }
137  };
138  using chare_type = ::Parallel::Algorithms::Singleton;
139  using const_global_cache_tags =
140  Parallel::get_const_global_cache_tags_from_actions<tmpl::list<
141  typename InterpolationTargetTag::compute_target_points,
142  typename InterpolationTargetTag::post_interpolation_callback>>;
143  using metavariables = Metavariables;
144  using phase_dependent_action_list = tmpl::list<
145  Parallel::PhaseActions<
146  typename Metavariables::Phase, Metavariables::Phase::Initialization,
147  tmpl::list<::Actions::SetupDataBox,
148  intrp::Actions::InitializeInterpolationTarget<
149  Metavariables, InterpolationTargetTag>,
150  Parallel::Actions::TerminatePhase>>,
151  Parallel::PhaseActions<
152  typename Metavariables::Phase, Metavariables::Phase::Register,
153  tmpl::list<
154  tmpl::conditional_t<
155  InterpolationTargetTag::compute_target_points::is_sequential::
156  value,
157  tmpl::list<>,
158  tmpl::list<
159  Actions::InterpolationTargetSendTimeIndepPointsToElements<
160  InterpolationTargetTag>>>,
161  tmpl::conditional_t<
162  std::is_same_v<
163  typename InterpolationTargetTag::
164  post_interpolation_callback::observation_types,
165  tmpl::list<>>,
166  tmpl::list<Parallel::Actions::TerminatePhase>,
167  tmpl::list<
168  ::observers::Actions::RegisterSingletonWithObserverWriter<
169  RegistrationHelper>,
170  Parallel::Actions::TerminatePhase>>>>>;
171 
172  using initialization_tags = Parallel::get_initialization_tags<
173  Parallel::get_initialization_actions_list<phase_dependent_action_list>>;
174 
175  static void execute_next_phase(
176  typename metavariables::Phase next_phase,
177  Parallel::CProxy_GlobalCache<metavariables>& global_cache) noexcept {
178  auto& local_cache = *(global_cache.ckLocalBranch());
179  Parallel::get_parallel_component<
180  InterpolationTarget<metavariables, InterpolationTargetTag>>(local_cache)
181  .start_phase(next_phase);
182  };
183 };
184 } // namespace intrp
