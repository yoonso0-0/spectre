31 /*!
32  * \ingroup CoordinateMapsGroup
33  * \brief Spatial rotation in two dimensions.
34  *
35  * Let \f$(R,\Phi)\f$ be the polar coordinates associated with
36  * \f$(\xi,\eta)\f$.
37  * Let \f$(r,\phi)\f$ be the polar coordinates associated with \f$(x,y)\f$.
38  * Applies the spatial rotation \f$\phi = \Phi + \alpha\f$.
39  *
40  * The formula for the mapping is:
41  *\f{eqnarray*}
42  x &=& \xi \cos \alpha - \eta \sin \alpha \\
43  y &=& \xi \sin \alpha + \eta \cos \alpha
44  \f}.
45  */

93 /*!
94  * \ingroup CoordinateMapsGroup
95  * \brief Spatial rotation in three dimensions using Euler angles
96  *
97  * Rotation angles should be specified in degrees.
98  * First rotation \f$\alpha\f$ is about z axis.
99  * Second rotation \f$\beta\f$ is about rotated y axis.
100  * Third rotation \f$\gamma\f$ is about rotated z axis.
101  * These rotations are of the \f$(\xi,\eta,\zeta)\f$ coordinate system with
102  * respect
103  * to the grid coordinates \f$(x,y,z)\f$.
104  *
105  * The formula for the mapping is:
106  * \f{eqnarray*}
107  * x &=& \xi (\cos\gamma \cos\beta \cos\alpha - \sin\gamma \sin\alpha)
108  * + \eta (-\sin\gamma \cos\beta \cos\alpha - \cos\gamma \sin\alpha)
109  * + \zeta \sin\beta \cos\alpha \\
110  * y &=& \xi (\cos\gamma \cos\beta \sin\alpha + \sin\gamma \cos\alpha)
111  * + \eta (-\sin\gamma \cos\beta \sin\alpha + \cos\gamma \cos\alpha)
112  * + \zeta \sin\beta \sin\alpha \\
113  * z &=& -\xi \cos\gamma \sin\beta + \eta \sin\gamma \sin\beta
114  * + \zeta \cos\beta
115  * \f}
116  */
117 template <>
