14 namespace TensorExpressions {
15 /*!
16  * \ingroup TensorExpressionsGroup
17  * \brief Evaluate a RHS tensor expression to a tensor with the LHS index order
18  * set in the template parameters
19  *
20  * \details Uses the right hand side (RHS) TensorExpression's index ordering
21  * (`T::args_list`) and the desired left hand side (LHS) tensor's index ordering
22  * (`LhsTensorIndices`) to construct a LHS Tensor with that LHS index ordering.
23  * This can carry out the evaluation of a RHS tensor expression to a LHS tensor
24  * with the same index ordering, such as \f$L_{ab} = R_{ab}\f$, or different
25  * ordering, such as \f$L_{ba} = R_{ab}\f$.
26  *
27  * ### Example usage
28  * Given two rank 2 Tensors `R` and `S` with index order (a, b), add them
29  * together and generate the resultant LHS Tensor `L` with index order (b, a):
30  * \code{.cpp}
31  * auto L = TensorExpressions::evaluate<ti_b, ti_a>(
32  * R(ti_a, ti_b) + S(ti_a, ti_b));
33  * \endcode
34  * \metareturns Tensor
35  *
36  * This represents evaluating: \f$L_{ba} = \R_{ab} + S_{ab}\f$
37  *
38  * Note: `LhsTensorIndices` must be passed by reference because non-type
39  * template parameters cannot be class types until C++20.
40  *
41  * @tparam LhsTensorIndices the TensorIndexs of the Tensor on the LHS of the
42  * tensor expression, e.g. `ti_a`, `ti_b`, `ti_c`
43  * @tparam T the type of the RHS TensorExpression
44  * @param rhs_te the RHS TensorExpression to be evaluated
45  * @return the LHS Tensor with index order specified by LhsTensorIndices
46  */
47 template <auto&... LhsTensorIndices, typename T,
48  Requires<std::is_base_of<Expression, T>::value> = nullptr>
49 auto evaluate(const T& rhs_te) {
50  static_assert(
51  sizeof...(LhsTensorIndices) == tmpl::size<typename T::args_list>::value,
52  "Must have the same number of indices on the LHS and RHS of a tensor "
53  "equation.");
54  using rhs = tmpl::transform<tmpl::remove_duplicates<typename T::args_list>,
55  std::decay<tmpl::_1>>;
56  static_assert(
57  tmpl::equal_members<
58  tmpl::list<std::decay_t<decltype(LhsTensorIndices)>...>, rhs>::value,
59  "All indices on the LHS of a Tensor Expression (that is, those specified "
60  "in evaluate<Indices::...>) must be present on the RHS of the expression "
61  "as well.");
62 
63  using rhs_tensorindex_list = typename T::args_list;
64  using lhs_tensorindex_list =
65  tmpl::list<std::decay_t<decltype(LhsTensorIndices)>...>;
66  using rhs_symmetry = typename T::symmetry;
67  using rhs_tensorindextype_list = typename T::index_list;
68 
69  // Stores (potentially reordered) symmetry and indices needed for constructing
70  // the LHS tensor with index order specified by LhsTensorIndices
71  using lhs_tensor =
72  LhsTensorSymmAndIndices<rhs_tensorindex_list, lhs_tensorindex_list,
73  rhs_symmetry, rhs_tensorindextype_list>;
74 
75  // Construct and return LHS tensor
76  return Tensor<typename T::type, typename lhs_tensor::symmetry,
77  typename lhs_tensor::tensorindextype_list>(
78  rhs_te, lhs_tensorindex_list{});
79 }
80 } // namespace TensorExpressions
