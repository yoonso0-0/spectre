47 /*!
48  * \ingroup DataStructuresGroup
49  * \brief Holds the extents of the mesh and the tensor components on the mesh.
50  *
51  * The extents is a `std::vector<size_t>` where each element is the number of
52  * grid points in the given dimension. The `TensorComponent`s must live on the
53  * grid of the size of the extents. We use runtime extents instead of the
54  * `Index` class because observers may write 1D, 2D, or 3D data in a 3D
55  * simulation.
56  */
57 struct ExtentsAndTensorVolumeData {
58  ExtentsAndTensorVolumeData() = default;
59  ExtentsAndTensorVolumeData(std::vector<size_t> extents_in,
60  std::vector<TensorComponent> components) noexcept
61  : extents(std::move(extents_in)),
62  tensor_components(std::move(components)) {}
63 
64  void pup(PUP::er& p) noexcept; // NOLINT
65  std::vector<size_t> extents{};
66  std::vector<TensorComponent> tensor_components{};
67 };
