33 /// These ComputeItems are different from those used in
34 /// GeneralizedHarmonic evolution because these live only on the
35 /// intrp::Actions::ApparentHorizon DataBox, not in the volume
36 /// DataBox. And these ComputeItems can do fewer allocations than the
37 /// volume ones, because (for example) Lapse, SpaceTimeNormalVector,
38 /// etc. can be inlined instead of being allocated as a separate
39 /// ComputeItem.
40 template <size_t Dim, typename Frame>
41 struct InverseSpatialMetricCompute : gr::Tags::InverseSpatialMetric<Dim, Frame>,
42  db::ComputeTag {
43  using return_type = tnsr::II<DataVector, Dim, Frame>;
44  static void function(
45  const gsl::not_null<tnsr::II<DataVector, Dim, Frame>*> result,
46  const tnsr::aa<DataVector, Dim, Frame>& psi) noexcept {
47  destructive_resize_components(result, psi.begin()->size());
48  *result = determinant_and_inverse(gr::spatial_metric(psi)).second;
49  };
50  using argument_tags = tmpl::list<gr::Tags::SpacetimeMetric<Dim, Frame>>;
51  using base = gr::Tags::InverseSpatialMetric<Dim, Frame>;
52 };
53 template <size_t Dim, typename Frame>
54 struct ExtrinsicCurvatureCompute : gr::Tags::ExtrinsicCurvature<Dim, Frame>,
55  db::ComputeTag {
56  using return_type = tnsr::ii<DataVector, Dim, Frame>;
57  static void function(
58  const gsl::not_null<tnsr::ii<DataVector, Dim, Frame>*> result,
59  const tnsr::aa<DataVector, Dim, Frame>& psi,
60  const tnsr::aa<DataVector, Dim, Frame>& pi,
61  const tnsr::iaa<DataVector, Dim, Frame>& phi,
62  const tnsr::II<DataVector, Dim, Frame>& inv_g) noexcept {
63  const auto shift = gr::shift(psi, inv_g);
64  destructive_resize_components(result, psi.begin()->size());
65  GeneralizedHarmonic::extrinsic_curvature(
66  result, gr::spacetime_normal_vector(gr::lapse(shift, psi), shift), pi,
67  phi);
68  }
69  using argument_tags = tmpl::list<gr::Tags::SpacetimeMetric<Dim, Frame>,
70  GeneralizedHarmonic::Tags::Pi<Dim, Frame>,
71  GeneralizedHarmonic::Tags::Phi<Dim, Frame>,
72  gr::Tags::InverseSpatialMetric<Dim, Frame>>;
73  using base = gr::Tags::ExtrinsicCurvature<Dim, Frame>;
74 };

76 struct SpatialChristoffelSecondKindCompute
77  : ::gr::Tags::SpatialChristoffelSecondKind<Dim, Frame>,

82  const tnsr::iaa<DataVector, Dim, Frame>& phi,
83  const tnsr::II<DataVector, Dim, Frame>& inv_g) noexcept {
