27 /*!
28  * \brief Abstract base class for storage and retrieval of generalized harmonic
29  * quantities communicated from a Cauchy simulation to the Cce system.
30  *
31  * \details The functions that are required to be overriden in the derived
32  * classes are:
33  * - `GhInterfaceManager::get_clone()`: should return a
34  * `std::unique_ptr<GhInterfaceManager>` with cloned state.
35  * - `GhInterfaceManager::insert_gh_data()`: should store the portions
36  * of the provided generalized harmonic data that are required to provide useful
37  * boundary values for the CCE evolution at requested timesteps.
38  * - `GhInterfaceManager::request_gh_data()`: should register requests
39  * from the CCE evolution for boundary data.
40  * - `GhInterfaceManager::retrieve_and_remove_first_ready_gh_data()`:
41  * should return a `boost::optional<std::tuple<TimeStepId,
42  * tnsr::aa<DataVector, 3>, tnsr::iaa<DataVector, 3>, tnsr::aa<DataVector, 3>>>`
43  * containing the boundary data associated with the oldest requested timestep if
44  * enough data has been supplied via `insert_gh_data()` to determine the
45  * boundary data. Otherwise, return a `boost::none` to indicate that the CCE
46  * system must continue waiting for generalized harmonic input.
47  * - `GhInterfaceManager::number_of_pending_requests()`: should return
48  * the number of requests that have been registered to the class that do not yet
49  * been retrieved via `retrieve_and_remove_first_ready_gh_data()`.
50  * - `GhInterfaceManager::number_of_gh_times()`: should return the
51  * number of time steps sent to `insert_gh_data()` that have not yet been
52  * retrieved via `retrieve_and_remove_first_ready_gh_data()`.
53  */
54 class GhInterfaceManager : public PUP::able {
55  public:
56  using gh_variables = Variables<
57  tmpl::list<gr::Tags::SpacetimeMetric<3, ::Frame::Inertial, DataVector>,
58  GeneralizedHarmonic::Tags::Pi<3, ::Frame::Inertial>,
59  GeneralizedHarmonic::Tags::Phi<3, ::Frame::Inertial>>>;
60 
61  using creatable_classes = tmpl::list<GhLocalTimeStepping, GhLockstep>;
62 
63  WRAPPED_PUPable_abstract(GhInterfaceManager); // NOLINT
64 
65  virtual std::unique_ptr<GhInterfaceManager> get_clone() const noexcept = 0;
66 
67  virtual void insert_gh_data(
68  TimeStepId time_id, const tnsr::aa<DataVector, 3>& spacetime_metric,
69  const tnsr::iaa<DataVector, 3>& phi, const tnsr::aa<DataVector, 3>& pi,
70  const tnsr::aa<DataVector, 3>& dt_spacetime_metric,
71  const tnsr::iaa<DataVector, 3>& dt_phi,
72  const tnsr::aa<DataVector, 3>& dt_pi) noexcept = 0;
73 
74  virtual void insert_next_gh_time(TimeStepId time_id,
75  TimeStepId next_time_id) noexcept = 0;
76 
77  virtual void request_gh_data(const TimeStepId&) noexcept = 0;
78 
79  virtual auto retrieve_and_remove_first_ready_gh_data() noexcept
80  -> boost::optional<std::tuple<TimeStepId, gh_variables>> = 0;
81 
82  virtual size_t number_of_pending_requests() const noexcept = 0;
83 
84  virtual size_t number_of_gh_times() const noexcept = 0;
85 
86  virtual InterpolationStrategy get_interpolation_strategy() const noexcept = 0;
87 };
