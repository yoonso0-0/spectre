36 // Global time stepping
37 template <typename Metavariables,
38  Requires<not Metavariables::local_time_stepping> = nullptr>
39 TimeDelta get_initial_time_step(
40  const Time& initial_time, const double initial_dt_value,
41  const Parallel::GlobalCache<Metavariables>& /*cache*/) noexcept {
42  return (initial_dt_value > 0.0 ? 1 : -1) * initial_time.slab().duration();
43 }
44 
45 // Local time stepping
46 template <typename Metavariables,
47  Requires<Metavariables::local_time_stepping> = nullptr>
48 TimeDelta get_initial_time_step(
49  const Time& initial_time, const double initial_dt_value,
50  const Parallel::GlobalCache<Metavariables>& cache) noexcept {
51  const auto& step_controller = Parallel::get<Tags::StepController>(cache);
52  return step_controller.choose_step(initial_time, initial_dt_value);
53 }

58 /// \ingroup InitializationGroup
59 /// \brief Initialize items related to time, such as the time step
60 ///
61 /// Since we have not started the evolution yet, we initialize the state
62 /// _before_ the initial time. So `Tags::TimeStepId` is undefined at this point,
63 /// and `Tags::Next<Tags::TimeStepId>` is the initial time.
64 ///
65 /// DataBox changes:
66 /// - Adds:
67 /// * Tags::TimeStepId
68 /// * `Tags::Next<Tags::TimeStepId>`
69 /// * Tags::TimeStep
70 /// * Tags::Time
71 /// - Removes: nothing
72 /// - Modifies: nothing
73 ///
74 /// \note HistoryEvolvedVariables is allocated, but needs to be initialized
75 ///
76 /// \note This action relies on the `SetupDataBox` aggregated initialization
77 /// mechanism, so `Actions::SetupDataBox` must be present in the
78 /// `Initialization` phase action list prior to this action.
79 template <typename Metavariables>
80 struct TimeAndTimeStep {
81  using initialization_tags =
82  tmpl::list<Tags::InitialTime, Tags::InitialTimeDelta,
83  Tags::InitialSlabSize<Metavariables::local_time_stepping>>;
84 
85  using simple_tags =
86  tmpl::list<::Tags::TimeStepId, ::Tags::Next<::Tags::TimeStepId>,
87  ::Tags::Time, ::Tags::TimeStep>;
88  using compute_tags = tmpl::list<::Tags::SubstepTimeCompute>;
89 
90  template <
91  typename DbTagsList, typename... InboxTags, typename ArrayIndex,
92  typename ActionList, typename ParallelComponent,
93  Requires<tmpl::list_contains_v<
94  typename db::DataBox<DbTagsList>::simple_item_tags,
95  Initialization::Tags::InitialTime> and
96 
97  tmpl::list_contains_v<
98  typename db::DataBox<DbTagsList>::simple_item_tags,
99  Initialization::Tags::InitialTimeDelta> and
100 
101  tmpl::list_contains_v<
102  typename db::DataBox<DbTagsList>::simple_item_tags,
103  Tags::InitialSlabSize<Metavariables::local_time_stepping>>> =
104  nullptr>
105  static auto apply(db::DataBox<DbTagsList>& box,
106  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
107  const Parallel::GlobalCache<Metavariables>& cache,
108  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
109  const ParallelComponent* const /*meta*/) noexcept {
110  const double initial_time_value = db::get<Tags::InitialTime>(box);
111  const double initial_dt_value = db::get<Tags::InitialTimeDelta>(box);
112  const double initial_slab_size =
113  db::get<Tags::InitialSlabSize<Metavariables::local_time_stepping>>(box);
114 
115  const bool time_runs_forward = initial_dt_value > 0.0;
116  const Slab initial_slab =
117  time_runs_forward
118  ? Slab::with_duration_from_start(initial_time_value,
119  initial_slab_size)
120  : Slab::with_duration_to_end(initial_time_value, initial_slab_size);
121  const Time initial_time =
122  time_runs_forward ? initial_slab.start() : initial_slab.end();
123  const TimeDelta initial_dt = Evolution_detail::get_initial_time_step(
124  initial_time, initial_dt_value, cache);
125 
126  // The slab number is increased in the self-start phase each
127  // time one order of accuracy is obtained, and the evolution
128  // proper starts with slab 0.
129  const auto& time_stepper = db::get<::Tags::TimeStepper<>>(box);
130 
131  const TimeStepId time_id(
132  time_runs_forward,
133  -static_cast<int64_t>(time_stepper.number_of_past_steps()),
134  initial_time);
135 
136  Initialization::mutate_assign<
137  tmpl::list<::Tags::TimeStepId, ::Tags::Next<::Tags::TimeStepId>,
138  ::Tags::Time, ::Tags::TimeStep>>(
139  make_not_null(&box), TimeStepId{}, time_id,
140  std::numeric_limits<double>::signaling_NaN(), initial_dt);
141  return std::make_tuple(std::move(box));
142  }

174 /// \ingroup InitializationGroup
175 /// \brief Initialize time-stepper items
176 ///
177 /// DataBox changes:
178 /// - Adds:
179 /// * `db::add_tag_prefix<Tags::dt, variables_tag>`
180 /// * `Tags::HistoryEvolvedVariables<variables_tag, dt_variables_tag>`
181 /// * Tags::ComputeDeriv (for non-conservative systems)
182 /// * Tags::ComputeDiv (for conservative systems)
183 /// - Removes: nothing
184 /// - Modifies: nothing
185 ///
186 /// \note HistoryEvolvedVariables is allocated, but needs to be initialized
187 ///
188 /// \note This action relies on the `SetupDataBox` aggregated initialization
189 /// mechanism, so `Actions::SetupDataBox` must be present in the
190 /// `Initialization` phase action list prior to this action.
191 template <typename Metavariables>
192 struct TimeStepperHistory {
193  using initialization_tags = tmpl::list<>;
194 
195  static constexpr size_t dim = Metavariables::volume_dim;
196  using variables_tag = typename Metavariables::system::variables_tag;
197  using dt_variables_tag = db::add_tag_prefix<::Tags::dt, variables_tag>;
198 
199  template <typename System, bool IsInFluxConservativeForm =
200  System::is_in_flux_conservative_form>
201  struct ComputeTags {
202  using type = db::AddComputeTags<::Tags::DerivCompute<
203  variables_tag,
204  domain::Tags::InverseJacobian<dim, Frame::Logical, Frame::Inertial>,
205  typename System::gradients_tags>>;
206  };
207 
208  template <typename System>
209  struct ComputeTags<System, true> {
210  using type = db::AddComputeTags<>;
211  };
212 
213  using simple_tags =
214  tmpl::list<dt_variables_tag,
215  ::Tags::HistoryEvolvedVariables<variables_tag>>;
216 
217  using compute_tags =
218  typename ComputeTags<typename Metavariables::system>::type;
219 
220  template <typename DbTagsList, typename... InboxTags, typename ArrayIndex,
221  typename ActionList, typename ParallelComponent,
222  Requires<tmpl::list_contains_v<
223  typename db::DataBox<DbTagsList>::simple_item_tags,
224  domain::Tags::Mesh<dim>> and
225 
226  tmpl::list_contains_v<
227  typename db::DataBox<DbTagsList>::simple_item_tags,
228  variables_tag>> = nullptr>
229  static auto apply(db::DataBox<DbTagsList>& box,
230  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
231  const Parallel::GlobalCache<Metavariables>& /*cache*/,
232  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
233  const ParallelComponent* const /*meta*/) noexcept {
234  using DtVars = typename dt_variables_tag::type;
235 
236  const size_t num_grid_points =
237  db::get<domain::Tags::Mesh<dim>>(box).number_of_grid_points();
238 
239  // Will be overwritten before use
240  DtVars dt_vars{num_grid_points};
241  typename ::Tags::HistoryEvolvedVariables<variables_tag>::type history;
242 
243  Initialization::mutate_assign<simple_tags>(
244  make_not_null(&box), std::move(dt_vars), std::move(history));
245 
246  return std::make_tuple(std::move(box));
247  }
