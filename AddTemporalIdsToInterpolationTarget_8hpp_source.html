21 /// \ingroup ActionsGroup
22 /// \brief Adds `temporal_id`s on which this InterpolationTarget
23 /// should be triggered.
24 ///
25 /// Uses:
26 /// - DataBox:
27 /// - `Tags::TemporalIds<TemporalId>`
28 /// - `Tags::CompletedTemporalIds<TemporalId>`
29 ///
30 /// DataBox changes:
31 /// - Adds: nothing
32 /// - Removes: nothing
33 /// - Modifies:
34 /// - `Tags::TemporalIds<TemporalId>`
35 template <typename InterpolationTargetTag>
36 struct AddTemporalIdsToInterpolationTarget {
37  template <typename ParallelComponent, typename DbTags, typename Metavariables,
38  typename ArrayIndex, typename TemporalId,
39  Requires<tmpl::list_contains_v<
40  DbTags, Tags::TemporalIds<TemporalId>>> = nullptr>
41  static void apply(db::DataBox<DbTags>& box,
42  Parallel::GlobalCache<Metavariables>& cache,
43  const ArrayIndex& /*array_index*/,
44  std::vector<TemporalId>&& temporal_ids) noexcept {
45  const bool temporal_id_added_for_the_first_time =
46  db::get<Tags::TemporalIds<TemporalId>>(box).empty();
47 
48  const std::vector<TemporalId> new_temporal_ids =
49  InterpolationTarget_detail::flag_temporal_ids_for_interpolation<
50  InterpolationTargetTag>(make_not_null(&box), temporal_ids);
51 
52  const auto& ids = db::get<Tags::TemporalIds<TemporalId>>(box);
53  if (InterpolationTargetTag::compute_target_points::is_sequential::value) {
54  // InterpolationTarget is sequential.
55  // Begin a single interpolation if one is not already in progress
56  // (i.e. waiting for data), and if there are temporal_ids to
57  // interpolate. If there's an interpolation in progress, then a
58  // later interpolation will be started as soon as the earlier one
59  // finishes (in InterpolationTargetReceiveVars).
60  if (temporal_id_added_for_the_first_time and not ids.empty()) {
61  auto& my_proxy =
62  Parallel::get_parallel_component<ParallelComponent>(cache);
63  Parallel::simple_action<
64  Actions::SendPointsToInterpolator<InterpolationTargetTag>>(
65  my_proxy, ids.front());
66  }
67  } else {
68  // InterpolationTarget is not sequential. So begin interpolation
69  // on every new temporal_id that has just been added.
70  auto& my_proxy =
71  Parallel::get_parallel_component<ParallelComponent>(cache);
72  for (const auto& id : new_temporal_ids) {
73  Parallel::simple_action<
74  Actions::SendPointsToInterpolator<InterpolationTargetTag>>(my_proxy,
75  id);
76  }
77  }
78  }
79 };
80 } // namespace Actions
81 } // namespace intrp
