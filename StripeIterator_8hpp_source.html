23  template <size_t Dim>
24  StripeIterator(const Index<Dim>& extents, size_t stripe_dim) noexcept;

26  /// Returns `true` if the iterator is valid
27  explicit operator bool() const noexcept { return offset_ < size_; }
28 
29  /// Increment to the next stripe.
30  StripeIterator& operator++() noexcept {
31  ++offset_;
32  ++stride_count_;
33  if (UNLIKELY(stride_count_ == stride_)) {
34  offset_ += jump_;
35  stride_count_ = 0;
36  }
37  return *this;
38  }
39 
40  /// Offset into DataVector for first element of stripe.
41  size_t offset() const noexcept { return offset_; }
42 
43  /// Stride of elements in DataVector for the stripe.
44  size_t stride() const noexcept { return stride_; }
45 
46  private:
47  size_t offset_ = std::numeric_limits<size_t>::max();
48  size_t size_ = std::numeric_limits<size_t>::max();
49  size_t stride_ = std::numeric_limits<size_t>::max();
50  size_t stride_count_ = std::numeric_limits<size_t>::max();
51  size_t jump_ = std::numeric_limits<size_t>::max();
52 };
