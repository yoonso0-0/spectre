21 /*!
22  * \brief Simple implementation of a `GhInterfaceManager` that only
23  * provides boundary data on matching `TimeStepId`s
24  *
25  * \details This version of the interface manager assumes that the CCE system
26  * and the generalized harmonic system that it communicates with evolve with an
27  * identical time stepper and on identical time step intervals (they evolve in
28  * 'lock step'). As a result, and to streamline communications, new data is
29  * always immediately 'ready' and requests are ignored to produce the behavior
30  * of just immediately gauge-transforming and sending the data to the CCE
31  * component as soon as it becomes available from the GH system.
32  *
33  * \warning Using this interface manager when the GH components and the CCE
34  * evolution are not identically stepped is considered undefined behavior. The
35  * outcome will likely be that CCE will fail to evolve and the boundary data
36  * will be continually inserted into a rapidly expanding inbox.
37  */
38 class GhLockstep : public GhInterfaceManager {
39  public:
40  using GhInterfaceManager::gh_variables;
41 
42  static constexpr Options::String help{
43  "Pass data between GH and CCE systems on matching timesteps only."};
44 
45  using options = tmpl::list<>;
46 
47  GhLockstep() = default;
48 
49  explicit GhLockstep(CkMigrateMessage* /*unused*/) noexcept {}
50 
51  WRAPPED_PUPable_decl_template(GhLockstep); // NOLINT
52 
53  std::unique_ptr<GhInterfaceManager> get_clone() const noexcept override;
54 
55  /// \brief Store a provided data set in a `std::deque`.
56  ///
57  /// \details The lock-step constraint ensures that only the generalized
58  /// harmonic variables `spacetime_metric`, `phi`, and `pi` are used. The
59  /// remaining variables are accepted to comply with the more general abstract
60  /// interface.
61  void insert_gh_data(
62  TimeStepId time_id, const tnsr::aa<DataVector, 3>& spacetime_metric,
63  const tnsr::iaa<DataVector, 3>& phi, const tnsr::aa<DataVector, 3>& pi,
64  const tnsr::aa<DataVector, 3>& dt_spacetime_metric = {},
65  const tnsr::iaa<DataVector, 3>& dt_phi = {},
66  const tnsr::aa<DataVector, 3>& dt_pi = {}) noexcept override;
67 
68  /// \brief next time information is ignored by this implementation, so this is
69  /// a no-op.
70  void insert_next_gh_time(TimeStepId /*time_id*/,
71  TimeStepId /*next_time_id*/) noexcept override {}
72 
73  /// \brief Requests are ignored by this implementation, so this is a no-op.
74  void request_gh_data(const TimeStepId& /*time_id*/) noexcept override {}
75 
76  /// \brief Return a `boost::optional<std::tuple>` of the least recently
77  /// submitted generalized harmonic boundary data if any exists and removes it
78  /// from the internal `std::deque`, otherwise returns `boost::none`.
79  auto retrieve_and_remove_first_ready_gh_data() noexcept
80  -> boost::optional<std::tuple<TimeStepId, gh_variables>> override;
81 
82  /// \brief This class ignores requests to ensure a one-way communication
83  /// pattern, so the number of requests is always 0.
84  size_t number_of_pending_requests() const noexcept override { return 0; }
85 
86  /// \brief The number of times at which data from a GH evolution have been
87  /// stored and not yet retrieved
88  size_t number_of_gh_times() const noexcept override {
89  return provided_data_.size();
90  }
91 
92  /// Serialization for Charm++.
93  void pup(PUP::er& p) noexcept override;
94 
95  InterpolationStrategy get_interpolation_strategy() const noexcept override {
96  return InterpolationStrategy::EverySubstep;
97  };
98 
99  private:
100  std::deque<std::tuple<TimeStepId, gh_variables>> provided_data_;
101 };
