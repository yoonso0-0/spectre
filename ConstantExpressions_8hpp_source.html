66 /*!
67  * \ingroup ConstantExpressionsGroup
68  * \brief Compute the falling factorial of \f$(x)_{n}\f$
69  *
70  * See http://mathworld.wolfram.com/FallingFactorial.html
71  * \note The largest representable factorial is 20!. It is up to the user to
72  * ensure this is satisfied
73  */
74 constexpr uint64_t falling_factorial(const uint64_t x,
75  const uint64_t n) noexcept {
76  // clang-tidy: don't warn about STL internals, I can't fix them
77  assert(n <= x); // NOLINT
78  uint64_t r = 1;
79  for (uint64_t k = 0; k < n; ++k) {
80  r *= (x - k);
81  }
82  return r;
83 }

97 // Implementation functions for the pow template function below which computes
98 // optimized powers where the exponent is a compile-time integer
99 
100 // base case power 0: Returns 1.0. This will need to be overloaded for types for
101 // which the simple return is inappropriate.
102 template <typename T>
103 SPECTRE_ALWAYS_INLINE constexpr decltype(auto) pow_impl(
104  const T& /*t*/, std::integral_constant<int, 0> /*meta*/,
105  std::bool_constant<true> /*exponent_was_positive*/) noexcept {
106  return static_cast<tt::get_fundamental_type_t<T>>(1.0);
107 }
108 
109 // special case power 1: acts as a direct identity function for efficiency
110 template <typename T>
111 SPECTRE_ALWAYS_INLINE constexpr decltype(auto) pow_impl(
112  const T& t, std::integral_constant<int, 1> /*meta*/,
113  std::bool_constant<true> /*exponent_was_positive*/) noexcept {
114  return t;
115 }
116 
117 // general case for positive powers: return the power via recursive inline call,
118 // which expands to a series of multiplication operations.
119 template <int N, typename T>
120 SPECTRE_ALWAYS_INLINE constexpr decltype(auto) pow_impl(
121  const T& t, std::integral_constant<int, N> /*meta*/,
122  std::bool_constant<true> /*exponent_was_positive*/) noexcept {
123  return t * pow_impl(t, std::integral_constant<int, N - 1>{},
124  std::bool_constant<true>{});
125 }
126 
127 // general case for negative powers: return the multiplicative inverse of the
128 // result from the recursive inline call, which expands to a series of
129 // multiplication operations. This assumes that division is supported with
130 // tt::get_fundamental_type_t<T> and T. If not, this utility will need further
131 // specialization.
132 template <int N, typename T>
133 SPECTRE_ALWAYS_INLINE constexpr decltype(auto) pow_impl(
134  const T& t, std::integral_constant<int, N> /*meta*/,
135  std::bool_constant<false> /*exponent_was_positive*/) noexcept {
136  return static_cast<tt::get_fundamental_type_t<T>>(1) /
137  (pow_impl(t, std::integral_constant<int, -N>{},
138  std::bool_constant<true>{}));
139 }

142 /// \ingroup ConstantExpressionsGroup
143 /// \brief Compute t^N where N is an integer (positive or negative)
144 ///
145 /// \warning If passing an integer this will do integer arithmetic, e.g.
146 /// `pow<-10>(2) == 0` evaluates to `true`
147 ///
148 /// \warning For optimization, it is assumed that the `pow<0>` of a vector type
149 /// (e.g. `DataVector`) will not be used for initialization or directly indexed,
150 /// so `pow<0>` returns simply `1.0`. In the case of use for initialization, a
151 /// constructor should be used instead, and in the case of a direct index, the
152 /// expression may be simplifyable to avoid the use of `pow<0>` altogether. If a
153 /// more complete treatment of `pow<0>` is required, further overloads may be
154 /// added to the `ConstantExpressions_detail` namespace.
155 ///
156 /// \tparam N the integer power being raised to in \f$t^N\f$
157 /// \param t the value being exponentiated
158 /// \return value \f$t^N\f$ determined via repeated multiplication
159 template <int N, typename T>
160 SPECTRE_ALWAYS_INLINE constexpr decltype(auto) pow(const T& t) noexcept {
161  return ConstantExpressions_detail::pow_impl(
162  t, std::integral_constant<int, N>{}, std::bool_constant<(N >= 0)>{});
163 }

243 /// \ingroup ConstantExpressionsGroup
244 /// \brief Returns `f(ic<0>{}) + f(ic<1>{}) + ... + f(ic<NumTerms-1>{})`
245 /// where `ic<N>` stands for `std::integral_constant<size_t, N>`.
246 /// This function allows the result types of each operation to be
247 /// different, and so works efficiently with expression templates.
248 /// \note When summing expression templates one must be careful of
249 /// referring to temporaries in `f`.
250 template <size_t NumTerms, typename Function, Requires<NumTerms == 1> = nullptr>
251 constexpr decltype(auto) constexpr_sum(Function&& f) noexcept {
252  return f(std::integral_constant<size_t, 0>{});
253 }
