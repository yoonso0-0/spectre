82  template <typename DbTags, typename... InboxTags, typename ArrayIndex,

105  template <typename DbTags, typename... InboxTags, typename ArrayIndex>
106  static bool is_ready(
107  const db::DataBox<DbTags>& box,
108  const tuples::TaggedTuple<InboxTags...>& inboxes,
109  const Parallel::GlobalCache<Metavariables>& /*cache*/,
110  const ArrayIndex& /*array_index*/) noexcept {
111  constexpr size_t volume_dim = Metavariables::system::volume_dim;
112  const auto& element = db::get<domain::Tags::Element<volume_dim>>(box);
113  const auto num_expected = element.neighbors().size();
114  // Edge case where we do not receive any data
115  if (UNLIKELY(num_expected == 0)) {
116  return true;
117  }
118  const auto& local_temporal_id =
119  db::get<typename Metavariables::temporal_id>(box);
120  const auto& inbox = tuples::get<limiter_comm_tag>(inboxes);
121  const auto& received = inbox.find(local_temporal_id);
122  // Check we have at least some data from correct time
123  if (received == inbox.end()) {
124  return false;
125  }
126  // Check data was received from each neighbor
127  const size_t num_neighbors_received = received->second.size();
128  return (num_neighbors_received == num_expected);
129  }

132 /// \ingroup ActionsGroup
133 /// \ingroup DiscontinuousGalerkinGroup
134 /// \ingroup LimitersGroup
135 /// \brief Send local data needed for limiting.
136 ///
137 /// Currently, is not tested for support of:
138 /// - h-refinement
139 /// Currently, does not support:
140 /// - Local time-stepping
141 ///
142 /// Uses:
143 /// - GlobalCache:
144 /// - Metavariables::limiter
145 /// - DataBox:
146 /// - Tags::Element<volume_dim>
147 /// - Metavariables::limiter::type::package_argument_tags
148 /// - Metavariables::temporal_id
149 ///
150 /// DataBox changes:
151 /// - Adds: nothing
152 /// - Removes: nothing
153 /// - Modifies: nothing
154 ///
155 /// \see ApplyLimiter
156 template <typename Metavariables>
157 struct SendData {
158  using const_global_cache_tags = tmpl::list<typename Metavariables::limiter>;
159  using limiter_comm_tag =
160  Limiters::Tags::LimiterCommunicationTag<Metavariables>;
161 
162  static_assert(
163  not Metavariables::local_time_stepping,
164  "Limiter communication actions do not yet support local time stepping");
165 
166  template <typename DbTags, typename... InboxTags, typename ArrayIndex,
167  typename ActionList, typename ParallelComponent>
168  static std::tuple<db::DataBox<DbTags>&&> apply(
169  db::DataBox<DbTags>& box, tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
170  Parallel::GlobalCache<Metavariables>& cache,
171  const ArrayIndex& /*array_index*/, const ActionList /*meta*/,
172  const ParallelComponent* const /*meta*/) noexcept {
173  constexpr size_t volume_dim = Metavariables::system::volume_dim;
174 
175  auto& receiver_proxy =
176  Parallel::get_parallel_component<ParallelComponent>(cache);
177  const auto& element = db::get<domain::Tags::Element<volume_dim>>(box);
178  const auto& temporal_id = db::get<typename Metavariables::temporal_id>(box);
179  const auto& limiter = get<typename Metavariables::limiter>(cache);
180 
181  for (const auto& direction_neighbors : element.neighbors()) {
182  const auto& direction = direction_neighbors.first;
183  const size_t dimension = direction.dimension();
184  const auto& neighbors_in_direction = direction_neighbors.second;
185  ASSERT(neighbors_in_direction.size() == 1,
186  "h-adaptivity is not supported yet.\nDirection: "
187  << direction << "\nDimension: " << dimension
188  << "\nNeighbors:\n"
189  << neighbors_in_direction);
190  const auto& orientation = neighbors_in_direction.orientation();
191  const auto direction_from_neighbor = orientation(direction.opposite());
192 
193  using argument_tags =
194  typename Metavariables::limiter::type::package_argument_tags;
195  const auto packaged_data = db::apply<argument_tags>(
196  [&limiter](const auto&... args) noexcept {
197  // Note: orientation is received as last element of pack `args`
198  typename Metavariables::limiter::type::PackagedData pack{};
199  limiter.package_data(make_not_null(&pack), args...);
200  return pack;
201  },
202  box, orientation);
203 
204  for (const auto& neighbor : neighbors_in_direction) {
205  Parallel::receive_data<limiter_comm_tag>(
206  receiver_proxy[neighbor], temporal_id,
207  std::make_pair(
208  std::make_pair(direction_from_neighbor, element.id()),
209  packaged_data));
210 
211  } // loop over neighbors_in_direction
212  } // loop over element.neighbors()
213 
214  return std::forward_as_tuple(std::move(box));
215  }
216 };
