40 /// \ingroup VariableFixingGroup
41 /// \brief Applies a pressure and density floor dependent on the distance
42 /// to the origin.
43 ///
44 /// Applies the floors:
45 /// \f$\rho(r) \geq \rho_{\mathrm{fl}}(r) = C_\rho r^{k_\rho}\f$
46 /// and \f$P(r) \geq P_{\mathrm{fl}}(r) = C_p r^{k_p}\f$
47 /// when \f$ r > r_{min}\f$, where \f$C_\rho\f$ is given by the option
48 /// `ScaleDensityFloor`, \f$k_\rho\f$ is given by the option
49 /// `PowerDensityFloor`, \f$C_p\f$ is given by the option
50 /// `ScalePressureFloor`, \f$k_p\f$ is given by the option
51 /// `PowerPressureFloor`, and \f$r_{min}\f$ is given by the option
52 /// `MinimumRadius`.
53 ///
54 /// \note In \cite Porth2016rfi, the following floors are applied:
55 /// \f$\rho(r) \geq \rho_{\mathrm{fl}}(r) = 10^{-5}r^{-3/2}\f$
56 /// and \f$P(r) \geq P_{\mathrm{fl}}(r) = \frac{1}{3} \times 10^{-7}r^{-5/2}\f$
57 template <size_t Dim>
58 class RadiallyFallingFloor {
59  public:
60  /// \brief The minimum radius at which to begin applying the floors on the
61  /// density and pressure.
62  struct MinimumRadius {
63  static constexpr Options::String help =
64  "The radius at which to begin applying the lower bound.";
65  using type = double;
66  static double lower_bound() noexcept { return 0.0; }
67  };
