46 // Implements the TVB troubled-cell indicator from Cockburn1999 for several
47 // tensors. Returns true if any component of any tensor needs limiting.
48 //
49 // Expects type `PackagedData` to contain, as in Limiters::Minmod:
50 // - a variable `means` that is a `TaggedTuple<Tags::Mean<Tags>...>`
51 // - a variable `element_size` that is a `std::array<double, VolumeDim>`
52 template <size_t VolumeDim, typename PackagedData, typename... Tags>
53 bool tvb_minmod_indicator(
54  const double tvb_constant, const typename Tags::type&... tensors,
55  const Mesh<VolumeDim>& mesh, const Element<VolumeDim>& element,
56  const std::array<double, VolumeDim>& element_size,
57  const std::unordered_map<
58  std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>, PackagedData,
59  boost::hash<std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>>>&
60  neighbor_data) noexcept {
61  Minmod_detail::BufferWrapper<VolumeDim> buffer(mesh);
62  const auto effective_neighbor_sizes =
63  Minmod_detail::compute_effective_neighbor_sizes(element, neighbor_data);
64 
65  // Ideally, as soon as one component is found that needs limiting, then we
66  // would exit the TCI early with return value `true`. But there is no natural
67  // way to return early from the pack expansion. So we use this bool to keep
68  // track of whether a previous component needed limiting... and if so, then
69  // we simply skip any work.
70  bool some_component_needs_limiting = false;
71  const auto wrap_tci_one_tensor = [&](auto tag, const auto tensor) noexcept {
72  if (some_component_needs_limiting) {
73  // Skip this tensor completely
74  return '0';
75  }
76 
77  for (size_t tensor_storage_index = 0; tensor_storage_index < tensor.size();
78  ++tensor_storage_index) {
79  const auto effective_neighbor_means =
80  Minmod_detail::compute_effective_neighbor_means<decltype(tag)>(
81  tensor_storage_index, element, neighbor_data);
82 
83  const DataVector& u = tensor[tensor_storage_index];
84  const bool component_needs_limiting = tvb_minmod_indicator(
85  make_not_null(&buffer), tvb_constant, u, mesh, element, element_size,
86  effective_neighbor_means, effective_neighbor_sizes);
87 
88  if (component_needs_limiting) {
89  some_component_needs_limiting = true;
90  // Skip remaining components of this tensor
91  return '0';
92  }
93  }
94  return '0';
95  };
96  expand_pack(wrap_tci_one_tensor(Tags{}, tensors)...);
97  return some_component_needs_limiting;
98 }
