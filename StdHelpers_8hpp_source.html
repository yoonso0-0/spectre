233 /*!
234  * \ingroup UtilitiesGroup
235  * \brief Format a string like printf
236  *
237  * Given a formatting string and arguments this returns the corresponding
238  * string. Similar to printf but using std::strings.
239  */
240 template <typename... Args>
241 std::string formatted_string(const std::string& fmt, Args... args) {
242 #pragma GCC diagnostic push
243 #pragma GCC diagnostic ignored "-Wformat-nonliteral"
244  // clang-tidy: do not use snprintf
245  auto requiredBytes = static_cast<size_t>(std::snprintf( // NOLINT
246  nullptr, 0, fmt.c_str(), args...)) +
247  1;
248  std::string rtn;
249  rtn.resize(requiredBytes);
250  // clang-tidy: do not use snprintf
251  std::snprintf(&rtn[0], requiredBytes, fmt.c_str(), args...); // NOLINT
252 #pragma GCC diagnostic pop
253  if (rtn[rtn.size() - 1] == '\0') {
254  rtn.resize(rtn.size() - 1);
255  }
256  return rtn;
257 }
