18 namespace Initialization {
19 /// The merge policy for resolving conflicts in `merge_into_databox` when adding
20 /// simple tags that already exist.
21 enum class MergePolicy {
22  /// Overwrite the existing simple tags
23  Overwrite,
24  /// Cause an error if a simple tag is already in the DataBox.
25  Error,
26  /// Cause an error if a simple tag is in the DataBox but does not compare
27  /// equal to the new value being added. Ignores types that do not have an
28  /// equivalence operator.
29  IgnoreIncomparable
30 };

33 template <typename AddingAction, typename SimpleTag, MergePolicy Policy,
34  typename DbTagsList,
35  Requires<Policy == MergePolicy::Overwrite> = nullptr>
36 void merge_simple_tag_value(
37  const gsl::not_null<db::DataBox<DbTagsList>*> box,
38  typename SimpleTag::type&& simple_tag_value) noexcept {
39  db::mutate<SimpleTag>(
40  box,
41  [](const gsl::not_null<typename SimpleTag::type*> stored_simple_tag_value,
42  typename SimpleTag::type&& local_simple_tag_value) noexcept {
43  *stored_simple_tag_value = std::move(local_simple_tag_value);
44  },
45  std::move(simple_tag_value));
46 }

48 template <typename AddingAction, typename SimpleTag, MergePolicy Policy,
49  typename DbTagsList,
50  Requires<(Policy == MergePolicy::Error or
51  Policy == MergePolicy::IgnoreIncomparable) and
52  tt::has_inequivalence_v<typename SimpleTag::type>> = nullptr>
53 void merge_simple_tag_value(const gsl::not_null<db::DataBox<DbTagsList>*> box,
54  typename SimpleTag::type&& simple_tag) noexcept {
55  if (db::get<SimpleTag>(*box) != simple_tag) {
56  ERROR("While adding the simple tag "
57  << db::tag_name<SimpleTag>()
58  << " that is already in the DataBox we found that the value being "
59  "set by the action "
60  << pretty_type::get_name<AddingAction>()
61  << " is not the same as what is already in the DataBox. The "
62  "value in the DataBox is: "
63  << db::get<SimpleTag>(*box) << " while the value being added is "
64  << simple_tag);
65  }
66 }

85 template <typename AddingAction, typename ComputeTagsList, MergePolicy Policy,
86  typename... SimpleTags, typename DbTagsList,
87  typename... SimpleTagsToAdd, typename... SimpleTagsToCheck>
88 auto merge_into_databox_impl(
89  db::DataBox<DbTagsList>&& box,
90  tuples::TaggedTuple<SimpleTags...>&& simple_tags,
91  tmpl::list<SimpleTagsToAdd...> /*meta*/,
92  tmpl::list<SimpleTagsToCheck...> /*meta*/) noexcept {
93  EXPAND_PACK_LEFT_TO_RIGHT(
94  merge_simple_tag_value<AddingAction, SimpleTagsToCheck, Policy>(
95  make_not_null(&box),
96  std::move(tuples::get<SimpleTagsToCheck>(simple_tags))));
97 
98  // Only add compute tags that are not in the DataBox.
99  using compute_tags_to_add = tmpl::remove_if<
100  ComputeTagsList,
101  tmpl::bind<tmpl::list_contains,
102  tmpl::pin<typename db::DataBox<DbTagsList>::compute_item_tags>,
103  tmpl::_1>>;
104 
105  return db::create_from<db::RemoveTags<>,
106  db::AddSimpleTags<SimpleTagsToAdd...>,
107  compute_tags_to_add>(
108  std::move(box), std::move(tuples::get<SimpleTagsToAdd>(simple_tags))...);
109 }

112 /*!
113  * \ingroup InitializationGroup
114  * \brief Add tags that are not yet in the DataBox.
115  *
116  * How duplicate tags are handled depends on the `MergePolicy` passed. The
117  * default merge policy is to error if the simple tag being added does not have
118  * an equivalence operator and is being added again.
119  * `MergePolicy::IgnoreIncomparable` means that nothing is done if the simple
120  * tag being added does not have an equivalence operator. Tags that have an
121  * equivalence operator are check that the value being added and that is already
122  * in the DataBox are the same. If they aren't, an error occurs. Finally,
123  * `MergePolicy::Overwrite` means the current simple tag in the DataBox is
124  * overwritten with the new value passed in.
125  *
126  * Compute tags that are not in the DataBox are added, ones that are in the
127  * DataBox already are ignored.
128  */
129 template <typename AddingAction, typename SimpleTagsList,
130  typename ComputeTagsList = tmpl::list<>,
131  MergePolicy Policy = MergePolicy::Error, typename... SimpleTags,
132  typename DbTagsList, typename... Args>
133 auto merge_into_databox(db::DataBox<DbTagsList>&& box,
134  Args&&... args) noexcept {
135  using simple_tags_to_check = tmpl::filter<
136  SimpleTagsList,
137  tmpl::bind<tmpl::list_contains,
138  tmpl::pin<typename db::DataBox<DbTagsList>::simple_item_tags>,
139  tmpl::_1>>;
140  using simple_tags_to_add =
141  tmpl::remove_if<SimpleTagsList,
142  tmpl::bind<tmpl::list_contains,
143  tmpl::pin<simple_tags_to_check>, tmpl::_1>>;
144  return detail::merge_into_databox_impl<AddingAction, ComputeTagsList, Policy>(
145  std::move(box),
146  tuples::tagged_tuple_from_typelist<SimpleTagsList>{
147  std::forward<Args>(args)...},
148  simple_tags_to_add{}, simple_tags_to_check{});
149 }
150 } // namespace Initialization
