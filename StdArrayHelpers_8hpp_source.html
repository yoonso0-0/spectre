95 /// \ingroup UtilitiesGroup
96 /// \brief Construct an array from an existing array omitting one element
97 template <typename T, size_t Dim>
98 inline std::array<T, Dim - 1> all_but_specified_element_of(
99  const std::array<T, Dim>& a, const size_t element_to_remove) noexcept {
100  ASSERT(element_to_remove < Dim, "Specified element does not exist");
101  std::array<T, Dim - 1> result{};
102  for (size_t i = 0; i < element_to_remove; ++i) {
103  gsl::at(result, i) = gsl::at(a, i);
104  }
105  for (size_t i = element_to_remove + 1; i < Dim; ++i) {
106  gsl::at(result, i - 1) = gsl::at(a, i);
107  }
108  return result;
109 }

111 /// \ingroup UtilitiesGroup
112 /// \brief Construct an array from an existing array adding one element
113 template <typename T, size_t Dim>
114 inline std::array<T, Dim + 1> insert_element(std::array<T, Dim> a,
115  const size_t element_to_add,
116  T value) noexcept {
117  ASSERT(element_to_add <= Dim, "Specified element is out of range");
118  std::array<T, Dim + 1> result{};
119  for (size_t i = 0; i < element_to_add; ++i) {
120  gsl::at(result, i) = std::move(gsl::at(a, i));
121  }
122  gsl::at(result, element_to_add) = std::move(value);
123  for (size_t i = element_to_add; i < Dim; ++i) {
124  gsl::at(result, i + 1) = std::move(gsl::at(a, i));
125  }
126  return result;
127 }

143 /// \ingroup UtilitiesGroup
144 /// \brief Euclidean magnitude of the elements of the array.
145 ///
146 /// \details If T is a container the magnitude is computed separately for each
147 /// element of the container.
148 ///
149 /// \requires If T is a container, T must have following mathematical operators:
150 /// abs(), sqrt(), and element-wise addition and multiplication. In addition,
151 /// each T in the array must have the same size.
152 template <typename T>
153 decltype(auto) magnitude(const std::array<T, 1>& a) noexcept {
154  using std::abs;
155  return abs(a[0]);
156 }

188 /*!
189  * \ingroup UtilitiesGroup
190  * \brief Declares a binary function on an array, intended for binary
191  * operators such as `+`
192  *
193  * \param RESULT_TYPE the `value_type` that is the result of the operation
194  * (e.g. `A` for resulting `std::array<A,2>`)
195  *
196  * \param LTYPE the `value_type` of the first argument of the function (so the
197  * left value if the function is an operator overload)
198  *
199  * \param RTYPE the `value_type` of the second argument of the function
200  *
201  * \param OP_FUNCTION_NAME the function which should be declared
202  * (e.g. `operator+`)
203  *
204  * \param BINARY_OP the binary function which should be applied elementwise to
205  * the pair of arrays. (e.g. `std::plus<>()`)
206  */
207 // clang-tidy: complaints about uninitialized member, but the transform will set
208 // data
209 #define DEFINE_STD_ARRAY_BINOP(RESULT_TYPE, LTYPE, RTYPE, OP_FUNCTION_NAME, \
210  BINARY_OP) \
211  template <size_t Dim> \
212  std::array<RESULT_TYPE, Dim> OP_FUNCTION_NAME( \
213  const std::array<LTYPE, Dim>& lhs, \
214  const std::array<RTYPE, Dim>& rhs) noexcept { \
215  std::array<RESULT_TYPE, Dim> result; /*NOLINT*/ \
216  std::transform(lhs.begin(), lhs.end(), rhs.begin(), result.begin(), \
217  BINARY_OP); \
218  return result; \
219  }
