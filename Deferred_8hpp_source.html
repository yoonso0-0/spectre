94 class deferred_assoc_state : public assoc_state<Rt> {
95  public:

97  deferred_assoc_state(const deferred_assoc_state& /*rhs*/) = delete;

101  ~deferred_assoc_state() override = default;
102

104  if (not evaluated_) {
105  apply(std::make_index_sequence<sizeof...(Args)>{});
106  evaluated_ = true;
107  }
108  return t_;
109  }
110

112 
113  void reset() noexcept override { evaluated_ = false; }
114 
115  // clang-tidy: no non-const references
116  void pack_unpack_lazy_function(PUP::er& p) noexcept override { // NOLINT
117  p | evaluated_;
118  if (evaluated_) {

121  }
122 
123  bool evaluated() const noexcept override { return evaluated_; }
124 
125  private:
126  const Fp func_;
127  std::tuple<std::decay_t<Args>...> args_;
128  mutable bool evaluated_ = false;
129  mutable Rt t_;

157 // Specialization to handle functions that return a `const Rt&`. We treat the
158 // return value as pointer to the data we actually want to have visible to us
159 // when we retrieve the data. The reason for using a pointer is because we need
160 // to be able to rebind in case the memory address of `const Rt&` changes (for
161 // example, if we point to a `GlobalCache` and we are migrated to a
162 // different node). Since lvalue references cannot be rebound, we store a
163 // pointer. The `get` function dereferences the pointer we store so that we have
164 // a const lvalue reference to work with when retrieving the data being pointed
165 // to. Dereferencing the pointer ensures that all functions that use the DataBox
166 // will be able to take a `const T& t` as input argument regardless of where the
167 // data is stored (in the DataBox or as a reference to somewhere else).
168 template <typename Rt, typename Fp, typename... Args>
169 class deferred_assoc_state<const Rt&, Fp, Args...> : public assoc_state<Rt> {
170  public:
171  explicit deferred_assoc_state(Fp f, Args... args) noexcept;
172  deferred_assoc_state(const deferred_assoc_state& /*rhs*/) = delete;
173  deferred_assoc_state& operator=(const deferred_assoc_state& /*rhs*/) = delete;
174  deferred_assoc_state(deferred_assoc_state&& /*rhs*/) = delete;
175  deferred_assoc_state& operator=(deferred_assoc_state&& /*rhs*/) = delete;
176  ~deferred_assoc_state() override = default;
177 
178  const Rt& get() const noexcept override {
179  if (not t_) {
180  apply(std::make_index_sequence<sizeof...(Args)>{});
181  }
182  return *t_;
183  }
184 
185  Rt& mutate() noexcept override { ERROR("Cannot mutate a compute tag."); }
186 
187  void reset() noexcept override { t_ = nullptr; }
188 
189  // clang-tidy: no non-const references
190  void pack_unpack_lazy_function(PUP::er& /*p*/) noexcept override {} // NOLINT
191 
192  bool evaluated() const noexcept override { return t_ != nullptr; }
193 
194  private:
195  const Fp func_;
196  std::tuple<std::decay_t<Args>...> args_;
197  mutable const Rt* t_ = nullptr;
198 
199  template <size_t... Is>
200  void apply(std::integer_sequence<size_t, Is...> /*meta*/) const noexcept {
201  t_ = &(func_(retrieve_from_deferred(std::get<Is>(args_))...));
202  }
203 };

211 /*!
212  * \ingroup DataBoxGroup
213  * \brief Provides deferred or lazy evaluation of a function or function object,
214  * as well as efficient storage of an object that is mutable.
215  *
216  * The class is similar to a std::shared_future that is able to hold and allow
217  * mutation of objects. std::shared_future allows lazy evaluation of functions
218  * but does not allow mutation of stored objects. Since mutation is only defined
219  * for storage of objects, not for lazily evaluated functions, attempts to
220  * mutate a lazily evaluated function's data is an error.
221  *
222  * To construct a Deferred for lazy evaluation use the make_deferred() function.
223  *
224  * \warning If passed a lazy function that returns a `const Rt& t` (a const
225  * lvalue reference) the underlying data stored is actually a pointer to `t`,
226  * which will be dereferenced upon retrieval. This could lead to a dangling
227  * pointer/reference if care isn't taken. The reason for this design is that:
228  * 1. We need to be able to retrieve data stored in the GlobalCache from
229  * the DataBox. The way to do this is to have a pointer to the
230  * GlobalCache inside the DataBox alongside compute items that return
231  * `const Rt&` to the GlobalCache data.
232  * 2. Functions used in the DataBox shouldn't return references, they are
233  * compute tags and should return by value. If we find an actual use-case for
234  * compute tags returning `const Rt&` where referencing behavior is undesired
235  * then this design needs to be reconsidered. It is currently the least
236  * breaking way to implement referencing DataBox members that can point to
237  * any memory.
238  *
239  * @tparam Rt the type being stored
240  */
241 template <typename Rt, typename MakeConstReference>
242 class Deferred {
243  public:
244  using value_type = std::remove_const_t<std::remove_reference_t<Rt>>;
245 
246  Deferred() = default;
247  template <typename Dummy = Rt,
248  Requires<std::is_same_v<Dummy, value_type>> = nullptr>
249  explicit Deferred(Rt t)
250  : state_(boost::make_shared<Deferred_detail::simple_assoc_state<Rt>>(
251  std::move(t))) {}
252  Deferred(const Deferred&) = default;
253  Deferred& operator=(const Deferred&) = default;
254  Deferred(Deferred&&) = default;
255  Deferred& operator=(Deferred&&) = default;
256  ~Deferred() = default;
257 
258  constexpr const value_type& get() const noexcept { return state_->get(); }
259 
260  constexpr value_type& mutate() noexcept { return state_->mutate(); }
261 
262  // clang-tidy: no non-const references
263  void pack_unpack_lazy_function(PUP::er& p) noexcept { // NOLINT
264  state_->pack_unpack_lazy_function(p);
265  }
266 
267  bool evaluated() const noexcept { return state_->evaluated(); }
268 
269  void reset() noexcept { state_->reset(); }
270 
271  explicit Deferred(
272  boost::shared_ptr<Deferred_detail::assoc_state<tmpl::conditional_t<
273  MakeConstReference::value, const value_type&, value_type>>>&&

289 /*!
290  * \ingroup DataBoxGroup
291  * \brief Create a deferred function call object
292  *
293  * If creating a Deferred with a function object the call operator of the
294  * function object must be marked `const` currently. Since the function object
295  * will only be evaluated once there currently seems to be no reason to allow
296  * mutating call operators.
297  *
298  * \example
299  * The examples below use the following functions:
300  * \snippet Test_Deferred.cpp functions_used
301  * To create a Deferred using a function object use:
302  * \snippet Test_Deferred.cpp make_deferred_with_function_object
303  * or using a regular function:
304  * \snippet Test_Deferred.cpp make_deferred_with_function
305  *
306  * It is also possible to pass Deferred objects to a deferred function call:
307  * \snippet Test_Deferred.cpp make_deferred_with_deferred_arg
308  * in which case the first function will be evaluated just before the second
309  * function is evaluated.
310  *
311  * In addition to functions that return by value, it is also possible to use
312  * functions that return by reference. The first argument of the function must
313  * then be a `gsl::not_null<Rt*>`, and can be mutated inside the function. The
314  * mutating functions are primarily useful if `Rt` performs heap allocations and
315  * is frequently recomputed in a manner where the heap allocation could be
316  * avoided.
317  *
318  * \tparam Rt the type of the object returned by the function
319  * @return Deferred object that will lazily evaluate the function
320  */
321 template <typename Rt, typename Fp, typename... Args>
322 Deferred<Rt> make_deferred(Fp f, Args&&... args) noexcept {
323  return Deferred<Rt>(boost::make_shared<Deferred_detail::deferred_assoc_state<
324  Rt, std::decay_t<Fp>, std::decay_t<Args>...>>(
325  f, std::forward<Args>(args)...));
326 }
