46 /*!
47  * \ingroup InitializationGroup
48  * \brief Initialize items related to the basic structure of the element
49  *
50  * GlobalCache:
51  * - Uses:
52  * - `domain::Tags::Domain<Dim, Frame::Inertial>`
53  * DataBox:
54  * - Uses:
55  * - `domain::Tags::InitialExtents<Dim>`
56  * - `domain::Tags::InitialFunctionsOfTime<Dim, OverrideCubicFunctionsOfTime>`
57  * - Adds:
58  * - `domain::Tags::Mesh<Dim>`
59  * - `domain::Tags::Element<Dim>`
60  * - `domain::Tags::ElementMap<Dim, Frame::Inertial>`
61  * - `domain::CoordinateMaps::Tags::CoordinateMap<Dim, Frame::Grid,
62  * Frame::Inertial>`
63  * - `domain::Tags::FunctionsOfTime`
64  * - `domain::Tags::CoordinatesMeshVelocityAndJacobiansCompute<
65  * CoordinateMap<Dim, Frame::Grid, Frame::Inertial>>`
66  * - `domain::Tags::Coordinates<Dim, Frame::Logical>`
67  * - `domain::Tags::Coordinates<Dim, Frame::Grid>`
68  * - `domain::Tags::Coordinates<Dim, Frame::Inertial>`
69  * - `domain::Tags::InverseJacobian<Dim, Frame::Logical, Frame::Grid>`
70  * - `domain::Tags::InverseJacobian<Dim, Frame::Logical, Frame::Inertial>`
71  * - `domain::Tags::DetInvJacobian<Frame::Logical, Frame::Inertial>`
72  * - `domain::Tags::MeshVelocity<Dim, Frame::Inertial>`
73  * - `domain::Tags::DivMeshVelocity`
74  * - `domain::Tags::MinimumGridSpacingCompute<Dim, Frame::Inertial>>`
75  * - Removes: nothing
76  * - Modifies: nothing
77  *
78  * \note This action relies on the `SetupDataBox` aggregated initialization
79  * mechanism, so `Actions::SetupDataBox` must be present in the `Initialization`
80  * phase action list prior to this action.
81  * \note If OverrideCubicFunctionsOfTime == true, then cubic functions
82  * of time are overriden via `read_spec_third_order_piecewise_polynomial()`
83  */
84 template <size_t Dim, bool OverrideCubicFunctionsOfTime = false>
85 struct Domain {
86  using initialization_tags =
87  tmpl::list<::domain::Tags::InitialExtents<Dim>,
88  ::domain::Tags::InitialRefinementLevels<Dim>,
89  ::domain::Tags::InitialFunctionsOfTime<
90  Dim, OverrideCubicFunctionsOfTime>>;
91  using const_global_cache_tags = tmpl::list<::domain::Tags::Domain<Dim>>;
92 
93  using simple_tags =
94  tmpl::list<::domain::Tags::Mesh<Dim>, ::domain::Tags::Element<Dim>,
95  ::domain::Tags::ElementMap<Dim, Frame::Grid>,
96  ::domain::CoordinateMaps::Tags::CoordinateMap<Dim, Frame::Grid,
97  Frame::Inertial>,
98  ::domain::Tags::FunctionsOfTime>;
99 
100  using compute_tags = tmpl::list<
101  ::domain::Tags::LogicalCoordinates<Dim>,
102  // Compute tags for Frame::Grid quantities
103  ::domain::Tags::MappedCoordinates<
104  ::domain::Tags::ElementMap<Dim, Frame::Grid>,
105  ::domain::Tags::Coordinates<Dim, Frame::Logical>>,
106  ::domain::Tags::InverseJacobianCompute<
107  ::domain::Tags::ElementMap<Dim, Frame::Grid>,
108  ::domain::Tags::Coordinates<Dim, Frame::Logical>>,
109  // Compute tags for Frame::Inertial quantities
110  ::domain::Tags::CoordinatesMeshVelocityAndJacobiansCompute<
111  ::domain::CoordinateMaps::Tags::CoordinateMap<Dim, Frame::Grid,
112  Frame::Inertial>>,
113 
114  ::domain::Tags::InertialFromGridCoordinatesCompute<Dim>,
115  ::domain::Tags::ElementToInertialInverseJacobian<Dim>,
116  ::domain::Tags::DetInvJacobianCompute<Dim, Frame::Logical,
117  Frame::Inertial>,
118  ::domain::Tags::InertialMeshVelocityCompute<Dim>,
119  evolution::domain::Tags::DivMeshVelocityCompute<Dim>,
120  // Compute tags for other mesh quantities
121  ::domain::Tags::MinimumGridSpacingCompute<Dim, Frame::Inertial>>;
122 
123  template <
124  typename DataBox, typename... InboxTags, typename Metavariables,
125  typename ActionList, typename ParallelComponent,
126  Requires<tmpl::all<initialization_tags,
127  tmpl::bind<db::tag_is_retrievable, tmpl::_1,
128  tmpl::pin<DataBox>>>::value> = nullptr>
129  static auto apply(DataBox& box,
130  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
131  const Parallel::GlobalCache<Metavariables>& /*cache*/,
132  const ElementId<Dim>& array_index,
133  const ActionList /*meta*/,
134  const ParallelComponent* const /*meta*/) noexcept {
135  const auto& initial_extents =
136  db::get<::domain::Tags::InitialExtents<Dim>>(box);
137  const auto& initial_refinement =
138  db::get<::domain::Tags::InitialRefinementLevels<Dim>>(box);
139  const auto& domain = db::get<::domain::Tags::Domain<Dim>>(box);
140 
141  const ElementId<Dim> element_id{array_index};
142  const auto& my_block = domain.blocks()[element_id.block_id()];
143  Mesh<Dim> mesh = ::domain::Initialization::create_initial_mesh(
144  initial_extents, element_id, Spectral::Quadrature::GaussLobatto);
145  Element<Dim> element = ::domain::Initialization::create_initial_element(
146  element_id, my_block, initial_refinement);
147  ElementMap<Dim, Frame::Grid> element_map{
148  element_id, my_block.is_time_dependent()
149  ? my_block.moving_mesh_logical_to_grid_map().get_clone()
150  : my_block.stationary_map().get_to_grid_frame()};
151 
152  const auto& initial_functions_of_time =
153  db::get<::domain::Tags::InitialFunctionsOfTime<Dim>>(box);
154 
155  std::unordered_map<
156  std::string, std::unique_ptr<::domain::FunctionsOfTime::FunctionOfTime>>
157  functions_of_time{};
158  for (const auto& name_and_function : initial_functions_of_time) {
159  functions_of_time.insert(std::make_pair(
160  name_and_function.first, name_and_function.second->get_clone()));
161  }
162 
163  std::unique_ptr<
164  ::domain::CoordinateMapBase<Frame::Grid, Frame::Inertial, Dim>>
165  grid_to_inertial_map;
166  if (my_block.is_time_dependent()) {
167  grid_to_inertial_map =
168  my_block.moving_mesh_grid_to_inertial_map().get_clone();
169  } else {
170  grid_to_inertial_map =
171  ::domain::make_coordinate_map_base<Frame::Grid, Frame::Inertial>(
172  ::domain::CoordinateMaps::Identity<Dim>{});
173  }
174  ::Initialization::mutate_assign<simple_tags>(
175  make_not_null(&box), std::move(mesh), std::move(element),
176  std::move(element_map), std::move(grid_to_inertial_map),
177  std::move(functions_of_time));
178 
179  return std::make_tuple(std::move(box));
180  }
181 
182  template <
183  typename DataBox, typename... InboxTags, typename Metavariables,
184  typename ArrayIndex, typename ActionList, typename ParallelComponent,
185  Requires<not tmpl::all<initialization_tags,
186  tmpl::bind<db::tag_is_retrievable, tmpl::_1,
187  tmpl::pin<DataBox>>>::value> = nullptr>
188  static std::tuple<DataBox&&> apply(
189  DataBox& /*box*/, const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
190  const Parallel::GlobalCache<Metavariables>& /*cache*/,
191  const ArrayIndex& /*array_index*/, ActionList /*meta*/,
192  const ParallelComponent* const /*meta*/) noexcept {
193  ERROR(
194  "Dependencies not fulfilled. Did you forget to terminate the phase "
195  "after removing options?");
196  }
197 };
198 } // namespace Initialization
