55 /*!
56  * \ingroup ObserversGroup
57  * \brief Send reduction data to the observer group.
58  *
59  * Once everything at a specific `ObservationId` has been contributed to the
60  * reduction, the groups reduce to their local nodegroup.
61  *
62  * The caller of this Action (which is to be invoked on the Observer parallel
63  * component) must pass in an `observation_id` used to uniquely identify the
64  * observation in time, the name of the `h5::Dat` subfile in the HDF5 file (e.g.
65  * `/element_data`, where the slash is important), a `std::vector<std::string>`
66  * of names of the quantities being reduced (e.g. `{"Time", "L1ErrorDensity",
67  * "L2ErrorDensity"}`), and the `Parallel::ReductionData` that holds the
68  * `ReductionDatums` containing info on how to do the reduction.
69  *
70  * The observer components need to know all expected reduction data types by
71  * compile-time, so they rely on the
72  * `Metavariables::observed_reduction_data_tags` alias to collect them in one
73  * place. To this end, each Action that contributes reduction data must expose
74  * the type alias as:
75  *
76  * \snippet ObserverHelpers.hpp make_reduction_data_tags
77  *
78  * Then, in the `Metavariables` collect them from all observing Actions using
79  * the `observers::collect_reduction_data_tags` metafunction.
80  */
81 struct ContributeReductionData {
82  template <typename ParallelComponent, typename DbTagsList,
83  typename Metavariables, typename ArrayIndex, typename... Ts>
84  static auto apply(db::DataBox<DbTagsList>& box,
85  Parallel::GlobalCache<Metavariables>& cache,
86  const ArrayIndex& array_index,
87  const observers::ObservationId& observation_id,
88  const ArrayComponentId& sender_array_id,
89  const std::string& subfile_name,
90  const std::vector<std::string>& reduction_names,
91  Parallel::ReductionData<Ts...>&& reduction_data) noexcept {
92  if constexpr (tmpl::list_contains_v<DbTagsList,
93  Tags::ReductionData<Ts...>> and
94  tmpl::list_contains_v<DbTagsList,
95  Tags::ReductionDataNames<Ts...>> and
96  tmpl::list_contains_v<DbTagsList,
97  Tags::ContributorsOfReductionData>) {
98  db::mutate<Tags::ReductionData<Ts...>, Tags::ReductionDataNames<Ts...>,
99  Tags::ContributorsOfReductionData>(
100  make_not_null(&box),
101  [&array_index, &cache, &observation_id,
102  reduction_data = std::move(reduction_data), &reduction_names,
103  &sender_array_id, &subfile_name](
104  const gsl::not_null<std::unordered_map<
105  ObservationId, Parallel::ReductionData<Ts...>>*>
106  reduction_data_map,
107  const gsl::not_null<
108  std::unordered_map<ObservationId, std::vector<std::string>>*>
109  reduction_names_map,
110  const gsl::not_null<std::unordered_map<
111  ObservationId, std::unordered_set<ArrayComponentId>>*>
112  reduction_observers_contributed,
113  const std::unordered_map<ObservationKey,
114  std::unordered_set<ArrayComponentId>>&
115  observations_registered) mutable noexcept {
116  ASSERT(observations_registered.find(
117  observation_id.observation_key()) !=
118  observations_registered.end(),
119  "Couldn't find registration key "
120  << observation_id.observation_key()
121  << " in the registered observers.");
122 
123  auto& contributed_array_ids =
124  (*reduction_observers_contributed)[observation_id];
125  if (UNLIKELY(contributed_array_ids.find(sender_array_id) !=
126  contributed_array_ids.end())) {
127  ERROR("Already received reduction data to observation id "
128  << observation_id << " from array component id "
129  << sender_array_id);
130  }
131  contributed_array_ids.insert(sender_array_id);
132 
133  if (reduction_data_map->count(observation_id) == 0) {
134  reduction_data_map->emplace(observation_id,
135  std::move(reduction_data));
136  reduction_names_map->emplace(observation_id, reduction_names);
137  } else {
138  if (UNLIKELY(reduction_names_map->at(observation_id) !=
139  reduction_names)) {
140  using ::operator<<;
141  ERROR("Reduction names differ at ObservationId "
142  << observation_id << " with the expected names being "
143  << reduction_names_map->at(observation_id)
144  << " and the received names being " << reduction_names);
145  }
146  reduction_data_map->operator[](observation_id)
147  .combine(std::move(reduction_data));
148  }
149 
150  // Check if we have received all reduction data from the registered
151  // elements. If so, we reduce to the local ObserverWriter nodegroup.
152  if (UNLIKELY(
153  contributed_array_ids.size() ==
154  observations_registered.at(observation_id.observation_key())
155  .size())) {
156  auto& local_writer = *Parallel::get_parallel_component<
157  ObserverWriter<Metavariables>>(cache)
158  .ckLocalBranch();
159  Parallel::threaded_action<
160  ThreadedActions::CollectReductionDataOnNode>(
161  local_writer, observation_id,
162  ArrayComponentId{
163  std::add_pointer_t<ParallelComponent>{nullptr},
164  Parallel::ArrayIndex<ArrayIndex>(array_index)},
165  subfile_name, (*reduction_names_map)[observation_id],
166  std::move((*reduction_data_map)[observation_id]));
167  reduction_data_map->erase(observation_id);
168  reduction_names_map->erase(observation_id);
169  reduction_observers_contributed->erase(observation_id);
170  }
171  },
172  db::get<Tags::ExpectedContributorsForObservations>(box));
173  } else {
174  ERROR("Could not find the tag "
175  << pretty_type::get_name<Tags::ReductionData<Ts...>>() << ' '
176  << pretty_type::get_name<Tags::ReductionDataNames<Ts...>>()
177  << " or "
178  << pretty_type::get_name<Tags::ContributorsOfReductionData>()
179  << " in the DataBox.");
180  }
181  }
182 };

187  * \brief Gathers all the reduction data from all processing elements/cores on a
188  * node.
189  */
190 struct CollectReductionDataOnNode {
191  public:
192  template <typename ParallelComponent, typename DbTagsList,
193  typename Metavariables, typename ArrayIndex,
194  typename... ReductionDatums>
195  static void apply(db::DataBox<DbTagsList>& box,
196  Parallel::GlobalCache<Metavariables>& cache,
197  const ArrayIndex& /*array_index*/,
198  const gsl::not_null<Parallel::NodeLock*> node_lock,
199  const observers::ObservationId& observation_id,
200  ArrayComponentId observer_group_id,
201  const std::string& subfile_name,
202  std::vector<std::string>&& reduction_names,
203  Parallel::ReductionData<ReductionDatums...>&&
204  received_reduction_data) noexcept {
205  if constexpr (tmpl::list_contains_v<
206  DbTagsList, Tags::ReductionData<ReductionDatums...>> and
207  tmpl::list_contains_v<DbTagsList, Tags::ReductionDataNames<
208  ReductionDatums...>> and
209  tmpl::list_contains_v<DbTagsList, Tags::ReductionDataLock>) {
210  // The below gymnastics with pointers is done in order to minimize the
211  // time spent locking the entire node, which is necessary because the
212  // DataBox does not allow any functions calls, both get and mutate, during
213  // a mutate. This design choice in DataBox is necessary to guarantee a
214  // consistent state throughout mutation. Here, however, we need to be
215  // reasonable efficient in parallel and so we manually guarantee that
216  // consistent state. To this end, we create pointers and assign to them
217  // the data in the DataBox which is guaranteed to be pointer stable. The
218  // data itself is guaranteed to be stable inside the ReductionDataLock.
219  std::unordered_map<observers::ObservationId,
220  Parallel::ReductionData<ReductionDatums...>>*
221  reduction_data = nullptr;
222  std::unordered_map<ObservationId, std::vector<std::string>>*
223  reduction_names_map = nullptr;
224  std::unordered_map<observers::ObservationId,
225  std::unordered_set<ArrayComponentId>>*
226  reduction_observers_contributed = nullptr;
227  Parallel::NodeLock* reduction_data_lock = nullptr;
228  size_t observations_registered_with_id =
229  std::numeric_limits<size_t>::max();
230 
231  node_lock->lock();
232  db::mutate<Tags::ReductionData<ReductionDatums...>,
233  Tags::ReductionDataNames<ReductionDatums...>,
234  Tags::ContributorsOfReductionData, Tags::ReductionDataLock>(
235  make_not_null(&box),
236  [&reduction_data, &reduction_names_map,
237  &reduction_observers_contributed, &reduction_data_lock,
238  &observation_id, &observer_group_id,
239  &observations_registered_with_id](
240  const gsl::not_null<std::unordered_map<
241  observers::ObservationId,
242  Parallel::ReductionData<ReductionDatums...>>*>
243  reduction_data_ptr,
244  const gsl::not_null<
245  std::unordered_map<ObservationId, std::vector<std::string>>*>
246  reduction_names_map_ptr,
247  const gsl::not_null<
248  std::unordered_map<observers::ObservationId,
249  std::unordered_set<ArrayComponentId>>*>
250  reduction_observers_contributed_ptr,
251  const gsl::not_null<Parallel::NodeLock*> reduction_data_lock_ptr,
252  const std::unordered_map<ObservationKey,
253  std::unordered_set<ArrayComponentId>>&
254  observations_registered) noexcept {
255  const ObservationKey& key{observation_id.observation_key()};
256  const auto& registered_group_ids = observations_registered.at(key);
257  if (UNLIKELY(registered_group_ids.find(observer_group_id) ==
258  registered_group_ids.end())) {
259  ERROR("The observer group id "
260  << observer_group_id
261  << " was not registered for the observation id "
262  << observation_id);
263  }
264  reduction_data = &*reduction_data_ptr;
265  reduction_names_map = &*reduction_names_map_ptr;
266  reduction_observers_contributed =
267  &*reduction_observers_contributed_ptr;
268  reduction_data_lock = &*reduction_data_lock_ptr;
269  observations_registered_with_id =
270  observations_registered.at(key).size();
271  },
272  db::get<Tags::ExpectedContributorsForObservations>(box));
273  node_lock->unlock();
274 
275  ASSERT(
276  observations_registered_with_id != std::numeric_limits<size_t>::max(),
277  "Failed to set observations_registered_with_id when mutating the "
278  "DataBox. This is a bug in the code.");
279 
280  // Now that we've retrieved pointers to the data in the DataBox we wish to
281  // manipulate, lock the data and manipulate it.
282  reduction_data_lock->lock();
283  auto& contributed_group_ids =
284  (*reduction_observers_contributed)[observation_id];
285 
286  if (UNLIKELY(contributed_group_ids.find(observer_group_id) !=
287  contributed_group_ids.end())) {
288  ERROR("Already received reduction data to observation id "
289  << observation_id << " from array component id "
290  << observer_group_id);
291  }
292  contributed_group_ids.insert(observer_group_id);
293 
294  if (reduction_data->find(observation_id) == reduction_data->end()) {
295  // This Action has been called for the first time,
296  // so all we need to do is move the input data to the
297  // reduction_data in the DataBox.
298  reduction_data->operator[](observation_id) =
299  std::move(received_reduction_data);
300  } else {
301  // This Action is being called at least the second time
302  // (but not the final time if on node 0).
303  reduction_data->at(observation_id)
304  .combine(std::move(received_reduction_data));
305  }
306 
307  if (UNLIKELY(reduction_names.empty())) {
308  ERROR(
309  "The reduction names, which is a std::vector of the names of "
310  "the columns in the file, must be non-empty.");
311  }
312  if (auto current_names = reduction_names_map->find(observation_id);
313  current_names == reduction_names_map->end()) {
314  reduction_names_map->emplace(observation_id,
315  std::move(reduction_names));
316  } else if (UNLIKELY(current_names->second != reduction_names)) {
317  ERROR(
318  "The reduction names passed in must match the currently "
319  "known reduction names.");
320  }
321 
322  // Check if we have received all reduction data from the Observer
323  // group. If so we reduce to node 0 for writing to disk. We use a bool
324  // `send_data` to allow us to defer the send call until after we've
325  // unlocked the lock.
326  bool send_data = false;
327  if (reduction_observers_contributed->at(observation_id).size() ==
328  observations_registered_with_id) {
329  send_data = true;
330  // We intentionally move the data out of the map and erase it
331  // before call `WriteReductionData` since if the call to
332  // `WriteReductionData` is inlined and we erase data from the maps
333  // afterward we would lose data.
334  reduction_names =
335  std::move(reduction_names_map->operator[](observation_id));
336  received_reduction_data =
337  std::move(reduction_data->operator[](observation_id));
338  reduction_observers_contributed->erase(observation_id);
339  reduction_data->erase(observation_id);
340  reduction_names_map->erase(observation_id);
341  }
342  reduction_data_lock->unlock();
343 
344  if (send_data) {
345  Parallel::threaded_action<WriteReductionData>(
346  Parallel::get_parallel_component<ObserverWriter<Metavariables>>(
347  cache)[0],
348  observation_id, static_cast<size_t>(Parallel::my_node()),
349  subfile_name,
350  // NOLINTNEXTLINE(bugprone-use-after-move)
351  std::move(reduction_names), std::move(received_reduction_data));
352  }
353  } else {
354  (void)node_lock;
355  (void)observer_group_id;
356  ERROR("Could not find one of the tags: "
357  << pretty_type::get_name<Tags::ReductionData<ReductionDatums...>>()
358  << ' '
359  << pretty_type::get_name<
360  Tags::ReductionDataNames<ReductionDatums...>>()
361  << " or Tags::ReductionDataLock");
362  }
363  }
364 };

370 struct WriteReductionData {
371  private:
372  static void append_to_reduction_data(
373  const gsl::not_null<std::vector<double>*> all_reduction_data,
374  const double t) noexcept {
375  all_reduction_data->push_back(t);
376  }
377 
378  static void append_to_reduction_data(
379  const gsl::not_null<std::vector<double>*> all_reduction_data,
380  const std::vector<double>& t) noexcept {
381  all_reduction_data->insert(all_reduction_data->end(), t.begin(), t.end());
382  }
383 
384  template <typename... Ts, size_t... Is>
385  static void write_data(const std::string& subfile_name,
386  std::vector<std::string>&& legend,
387  std::tuple<Ts...>&& data,
388  const std::string& file_prefix,
389  std::index_sequence<Is...> /*meta*/) noexcept {
390  static_assert(sizeof...(Ts) > 0,
391  "Must be reducing at least one piece of data");
392  std::vector<double> data_to_append{};
393  EXPAND_PACK_LEFT_TO_RIGHT(
394  append_to_reduction_data(&data_to_append, std::get<Is>(data)));
395 
396  h5::H5File<h5::AccessType::ReadWrite> h5file(file_prefix + ".h5", true);
397  constexpr size_t version_number = 0;
398  auto& time_series_file = h5file.try_insert<h5::Dat>(
399  subfile_name, std::move(legend), version_number);
400  time_series_file.append(data_to_append);
401  }

436  reduction_data = nullptr;

438  reduction_names_map = nullptr;

459  const gsl::not_null<

475  nodes_registered_for_reductions.at(key);

477  if (UNLIKELY(registered_nodes.find(sender_node_number) ==
478  registered_nodes.end())) {
479  ERROR("Node " << sender_node_number

494 
495  ASSERT(
496  observations_registered_with_id != std::numeric_limits<size_t>::max(),
497  "Failed to set observations_registered_with_id when mutating the "
498  "DataBox. This is a bug in the code.");
499 
500  // Now that we've retrieved pointers to the data in the DataBox we wish to
501  // manipulate, lock the data and manipulate it.
502  reduction_data_lock->lock();
503  auto& nodes_contributed_to_observation =
504  (*nodes_contributed)[observation_id];
505  if (nodes_contributed_to_observation.find(sender_node_number) !=
506  nodes_contributed_to_observation.end()) {
507  ERROR("Already received reduction data at observation id "
508  << observation_id << " from node " << sender_node_number);
509  }
510  nodes_contributed_to_observation.insert(sender_node_number);
511 
512  if (UNLIKELY(reduction_names.empty())) {
513  ERROR(
514  "The reduction names, which is a std::vector of the names of "
515  "the columns in the file, must be non-empty.");
516  }
517  if (auto current_names = reduction_names_map->find(observation_id);
518  current_names == reduction_names_map->end()) {
519  reduction_names_map->emplace(observation_id,
520  std::move(reduction_names));
521  } else if (UNLIKELY(current_names->second != reduction_names)) {
522  using ::operator<<;
523  ERROR(
524  "The reduction names passed in must match the currently "
525  "known reduction names. Current ones are "
526  << current_names->second << " while the received are "
527  << reduction_names);
528  }
529 
530  if (reduction_data->find(observation_id) == reduction_data->end()) {
531  // This Action has been called for the first time,
532  // so all we need to do is move the input data to the
533  // reduction_data in the DataBox.
534  reduction_data->operator[](observation_id) =
535  std::move(received_reduction_data);
536  } else {
537  // This Action is being called at least the second time
538  // (but not the final time if on node 0).
539  reduction_data->at(observation_id)
540  .combine(std::move(received_reduction_data));
541  }
542 
543  // We use a bool `write_to_disk` to allow us to defer the data writing
544  // until after we've unlocked the lock. For the same reason, we move the
545  // final, reduced result into `received_reduction_data` and
546  // `reduction_names`.
547  bool write_to_disk = false;
548  if (nodes_contributed_to_observation.size() ==
549  observations_registered_with_id) {
550  write_to_disk = true;
551  received_reduction_data =
552  std::move(reduction_data->operator[](observation_id));
553  reduction_names =
